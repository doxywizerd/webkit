98 // Work around a bug in Win, where atan2(+-infinity, +-infinity) yields NaN instead of specific values.
99 extern "C" inline double wtf_atan2(double x, double y)
100 {
101  double posInf = std::numeric_limits<double>::infinity();
102  double negInf = -std::numeric_limits<double>::infinity();
103  double nan = std::numeric_limits<double>::quiet_NaN();
104 
105  double result = nan;
106 
107  if (x == posInf && y == posInf)
108  result = piOverFourDouble;
109  else if (x == posInf && y == negInf)
110  result = 3 * piOverFourDouble;
111  else if (x == negInf && y == posInf)
112  result = -piOverFourDouble;
113  else if (x == negInf && y == negInf)
114  result = -3 * piOverFourDouble;
115  else
116  result = ::atan2(x, y);
117 
118  return result;
119 }

291 // decompose 'number' to its sign, exponent, and mantissa components.
292 // The result is interpreted as:
293 // (sign ? -1 : 1) * pow(2, exponent) * (mantissa / (1 << 52))
294 inline void decomposeDouble(double number, bool& sign, int32_t& exponent, uint64_t& mantissa)
295 {
296  ASSERT(std::isfinite(number));
297 
298  sign = std::signbit(number);
299 
300  uint64_t bits = WTF::bitwise_cast<uint64_t>(number);
301  exponent = (static_cast<int32_t>(bits >> 52) & 0x7ff) - 0x3ff;
302  mantissa = bits & 0xFFFFFFFFFFFFFull;
303 
304  // Check for zero/denormal values; if so, adjust the exponent,
305  // if not insert the implicit, omitted leading 1 bit.
306  if (exponent == -0x3ff)
307  exponent = mantissa ? -0x3fe : 0;
308  else
309  mantissa |= 0x10000000000000ull;
310 }

312 // Calculate d % 2^{64}.
313 inline void doubleToInteger(double d, unsigned long long& value)
314 {
315  if (std::isnan(d) || std::isinf(d))
316  value = 0;
317  else {
318  // -2^{64} < fmodValue < 2^{64}.
319  double fmodValue = fmod(trunc(d), std::numeric_limits<unsigned long long>::max() + 1.0);
320  if (fmodValue >= 0) {
321  // 0 <= fmodValue < 2^{64}.
322  // 0 <= value < 2^{64}. This cast causes no loss.
323  value = static_cast<unsigned long long>(fmodValue);
324  } else {
325  // -2^{64} < fmodValue < 0.
326  // 0 < fmodValueInUnsignedLongLong < 2^{64}. This cast causes no loss.
327  unsigned long long fmodValueInUnsignedLongLong = static_cast<unsigned long long>(-fmodValue);
328  // -1 < (std::numeric_limits<unsigned long long>::max() - fmodValueInUnsignedLongLong) < 2^{64} - 1.
329  // 0 < value < 2^{64}.
330  value = std::numeric_limits<unsigned long long>::max() - fmodValueInUnsignedLongLong + 1;
331  }
332  }

350 inline unsigned fastLog2(unsigned i)
351 {
352  unsigned log2 = 0;
353  if (i & (i - 1))
354  log2 += 1;
355  if (i >> 16)
356  log2 += 16, i >>= 16;
357  if (i >> 8)
358  log2 += 8, i >>= 8;
359  if (i >> 4)
360  log2 += 4, i >>= 4;
361  if (i >> 2)
362  log2 += 2, i >>= 2;
363  if (i >> 1)
364  log2 += 1;
365  return log2;

387 // Floating point numbers comparison:
388 // u is "essentially equal" [1][2] to v if: | u - v | / |u| <= e and | u - v | / |v| <= e
389 //
390 // [1] Knuth, D. E. "Accuracy of Floating Point Arithmetic." The Art of Computer Programming. 3rd ed. Vol. 2.
391 // Boston: Addison-Wesley, 1998. 229-45.
392 // [2] http://www.boost.org/doc/libs/1_34_0/libs/test/doc/components/test_tools/floating_point_comparison.html
393 template <typename T>
394 inline typename std::enable_if<std::is_floating_point<T>::value, bool>::type areEssentiallyEqual(T u, T v, T epsilon = std::numeric_limits<T>::epsilon())
395 {
396  if (u == v)
397  return true;
398 
399  const T delta = std::abs(u - v);
400  return safeFPDivision(delta, std::abs(u)) <= epsilon && safeFPDivision(delta, std::abs(v)) <= epsilon;
401 }
