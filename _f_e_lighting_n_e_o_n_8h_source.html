94 inline void FELighting::platformApplyNeon(LightingData& data, LightSource::PaintingData& paintingData)
95 {
96  FELightingFloatArgumentsForNeon floatArguments __attribute__((__aligned__(16)));
97  FELightingPaintingDataForNeon neonData = {
98  data.pixels->data(),
99  1,
100  data.widthDecreasedByOne - 1,
101  data.heightDecreasedByOne - 1,
102  0,
103  0,
104  0,
105  &floatArguments,
106  feLightingConstantsForNeon()
107  };
108 
109  // Set light source arguments.
110  floatArguments.constOne = 1;
111 
112  floatArguments.colorRed = m_lightingColor.red();
113  floatArguments.colorGreen = m_lightingColor.green();
114  floatArguments.colorBlue = m_lightingColor.blue();
115  floatArguments.padding4 = 0;
116 
117  if (m_lightSource->type() == LS_POINT) {
118  neonData.flags |= FLAG_POINT_LIGHT;
119  PointLightSource* pointLightSource = static_cast<PointLightSource*>(m_lightSource.get());
120  floatArguments.lightX = pointLightSource->position().x();
121  floatArguments.lightY = pointLightSource->position().y();
122  floatArguments.lightZ = pointLightSource->position().z();
123  floatArguments.padding2 = 0;
124  } else if (m_lightSource->type() == LS_SPOT) {
125  neonData.flags |= FLAG_SPOT_LIGHT;
126  SpotLightSource* spotLightSource = static_cast<SpotLightSource*>(m_lightSource.get());
127  floatArguments.lightX = spotLightSource->position().x();
128  floatArguments.lightY = spotLightSource->position().y();
129  floatArguments.lightZ = spotLightSource->position().z();
130  floatArguments.padding2 = 0;
131 
132  floatArguments.directionX = paintingData.directionVector.x();
133  floatArguments.directionY = paintingData.directionVector.y();
134  floatArguments.directionZ = paintingData.directionVector.z();
135  floatArguments.padding3 = 0;
136 
137  floatArguments.coneCutOffLimit = paintingData.coneCutOffLimit;
138  floatArguments.coneFullLight = paintingData.coneFullLight;
139  floatArguments.coneCutOffRange = paintingData.coneCutOffLimit - paintingData.coneFullLight;
140  neonData.coneExponent = getPowerCoefficients(spotLightSource->specularExponent());
141  if (spotLightSource->specularExponent() == 1)
142  neonData.flags |= FLAG_CONE_EXPONENT_IS_1;
143  } else {
144  ASSERT(m_lightSource->type() == LS_DISTANT);
145  floatArguments.lightX = paintingData.lightVector.x();
146  floatArguments.lightY = paintingData.lightVector.y();
147  floatArguments.lightZ = paintingData.lightVector.z();
148  floatArguments.padding2 = 1;
149  }
150 
151  // Set lighting arguments.
152  floatArguments.surfaceScale = data.surfaceScale;
153  floatArguments.minusSurfaceScaleDividedByFour = -data.surfaceScale / 4;
154  if (m_lightingType == FELighting::DiffuseLighting)
155  floatArguments.diffuseConstant = m_diffuseConstant;
156  else {
157  neonData.flags |= FLAG_SPECULAR_LIGHT;
158  floatArguments.diffuseConstant = m_specularConstant;
159  neonData.specularExponent = getPowerCoefficients(m_specularExponent);
160  if (m_specularExponent == 1)
161  neonData.flags |= FLAG_SPECULAR_EXPONENT_IS_1;
162  }
163  if (floatArguments.diffuseConstant == 1)
164  neonData.flags |= FLAG_DIFFUSE_CONST_IS_1;
165 
166  int optimalThreadNumber = ((data.widthDecreasedByOne - 1) * (data.heightDecreasedByOne - 1)) / s_minimalRectDimension;
167  if (optimalThreadNumber > 1) {
168  // Initialize parallel jobs
169  ParallelJobs<FELightingPaintingDataForNeon> parallelJobs(&WebCore::FELighting::platformApplyNeonWorker, optimalThreadNumber);
170 
171  // Fill the parameter array
172  int job = parallelJobs.numberOfJobs();
173  if (job > 1) {
174  int yStart = 1;
175  int yStep = (data.heightDecreasedByOne - 1) / job;
176  for (--job; job >= 0; --job) {
177  FELightingPaintingDataForNeon& params = parallelJobs.parameter(job);
178  params = neonData;
179  params.yStart = yStart;
180  params.pixels += (yStart - 1) * (data.widthDecreasedByOne + 1) * 4;
181  if (job > 0) {
182  params.absoluteHeight = yStep;
183  yStart += yStep;
184  } else
185  params.absoluteHeight = data.heightDecreasedByOne - yStart;
186  }
187  parallelJobs.execute();
188  return;
189  }
190  }
191 
192  neonDrawLighting(&neonData);
193 }
