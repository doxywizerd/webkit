233 template<typename CharacterType> inline bool isASCIIAlphaCaselessEqual(CharacterType inputCharacter, char expectedASCIILowercaseLetter)
234 {
235  // Name of this argument says this must be a lowercase letter, but it can actually be:
236  // - a lowercase letter
237  // - a numeric digit
238  // - a space
239  // - punctuation in the range 0x21-0x3F, including "-", "/", and "+"
240  // It cannot be:
241  // - an uppercase letter
242  // - a non-ASCII character
243  // - other punctuation, such as underscore and backslash
244  // - a control character such as "\n"
245  // FIXME: Would be nice to make both the function name and expectedASCIILowercaseLetter argument name clearer.
246  ASSERT(toASCIILowerUnchecked(expectedASCIILowercaseLetter) == expectedASCIILowercaseLetter);
247  return LIKELY(toASCIILowerUnchecked(inputCharacter) == expectedASCIILowercaseLetter);
248 }
