154  template <typename Functor> void forEachRelatedCodeBlock(Functor&& functor)
155  {
156  Functor f(std::forward<Functor>(functor));
157  Vector<CodeBlock*, 4> codeBlocks;
158  codeBlocks.append(this);
159 
160  while (!codeBlocks.isEmpty()) {
161  CodeBlock* currentCodeBlock = codeBlocks.takeLast();
162  f(currentCodeBlock);
163 
164  if (CodeBlock* alternative = currentCodeBlock->alternative())
165  codeBlocks.append(alternative);
166  if (CodeBlock* osrEntryBlock = currentCodeBlock->specialOSREntryBlockOrNull())
167  codeBlocks.append(osrEntryBlock);
168  }
169  }
