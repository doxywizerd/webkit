209 inline float RunResolver::Run::computeBaselinePosition() const
210 {
211  auto& resolver = m_iterator.resolver();
212  auto offset = resolver.m_borderAndPaddingBefore + resolver.m_lineHeight * lineIndex();
213  if (!resolver.m_layout.hasLineStruts())
214  return offset + resolver.m_baseline;
215  for (auto& strutEntry : resolver.m_layout.struts()) {
216  if (strutEntry.lineBreak > lineIndex())
217  break;
218  offset += strutEntry.offset;
219  }
220  return offset + resolver.m_baseline;
221 }
