46 // Class representation of an ICE candidate.

51 // An instance can be created by CreateIceCandidate.
52 class IceCandidateInterface {

56  // m= section in SDP, which identifies the m= section.
57  virtual std::string sdp_mid() const = 0;

59  // is assocated with. Needed when an endpoint doesn't support MIDs.

61  // Only for use internally.
62  virtual const cricket::Candidate& candidate() const = 0;

64  // TODO(zhihuang): Remove the default implementation once the subclasses
65  // implement this method.

67  // Creates a SDP-ized form of this candidate.
