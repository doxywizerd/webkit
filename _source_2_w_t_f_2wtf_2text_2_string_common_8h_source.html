178  bool isEqual = false;
179  uint32_t aValue;
180  uint32_t bValue;
181  asm("subs %[length], #4\n"
182  "blo 2f\n"
183 
184  "0:\n" // Label 0 = Start of loop over 32 bits.
185  "ldr %[aValue], [%[a]], #4\n"

502 template <typename SearchCharacterType, typename MatchCharacterType>
503 ALWAYS_INLINE static size_t findInner(const SearchCharacterType* searchCharacters, const MatchCharacterType* matchCharacters, unsigned index, unsigned searchLength, unsigned matchLength)
504 {
505  // Optimization: keep a running hash of the strings,
506  // only call equal() if the hashes match.
507 
508  // delta is the number of additional times to test; delta == 0 means test only once.
509  unsigned delta = searchLength - matchLength;
510 
511  unsigned searchHash = 0;
512  unsigned matchHash = 0;
513 
514  for (unsigned i = 0; i < matchLength; ++i) {
515  searchHash += searchCharacters[i];
516  matchHash += matchCharacters[i];
517  }
518 
519  unsigned i = 0;
520  // keep looping until we match
521  while (searchHash != matchHash || !equal(searchCharacters + i, matchCharacters, matchLength)) {
522  if (i == delta)
523  return notFound;
524  searchHash += searchCharacters[i + matchLength];
525  searchHash -= searchCharacters[i];
526  ++i;
527  }
528  return index + i;
529 }
