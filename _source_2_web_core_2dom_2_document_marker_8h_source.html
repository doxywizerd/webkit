34 // A range of a node within a document that is "marked", such as the range of a misspelled word.
35 // It optionally includes a description that could be displayed in the user interface.
36 // It also optionally includes a flag specifying whether the match is active, which is ignored
37 // for all types other than type TextMatch.
38 class DocumentMarker {
39 public:
40  enum MarkerType {
41  Spelling = 1 << 0,
42  Grammar = 1 << 1,
43  TextMatch = 1 << 2,
44  // Text has been modified by spell correction, reversion of spell correction or other type of substitution.
45  // On some platforms, this prevents the text from being autocorrected again. On post Snow Leopard Mac OS X,
46  // if a Replacement marker contains non-empty description, a reversion UI will be shown.
47  Replacement = 1 << 3,
48  // Renderer needs to add underline indicating that the text has been modified by spell
49  // correction. Text with Replacement marker doesn't necessarily has CorrectionIndicator
50  // marker. For instance, after some text has been corrected, it will have both Replacement
51  // and CorrectionIndicator. However, if user further modifies such text, we would remove
52  // CorrectionIndicator marker, but retain Replacement marker.
53  CorrectionIndicator = 1 << 4,
54  // Correction suggestion has been offered, but got rejected by user.
55  RejectedCorrection = 1 << 5,
56  // Text has been modified by autocorrection. The description of this marker is the original text before autocorrection.
57  Autocorrected = 1 << 6,
58  // On some platforms, this prevents the text from being spellchecked again.
59  SpellCheckingExemption = 1 << 7,
60  // This marker indicates user has deleted an autocorrection starting at the end of the
61  // range that bears this marker. In some platforms, if the user later inserts the same original
62  // word again at this position, it will not be autocorrected again. The description of this
63  // marker is the original word before autocorrection was applied.
64  DeletedAutocorrection = 1 << 8,
65  // This marker indicates that the range of text spanned by the marker is entered by voice dictation,
66  // and it has alternative text.
67  DictationAlternatives = 1 << 9,
68 #if ENABLE(TELEPHONE_NUMBER_DETECTION)
69  TelephoneNumber = 1 << 10,
70 #endif
71 #if PLATFORM(IOS)
72  // FIXME: iOS should share the same dictation mark system with the other platforms <rdar://problem/9431249>.
73  DictationPhraseWithAlternatives = 1 << 11,
74  DictationResult = 1 << 12,
75 #endif
76  // This marker indicates that the user has selected a text candidate.
77  AcceptedCandidate = 1 << 13,
78  };

82  // This constructor is left implicit to allow conversion from result of a bit-wise or of enumeration values.
83  MarkerTypes(unsigned mask) : m_mask(mask) { }

85  bool contains(MarkerType type) const { return m_mask & type; }
86  bool intersects(MarkerTypes types) const { return m_mask & types.m_mask; }
87  bool operator==(MarkerTypes other) const { return m_mask == other.m_mask; }
88 
89  void add(MarkerTypes types) { m_mask |= types.m_mask; }
