149  os << "{id: " << ac.id;

151  os << ", clockrate: " << ac.clockrate;
152  os << ", bitrate: " << ac.bitrate;
153  os << ", channels: " << ac.channels;
154  os << ", params: {";
155  const char* sep = "";

157  os << sep << kv.first << ": " << kv.second;
158  sep = ", ";

160  os << "}, feedback_params: {";
