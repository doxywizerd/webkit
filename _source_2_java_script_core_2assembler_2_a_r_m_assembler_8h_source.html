294  void emitInstruction(ARMWord op, int rd, int rn, ARMWord op2)
295  {
296  ASSERT(((op2 & ~Op2Immediate) <= 0xfff) || (((op2 & ~ImmediateForHalfWordTransfer) <= 0xfff)));
297  m_buffer.putInt(op | RN(rn) | RD(rd) | op2);
298  }

300  void emitDoublePrecisionInstruction(ARMWord op, int dd, int dn, int dm)
301  {
302  ASSERT((dd >= 0 && dd <= 31) && (dn >= 0 && dn <= 31) && (dm >= 0 && dm <= 31));
303  m_buffer.putInt(op | ((dd & 0xf) << 12) | ((dd & 0x10) << (22 - 4))
304  | ((dn & 0xf) << 16) | ((dn & 0x10) << (7 - 4))
305  | (dm & 0xf) | ((dm & 0x10) << (5 - 4)));
306  }
307 
308  void emitSinglePrecisionInstruction(ARMWord op, int sd, int sn, int sm)
309  {
310  ASSERT((sd >= 0 && sd <= 31) && (sn >= 0 && sn <= 31) && (sm >= 0 && sm <= 31));
311  m_buffer.putInt(op | ((sd >> 1) << 12) | ((sd & 0x1) << 22)
312  | ((sn >> 1) << 16) | ((sn & 0x1) << 7)
313  | (sm >> 1) | ((sm & 0x1) << 5));
314  }
