217 struct SVGAttributeHashTranslator {
218  static unsigned hash(const QualifiedName& key)
219  {
220  if (key.hasPrefix()) {
221  QualifiedNameComponents components = { nullAtom.impl(), key.localName().impl(), key.namespaceURI().impl() };
222  return hashComponents(components);
223  }
224  return DefaultHash<QualifiedName>::Hash::hash(key);
225  }
226  static bool equal(const QualifiedName& a, const QualifiedName& b) { return a.matches(b); }
