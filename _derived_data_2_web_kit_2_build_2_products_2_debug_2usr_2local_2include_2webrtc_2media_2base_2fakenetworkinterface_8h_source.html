114  virtual bool SendPacket(rtc::CopyOnWriteBuffer* packet,
115  const rtc::PacketOptions& options) {
116  rtc::CritScope cs(&crit_);
117 
118  uint32_t cur_ssrc = 0;
119  if (!GetRtpSsrc(packet->data(), packet->size(), &cur_ssrc)) {
120  return false;
121  }
122  sent_ssrcs_[cur_ssrc]++;
123 
124  rtp_packets_.push_back(*packet);
125  if (conf_) {
126  for (size_t i = 0; i < conf_sent_ssrcs_.size(); ++i) {
127  if (!SetRtpSsrc(packet->data(), packet->size(),
128  conf_sent_ssrcs_[i])) {
129  return false;
130  }
131  PostMessage(ST_RTP, *packet);
132  }
133  } else {
134  PostMessage(ST_RTP, *packet);
135  }
136  return true;

166  virtual void OnMessage(rtc::Message* msg) {
167  rtc::TypedMessageData<rtc::CopyOnWriteBuffer>* msg_data =
168  static_cast<rtc::TypedMessageData<rtc::CopyOnWriteBuffer>*>(
169  msg->pdata);

172  dest_->OnPacketReceived(&msg_data->data(),
173  rtc::CreatePacketTime(0));

176  rtc::CreatePacketTime(0));
177  }
178  }
179  delete msg_data;

183  void GetNumRtpBytesAndPackets(uint32_t ssrc, int* bytes, int* packets) {
184  if (bytes) {
185  *bytes = 0;
186  }
187  if (packets) {
188  *packets = 0;
189  }
190  uint32_t cur_ssrc = 0;
191  for (size_t i = 0; i < rtp_packets_.size(); ++i) {
192  if (!GetRtpSsrc(rtp_packets_[i].data(), rtp_packets_[i].size(),
193  &cur_ssrc)) {
194  return;
195  }
196  if (ssrc == cur_ssrc) {
197  if (bytes) {
198  *bytes += static_cast<int>(rtp_packets_[i].size());
199  }
200  if (packets) {
201  ++(*packets);
202  }
203  }
204  }
