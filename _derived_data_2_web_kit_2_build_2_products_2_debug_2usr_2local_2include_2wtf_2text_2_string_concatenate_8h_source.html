305 template<typename StringTypeAdapter, typename... StringTypeAdapters>
306 String tryMakeStringFromAdapters(StringTypeAdapter adapter, StringTypeAdapters ...adapters)
307 {
308  bool overflow = false;
309  unsigned length = adapter.length();
310  sumWithOverflow(overflow, length, adapters.length()...);
311  if (overflow)
312  return String();
313 
314  if (are8Bit(adapter, adapters...)) {
315  LChar* buffer;
316  RefPtr<StringImpl> resultImpl = StringImpl::tryCreateUninitialized(length, buffer);
317  if (!resultImpl)
318  return String();
319 
320  makeStringAccumulator(buffer, adapter, adapters...);
321 
322  return WTFMove(resultImpl);
323  }
324
