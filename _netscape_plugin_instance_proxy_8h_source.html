251  template <typename T>
252  std::unique_ptr<T> waitForReply(uint32_t requestID)
253  {
254  Ref<NetscapePluginInstanceProxy> protect(*this); // Plug-in host may crash while we are waiting for reply, releasing all instances to the instance proxy.
255 
256  willCallPluginFunction();
257  m_waitingForReply = true;
258 
259  auto reply = processRequestsAndWaitForReply(requestID);
260  if (reply)
261  ASSERT(reply->m_type == T::ReplyType);
262 
263  m_waitingForReply = false;
264 
265  bool stopped = false;
266  didCallPluginFunction(stopped);
267  if (stopped) {
268  // The instance proxy may have been deleted from didCallPluginFunction(), so a null reply needs to be returned.
269  return nullptr;
270  }
271 
272  return std::unique_ptr<T>(static_cast<T*>(reply.release()));
273  }

344  // All NPRuntime functions will return false when destroying a plug-in. This is necessary because there may be unhandled messages waiting,
345  // and spinning in processRequests() will unexpectedly execute them from inside destroy(). That's not a good time to execute arbitrary JavaScript,
