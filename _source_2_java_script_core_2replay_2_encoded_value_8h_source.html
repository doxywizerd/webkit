106 template<typename T, size_t inlineCapacity, typename OverflowHandler>
107 struct EncodingTraits<Vector<T, inlineCapacity, OverflowHandler>> {
108  typedef Vector<typename EncodingTraits<T>::DecodedType, inlineCapacity, OverflowHandler> DecodedType;
109 
110  static EncodedValue encodeValue(const DecodedType& vectorOfValues)
111  {
112  EncodedValue encodedVector = EncodedValue::createArray();
113  for (const typename EncodingTraits<T>::DecodedType& value : vectorOfValues)
114  encodedVector.append<T>(value);
115 
116  return encodedVector;
117  }
118 
119  static bool decodeValue(EncodedValue& encodedVector, DecodedType& decodedValue)
120  {
121  RefPtr<Inspector::InspectorArray> inspectorArray = encodedVector.asArray();
122  decodedValue = Vector<typename EncodingTraits<T>::DecodedType, inlineCapacity, OverflowHandler>(inspectorArray->length());
123  for (size_t i = 0; i < inspectorArray->length(); ++i) {
124  EncodedValue encodedElement(inspectorArray->get(i));
125  if (!EncodingTraits<T>::decodeValue(encodedElement, decodedValue.at(i)))
126  return false;
127  }
128  return true;
129  }
130 };
