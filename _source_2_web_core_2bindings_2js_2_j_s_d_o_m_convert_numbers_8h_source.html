77 template<> struct Converter<IDLByte> : DefaultConverter<IDLByte> {
78  static int8_t convert(JSC::ExecState& state, JSC::JSValue value, IntegerConversionConfiguration configuration = IntegerConversionConfiguration::Normal)
79  {
80  switch (configuration) {
81  case IntegerConversionConfiguration::Normal:
82  break;
83  case IntegerConversionConfiguration::EnforceRange:
84  return toInt8EnforceRange(state, value);
85  case IntegerConversionConfiguration::Clamp:
86  return toInt8Clamp(state, value);
87  }
88  return toInt8(state, value);
89  }
90 };

307 template<> struct Converter<IDLFloat> : DefaultConverter<IDLFloat> {
308 
309  static inline float convert(JSC::ExecState& state, JSC::ThrowScope& scope, double number)
310  {
311  if (UNLIKELY(!std::isfinite(number)))
312  throwNonFiniteTypeError(state, scope);
313  return static_cast<float>(number);
314  }
315 
316  static float convert(JSC::ExecState& state, JSC::JSValue value)
317  {
318  JSC::VM& vm = state.vm();
319  auto scope = DECLARE_THROW_SCOPE(vm);
320  double number = value.toNumber(&state);
321  if (UNLIKELY(!std::isfinite(number)))
322  throwNonFiniteTypeError(state, scope);
323  return static_cast<float>(number);
324  }
325 };
