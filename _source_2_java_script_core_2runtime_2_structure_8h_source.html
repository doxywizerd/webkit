303  static unsigned outOfLineCapacity(PropertyOffset lastOffset)
304  {
305  unsigned outOfLineSize = Structure::outOfLineSize(lastOffset);
306 
307  // This algorithm completely determines the out-of-line property storage growth algorithm.
308  // The JSObject code will only trigger a resize if the value returned by this algorithm
309  // changed between the new and old structure. So, it's important to keep this simple because
310  // it's on a fast path.
311 
312  if (!outOfLineSize)
313  return 0;
314 
315  if (outOfLineSize <= initialOutOfLineCapacity)
316  return initialOutOfLineCapacity;
317 
318  ASSERT(outOfLineSize > initialOutOfLineCapacity);
319  COMPILE_ASSERT(outOfLineGrowthFactor == 2, outOfLineGrowthFactor_is_two);
320  return WTF::roundUpToPowerOfTwo(outOfLineSize);
321  }

483  bool dfgShouldWatchIfPossible() const

485  // FIXME: We would like to not watch things that are unprofitable to watch, like
486  // dictionaries. Unfortunately, we can't do such things: a dictionary could get flattened,
487  // in which case it will start to appear watchable and so the DFG will think that it is
488  // watching it. We should come up with a comprehensive story for not watching things that
489  // aren't profitable to watch.
490  // https://bugs.webkit.org/show_bug.cgi?id=133625
491 
492  // - We don't watch Structures that either decided not to be watched, or whose predecessors
493  // decided not to be watched. This happens when a transition is fired while being watched.
494  if (transitionWatchpointIsLikelyToBeFired())
495  return false;
496 
497  // - Don't watch Structures that had been dictionaries.
498  if (hasBeenDictionary())
499  return false;
500 
501  return true;
