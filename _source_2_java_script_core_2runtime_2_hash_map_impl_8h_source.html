195 ALWAYS_INLINE static bool areKeysEqual(ExecState* exec, JSValue a, JSValue b)
196 {
197  // We want +0 and -0 to be compared to true here. sameValue() itself doesn't
198  // guarantee that, however, we normalize all keys before comparing and storing
199  // them in the map. The normalization will convert -0.0 and 0.0 to the integer
200  // representation for 0.
201  return sameValue(exec, a, b);
202 }

204 ALWAYS_INLINE JSValue normalizeMapKey(JSValue key)
205 {
206  if (!key.isNumber())
207  return key;
208 
209  if (key.isInt32())
210  return key;
211 
212  double d = key.asDouble();
213  if (std::isnan(d))
214  return key;
215 
216  int i = static_cast<int>(d);
217  if (i == d) {
218  // When a key is -0, we convert it to positive zero.
219  // When a key is the double representation for an integer, we convert it to an integer.
220  return jsNumber(i);
221  }
222  // This means key is definitely not negative zero, and it's definitely not a double representation of an integer.
223  return key;
224 }

226 static ALWAYS_INLINE uint32_t wangsInt64Hash(uint64_t key)
227 {
228  key += ~(key << 32);
229  key ^= (key >> 22);
230  key += ~(key << 13);
231  key ^= (key >> 8);
232  key += (key << 3);
233  key ^= (key >> 15);
234  key += ~(key << 27);
235  key ^= (key >> 31);
236  return static_cast<unsigned>(key);
237 }

239 ALWAYS_INLINE uint32_t jsMapHash(ExecState* exec, VM& vm, JSValue value)
240 {
241  ASSERT_WITH_MESSAGE(normalizeMapKey(value) == value, "We expect normalized values flowing into this function.");
242 
243  if (value.isString()) {
244  auto scope = DECLARE_THROW_SCOPE(vm);
245  const String& wtfString = asString(value)->value(exec);
246  RETURN_IF_EXCEPTION(scope, UINT_MAX);
247  return wtfString.impl()->hash();
248  }
249 
250  return wangsInt64Hash(JSValue::encode(value));
251 }

254 {

396  ALWAYS_INLINE void add(ExecState* exec, JSValue key, JSValue value = JSValue())
397  {
398  key = normalizeMapKey(key);
399 
400  VM& vm = exec->vm();
401  auto scope = DECLARE_THROW_SCOPE(vm);
402 
403  const uint32_t mask = m_capacity - 1;
404  uint32_t index = jsMapHash(exec, vm, key) & mask;
405  RETURN_IF_EXCEPTION(scope, void());
406  HashMapBucketType** buffer = this->buffer();
407  HashMapBucketType* bucket = buffer[index];
408  while (!isEmpty(bucket)) {
409  if (!isDeleted(bucket) && areKeysEqual(exec, key, bucket->key())) {
410  bucket->setValue(vm, value);
411  return;
412  }
413  index = (index + 1) & mask;
414  bucket = buffer[index];
415  }
416 
417  HashMapBucketType* newEntry = m_tail.get();
418  buffer[index] = newEntry;
419  newEntry->setKey(vm, key);

539  void rehash(ExecState* exec)
540  {
541  VM& vm = exec->vm();
542  auto scope = DECLARE_THROW_SCOPE(vm);
543 
544  uint32_t oldCapacity = m_capacity;
545  if (shouldShrink()) {
546  m_capacity = m_capacity / 2;
547  ASSERT(m_capacity >= 4);
548  } else if (3 * m_keyCount <= m_capacity && m_capacity > 64) {
549  // We stay at the same size if rehashing would cause us to be no more than
550  // 1/3rd full. This comes up for programs like this:
551  // Say the hash table grew to a key count of 64, causing it to grow to a capacity of 256.
552  // Then, the table added 63 items. The load is now 127. Then, 63 items are deleted.
553  // The load is still 127. Then, another item is added. The load is now 128, and we
554  // decide that we need to rehash. The key count is 65, almost exactly what it was
555  // when we grew to a capacity of 256. We don't really need to grow to a capacity
556  // of 512 in this situation. Instead, we choose to rehash at the same size. This
557  // will bring the load down to 65. We rehash into the same size when we determine
558  // that the new load ratio will be under 1/3rd. (We also pick a minumum capacity
559  // at which this rule kicks in because otherwise we will be too sensitive to rehashing
560  // at the same capacity).
561  } else
562  m_capacity = (Checked<uint32_t>(m_capacity) * 2).unsafeGet();
563 
564  if (m_capacity != oldCapacity) {
565  makeAndSetNewBuffer(exec, vm);
566  RETURN_IF_EXCEPTION(scope, void());
567  } else {
568  m_buffer.get()->reset(m_capacity);
569  assertBufferIsEmpty();
570  }
571 
572  HashMapBucketType* iter = m_head->next();
573  HashMapBucketType* end = m_tail.get();
574  const uint32_t mask = m_capacity - 1;
575  RELEASE_ASSERT(!(m_capacity & (m_capacity - 1)));
576  HashMapBucketType** buffer = this->buffer();
577  while (iter != end) {
578  uint32_t index = jsMapHash(exec, vm, iter->key()) & mask;
579  ASSERT_WITH_MESSAGE(!scope.exception(), "All keys should already be hashed before, so this should not throw because it won't resolve ropes.");
580  {
581  HashMapBucketType* bucket = buffer[index];
582  while (!isEmpty(bucket)) {
583  index = (index + 1) & mask;
584  bucket = buffer[index];
585  }
586  }
587  buffer[index] = iter;
588  iter = iter->next();
589  }
590 
591  m_deleteCount = 0;
592 
593  checkConsistency();
594  }
595
