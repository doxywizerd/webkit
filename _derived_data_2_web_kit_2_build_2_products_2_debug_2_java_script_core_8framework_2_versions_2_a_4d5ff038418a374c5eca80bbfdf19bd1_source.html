121  template<typename Functor>
122  void loadArguments(const Signature* signature, B3::Procedure& proc, B3::BasicBlock* block, B3::Origin origin, const Functor& functor) const
123  {
124  B3::Value* framePointer = block->appendNew<B3::Value>(proc, B3::FramePointer, origin);
125 
126  size_t gpArgumentCount = 0;
127  size_t fpArgumentCount = 0;
128  size_t stackOffset = headerSize;
129 
130  for (size_t i = 0; i < signature->argumentCount(); ++i) {
131  B3::Type type = toB3Type(signature->argument(i));
132  B3::Value* argument;
133  B3::ValueRep rep = marshallArgument(type, gpArgumentCount, fpArgumentCount, stackOffset);
134  if (rep.isReg()) {
135  argument = block->appendNew<B3::ArgumentRegValue>(proc, origin, rep.reg());
136  if (type == B3::Int32 || type == B3::Float)
137  argument = block->appendNew<B3::Value>(proc, B3::Trunc, origin, argument);
138  } else {
139  ASSERT(rep.isStackArgument());
140  B3::Value* address = block->appendNew<B3::Value>(proc, B3::Add, origin, framePointer,
141  block->appendNew<B3::Const64Value>(proc, origin, rep.offsetFromSP()));
142  argument = block->appendNew<B3::MemoryValue>(proc, B3::Load, type, origin, address);
143  }
144  functor(argument, i);
145  }
146  }
