41 // This is a fully adaptive mutex that only requires 1 byte of storage. It has fast paths that are
42 // competetive to a spinlock (uncontended locking is inlined and is just a CAS, microcontention is
43 // handled by spinning and yielding), and a slow path that is competetive to std::mutex (if a lock
44 // cannot be acquired in a short period of time, the thread is put to sleep until the lock is
45 // available again). It uses less memory than a std::mutex. This lock guarantees eventual stochastic
46 // fairness, even in programs that relock the lock immediately after unlocking it. Except when there
47 // are collisions between this lock and other locks in the ParkingLot, this lock will guarantee that
48 // at worst one call to unlock() per millisecond will do a direct hand-off to the thread that is at
49 // the head of the queue. When there are collisions, each collision increases the fair unlock delay
50 // by one millisecond in the worst case.
51 
52 // This is a struct without a constructor or destructor so that it can be statically initialized.
53 // Use Lock in instance variables.
54 struct LockBase {
55  void lock()
56  {
57  if (UNLIKELY(!DefaultLockAlgorithm::lockFastAssumingZero(m_byte)))
58  lockSlow();
59  }
60 
61  bool tryLock()
62  {
63  return DefaultLockAlgorithm::tryLock(m_byte);
64  }
65 
66  // Need this version for std::unique_lock.
67  bool try_lock()
68  {
69  return tryLock();
70  }
71 
72  // Relinquish the lock. Either one of the threads that were waiting for the lock, or some other
73  // thread that happens to be running, will be able to grab the lock. This bit of unfairness is
74  // called barging, and we allow it because it maximizes throughput. However, we bound how unfair
75  // barging can get by ensuring that every once in a while, when there is a thread waiting on the
76  // lock, we hand the lock to that thread directly. Every time unlock() finds a thread waiting,
77  // we check if the last time that we did a fair unlock was more than roughly 1ms ago; if so, we
78  // unlock fairly. Fairness matters most for long critical sections, and this virtually
79  // guarantees that long critical sections always get a fair lock.
80  void unlock()
81  {
82  if (UNLIKELY(!DefaultLockAlgorithm::unlockFastAssumingZero(m_byte)))
83  unlockSlow();
84  }
85 
86  // This is like unlock() but it guarantees that we unlock the lock fairly. For short critical
87  // sections, this is much slower than unlock(). For long critical sections, unlock() will learn
88  // to be fair anyway. However, if you plan to relock the lock right after unlocking and you want
89  // to ensure that some other thread runs in the meantime, this is probably the function you
90  // want.
91  void unlockFairly()
92  {
93  if (UNLIKELY(!DefaultLockAlgorithm::unlockFastAssumingZero(m_byte)))
94  unlockFairlySlow();
95  }
96 
97  void safepoint()
98  {
99  if (UNLIKELY(!DefaultLockAlgorithm::safepointFast(m_byte)))
100  safepointSlow();
101  }
102 
103  bool isHeld() const
104  {
105  return DefaultLockAlgorithm::isLocked(m_byte);
106  }
107 
108  bool isLocked() const
109  {
110  return isHeld();
111  }
112 
113 protected:
114  friend struct TestWebKitAPI::LockInspector;

117  static const uint8_t hasParkedBit = 2;
