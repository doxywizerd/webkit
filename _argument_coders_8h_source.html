209 template<typename T, size_t inlineCapacity> struct VectorArgumentCoder<false, T, inlineCapacity> {
210  static void encode(Encoder& encoder, const Vector<T, inlineCapacity>& vector)
211  {
212  encoder << static_cast<uint64_t>(vector.size());
213  for (size_t i = 0; i < vector.size(); ++i)
214  encoder << vector[i];
215  }
216 
217  static bool decode(Decoder& decoder, Vector<T, inlineCapacity>& vector)
218  {
219  uint64_t size;
220  if (!decoder.decode(size))
221  return false;
222 
223  Vector<T, inlineCapacity> tmp;
224  for (size_t i = 0; i < size; ++i) {
225  T element;
226  if (!decoder.decode(element))
227  return false;
228 
229  tmp.append(WTFMove(element));
230  }
231 
232  tmp.shrinkToFit();
233  vector.swap(tmp);
234  return true;
235  }

238 template<typename T, size_t inlineCapacity> struct VectorArgumentCoder<true, T, inlineCapacity> {
239  static void encode(Encoder& encoder, const Vector<T, inlineCapacity>& vector)
240  {
241  encoder << static_cast<uint64_t>(vector.size());
242  encoder.encodeFixedLengthData(reinterpret_cast<const uint8_t*>(vector.data()), vector.size() * sizeof(T), alignof(T));
243  }
244 
245  static bool decode(Decoder& decoder, Vector<T, inlineCapacity>& vector)
246  {
247  uint64_t size;
248  if (!decoder.decode(size))
249  return false;
250 
251  // Since we know the total size of the elements, we can allocate the vector in
252  // one fell swoop. Before allocating we must however make sure that the decoder buffer
253  // is big enough.
254  if (!decoder.bufferIsLargeEnoughToContain<T>(size)) {
255  decoder.markInvalid();
256  return false;
257  }
258 
259  Vector<T, inlineCapacity> temp;
260  temp.resize(size);
261 
262  decoder.decodeFixedLengthData(reinterpret_cast<uint8_t*>(temp.data()), size * sizeof(T), alignof(T));
263 
264  vector.swap(temp);
265  return true;
266  }

273 
274  static void encode(Encoder& encoder, const HashMapType& hashMap)
275  {
276  encoder << static_cast<uint64_t>(hashMap.size());
277  for (typename HashMapType::const_iterator it = hashMap.begin(), end = hashMap.end(); it != end; ++it)
278  encoder << *it;

285  return false;

294  return false;

297  // The hash map already has the specified key, bail.
