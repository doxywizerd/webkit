192 template <class Decoder>
193 bool HTTPHeaderMap::decode(Decoder& decoder, HTTPHeaderMap& headerMap)
194 {
195  uint64_t commonHeadersSize;
196  if (!decoder.decode(commonHeadersSize))
197  return false;
198  for (size_t i = 0; i < commonHeadersSize; ++i) {
199  HTTPHeaderName name;
200  if (!decoder.decodeEnum(name))
201  return false;
202  String value;
203  if (!decoder.decode(value))
204  return false;
205  headerMap.m_commonHeaders.add(name, value);
206  }
207 
208  if (!decoder.decode(headerMap.m_uncommonHeaders))
209  return false;
210 
211  return true;
212 }
