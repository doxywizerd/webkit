320  void finishCreation(VM& vm, ExecState* exec, JSString* base, unsigned offset, unsigned length)
321  {
322  Base::finishCreation(vm);
323  RELEASE_ASSERT(!sumOverflows<int32_t>(offset, length));
324  RELEASE_ASSERT(offset + length <= base->length());
325  setLength(length);
326  setIs8Bit(base->is8Bit());
327  setIsSubstring(true);
328  if (base->isSubstring()) {
329  JSRopeString* baseRope = jsCast<JSRopeString*>(base);
330  substringBase().set(vm, this, baseRope->substringBase().get());
331  substringOffset() = baseRope->substringOffset() + offset;
332  } else {
333  substringBase().set(vm, this, base);
334  substringOffset() = offset;
335 
336  // For now, let's not allow substrings with a rope base.
337  // Resolve non-substring rope bases so we don't have to deal with it.
338  // FIXME: Evaluate if this would be worth adding more branches.
339  if (base->isRope())
340  jsCast<JSRopeString*>(base)->resolveRope(exec);
341  }
342  }
