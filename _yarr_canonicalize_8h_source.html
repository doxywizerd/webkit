33 // This set of data provides information for each UCS2 code point as to the set of code points
34 // that it should match under the ES6 case insensitive RegExp matching rules, specified in 21.2.2.8.2.
35 // The non-Unicode tables are autogenerated using YarrCanonicalize.js into YarrCanonicalize.cpp.
36 // The Unicode tables are autogenerated using the python script generateYarrCanonicalizeUnicode
37 // which creates YarrCanonicalizeUnicode.cpp.
38 enum UCS2CanonicalizationType {
39  CanonicalizeUnique, // No canonically equal values, e.g. 0x0.
40  CanonicalizeSet, // Value indicates a set in characterSetInfo.
41  CanonicalizeRangeLo, // Value is positive delta to pair, E.g. 0x41 has value 0x20, -> 0x61.
42  CanonicalizeRangeHi, // Value is positive delta to pair, E.g. 0x61 has value 0x20, -> 0x41.
43  CanonicalizeAlternatingAligned, // Aligned consequtive pair, e.g. 0x1f4,0x1f5.
44  CanonicalizeAlternatingUnaligned, // Unaligned consequtive pair, e.g. 0x241,0x242.
45 };
46 struct CanonicalizationRange {
47  UChar32 begin;
48  UChar32 end;
49  UChar32 value;
50  UCS2CanonicalizationType type;
51 };

69 // This searches in log2 time over ~400-600 entries, so should typically result in 9 compares.
70 inline const CanonicalizationRange* canonicalRangeInfoFor(UChar32 ch, CanonicalMode canonicalMode = CanonicalMode::UCS2)
71 {
72  const CanonicalizationRange* info = canonicalMode == CanonicalMode::UCS2 ? ucs2RangeInfo : unicodeRangeInfo;
73  size_t entries = canonicalMode == CanonicalMode::UCS2 ? UCS2_CANONICALIZATION_RANGES : UNICODE_CANONICALIZATION_RANGES;
74 
75  while (true) {
76  size_t candidate = entries >> 1;
77  const CanonicalizationRange* candidateInfo = info + candidate;
78  if (ch < candidateInfo->begin)
79  entries = candidate;
80  else if (ch <= candidateInfo->end)
81  return candidateInfo;
82  else {
83  info = candidateInfo + 1;
84  entries -= (candidate + 1);
85  }
86  }
87 }

89 // Should only be called for characters that have one canonically matching value.
90 inline UChar32 getCanonicalPair(const CanonicalizationRange* info, UChar32 ch)
91 {
92  ASSERT(ch >= info->begin && ch <= info->end);
93  switch (info->type) {
94  case CanonicalizeRangeLo:
95  return ch + info->value;
96  case CanonicalizeRangeHi:
97  return ch - info->value;
98  case CanonicalizeAlternatingAligned:
99  return ch ^ 1;
100  case CanonicalizeAlternatingUnaligned:
101  return ((ch - 1) ^ 1) + 1;
102  default:
103  RELEASE_ASSERT_NOT_REACHED();
104  }
105  RELEASE_ASSERT_NOT_REACHED();
106  return 0;
107 }

109 // Returns true if no other UCS2 codepoint can match this value.
110 inline bool isCanonicallyUnique(UChar32 ch, CanonicalMode canonicalMode = CanonicalMode::UCS2)
111 {

115 // Returns true if values are equal, under the canonicalization rules.
116 inline bool areCanonicallyEquivalent(UChar32 a, UChar32 b, CanonicalMode canonicalMode = CanonicalMode::UCS2)
117 {
118  const CanonicalizationRange* info = canonicalRangeInfoFor(a, canonicalMode);
119  switch (info->type) {
120  case CanonicalizeUnique:
121  return a == b;
122  case CanonicalizeSet: {
123  for (const UChar32* set = canonicalCharacterSetInfo(info->value, canonicalMode); (a = *set); ++set) {
124  if (a == b)
125  return true;
126  }
127  return false;
128  }
129  case CanonicalizeRangeLo:
130  return (a == b) || (a + info->value == b);
131  case CanonicalizeRangeHi:
132  return (a == b) || (a - info->value == b);
133  case CanonicalizeAlternatingAligned:
134  return (a | 1) == (b | 1);
135  case CanonicalizeAlternatingUnaligned:
136  return ((a - 1) | 1) == ((b - 1) | 1);
137  }
138 
139  RELEASE_ASSERT_NOT_REACHED();
140  return false;
141 }
