558  ScratchBuffer* scratchBufferForSize(size_t size)
559  {
560  if (!size)
561  return 0;
562 
563  if (size > sizeOfLastScratchBuffer) {
564  // Protect against a N^2 memory usage pathology by ensuring
565  // that at worst, we get a geometric series, meaning that the
566  // total memory usage is somewhere around
567  // max(scratch buffer size) * 4.
568  sizeOfLastScratchBuffer = size * 2;
569 
570  ScratchBuffer* newBuffer = ScratchBuffer::create(sizeOfLastScratchBuffer);
571  RELEASE_ASSERT(newBuffer);
572  scratchBuffers.append(newBuffer);
573  }
574 
575  ScratchBuffer* result = scratchBuffers.last();
576  result->setActiveLength(0);
577  return result;
578  }
