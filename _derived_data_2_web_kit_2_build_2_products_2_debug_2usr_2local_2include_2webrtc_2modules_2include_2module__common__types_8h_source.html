457 inline bool IsNewerSequenceNumber(uint16_t sequence_number,
458  uint16_t prev_sequence_number) {
459  // Distinguish between elements that are exactly 0x8000 apart.
460  // If s1>s2 and |s1-s2| = 0x8000: IsNewer(s1,s2)=true, IsNewer(s2,s1)=false
461  // rather than having IsNewer(s1,s2) = IsNewer(s2,s1) = false.
462  if (static_cast<uint16_t>(sequence_number - prev_sequence_number) == 0x8000) {
463  return sequence_number > prev_sequence_number;
464  }
465  return sequence_number != prev_sequence_number &&
466  static_cast<uint16_t>(sequence_number - prev_sequence_number) < 0x8000;
467 }

469 inline bool IsNewerTimestamp(uint32_t timestamp, uint32_t prev_timestamp) {
470  // Distinguish between elements that are exactly 0x80000000 apart.
471  // If t1>t2 and |t1-t2| = 0x80000000: IsNewer(t1,t2)=true,
472  // IsNewer(t2,t1)=false
473  // rather than having IsNewer(t1,t2) = IsNewer(t2,t1) = false.
474  if (static_cast<uint32_t>(timestamp - prev_timestamp) == 0x80000000) {
475  return timestamp > prev_timestamp;
476  }
477  return timestamp != prev_timestamp &&
478  static_cast<uint32_t>(timestamp - prev_timestamp) < 0x80000000;
479 }

492 // Utility class to unwrap a sequence number to a larger type, for easier
493 // handling large ranges. Note that sequence numbers will never be unwrapped
494 // to a negative value.
495 class SequenceNumberUnwrapper {
496  public:
497  SequenceNumberUnwrapper() : last_seq_(-1) {}
498 
499  // Get the unwrapped sequence, but don't update the internal state.
500  int64_t UnwrapWithoutUpdate(uint16_t sequence_number) {
501  if (last_seq_ == -1)
502  return sequence_number;
503 
504  uint16_t cropped_last = static_cast<uint16_t>(last_seq_);
505  int64_t delta = sequence_number - cropped_last;
506  if (IsNewerSequenceNumber(sequence_number, cropped_last)) {
507  if (delta < 0)
508  delta += (1 << 16); // Wrap forwards.
509  } else if (delta > 0 && (last_seq_ + delta - (1 << 16)) >= 0) {
510  // If sequence_number is older but delta is positive, this is a backwards
511  // wrap-around. However, don't wrap backwards past 0 (unwrapped).
512  delta -= (1 << 16);
513  }
514 
515  return last_seq_ + delta;
516  }
517 
518  // Only update the internal state to the specified last (unwrapped) sequence.
519  void UpdateLast(int64_t last_sequence) { last_seq_ = last_sequence; }
520 
521  // Unwrap the sequence number and update the internal state.
522  int64_t Unwrap(uint16_t sequence_number) {
523  int64_t unwrapped = UnwrapWithoutUpdate(sequence_number);
524  UpdateLast(unwrapped);
525  return unwrapped;
526  }
527 
528  private:
529  int64_t last_seq_;
530 };
