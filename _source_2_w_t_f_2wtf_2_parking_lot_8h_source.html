42  // ParkingLot will accept any kind of time and convert it internally, but this typedef tells
43  // you what kind of time ParkingLot would be able to use without conversions. It's sad that
44  // this is WallTime not MonotonicTime, but that's just how OS wait functions work. However,
45  // because ParkingLot evaluates whether it should wait by checking if your time has passed
46  // using whatever clock you used, specifying timeouts in MonotonicTime is semantically better.
47  // For example, if the user sets his computer's clock back during the time that you wanted to
48  // wait for one second, and you specified the timeout using the MonotonicTime, then ParkingLot
49  // will be smart enough to know that your one second has elapsed.
50  typedef WallTime Time;
51 
52  // Parks the thread in a queue associated with the given address, which cannot be null. The
53  // parking only succeeds if the validation function returns true while the queue lock is held.
54  //
55  // If validation returns false, it will unlock the internal parking queue and then it will
56  // return a null ParkResult (wasUnparked = false, token = 0) without doing anything else.
57  //
58  // If validation returns true, it will enqueue the thread, unlock the parking queue lock, call
59  // the beforeSleep function, and then it will sleep so long as the thread continues to be on the
60  // queue and the timeout hasn't fired. Finally, this returns wasUnparked = true if we actually
61  // got unparked or wasUnparked = false if the timeout was hit. When wasUnparked = true, the
62  // token will contain whatever token was returned from the callback to unparkOne(), or 0 if the
63  // thread was unparked using unparkAll() or the form of unparkOne() that doesn't take a
64  // callback.
65  //
66  // Note that beforeSleep is called with no locks held, so it's OK to do pretty much anything so
67  // long as you don't recursively call parkConditionally(). You can call unparkOne()/unparkAll()
68  // though. It's useful to use beforeSleep() to unlock some mutex in the implementation of
69  // Condition::wait().
70  struct ParkResult {
71  bool wasUnparked { false };
72  intptr_t token { 0 };
73  };
74  template<typename ValidationFunctor, typename BeforeSleepFunctor>
75  static ParkResult parkConditionally(
76  const void* address,
77  const ValidationFunctor& validation,
78  const BeforeSleepFunctor& beforeSleep,
79  const TimeWithDynamicClockType& timeout)
80  {
81  return parkConditionallyImpl(
82  address,
83  scopedLambdaRef<bool()>(validation),
84  scopedLambdaRef<void()>(beforeSleep),
85  timeout);
86  }

88  // Simple version of parkConditionally() that covers the most common case: you want to park
89  // indefinitely so long as the value at the given address hasn't changed.
90  template<typename T, typename U>
91  static ParkResult compareAndPark(const Atomic<T>* address, U expected)
92  {
93  return parkConditionally(
94  address,
95  [address, expected] () -> bool {
96  U value = address->load();
97  return value == expected;
98  },
99  [] () { },
100  Time::infinity());
101  }

103  // Unparking status given to you anytime you unparkOne().
104  struct UnparkResult {
105  // True if some thread was unparked.
106  bool didUnparkThread { false };
107  // True if there may be more threads on this address. This may be conservatively true.
108  bool mayHaveMoreThreads { false };
109  // This bit is randomly set to true indicating that it may be profitable to unlock the lock
110  // using a fair unlocking protocol. This is most useful when used in conjunction with
111  // unparkOne(address, callback).
112  bool timeToBeFair { false };
113  };

115  // Unparks one thread from the queue associated with the given address, which cannot be null.
116  // Returns true if there may still be other threads on that queue, or false if there definitely
117  // are no more threads on the queue.
118  WTF_EXPORT_PRIVATE static UnparkResult unparkOne(const void* address);
119 
120  // This is an expert-mode version of unparkOne() that allows for really good thundering herd
121  // avoidance and eventual stochastic fairness in adaptive mutexes.
122  //
123  // Unparks one thread from the queue associated with the given address, and calls the given
124  // callback while the address is locked. Reports to the callback whether any thread got
125  // unparked, whether there may be any other threads still on the queue, and whether this may be
126  // a good time to do fair unlocking. The callback returns an intptr_t token, which is returned
127  // to the unparked thread via ParkResult::token.
128  //
129  // WTF::Lock and WTF::Condition both use this form of unparkOne() because it allows them to use
130  // the ParkingLot's internal queue lock to serialize some decision-making. For example, if
131  // UnparkResult::mayHaveMoreThreads is false inside the callback, then we know that at that
132  // moment nobody can add any threads to the queue because the queue lock is still held. Also,
133  // WTF::Lock uses the timeToBeFair and token mechanism to implement eventual fairness.
134  template<typename Callback>
135  static void unparkOne(const void* address, const Callback& callback)
136  {
137  unparkOneImpl(address, scopedLambdaRef<intptr_t(UnparkResult)>(callback));
138  }

140  WTF_EXPORT_PRIVATE static unsigned unparkCount(const void* address, unsigned count);

142  // Unparks every thread from the queue associated with the given address, which cannot be null.
143  WTF_EXPORT_PRIVATE static void unparkAll(const void* address);

145  // Locks the parking lot and walks all of the parked threads and the addresses they are waiting
146  // on. Threads that are on the same queue are guaranteed to be walked from first to last, but the
147  // queues may be randomly interleaved. For example, if the queue for address A1 has T1 and T2 and
148  // the queue for address A2 has T3 and T4, then you might see iteration orders like:
149  //
150  // A1,T1 A1,T2 A2,T3 A2,T4
151  // A2,T3 A2,T4 A1,T1 A1,T2
152  // A1,T1 A2,T3 A1,T2 A2,T4
153  // A1,T1 A2,T3 A2,T4 A1,T2
154  //
155  // As well as many other possible interleavings that all have T1 before T2 and T3 before T4 but are
156  // otherwise unconstrained. This method is useful primarily for debugging. It's also used by unit
157  // tests.
158  template<typename Func>
159  static void forEach(const Func& func)
160  {
161  forEachImpl(scopedLambdaRef<void(ThreadIdentifier, const void*)>(func));
162  }
