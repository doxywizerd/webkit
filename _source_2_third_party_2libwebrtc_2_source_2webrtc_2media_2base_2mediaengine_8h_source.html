46 // MediaEngineInterface is an abstraction of a media engine which can be
47 // subclassed to support different media componentry backends.
48 // It supports voice and video operations in the same class to facilitate
49 // proper synchronization between both media types.
50 class MediaEngineInterface {
51  public:
52  virtual ~MediaEngineInterface() {}
53 
54  // Initialization
55  // Starts the engine.
56  virtual bool Init() = 0;
57  // TODO(solenberg): Remove once VoE API refactoring is done.
58  virtual rtc::scoped_refptr<webrtc::AudioState> GetAudioState() const = 0;
59 
60  // MediaChannel creation
61  // Creates a voice media channel. Returns NULL on failure.
62  virtual VoiceMediaChannel* CreateChannel(webrtc::Call* call,
63  const MediaConfig& config,
64  const AudioOptions& options) = 0;
65  // Creates a video media channel, paired with the specified voice channel.
66  // Returns NULL on failure.
67  virtual VideoMediaChannel* CreateVideoChannel(
68  webrtc::Call* call,
69  const MediaConfig& config,
70  const VideoOptions& options) = 0;
71 
72  // Gets the current microphone level, as a value between 0 and 10.
73  virtual int GetInputLevel() = 0;
74 
75  virtual const std::vector<AudioCodec>& audio_send_codecs() = 0;
76  virtual const std::vector<AudioCodec>& audio_recv_codecs() = 0;
77  virtual RtpCapabilities GetAudioCapabilities() = 0;
78  virtual std::vector<VideoCodec> video_codecs() = 0;
79  virtual RtpCapabilities GetVideoCapabilities() = 0;
80 
81  // Starts AEC dump using existing file, a maximum file size in bytes can be
82  // specified. Logging is stopped just before the size limit is exceeded.
83  // If max_size_bytes is set to a value <= 0, no limit will be used.
84  virtual bool StartAecDump(rtc::PlatformFile file, int64_t max_size_bytes) = 0;
85 
86  // Stops recording AEC dump.
87  virtual void StopAecDump() = 0;
