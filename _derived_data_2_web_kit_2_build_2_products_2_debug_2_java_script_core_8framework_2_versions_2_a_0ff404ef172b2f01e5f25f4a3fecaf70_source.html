28 // Overview of JSArray
29 //
30 // Properties of JSArray objects may be stored in one of three locations:
31 // * The regular JSObject property map.
32 // * A storage vector.
33 // * A sparse map of array entries.
34 //
35 // Properties with non-numeric identifiers, with identifiers that are not representable
36 // as an unsigned integer, or where the value is greater than MAX_ARRAY_INDEX
37 // (specifically, this is only one property - the value 0xFFFFFFFFU as an unsigned 32-bit
38 // integer) are not considered array indices and will be stored in the JSObject property map.
39 //
40 // All properties with a numeric identifier, representable as an unsigned integer i,
41 // where (i <= MAX_ARRAY_INDEX), are an array index and will be stored in either the
42 // storage vector or the sparse map. An array index i will be handled in the following
43 // fashion:
44 //
45 // * Where (i < MIN_SPARSE_ARRAY_INDEX) the value will be stored in the storage vector,
46 // unless the array is in SparseMode in which case all properties go into the map.
47 // * Where (MIN_SPARSE_ARRAY_INDEX <= i <= MAX_STORAGE_VECTOR_INDEX) the value will either
48 // be stored in the storage vector or in the sparse array, depending on the density of
49 // data that would be stored in the vector (a vector being used where at least
50 // (1 / minDensityMultiplier) of the entries would be populated).
51 // * Where (MAX_STORAGE_VECTOR_INDEX < i <= MAX_ARRAY_INDEX) the value will always be stored
52 // in the sparse array.
53 
54 // Define the maximum storage vector length to be 2^32 / sizeof(JSValue) / 2 to ensure that
55 // there is no risk of overflow.
56 #define MAX_STORAGE_VECTOR_LENGTH (static_cast<unsigned>(IndexingHeader::maximumLength))
57 
58 // These values have to be macros to be used in max() and min() without introducing
59 // a PIC branch in Mach-O binaries, see <rdar://problem/5971391>.
60 
61 // If you grow an ArrayStorage array by more than this, then the array will go sparse. Note that we
62 // could probably make this smaller (it's large because it used to be conflated with
63 // MIN_ARRAY_STORAGE_CONSTRUCTION_LENGTH).
64 #define MIN_SPARSE_ARRAY_INDEX 100000U
65 // If you try to allocate a contiguous array larger than this, then we will allocate an ArrayStorage
66 // array instead. We allow for an array that occupies 1GB of VM.
67 #define MIN_ARRAY_STORAGE_CONSTRUCTION_LENGTH 1024 * 1024 * 1024 / 8
68 #define MAX_STORAGE_VECTOR_INDEX (MAX_STORAGE_VECTOR_LENGTH - 1)
69 // 0xFFFFFFFF is a bit weird -- is not an array index even though it's an integer.
70 #define MAX_ARRAY_INDEX 0xFFFFFFFEU
71 
72 // The value BASE_XXX_VECTOR_LEN is the maximum number of vector elements we'll allocate
73 // for an array that was created with a sepcified length (e.g. a = new Array(123))
74 #define BASE_CONTIGUOUS_VECTOR_LEN 3U
75 #define BASE_CONTIGUOUS_VECTOR_LEN_EMPTY 5U
76 #define BASE_ARRAY_STORAGE_VECTOR_LEN 4U
77 
78 // The upper bound to the size we'll grow a zero length array when the first element
79 // is added.
80 #define FIRST_ARRAY_STORAGE_VECTOR_GROW 4U

82 #define MIN_BEYOND_LENGTH_SPARSE_INDEX 1000

84 // Our policy for when to use a vector and when to use a sparse map.
85 // For all array indices under MIN_SPARSE_ARRAY_INDEX, we always use a vector.
86 // When indices greater than MIN_SPARSE_ARRAY_INDEX are involved, we use a vector
87 // as long as it is 1/8 full. If more sparse than that, we use a map.
88 static const unsigned minDensityMultiplier = 8;
