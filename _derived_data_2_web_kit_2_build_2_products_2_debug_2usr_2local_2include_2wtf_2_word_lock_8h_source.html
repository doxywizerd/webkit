40 // A WordLock is a fully adaptive mutex that uses sizeof(void*) storage. It has a fast path that is
41 // similar to a spinlock, and a slow path that is similar to std::mutex. In most cases, you should use
42 // Lock instead. WordLock sits lower in the stack and is used to implement Lock, so Lock is the main
43 // client of WordLock.
44 
45 // NOTE: This is also a great lock to use if you are very low in the stack. For example,
46 // PrintStream uses this so that ParkingLot and Lock can use PrintStream. This means that if you
47 // try to use dataLog to debug this code, you will have a bad time.
48 
49 struct WordLockBase {
50  void lock()
51  {
52  if (LIKELY(m_word.compareExchangeWeak(0, isLockedBit, std::memory_order_acquire))) {
53  // WordLock acquired!
54  return;
55  }
56 
57  lockSlow();
58  }
59 
60  void unlock()
61  {
62  if (LIKELY(m_word.compareExchangeWeak(isLockedBit, 0, std::memory_order_release))) {
63  // WordLock released, and nobody was waiting!
64  return;
65  }
66 
67  unlockSlow();
68  }
69 
70  bool isHeld() const
71  {
72  return m_word.load(std::memory_order_acquire) & isLockedBit;
73  }
74 
75  bool isLocked() const
76  {
77  return isHeld();
78  }
79 
80 protected:
81  friend struct TestWebKitAPI::LockInspector;

83  static const uintptr_t isLockedBit = 1;
84  static const uintptr_t isQueueLockedBit = 2;
85  static const uintptr_t queueHeadMask = 3;
86 
87  WTF_EXPORT_PRIVATE void lockSlow();
88  WTF_EXPORT_PRIVATE void unlockSlow();
89 
90  // Method used for testing only.
91  bool isFullyReset() const
92  {
93  return !m_word.load();
94  }

97 };
