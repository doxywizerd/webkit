70 void XPCServiceInitializer(OSObjectPtr<xpc_connection_t> connection, xpc_object_t initializerMessage, xpc_object_t priorityBoostMessage)
71 {
72  XPCServiceInitializerDelegateType delegate(WTFMove(connection), initializerMessage);
73 
74  // We don't want XPC to be in charge of whether the process should be terminated or not,
75  // so ensure that we have an outstanding transaction here.
76  xpc_transaction_begin();
77 
78  InitializeWebKit2();
79 
80  if (!delegate.checkEntitlements())
81  exit(EXIT_FAILURE);
82 
83  ChildProcessInitializationParameters parameters;
84  if (priorityBoostMessage)
85  parameters.priorityBoostMessage = adoptOSObject(xpc_retain(priorityBoostMessage));
86 
87  if (!delegate.getConnectionIdentifier(parameters.connectionIdentifier))
88  exit(EXIT_FAILURE);
89 
90  if (!delegate.getClientIdentifier(parameters.clientIdentifier))
91  exit(EXIT_FAILURE);
92 
93  if (!delegate.getClientProcessName(parameters.uiProcessName))
94  exit(EXIT_FAILURE);
95 
96  if (!delegate.getExtraInitializationData(parameters.extraInitializationData))
97  exit(EXIT_FAILURE);
98 
99 #if HAVE(VOUCHERS)
100  // Set the task default voucher to the current value (as propagated by XPC).
101  voucher_replace_default_voucher();
102 #endif
103 
104 #if HAVE(QOS_CLASSES)
105  if (parameters.extraInitializationData.contains(ASCIILiteral("always-runs-at-background-priority")))
106  setGlobalMaxQOSClass(QOS_CLASS_UTILITY);
107 #endif
108 
109  XPCServiceType::singleton().initialize(parameters);
110 }
