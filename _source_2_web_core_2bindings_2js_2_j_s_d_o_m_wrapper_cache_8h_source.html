168 template<typename DOMClass, typename WrapperClass> inline void cacheWrapper(DOMWrapperWorld& world, DOMClass* domObject, WrapperClass* wrapper)
169 {
170  JSC::WeakHandleOwner* owner = wrapperOwner(world, domObject);
171  if (setInlineCachedWrapper(world, domObject, wrapper, owner))
172  return;
173  weakAdd(world.m_wrappers, wrapperKey(domObject), JSC::Weak<JSC::JSObject>(wrapper, owner, &world));
174 }

176 template<typename DOMClass, typename WrapperClass> inline void uncacheWrapper(DOMWrapperWorld& world, DOMClass* domObject, WrapperClass* wrapper)
177 {
178  if (clearInlineCachedWrapper(world, domObject, wrapper))
179  return;
180  weakRemove(world.m_wrappers, wrapperKey(domObject), wrapper);
