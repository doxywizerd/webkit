33 // Call this "walker" instead of iterator so people won't expect Qt or STL-style iterator interface.
34 // Just keep calling next() on this. It's safe from deletions of items.
35 template<typename T> class CachedResourceClientWalker {
36 public:
37  CachedResourceClientWalker(const HashCountedSet<CachedResourceClient*>& set)
38  : m_clientSet(set), m_clientVector(set.size()), m_index(0)
39  {
40  typedef HashCountedSet<CachedResourceClient*>::const_iterator Iterator;
41  Iterator end = set.end();
42  size_t clientIndex = 0;
43  for (Iterator current = set.begin(); current != end; ++current)
44  m_clientVector[clientIndex++] = current->key;
45  }
46 
47  T* next()
48  {
49  size_t size = m_clientVector.size();
50  while (m_index < size) {
51  CachedResourceClient* next = m_clientVector[m_index++];
52  if (m_clientSet.contains(next)) {
53  ASSERT_WITH_SECURITY_IMPLICATION(T::expectedType() == CachedResourceClient::expectedType() || next->resourceClientType() == T::expectedType());
54  return static_cast<T*>(next);
55  }
56  }
57 
58  return nullptr;
59  }
60 private:
61  const HashCountedSet<CachedResourceClient*>& m_clientSet;
62  Vector<CachedResourceClient*> m_clientVector;
63  size_t m_index;
64 };
