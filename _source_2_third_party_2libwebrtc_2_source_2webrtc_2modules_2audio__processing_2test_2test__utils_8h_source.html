109 template <typename T>
110 float ComputeSNR(const T* ref, const T* test, size_t length, float* variance) {
111  float mse = 0;
112  float mean = 0;
113  *variance = 0;
114  for (size_t i = 0; i < length; ++i) {
115  T error = ref[i] - test[i];
116  mse += error * error;
117  *variance += ref[i] * ref[i];
118  mean += ref[i];
119  }
120  mse /= length;
121  *variance /= length;
122  mean /= length;
123  *variance -= mean * mean;
124 
125  float snr = 100; // We assign 100 dB to the zero-error case.
126  if (mse > 0)
127  snr = 10 * log10(*variance / mse);
128  return snr;
129 }
