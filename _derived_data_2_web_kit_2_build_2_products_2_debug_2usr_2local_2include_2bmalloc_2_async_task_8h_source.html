92 NO_INLINE void AsyncTask<Object, Function>::runSlowCase()
93 {

95  if (oldState == RunRequested || oldState == Running)

113 template<typename Object, typename Function>
114 void AsyncTask<Object, Function>::threadRunLoop()
115 {
116  // This loop ratchets downward from most active to least active state. While
117  // we ratchet downward, any other thread may reset our state.
118 
119  // We require any state change while we are sleeping to signal to our
120  // condition variable and wake us up.
121 
122  while (1) {
123  State expectedState = RunRequested;
124  if (m_state.compare_exchange_weak(expectedState, Running))
125  (m_object.*m_function)();
126 
127  expectedState = Running;
128  if (m_state.compare_exchange_weak(expectedState, Sleeping)) {
129  std::unique_lock<Mutex> lock(m_conditionMutex);
130  m_condition.wait(lock, [&]() { return m_state != Sleeping; });
131  }
132  }
133 }
