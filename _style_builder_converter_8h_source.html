174 inline Length StyleBuilderConverter::convertLength(StyleResolver& styleResolver, const CSSValue& value)
175 {
176  auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
177  CSSToLengthConversionData conversionData = styleResolver.useSVGZoomRulesForLength() ?
178  styleResolver.state().cssToLengthConversionData().copyWithAdjustedZoom(1.0f)
179  : styleResolver.state().cssToLengthConversionData();
180 
181  if (primitiveValue.isLength()) {
182  Length length = primitiveValue.computeLength<Length>(conversionData);
183  length.setHasQuirk(primitiveValue.isQuirkValue());
184  return length;
185  }
186 
187  if (primitiveValue.isPercentage())
188  return Length(primitiveValue.doubleValue(), Percent);
189 
190  if (primitiveValue.isCalculatedPercentageWithLength())
191  return Length(primitiveValue.cssCalcValue()->createCalculationValue(conversionData));
192 
193  ASSERT_NOT_REACHED();
194  return Length(0, Fixed);
195 }

243 template<typename T>
244 inline T StyleBuilderConverter::convertLineWidth(StyleResolver& styleResolver, const CSSValue& value)
245 {
246  auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
247  switch (primitiveValue.valueID()) {
248  case CSSValueThin:
249  return 1;
250  case CSSValueMedium:
251  return 3;
252  case CSSValueThick:
253  return 5;
254  case CSSValueInvalid: {
255  // Any original result that was >= 1 should not be allowed to fall below 1.
256  // This keeps border lines from vanishing.
257  T result = convertComputedLength<T>(styleResolver, value);
258  if (styleResolver.state().style()->effectiveZoom() < 1.0f && result < 1.0) {
259  T originalLength = primitiveValue.computeLength<T>(styleResolver.state().cssToLengthConversionData().copyWithAdjustedZoom(1.0));
260  if (originalLength >= 1.0)
261  return 1;
262  }
263  float minimumLineWidth = 1 / styleResolver.document().deviceScaleFactor();
264  if (result > 0 && result < minimumLineWidth)
265  return minimumLineWidth;
266  return floorToDevicePixel(result, styleResolver.document().deviceScaleFactor());
267  }
268  default:
269  ASSERT_NOT_REACHED();
270  return 0;
271  }
272 }

334 template<CSSValueID cssValueFor0, CSSValueID cssValueFor100>
335 inline Length StyleBuilderConverter::convertPositionComponent(StyleResolver& styleResolver, const CSSPrimitiveValue& value)
336 {
337  Length length;
338 
339  auto* lengthValue = &value;
340  bool relativeToTrailingEdge = false;
341 
342  if (value.isPair()) {
343  auto& first = *value.pairValue()->first();
344  if (first.valueID() == CSSValueRight || first.valueID() == CSSValueBottom)
345  relativeToTrailingEdge = true;
346  lengthValue = value.pairValue()->second();
347  }
348 
349  if (value.isValueID()) {
350  switch (value.valueID()) {
351  case cssValueFor0:
352  return Length(0, Percent);
353  case cssValueFor100:
354  return Length(100, Percent);
355  case CSSValueCenter:
356  return Length(50, Percent);
357  default:
358  ASSERT_NOT_REACHED();
359  }
360  }
361 
362  length = convertLength(styleResolver, *lengthValue);
363 
364  if (relativeToTrailingEdge)
365  length = convertTo100PercentMinusLength(length);
366 
367  return length;
368 }

512 inline RefPtr<ClipPathOperation> StyleBuilderConverter::convertClipPath(StyleResolver& styleResolver, const CSSValue& value)
513 {
514  if (is<CSSPrimitiveValue>(value)) {
515  auto& primitiveValue = downcast<CSSPrimitiveValue>(value);
516  if (primitiveValue.primitiveType() == CSSPrimitiveValue::CSS_URI) {
517  String cssURLValue = primitiveValue.stringValue();
518  URL url = styleResolver.document().completeURL(cssURLValue);
519  // FIXME: It doesn't work with external SVG references (see https://bugs.webkit.org/show_bug.cgi?id=126133)
520  return ReferenceClipPathOperation::create(cssURLValue, url.fragmentIdentifier());
521  }
522  ASSERT(primitiveValue.valueID() == CSSValueNone);
523  return nullptr;
524  }
525 
526  CSSBoxType referenceBox = BoxMissing;
527  RefPtr<ClipPathOperation> operation;
528 
529  for (auto& currentValue : downcast<CSSValueList>(value)) {
530  auto& primitiveValue = downcast<CSSPrimitiveValue>(currentValue.get());
531  if (primitiveValue.isShape()) {
532  ASSERT(!operation);
533  operation = ShapeClipPathOperation::create(basicShapeForValue(styleResolver.state().cssToLengthConversionData(), *primitiveValue.shapeValue()));
534  } else {
535  ASSERT(primitiveValue.valueID() == CSSValueContentBox
536  || primitiveValue.valueID() == CSSValueBorderBox
537  || primitiveValue.valueID() == CSSValuePaddingBox
538  || primitiveValue.valueID() == CSSValueMarginBox
539  || primitiveValue.valueID() == CSSValueFill
540  || primitiveValue.valueID() == CSSValueStroke
541  || primitiveValue.valueID() == CSSValueViewBox);
542  ASSERT(referenceBox == BoxMissing);
543  referenceBox = primitiveValue;
544  }
545  }
546  if (operation)
547  downcast<ShapeClipPathOperation>(*operation).setReferenceBox(referenceBox);
548  else {
549  ASSERT(referenceBox != BoxMissing);
550  operation = BoxClipPathOperation::create(referenceBox);
551  }
552 
553  return operation;
554 }
