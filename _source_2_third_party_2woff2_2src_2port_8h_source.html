26 inline int Log2Floor(uint32 n) {
27 #if defined(__GNUC__)
28  return n == 0 ? -1 : 31 ^ __builtin_clz(n);
29 #else
30  if (n == 0)
31  return -1;
32  int log = 0;
33  uint32 value = n;
34  for (int i = 4; i >= 0; --i) {
35  int shift = (1 << i);
36  uint32 x = value >> shift;
37  if (x != 0) {
38  value = x;
39  log += shift;
40  }
41  }
42  assert(value == 1);
43  return log;
44 #endif
45 }
