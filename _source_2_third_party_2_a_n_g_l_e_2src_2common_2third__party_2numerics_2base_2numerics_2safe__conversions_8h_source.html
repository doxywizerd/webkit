44 // checked_cast<> is analogous to static_cast<> for numeric types,

77 // This wrapper is used for C++11 constexpr support by avoiding the declaration
78 // of local variables in the saturated_cast template function.
79 template <typename Dst, class NaNHandler, typename Src>
80 constexpr Dst saturated_cast_impl(const Src value, const RangeConstraint constraint)
81 {
82  return constraint == RANGE_VALID
83  ? static_cast<Dst>(value)
84  : (constraint == RANGE_UNDERFLOW
85  ? std::numeric_limits<Dst>::min()
86  : (constraint == RANGE_OVERFLOW
87  ? std::numeric_limits<Dst>::max()
88  : (constraint == RANGE_INVALID
89  ? NaNHandler::template HandleNaN<Dst>()
90  : (NOTREACHED(), static_cast<Dst>(value)))));
91 }

94 // saturated_cast<> is analogous to static_cast<> for numeric types, except
95 // that the specified numeric conversion will saturate rather than overflow or
96 // underflow. NaN assignment to an integral will defer the behavior to a
97 // specified class. By default, it will return 0.
98 template <typename Dst, class NaNHandler = SaturatedCastNaNBehaviorReturnZero, typename Src>
99 constexpr Dst saturated_cast(Src value)
100 {
101  return std::numeric_limits<Dst>::is_iec559
102  ? static_cast<Dst>(value) // Floating point optimization.
103  : internal::saturated_cast_impl<Dst, NaNHandler>(
104  value, internal::DstRangeRelationToSrcRange<Dst>(value));
105 }

107 // strict_cast<> is analogous to static_cast<> for numeric types, except that
108 // it will cause a compile failure if the destination type is not large enough
109 // to contain any value in the source type. It performs no runtime checking.
110 template <typename Dst, typename Src>
111 constexpr Dst strict_cast(Src value)
112 {
113  static_assert(std::numeric_limits<Src>::is_specialized, "Argument must be numeric.");
114  static_assert(std::numeric_limits<Dst>::is_specialized, "Result must be numeric.");
115  static_assert((internal::StaticDstRangeRelationToSrcRange<Dst, Src>::value ==
116  internal::NUMERIC_RANGE_CONTAINED),
117  "The numeric conversion is out of range for this type. You "
118  "should probably use one of the following conversion "
119  "mechanisms on the value you want to pass:\n"
120  "- base::checked_cast\n"
121  "- base::saturated_cast\n"
122  "- base::CheckedNumeric");
123 
124  return static_cast<Dst>(value);
125 }

127 // StrictNumeric implements compile time range checking between numeric types by
128 // wrapping assignment operations in a strict_cast. This class is intended to be
129 // used for function arguments and return types, to ensure the destination type
130 // can always contain the source type. This is essentially the same as enforcing
131 // -Wconversion in gcc and C4302 warnings on MSVC, but it can be applied
132 // incrementally at API boundaries, making it easier to convert code so that it
133 // compiles cleanly with truncation warnings enabled.
134 // This template should introduce no runtime overhead, but it also provides no
135 // runtime checking of any of the associated mathematical operations. Use
136 // CheckedNumeric for runtime range checks of the actual value being assigned.
137 template <typename T>
138 class StrictNumeric
139 {
140  public:
141  typedef T type;
142 
143  constexpr StrictNumeric() : value_(0) {}
144 
145  // Copy constructor.
146  template <typename Src>
147  constexpr StrictNumeric(const StrictNumeric<Src> &rhs) : value_(strict_cast<T>(rhs.value_))
148  {
149  }
150 
151  // This is not an explicit constructor because we implicitly upgrade regular
152  // numerics to StrictNumerics to make them easier to use.
153  template <typename Src>
154  constexpr StrictNumeric(Src value) : value_(strict_cast<T>(value))
155  {
156  }
157 
158  // The numeric cast operator basically handles all the magic.
159  template <typename Dst>
160  constexpr operator Dst() const
161  {
162  return strict_cast<Dst>(value_);
163  }
164 
165  private:
166  const T value_;
167 };
