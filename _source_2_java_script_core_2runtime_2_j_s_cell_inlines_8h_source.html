59 inline void JSCell::finishCreation(VM& vm)
60 {
61  // This object is ready to be escaped so the concurrent GC may see it at any time. We have
62  // to make sure that none of our stores sink below here.
63  vm.heap.mutatorFence();
64 #if ENABLE(GC_VALIDATION)
65  ASSERT(vm.isInitializingObject());
66  vm.setInitializingObjectClass(0);
67 #else
68  UNUSED_PARAM(vm);
69 #endif
70  ASSERT(m_structureID);
71 }

144 template<typename T>
145 void* allocateCell(Heap& heap, size_t size)
146 {
147  ASSERT(!DisallowGC::isGCDisallowedOnCurrentThread());
148  ASSERT(size >= sizeof(T));
149  JSCell* result = static_cast<JSCell*>(subspaceFor<T>(*heap.vm())->allocate(size));
150 #if ENABLE(GC_VALIDATION)
151  ASSERT(!heap.vm()->isInitializingObject());
152  heap.vm()->setInitializingObjectClass(T::info());
153 #endif
154  result->clearStructure();
155  return result;

218 ALWAYS_INLINE void JSCell::setStructure(VM& vm, Structure* structure)
219 {
220  ASSERT(structure->classInfo() == this->structure()->classInfo());
221  ASSERT(!this->structure()
222  || this->structure()->transitionWatchpointSetHasBeenInvalidated()
223  || Heap::heap(this)->structureIDTable().get(structure->id()) == structure);
224  m_structureID = structure->id();
225  m_flags = structure->typeInfo().inlineTypeFlags();
226  m_type = structure->typeInfo().type();
227  IndexingType newIndexingType = structure->indexingTypeIncludingHistory();
228  if (m_indexingTypeAndMisc != newIndexingType) {
229  ASSERT(!(newIndexingType & ~AllArrayTypesAndHistory));
230  for (;;) {
231  IndexingType oldValue = m_indexingTypeAndMisc;
232  IndexingType newValue = (oldValue & ~AllArrayTypesAndHistory) | structure->indexingTypeIncludingHistory();
233  if (WTF::atomicCompareExchangeWeakRelaxed(&m_indexingTypeAndMisc, oldValue, newValue))
234  break;
235  }
236  }
237  vm.heap.writeBarrier(this, structure);
238 }

276 ALWAYS_INLINE const ClassInfo* JSCell::classInfo(VM& vm) const
277 {
278  // What we really want to assert here is that we're not currently destructing this object (which makes its classInfo
279  // invalid). If mutatorState() == MutatorState::Running, then we're not currently sweeping, and therefore cannot be
280  // destructing the object. The GC thread or JIT threads, unlike the mutator thread, are able to access classInfo
281  // independent of whether the mutator thread is sweeping or not. Hence, we also check for !currentThreadIsHoldingAPILock()
282  // to allow the GC thread or JIT threads to pass this assertion.
283  ASSERT(vm.heap.mutatorState() != MutatorState::Sweeping || !vm.currentThreadIsHoldingAPILock());
284  return structure(vm)->classInfo();
285 }

303 inline void JSCell::callDestructor(VM& vm)
304 {
305  if (isZapped())
306  return;
307  ASSERT(structureID());
308  if (inlineTypeFlags() & StructureIsImmortal) {
309  Structure* structure = this->structure(vm);
310  const ClassInfo* classInfo = structure->classInfo();
311  MethodTable::DestroyFunctionPtr destroy = classInfo->methodTable.destroy;
312  destroy(this);
313  } else
314  static_cast<JSDestructibleObject*>(this)->classInfo()->methodTable.destroy(this);
315  zap();
316 }
