39 struct GenericSequenceConverter {
40  using ReturnType = Vector<typename IDLType::ImplementationType>;
41 
42  static ReturnType convert(JSC::ExecState& state, JSC::JSObject* jsObject)
43  {
44  ReturnType result;
45  forEachInIterable(&state, jsObject, [&result](JSC::VM& vm, JSC::ExecState* state, JSC::JSValue jsValue) {
46  auto scope = DECLARE_THROW_SCOPE(vm);
47 
48  auto convertedValue = Converter<IDLType>::convert(*state, jsValue);
49  if (UNLIKELY(scope.exception()))
50  return;
51  result.append(WTFMove(convertedValue));
52  });
53  return result;
54  }
55 };

57 // Specialization for numeric types
58 // FIXME: This is only implemented for the IDLFloatingPointTypes and IDLLong. To add
59 // support for more numeric types, add an overload of Converter<IDLType>::convert that
60 // takes an ExecState, ThrowScope, double as its arguments.
61 template<typename IDLType>
62 struct NumericSequenceConverter {
63  using GenericConverter = GenericSequenceConverter<IDLType>;
64  using ReturnType = typename GenericConverter::ReturnType;
65 
66  static ReturnType convert(JSC::ExecState& state, JSC::JSValue value)
67  {
68  auto& vm = state.vm();
69  auto scope = DECLARE_THROW_SCOPE(vm);
70 
71  if (!value.isObject()) {
72  throwSequenceTypeError(state, scope);
73  return { };
74  }
75 
76  JSC::JSObject* object = JSC::asObject(value);
77  if (!JSC::isJSArray(object))
78  return GenericConverter::convert(state, object);
79 
80  JSC::JSArray* array = JSC::asArray(object);
81  if (!array->globalObject()->isArrayIteratorProtocolFastAndNonObservable())
82  return GenericConverter::convert(state, object);
83 
84  unsigned length = array->length();
85 
86  ReturnType result;
87  if (!result.tryReserveCapacity(length)) {
88  // FIXME: Is the right exception to throw?
89  throwTypeError(&state, scope);
90  return { };
91  }
92 
93  JSC::IndexingType indexingType = array->indexingType() & JSC::IndexingShapeMask;
94 
95  if (indexingType == JSC::ContiguousShape) {
96  for (unsigned i = 0; i < length; i++) {
97  auto indexValue = array->butterfly()->contiguous()[i].get();
98  if (!indexValue)
99  result.uncheckedAppend(0);
100  else {
101  auto convertedValue = Converter<IDLType>::convert(state, indexValue);
102  RETURN_IF_EXCEPTION(scope, { });
103 
104  result.uncheckedAppend(convertedValue);
105  }
106  }
107  return result;
108  }
109 
110  if (indexingType == JSC::Int32Shape) {
111  for (unsigned i = 0; i < length; i++) {
112  auto indexValue = array->butterfly()->contiguousInt32()[i].get();
113  ASSERT(!indexValue || indexValue.isInt32());
114  if (!indexValue)
115  result.uncheckedAppend(0);
116  else
117  result.uncheckedAppend(indexValue.asInt32());
118  }
119  return result;
120  }
121 
122  if (indexingType == JSC::DoubleShape) {
123  for (unsigned i = 0; i < length; i++) {
124  auto doubleValue = array->butterfly()->contiguousDouble()[i];
125  if (std::isnan(doubleValue))
126  result.uncheckedAppend(0);
127  else {
128  auto convertedValue = Converter<IDLType>::convert(state, scope, doubleValue);
129  RETURN_IF_EXCEPTION(scope, { });
130 
131  result.uncheckedAppend(convertedValue);
132  }
133  }
134  return result;
135  }
136 
137  for (unsigned i = 0; i < length; i++) {
138  auto indexValue = array->getDirectIndex(&state, i);
139  RETURN_IF_EXCEPTION(scope, { });
140 
141  if (!indexValue)
142  result.uncheckedAppend(0);
143  else {
144  auto convertedValue = Converter<IDLType>::convert(state, indexValue);
145  RETURN_IF_EXCEPTION(scope, { });
146 
147  result.uncheckedAppend(convertedValue);
148  }
149  }
150  return result;
151  }
152 };
