137  static void customDeleteBucket(std::unique_ptr<T, Deleter>& value)
138  {
139  // The custom delete function exists to avoid a dead store before the value is destructed.
140  // The normal destruction sequence of a bucket would be:
141  // 1) Call the destructor of unique_ptr.
142  // 2) unique_ptr store a zero for its internal pointer.
143  // 3) unique_ptr destroys its value.
144  // 4) Call constructDeletedValue() to set the bucket as destructed.
145  //
146  // The problem is the call in (3) prevents the compile from eliminating the dead store in (2)
147  // becase a side effect of free() could be observing the value.
148  //
149  // This version of deleteBucket() ensures the dead 2 stores changing "value"
150  // are on the same side of the function call.
151  ASSERT(!isDeletedValue(value));
152  T* pointer = value.release();
153  constructDeletedValue(value);
154 
155  // The null case happens if a caller uses std::move() to remove the pointer before calling remove()
156  // with an iterator. This is very uncommon.
157  if (LIKELY(pointer))
158  Deleter()(pointer);
159  }
160 };
