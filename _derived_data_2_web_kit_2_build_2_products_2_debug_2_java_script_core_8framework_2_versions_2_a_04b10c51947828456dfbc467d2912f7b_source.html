83 // Helper for producing a JSString for 'string', where 'string' was been produced by
84 // calling ToString on 'originalValue'. In cases where 'originalValue' already was a
85 // string primitive we can just use this, otherwise we need to allocate a new JSString.
86 // FIXME: Basically any use of this is bad. toString() returns a JSString* so we don't need to
87 // pass around the originalValue; we could just pass around the JSString*. Then you don't need
88 // this function. You just use the JSString* that toString() returned.
89 static inline JSString* jsStringWithReuse(ExecState* exec, JSValue originalValue, const String& string)
90 {
91  if (originalValue.isString()) {
92  ASSERT(asString(originalValue)->value(exec) == string);
93  return asString(originalValue);
94  }
95  return jsString(exec, string);
96 }
97 
98 // Helper that tries to use the JSString substring sharing mechanism if 'originalValue' is a JSString.
99 // FIXME: Basically any use of this is bad. toString() returns a JSString* so we don't need to
100 // pass around the originalValue; we could just pass around the JSString*. And since we've
101 // resolved it, we know that we can just allocate the substring cell directly.
102 // https://bugs.webkit.org/show_bug.cgi?id=158140
103 static inline JSString* jsSubstring(ExecState* exec, JSValue originalValue, const String& string, unsigned offset, unsigned length)
104 {
105  if (originalValue.isString()) {
106  ASSERT(asString(originalValue)->value(exec) == string);
107  return jsSubstring(exec, asString(originalValue), offset, length);
108  }
109  return jsSubstring(exec, string, offset, length);
110 }
