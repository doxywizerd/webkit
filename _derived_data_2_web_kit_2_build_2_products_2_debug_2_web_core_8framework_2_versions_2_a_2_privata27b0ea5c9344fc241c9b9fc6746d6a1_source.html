230 template<class Encoder>
231 void ResourceResponseBase::encode(Encoder& encoder) const
232 {
233  encoder << m_isNull;
234  if (m_isNull)
235  return;
236  lazyInit(AllFields);
237 
238  encoder << m_url;
239  encoder << m_mimeType;
240  encoder << static_cast<int64_t>(m_expectedContentLength);
241  encoder << m_textEncodingName;
242  encoder << m_httpStatusText;
243  encoder << m_httpVersion;
244  encoder << m_httpHeaderFields;
245 
246  // We don't want to put the networkLoadMetrics info
247  // into the disk cache, because we will never use the old info.
248  if (Encoder::isIPCEncoder)
249  encoder << m_networkLoadMetrics;
250 
251  encoder << m_httpStatusCode;
252  encoder << m_certificateInfo;
253  encoder.encodeEnum(m_source);
254  encoder << m_cacheBodyKey;
255  encoder.encodeEnum(m_type);
256  encoder << m_isRedirected;
257 }

259 template<class Decoder>
260 bool ResourceResponseBase::decode(Decoder& decoder, ResourceResponseBase& response)
261 {
262  ASSERT(response.m_isNull);
263  bool responseIsNull;
264  if (!decoder.decode(responseIsNull))
265  return false;
266  if (responseIsNull)
267  return true;
268 
269  if (!decoder.decode(response.m_url))
270  return false;
271  if (!decoder.decode(response.m_mimeType))
272  return false;
273  int64_t expectedContentLength;
274  if (!decoder.decode(expectedContentLength))
275  return false;
276  response.m_expectedContentLength = expectedContentLength;
277  if (!decoder.decode(response.m_textEncodingName))
278  return false;
279  if (!decoder.decode(response.m_httpStatusText))
280  return false;
281  if (!decoder.decode(response.m_httpVersion))
282  return false;
283  if (!decoder.decode(response.m_httpHeaderFields))
284  return false;
285  // The networkLoadMetrics info is only send over IPC and not stored in disk cache.
286  if (Decoder::isIPCDecoder && !decoder.decode(response.m_networkLoadMetrics))
287  return false;
288  if (!decoder.decode(response.m_httpStatusCode))
289  return false;
290  if (!decoder.decode(response.m_certificateInfo))
291  return false;
292  if (!decoder.decodeEnum(response.m_source))
293  return false;
294  if (!decoder.decode(response.m_cacheBodyKey))
295  return false;
296  if (!decoder.decodeEnum(response.m_type))
297  return false;
298  if (!decoder.decode(response.m_isRedirected))
299  return false;
300  response.m_isNull = false;
301 
302  return true;
303 }
