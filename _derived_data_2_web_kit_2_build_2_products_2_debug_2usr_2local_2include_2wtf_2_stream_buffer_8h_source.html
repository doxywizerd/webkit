54  void append(const T* data, size_t size)
55  {
56  if (!size)
57  return;
58 
59  m_size += size;
60  while (size) {
61  if (!m_buffer.size() || m_buffer.last()->size() == BlockSize)
62  m_buffer.append(std::make_unique<Block>());
63  size_t appendSize = std::min(BlockSize - m_buffer.last()->size(), size);
64  m_buffer.last()->append(data, appendSize);
65  data += appendSize;
66  size -= appendSize;
67  }
68  }

70  // This function consume data in the fist block.

72  void consume(size_t size)
73  {
74  ASSERT(m_size >= size);
75  if (!m_size)
76  return;
77 
78  ASSERT(m_buffer.size() > 0);
79  ASSERT(m_readOffset + size <= m_buffer.first()->size());
80  m_readOffset += size;
81  m_size -= size;
82  if (m_readOffset >= m_buffer.first()->size()) {
83  m_readOffset = 0;
84  m_buffer.removeFirst();
85  }
86  }
