48 template <typename CallBackType>
49 void forEachInIterable(ExecState* exec, JSValue iterable, const CallBackType& callback)
50 {
51  auto& vm = exec->vm();
52  auto scope = DECLARE_THROW_SCOPE(vm);
53 
54  JSValue iterator = iteratorForIterable(exec, iterable);
55  RETURN_IF_EXCEPTION(scope, void());
56  while (true) {
57  JSValue next = iteratorStep(exec, iterator);
58  if (UNLIKELY(scope.exception()) || next.isFalse())
59  return;
60 
61  JSValue nextValue = iteratorValue(exec, next);
62  RETURN_IF_EXCEPTION(scope, void());
63 
64  callback(vm, exec, nextValue);
65  if (UNLIKELY(scope.exception())) {
66  scope.release();
67  iteratorClose(exec, iterator);
68  return;
69  }
70  }
71 }
