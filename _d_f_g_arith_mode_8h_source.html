32 // Arith::Mode describes the mode of an arithmetic operation that speculates integer.
33 // Note that not all modes are valid for all operations.
34 namespace Arith {
35 enum Mode {
36  NotSet, // Arithmetic mode is either not relevant because we're using doubles anyway or we are at a phase in compilation where we don't know what we're doing, yet. Should never see this after FixupPhase except for nodes that take doubles as inputs already.
37  Unchecked, // Don't check anything and just do the direct hardware operation.
38  CheckOverflow, // Check for overflow but don't bother with negative zero.
39  CheckOverflowAndNegativeZero, // Check for both overflow and negative zero.
40  DoOverflow // Up-convert to the smallest type that soundly represents all possible results after input type speculation.
41 };

43 // Define the type of operation the rounding operation will perform.
44 enum class RoundingMode {

46  Int32WithNegativeZeroCheck, // The round operation produces a integer and checks for -0.
47  Double // The round operation produce a double. The result can be -0, NaN or (+/-)Infinity.
48 };

52 inline bool doesOverflow(Arith::Mode mode)
53 {
54  switch (mode) {
55  case Arith::NotSet:
56  ASSERT_NOT_REACHED();
57 #if ASSERT_DISABLED
58  FALLTHROUGH;
59 #endif
60  case Arith::Unchecked:
61  case Arith::CheckOverflow:
62  case Arith::CheckOverflowAndNegativeZero:
63  return false;
64  case Arith::DoOverflow:
65  return true;
66  }
67  ASSERT_NOT_REACHED();
68  return true;
69 }

107 inline bool subsumes(Arith::Mode earlier, Arith::Mode later)
108 {
109  switch (earlier) {
110  case Arith::CheckOverflow:
111  switch (later) {
112  case Arith::Unchecked:
113  case Arith::CheckOverflow:
114  return true;
115  default:
116  return false;
117  }
118  case Arith::CheckOverflowAndNegativeZero:
119  switch (later) {
120  case Arith::Unchecked:
121  case Arith::CheckOverflow:
122  case Arith::CheckOverflowAndNegativeZero:
123  return true;
124  default:
125  return false;
126  }
127  default:
128  return earlier == later;
129  }
