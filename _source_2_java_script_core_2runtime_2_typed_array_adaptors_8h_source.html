82  static std::optional<Type> toNativeFromInt32WithoutCoercion(int32_t value)
83  {
84  if ((value >= 0 && static_cast<uint32_t>(value) > static_cast<uint32_t>(maxValue)) || value < static_cast<int32_t>(minValue))
85  return std::nullopt;

87  }
88 
89  static std::optional<Type> toNativeFromUint32WithoutCoercion(uint32_t value)
90  {

92  return std::nullopt;
93 
94  return static_cast<Type>(value);
95  }
96 
97  static std::optional<Type> toNativeFromDoubleWithoutCoercion(double value)
98  {
99  Type integer = static_cast<Type>(value);
100  if (static_cast<double>(integer) != value)
101  return std::nullopt;
102
