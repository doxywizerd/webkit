75 template<typename WrappedType, typename ToJSFunction>
76 void toWrapper(CCallHelpers& jit, JSC::DOMJIT::PatchpointParams& params, GPRReg wrapped, GPRReg globalObject, JSValueRegs result, ToJSFunction function, JSC::JSValue globalObjectConstant)
77 {
78  ASSERT(wrapped != result.payloadGPR());
79  ASSERT(globalObject != result.payloadGPR());
80  GPRReg payloadGPR = result.payloadGPR();
81  CCallHelpers::JumpList slowCases;
82 
83  if (globalObjectConstant) {
84  if (!JSC::jsCast<JSDOMGlobalObject*>(globalObjectConstant)->worldIsNormal()) {
85  slowCases.append(jit.jump());
86  params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
87  return;
88  }
89  } else
90  slowCases.append(branchIfNotWorldIsNormal(jit, globalObject));
91 
92  tryLookUpWrapperCache<WrappedType>(jit, slowCases, wrapped, payloadGPR);
93  jit.boxCell(payloadGPR, result);
94  params.addSlowPathCall(slowCases, jit, function, result, globalObject, wrapped);
95 }
