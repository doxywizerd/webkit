59 inline double parseDouble(const UChar* string, size_t length, size_t& parsedLength)
60 {
61  const size_t conversionBufferSize = 64;
62  if (length > conversionBufferSize)
63  return Internal::parseDoubleFromLongString(string, length, parsedLength);
64  LChar conversionBuffer[conversionBufferSize];
65  for (int i = 0; i < static_cast<int>(length); ++i)
66  conversionBuffer[i] = isASCII(string[i]) ? string[i] : 0;
67  return parseDouble(conversionBuffer, length, parsedLength);
68 }
