45  // Simulates the two transports connecting to each other.
46  // If |asymmetric| is true this method only affects this FakePacketTransport.
47  // If false, it affects |dest| as well.
48  void SetDestination(FakePacketTransport* dest, bool asymmetric) {
49  if (dest) {
50  dest_ = dest;
51  set_writable(true);
52  if (!asymmetric) {
53  dest->SetDestination(this, true);
54  }
55  } else {
56  // Simulates loss of connectivity, by asymmetrically forgetting dest_.
57  dest_ = nullptr;
58  set_writable(false);
59  }
60  }
