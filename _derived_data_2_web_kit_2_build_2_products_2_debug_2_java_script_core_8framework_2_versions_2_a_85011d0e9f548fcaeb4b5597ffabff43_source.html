147  static CodeOrigin* computeCallerSkippingTailCalls(InlineCallFrame* inlineCallFrame, Kind* callerCallKind = nullptr)
148  {
149  CodeOrigin* codeOrigin;
150  bool tailCallee;
151  int callKind;
152  do {
153  tailCallee = inlineCallFrame->isTail();
154  callKind = inlineCallFrame->kind;
155  codeOrigin = &inlineCallFrame->directCaller;
156  inlineCallFrame = codeOrigin->inlineCallFrame;
157  } while (inlineCallFrame && tailCallee);

161

164 
165  return codeOrigin;
166  }
