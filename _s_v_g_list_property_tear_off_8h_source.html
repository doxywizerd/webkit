160  bool processIncomingListItemWrapper(Ref<ListItemTearOff>& newItem, unsigned* indexToModify) override
161  {
162  SVGAnimatedProperty* animatedPropertyOfItem = newItem->animatedProperty();
163 
164  // newItem has been created manually, it doesn't belong to any SVGElement.
165  // (for example: "textElement.x.baseVal.appendItem(svgsvgElement.createSVGLength())")
166  if (!animatedPropertyOfItem)
167  return true;
168 
169  // newItem belongs to a SVGElement, but its associated SVGAnimatedProperty is not an animated list tear off.
170  // (for example: "textElement.x.baseVal.appendItem(rectElement.width.baseVal)")
171  if (!animatedPropertyOfItem->isAnimatedListTearOff()) {
172  // We have to copy the incoming newItem, as we're not allowed to insert this tear off as is into our wrapper cache.
173  // Otherwhise we'll end up having two SVGAnimatedPropertys that operate on the same SVGPropertyTearOff. Consider the example above:
174  // SVGRectElements SVGAnimatedLength 'width' property baseVal points to the same tear off object
175  // that's inserted into SVGTextElements SVGAnimatedLengthList 'x'. textElement.x.baseVal.getItem(0).value += 150 would
176  // mutate the rectElement width _and_ the textElement x list. That's obviously wrong, take care of that.
177  newItem = ListItemTearOff::create(newItem->propertyReference());
178  return true;
179  }
180 
181  // Spec: If newItem is already in a list, it is removed from its previous list before it is inserted into this list.
182  // 'newItem' is already living in another list. If it's not our list, synchronize the other lists wrappers after the removal.
183  bool livesInOtherList = animatedPropertyOfItem != m_animatedProperty.ptr();
184  AnimatedListPropertyTearOff* propertyTearOff = static_cast<AnimatedListPropertyTearOff*>(animatedPropertyOfItem);
185  int indexToRemove = propertyTearOff->findItem(newItem.ptr());
186  ASSERT(indexToRemove != -1);
187 
188  // Do not remove newItem if already in this list at the target index.
189  if (!livesInOtherList && indexToModify && static_cast<unsigned>(indexToRemove) == *indexToModify)
190  return false;
191 
192  propertyTearOff->removeItemFromList(indexToRemove, true);
193 
194  if (!indexToModify)
195  return true;
196 
197  // If the item lived in our list, adjust the insertion index.
198  if (!livesInOtherList) {
199  unsigned& index = *indexToModify;
200  // Spec: If the item is already in this list, note that the index of the item to (replace|insert before) is before the removal of the item.
201  if (static_cast<unsigned>(indexToRemove) < index)
202  --index;
203  }
204 
205  return true;
206  }
207 
208  // Back pointer to the animated property that created us
209  // For example (text.x.baseVal): m_animatedProperty points to the 'x' SVGAnimatedLengthList object
210  Ref<AnimatedListPropertyTearOff> m_animatedProperty;
211 };
