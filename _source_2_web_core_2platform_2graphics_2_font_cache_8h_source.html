69 // This key contains the FontDescription fields other than family that matter when fetching FontDatas (platform fonts).
70 struct FontDescriptionKey {
71  FontDescriptionKey() = default;
72 
73  FontDescriptionKey(const FontDescription& description)
74  : m_size(description.computedPixelSize())
75  , m_fontSelectionRequest(description.fontSelectionRequest())
76  , m_flags(makeFlagsKey(description))
77  , m_featureSettings(description.featureSettings())
78 #if ENABLE(VARIATION_FONTS)
79  , m_variationSettings(description.variationSettings())
80 #endif
81  { }
82 
83  explicit FontDescriptionKey(WTF::HashTableDeletedValueType)
84  : m_size(cHashTableDeletedSize)
85  { }
86 
87  bool operator==(const FontDescriptionKey& other) const
88  {
89  return m_size == other.m_size
90  && m_fontSelectionRequest == other.m_fontSelectionRequest
91  && m_flags == other.m_flags
92 #if ENABLE(VARIATION_FONTS)
93  && m_variationSettings == other.m_variationSettings
94 #endif
95  && m_featureSettings == other.m_featureSettings;
96  }
97 
98  bool operator!=(const FontDescriptionKey& other) const
99  {
100  return !(*this == other);
101  }
102 
103  bool isHashTableDeletedValue() const { return m_size == cHashTableDeletedSize; }
104 
105  inline unsigned computeHash() const
106  {
107  IntegerHasher hasher;
108  hasher.add(m_size);
109  hasher.add(m_fontSelectionRequest.weight);
110  hasher.add(m_fontSelectionRequest.width);
111  hasher.add(m_fontSelectionRequest.slope);
112  for (unsigned flagItem : m_flags)
113  hasher.add(flagItem);
114  hasher.add(m_featureSettings.hash());
115 #if ENABLE(VARIATION_FONTS)
116  hasher.add(m_variationSettings.hash());
117 #endif
118  return hasher.hash();
119  }
120 
121 private:
122  static std::array<unsigned, 2> makeFlagsKey(const FontDescription& description)
123  {
124  static_assert(USCRIPT_CODE_LIMIT < 0x1000, "Script code must fit in an unsigned along with the other flags");
125  unsigned first = static_cast<unsigned>(description.script()) << 11
126  | static_cast<unsigned>(description.textRenderingMode()) << 9
127  | static_cast<unsigned>(description.fontSynthesis()) << 6
128  | static_cast<unsigned>(description.widthVariant()) << 4
129  | static_cast<unsigned>(description.nonCJKGlyphOrientation()) << 3
130  | static_cast<unsigned>(description.orientation()) << 2
131  | static_cast<unsigned>(description.renderingMode());
132  unsigned second = static_cast<unsigned>(description.variantEastAsianRuby()) << 27
133  | static_cast<unsigned>(description.variantEastAsianWidth()) << 25
134  | static_cast<unsigned>(description.variantEastAsianVariant()) << 22
135  | static_cast<unsigned>(description.variantAlternates()) << 21
136  | static_cast<unsigned>(description.variantNumericSlashedZero()) << 20
137  | static_cast<unsigned>(description.variantNumericOrdinal()) << 19
138  | static_cast<unsigned>(description.variantNumericFraction()) << 17
139  | static_cast<unsigned>(description.variantNumericSpacing()) << 15
140  | static_cast<unsigned>(description.variantNumericFigure()) << 13
141  | static_cast<unsigned>(description.variantCaps()) << 10
142  | static_cast<unsigned>(description.variantPosition()) << 8
143  | static_cast<unsigned>(description.variantContextualAlternates()) << 6
144  | static_cast<unsigned>(description.variantHistoricalLigatures()) << 4
145  | static_cast<unsigned>(description.variantDiscretionaryLigatures()) << 2
146  | static_cast<unsigned>(description.variantCommonLigatures());
147  return {{ first, second }};
148  }
149 
150  static const unsigned cHashTableDeletedSize = 0xFFFFFFFFU;
151 
152  // FontCascade::locale() is explicitly not included in this struct.
153  unsigned m_size { 0 };
154  FontSelectionRequest m_fontSelectionRequest;
155  std::array<unsigned, 2> m_flags {{ 0, 0 }};
156  FontFeatureSettings m_featureSettings;
157 #if ENABLE(VARIATION_FONTS)
158  FontVariationSettings m_variationSettings;
159 #endif
160 };
