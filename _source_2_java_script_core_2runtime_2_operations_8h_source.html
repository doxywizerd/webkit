87 {

92  int32_t length2 = u2.length();

122 {

158 // See ES5 11.8.1/11.8.2/11.8.5 for definition of leftFirst, this value ensures correct
159 // evaluation ordering for argument conversions for '<' and '>'. For '<' pass the value
160 // true, for leftFirst, for '>' pass the value false (and reverse operand order).
161 template<bool leftFirst>
162 ALWAYS_INLINE bool jsLess(CallFrame* callFrame, JSValue v1, JSValue v2)
163 {
164  VM& vm = callFrame->vm();
165  auto scope = DECLARE_THROW_SCOPE(vm);
166 
167  if (v1.isInt32() && v2.isInt32())
168  return v1.asInt32() < v2.asInt32();
169 
170  if (v1.isNumber() && v2.isNumber())
171  return v1.asNumber() < v2.asNumber();
172 
173  if (isJSString(v1) && isJSString(v2))
174  return codePointCompareLessThan(asString(v1)->value(callFrame), asString(v2)->value(callFrame));
175 
176  double n1;
177  double n2;
178  JSValue p1;
179  JSValue p2;
180  bool wasNotString1;
181  bool wasNotString2;
182  if (leftFirst) {
183  wasNotString1 = v1.getPrimitiveNumber(callFrame, n1, p1);
184  RETURN_IF_EXCEPTION(scope, false);
185  wasNotString2 = v2.getPrimitiveNumber(callFrame, n2, p2);
186  } else {
187  wasNotString2 = v2.getPrimitiveNumber(callFrame, n2, p2);
188  RETURN_IF_EXCEPTION(scope, false);
189  wasNotString1 = v1.getPrimitiveNumber(callFrame, n1, p1);
190  }
191  RETURN_IF_EXCEPTION(scope, false);
192 
193  if (wasNotString1 | wasNotString2)
194  return n1 < n2;
195  return codePointCompareLessThan(asString(p1)->value(callFrame), asString(p2)->value(callFrame));
196 }

247 ALWAYS_INLINE JSValue jsAdd(CallFrame* callFrame, JSValue v1, JSValue v2)
248 {
249  if (v1.isNumber() && v2.isNumber())
250  return jsNumber(v1.asNumber() + v2.asNumber());
251 
252  if (v1.isString() && !v2.isObject())
253  return jsString(callFrame, asString(v1), v2.toString(callFrame));
254 
255  // All other cases are pretty uncommon
256  return jsAddSlowCase(callFrame, v1, v2);
257 }
