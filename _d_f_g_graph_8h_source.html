752  // Quickly get all of the non-argument locals live at the given point. This doesn't give you
753  // any arguments because those are all presumed live. You can call forAllLiveInBytecode() to
754  // also get the arguments. This is much faster than calling isLiveInBytecode() for each local.
755  template<typename Functor>
756  void forAllLocalsLiveInBytecode(CodeOrigin codeOrigin, const Functor& functor)
757  {
758  // Support for not redundantly reporting arguments. Necessary because in case of a varargs
759  // call, only the callee knows that arguments are live while in the case of a non-varargs
760  // call, both callee and caller will see the variables live.
761  VirtualRegister exclusionStart;
762  VirtualRegister exclusionEnd;
763 
764  CodeOrigin* codeOriginPtr = &codeOrigin;
765 
766  for (;;) {
767  InlineCallFrame* inlineCallFrame = codeOriginPtr->inlineCallFrame;
768  VirtualRegister stackOffset(inlineCallFrame ? inlineCallFrame->stackOffset : 0);
769 
770  if (inlineCallFrame) {
771  if (inlineCallFrame->isClosureCall)
772  functor(stackOffset + CallFrameSlot::callee);
773  if (inlineCallFrame->isVarargs())
774  functor(stackOffset + CallFrameSlot::argumentCount);
775  }
776 
777  CodeBlock* codeBlock = baselineCodeBlockFor(inlineCallFrame);
778  FullBytecodeLiveness& fullLiveness = livenessFor(codeBlock);
779  const FastBitVector& liveness = fullLiveness.getLiveness(codeOriginPtr->bytecodeIndex);
780  for (unsigned relativeLocal = codeBlock->m_numCalleeLocals; relativeLocal--;) {
781  VirtualRegister reg = stackOffset + virtualRegisterForLocal(relativeLocal);
782 
783  // Don't report if our callee already reported.
784  if (reg >= exclusionStart && reg < exclusionEnd)
785  continue;
786 
787  if (liveness[relativeLocal])
788  functor(reg);
789  }
790 
791  if (!inlineCallFrame)
792  break;
793 
794  // Arguments are always live. This would be redundant if it wasn't for our
795  // op_call_varargs inlining. See the comment above.
796  exclusionStart = stackOffset + CallFrame::argumentOffsetIncludingThis(0);
797  exclusionEnd = stackOffset + CallFrame::argumentOffsetIncludingThis(inlineCallFrame->arguments.size());
798 
799  // We will always have a "this" argument and exclusionStart should be a smaller stack
800  // offset than exclusionEnd.
801  ASSERT(exclusionStart < exclusionEnd);
802 
803  for (VirtualRegister reg = exclusionStart; reg < exclusionEnd; reg += 1)
804  functor(reg);
805 
806  codeOriginPtr = inlineCallFrame->getCallerSkippingTailCalls();
807 
808  // The first inline call frame could be an inline tail call
809  if (!codeOriginPtr)
810  break;
811  }
812  }
