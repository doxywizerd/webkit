225  {
226  if (imm >= -32768 && imm <= 32767)

229  ori(dest, MIPSRegisters::zero, imm);

233  ori(dest, dest, imm);

949  void relocateJumps(void* oldBase, void* newBase)
950  {
951  // Check each jump
952  for (Jumps::Iterator iter = m_jumps.begin(); iter != m_jumps.end(); ++iter) {
953  int pos = iter->m_offset;
954  MIPSWord* insn = reinterpret_cast<MIPSWord*>(reinterpret_cast<intptr_t>(newBase) + pos);
955  insn = insn + 2;
956  // Need to make sure we have 5 valid instructions after pos
957  if ((unsigned)pos >= m_buffer.codeSize() - 5 * sizeof(MIPSWord))
958  continue;
959 
960  if ((*insn & 0xfc000000) == 0x08000000) { // j
961  int offset = *insn & 0x03ffffff;
962  int oldInsnAddress = (int)insn - (int)newBase + (int)oldBase;
963  int topFourBits = (oldInsnAddress + 4) >> 28;
964  int oldTargetAddress = (topFourBits << 28) | (offset << 2);
965  int newTargetAddress = oldTargetAddress - (int)oldBase + (int)newBase;
966  int newInsnAddress = (int)insn;
967  if (((newInsnAddress + 4) >> 28) == (newTargetAddress >> 28))
968  *insn = 0x08000000 | ((newTargetAddress >> 2) & 0x3ffffff);
969  else {
970  /* lui */
971  *insn = 0x3c000000 | (MIPSRegisters::t9 << OP_SH_RT) | ((newTargetAddress >> 16) & 0xffff);
972  /* ori */
973  *(insn + 1) = 0x34000000 | (MIPSRegisters::t9 << OP_SH_RT) | (MIPSRegisters::t9 << OP_SH_RS) | (newTargetAddress & 0xffff);
974  /* jr */
975  *(insn + 2) = 0x00000008 | (MIPSRegisters::t9 << OP_SH_RS);
976  }
977  } else if ((*insn & 0xffe00000) == 0x3c000000) { // lui
978  int high = (*insn & 0xffff) << 16;
979  int low = *(insn + 1) & 0xffff;
980  int oldTargetAddress = high | low;
981  int newTargetAddress = oldTargetAddress - (int)oldBase + (int)newBase;
982  /* lui */
983  *insn = 0x3c000000 | (MIPSRegisters::t9 << OP_SH_RT) | ((newTargetAddress >> 16) & 0xffff);
984  /* ori */
985  *(insn + 1) = 0x34000000 | (MIPSRegisters::t9 << OP_SH_RT) | (MIPSRegisters::t9 << OP_SH_RS) | (newTargetAddress & 0xffff);

991  static int linkWithOffset(MIPSWord* insn, void* to)
992  {
993  ASSERT((*insn & 0xfc000000) == 0x10000000 // beq
994  || (*insn & 0xfc000000) == 0x14000000 // bne
995  || (*insn & 0xffff0000) == 0x45010000 // bc1t
996  || (*insn & 0xffff0000) == 0x45000000); // bc1f
997  intptr_t diff = (reinterpret_cast<intptr_t>(to) - reinterpret_cast<intptr_t>(insn) - 4) >> 2;
998 
999  if (diff < -32768 || diff > 32767 || *(insn + 2) != 0x10000003) {
1000  /*
1001  Convert the sequence:
1002  beq $2, $3, target
1003  nop
1004  b 1f
1005  nop
1006  nop
1007  nop
1008  1:
1009 
1010  to the new sequence if possible:
1011  bne $2, $3, 1f
1012  nop
1013  j target
1014  nop
1015  nop
1016  nop
1017  1:
1018 
1019  OR to the new sequence:
1020  bne $2, $3, 1f
1021  nop
1022  lui $25, target >> 16
1023  ori $25, $25, target & 0xffff
1024  jr $25
1025  nop
1026  1:
1027 
1028  Note: beq/bne/bc1t/bc1f are converted to bne/beq/bc1f/bc1t.
1029  */
1030 
1031  if (*(insn + 2) == 0x10000003) {
1032  if ((*insn & 0xfc000000) == 0x10000000) // beq
1033  *insn = (*insn & 0x03ff0000) | 0x14000005; // bne
1034  else if ((*insn & 0xfc000000) == 0x14000000) // bne
1035  *insn = (*insn & 0x03ff0000) | 0x10000005; // beq
1036  else if ((*insn & 0xffff0000) == 0x45010000) // bc1t
1037  *insn = 0x45000005; // bc1f
1038  else if ((*insn & 0xffff0000) == 0x45000000) // bc1f
1039  *insn = 0x45010005; // bc1t
1040  else
1041  ASSERT(0);
1042  }
1043 
1044  insn = insn + 2;
1045  if ((reinterpret_cast<intptr_t>(insn) + 4) >> 28
1046  == reinterpret_cast<intptr_t>(to) >> 28) {
1047  *insn = 0x08000000 | ((reinterpret_cast<intptr_t>(to) >> 2) & 0x3ffffff);
1048  *(insn + 1) = 0;
1049  return 4 * sizeof(MIPSWord);
1050  }
1051 
1052  intptr_t newTargetAddress = reinterpret_cast<intptr_t>(to);
1053  /* lui */
1054  *insn = 0x3c000000 | (MIPSRegisters::t9 << OP_SH_RT) | ((newTargetAddress >> 16) & 0xffff);
1055  /* ori */
1056  *(insn + 1) = 0x34000000 | (MIPSRegisters::t9 << OP_SH_RT) | (MIPSRegisters::t9 << OP_SH_RS) | (newTargetAddress & 0xffff);

1059  return 5 * sizeof(MIPSWord);
1060  }
1061

1063  return sizeof(MIPSWord);
1064  }
