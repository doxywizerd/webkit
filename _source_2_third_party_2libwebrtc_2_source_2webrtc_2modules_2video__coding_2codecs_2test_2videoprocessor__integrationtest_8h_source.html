335  // Reset quantities after each encoder update, update the target
336  // per-frame bandwidth.
337  void ResetRateControlMetrics(int num_frames) {
338  for (int i = 0; i < num_temporal_layers_; i++) {
339  num_frames_per_update_[i] = 0;
340  sum_frame_size_mismatch_[i] = 0.0f;
341  sum_encoded_frame_size_[i] = 0.0f;
342  encoding_bitrate_[i] = 0.0f;
343  // Update layer per-frame-bandwidth.
344  per_frame_bandwidth_[i] = static_cast<float>(bit_rate_layer_[i]) /
345  static_cast<float>(frame_rate_layer_[i]);
346  }
347  // Set maximum size of key frames, following setting in the VP8 wrapper.
348  float max_key_size = kScaleKeyFrameSize * kOptimalBufferSize * frame_rate_;
349  // We don't know exact target size of the key frames (except for first one),
350  // but the minimum in libvpx is ~|3 * per_frame_bandwidth| and maximum is
351  // set by |max_key_size_ * per_frame_bandwidth|. Take middle point/average
352  // as reference for mismatch. Note key frames always correspond to base
353  // layer frame in this test.
354  target_size_key_frame_ = 0.5 * (3 + max_key_size) * per_frame_bandwidth_[0];
355  num_frames_total_ = 0;
356  sum_encoded_frame_size_total_ = 0.0f;
357  encoding_bitrate_total_ = 0.0f;
358  perc_encoding_rate_mismatch_ = 0.0f;
359  num_frames_to_hit_target_ = num_frames;
360  encoding_rate_within_target_ = false;
361  sum_key_frame_size_mismatch_ = 0.0;
362  num_key_frames_ = 0;
363  }

365  // For every encoded frame, update the rate control metrics.
366  void UpdateRateControlMetrics(int frame_num, FrameType frame_type) {
367  float encoded_size_kbits = processor_->EncodedFrameSize() * 8.0f / 1000.0f;
368  // Update layer data.
369  // Update rate mismatch relative to per-frame bandwidth for delta frames.
370  if (frame_type == kVideoFrameDelta) {
371  // TODO(marpan): Should we count dropped (zero size) frames in mismatch?
372  sum_frame_size_mismatch_[layer_] +=
373  fabs(encoded_size_kbits - per_frame_bandwidth_[layer_]) /
374  per_frame_bandwidth_[layer_];
375  } else {
376  float target_size = (frame_num == 1) ? target_size_key_frame_initial_
377  : target_size_key_frame_;
378  sum_key_frame_size_mismatch_ +=
379  fabs(encoded_size_kbits - target_size) / target_size;
380  num_key_frames_ += 1;
381  }
382  sum_encoded_frame_size_[layer_] += encoded_size_kbits;
383  // Encoding bitrate per layer: from the start of the update/run to the
384  // current frame.
385  encoding_bitrate_[layer_] = sum_encoded_frame_size_[layer_] *
386  frame_rate_layer_[layer_] /
387  num_frames_per_update_[layer_];
388  // Total encoding rate: from the start of the update/run to current frame.
389  sum_encoded_frame_size_total_ += encoded_size_kbits;
390  encoding_bitrate_total_ =
391  sum_encoded_frame_size_total_ * frame_rate_ / num_frames_total_;
392  perc_encoding_rate_mismatch_ =
393  100 * fabs(encoding_bitrate_total_ - bit_rate_) / bit_rate_;
394  if (perc_encoding_rate_mismatch_ < kPercTargetvsActualMismatch &&
395  !encoding_rate_within_target_) {
396  num_frames_to_hit_target_ = num_frames_total_;
397  encoding_rate_within_target_ = true;
398  }
399  }

437  int perc_frame_size_mismatch =
438  100 * sum_frame_size_mismatch_[i] / num_frames_per_update_[i];
439  int perc_encoding_rate_mismatch =

446  " Layer Encoding bit rate: %f, \n"

449  " Number of frame processed per layer: %d \n",
450  bit_rate_layer_[i], frame_rate_layer_[i], per_frame_bandwidth_[i],
451  encoding_bitrate_[i], perc_frame_size_mismatch,
452  perc_encoding_rate_mismatch, num_frames_per_update_[i]);
453  EXPECT_LE(perc_frame_size_mismatch, max_delta_frame_size_mismatch);
454  EXPECT_LE(perc_encoding_rate_mismatch, max_encoding_rate_mismatch);
455  }
456  printf("\n");
457  EXPECT_LE(num_frames_to_hit_target_, max_time_hit_target);
458  EXPECT_LE(num_dropped_frames, max_num_dropped_frames);
459  EXPECT_EQ(num_resize_actions, num_spatial_resizes);
460  EXPECT_EQ(num_key_frames_, num_key_frames);
461  }

472  // Layer index corresponding to frame number, for up to 3 layers.
473  void LayerIndexForFrame(int frame_number) {
474  if (num_temporal_layers_ == 1) {
475  layer_ = 0;
476  } else if (num_temporal_layers_ == 2) {
477  // layer 0: 0 2 4 ...
478  // layer 1: 1 3
479  if (frame_number % 2 == 0) {
480  layer_ = 0;
481  } else {
482  layer_ = 1;
483  }
484  } else if (num_temporal_layers_ == 3) {
485  // layer 0: 0 4 8 ...
486  // layer 1: 2 6
487  // layer 2: 1 3 5 7
488  if (frame_number % 4 == 0) {
489  layer_ = 0;
490  } else if ((frame_number + 2) % 4 == 0) {
491  layer_ = 1;
492  } else if ((frame_number + 1) % 2 == 0) {
493  layer_ = 2;
494  }
495  } else {
496  RTC_NOTREACHED() << "Max 3 layers are supported.";
497  }
498  }

500  // Set the bitrate and frame rate per layer, for up to 3 layers.
501  void SetLayerRates() {
502  RTC_DCHECK_LE(num_temporal_layers_, 3);
503  for (int i = 0; i < num_temporal_layers_; i++) {
504  float bit_rate_ratio =
505  kVp8LayerRateAlloction[num_temporal_layers_ - 1][i];
506  if (i > 0) {
507  float bit_rate_delta_ratio =
508  kVp8LayerRateAlloction[num_temporal_layers_ - 1][i] -
509  kVp8LayerRateAlloction[num_temporal_layers_ - 1][i - 1];
510  bit_rate_layer_[i] = bit_rate_ * bit_rate_delta_ratio;
511  } else {
512  bit_rate_layer_[i] = bit_rate_ * bit_rate_ratio;
513  }
514  frame_rate_layer_[i] =
515  frame_rate_ / static_cast<float>(1 << (num_temporal_layers_ - 1));
516  }
517  if (num_temporal_layers_ == 3) {
518  frame_rate_layer_[2] = frame_rate_ / 2.0f;
519  }
520  }
