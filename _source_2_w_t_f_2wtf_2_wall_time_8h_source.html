37 // The current time according to a wall clock (aka real time clock). This uses floating point
38 // internally so that you can reason about infinity and other things that arise in math. It's
39 // acceptable to use this to wrap NaN times, negative times, and infinite times, so long as they
40 // are relative to the same clock. Specifically, WallTime should be used in agreement with the
41 // principle that WallTime::now().secondsSinceEpoch().value() is the same as WTF::currentTime().
42 class WallTime {
43 public:
44  static const ClockType clockType = ClockType::Wall;
45 
46  // This is the epoch. So, x.secondsSinceEpoch() should be the same as x - WallTime().
47  WallTime() { }

49  // Call this if you know for sure that the double represents time according to
50  // WTF::currentTime(). It must be in seconds and it must be from the same time source.
51  static WallTime fromRawSeconds(double value)
52  {
53  WallTime result;
54  result.m_value = value;
55  return result;
56  }
