321 ALWAYS_INLINE FloatRect Font::boundsForGlyph(Glyph glyph) const
322 {
323  if (isZeroWidthSpaceGlyph(glyph))
324  return FloatRect();
325 
326  FloatRect bounds;
327  if (m_glyphToBoundsMap) {
328  bounds = m_glyphToBoundsMap->metricsForGlyph(glyph);
329  if (bounds.width() != cGlyphSizeUnknown)
330  return bounds;
331  }
332 
333  bounds = platformBoundsForGlyph(glyph);
334  if (!m_glyphToBoundsMap)
335  m_glyphToBoundsMap = std::make_unique<GlyphMetricsMap<FloatRect>>();
336  m_glyphToBoundsMap->setMetricsForGlyph(glyph, bounds);
337  return bounds;
338 }

340 ALWAYS_INLINE float Font::widthForGlyph(Glyph glyph) const
341 {
342  // The optimization of returning 0 for the zero-width-space glyph is incorrect for the LastResort font,
343  // used in place of the actual font when isLoading() is true on both macOS and iOS.
344  // The zero-width-space glyph in that font does not have a width of zero and, further, that glyph is used
345  // for many other characters and must not be zero width when used for them.
346  if (isZeroWidthSpaceGlyph(glyph) && !isLoading())
347  return 0;
348 
349  float width = m_glyphToWidthMap.metricsForGlyph(glyph);
350  if (width != cGlyphSizeUnknown)
351  return width;
352 
353 #if ENABLE(OPENTYPE_VERTICAL)
354  if (m_verticalData) {
355 #if USE(CG) || USE(DIRECT2D) || USE(CAIRO)
356  width = m_verticalData->advanceHeight(this, glyph) + m_syntheticBoldOffset;
357 #else
358  width = m_verticalData->advanceHeight(this, glyph);
359 #endif
360  } else
361 #endif
362  width = platformWidthForGlyph(glyph);
363 
364  m_glyphToWidthMap.setMetricsForGlyph(glyph, width);
365  return width;
366 }
