48 inline JSArrayBufferView* speciesConstruct(ExecState* exec, JSObject* exemplar, MarkedArgumentBuffer& args, const Functor& defaultConstructor)
49 {
50  VM& vm = exec->vm();
51  auto scope = DECLARE_THROW_SCOPE(vm);
52 
53  JSValue constructor = exemplar->get(exec, vm.propertyNames->constructor);
54  RETURN_IF_EXCEPTION(scope, nullptr);
55 
56  if (constructor.isUndefined()) {
57  scope.release();
58  return defaultConstructor();
59  }
60  if (!constructor.isObject()) {
61  throwTypeError(exec, scope, ASCIILiteral("constructor Property should not be null"));
62  return nullptr;
63  }
64 
65  JSValue species = constructor.get(exec, vm.propertyNames->speciesSymbol);
66  RETURN_IF_EXCEPTION(scope, nullptr);
67 
68  if (species.isUndefinedOrNull()) {
69  scope.release();
70  return defaultConstructor();
71  }
72 
73  JSValue result = construct(exec, species, args, "species is not a constructor");
74  RETURN_IF_EXCEPTION(scope, nullptr);
75 
76  if (JSArrayBufferView* view = jsDynamicCast<JSArrayBufferView*>(vm, result)) {
77  if (!view->isNeutered())
78  return view;
79 
80  throwTypeError(exec, scope, ASCIILiteral(typedArrayBufferHasBeenDetachedErrorMessage));
81  return nullptr;
82  }
83 
84  throwTypeError(exec, scope, ASCIILiteral("species constructor did not return a TypedArray View"));
85  return nullptr;
86 }

88 inline unsigned argumentClampedIndexFromStartOrEnd(ExecState* exec, int argument, unsigned length, unsigned undefinedValue = 0)
89 {
90  JSValue value = exec->argument(argument);
91  if (value.isUndefined())
92  return undefinedValue;
93

95  if (indexDouble < 0) {
96  indexDouble += length;
97  return indexDouble < 0 ? 0 : static_cast<unsigned>(indexDouble);
98  }
99  return indexDouble > length ? length : static_cast<unsigned>(indexDouble);
