67 /* This header file contains some common macros for enforcing type checking.
68  * Several, common OpenSSL structures (i.e. stack and lhash) operate on void
69  * pointers, but we wish to have type checking when they are used with a
70  * specific type. */
71 
72 /* CHECKED_CAST casts |p| from type |from| to type |to|. */
73 #define CHECKED_CAST(to, from, p) ((to) (1 ? (p) : (from)0))
74 
75 /* CHECKED_PTR_OF casts a given pointer to void* and statically checks that it
76  * was a pointer to |type|. */
77 #define CHECKED_PTR_OF(type, p) CHECKED_CAST(void*, type*, (p))
