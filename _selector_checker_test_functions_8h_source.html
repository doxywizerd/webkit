75 ALWAYS_INLINE bool isChecked(const Element& element)
76 {
77  // Even though WinIE allows checked and indeterminate to co-exist, the CSS selector spec says that
78  // you can't be both checked and indeterminate. We will behave like WinIE behind the scenes and just
79  // obey the CSS spec here in the test for matching the pseudo.
80  if (is<HTMLInputElement>(element)) {

83  }
84  if (is<HTMLOptionElement>(element))
85  return const_cast<HTMLOptionElement&>(downcast<HTMLOptionElement>(element)).selected();
86 
87  return false;

128 ALWAYS_INLINE bool containslanguageSubtagMatchingRange(StringView language, StringView range, unsigned languageLength, unsigned& position)
129 {
130  unsigned languageSubtagsStartIndex = position;
131  unsigned languageSubtagsEndIndex = languageLength;
132  bool isAsteriskRange = range == "*";
133  do {
134  if (languageSubtagsStartIndex > 0)
135  languageSubtagsStartIndex += 1;
136 
137  languageSubtagsEndIndex = std::min<unsigned>(language.find('-', languageSubtagsStartIndex), languageLength);
138 
139  if (languageSubtagsStartIndex > languageSubtagsEndIndex)
140  return false;
141 
142  StringView languageSubtag = language.substring(languageSubtagsStartIndex, languageSubtagsEndIndex - languageSubtagsStartIndex);
143  bool isEqual = equalIgnoringASCIICase(range, languageSubtag);
144  if (!isAsteriskRange) {
145  if ((!isEqual && !languageSubtagsStartIndex) || (languageSubtag.length() == 1 && languageSubtagsStartIndex > 0))
146  return false;
147  }
148  languageSubtagsStartIndex = languageSubtagsEndIndex;
149  if (isEqual || isAsteriskRange) {
150  position = languageSubtagsStartIndex;
151  return true;
152  }
153 
154  } while (languageSubtagsStartIndex < languageLength);
155  return false;
156 }

158 ALWAYS_INLINE bool matchesLangPseudoClass(const Element& element, const Vector<AtomicString>& argumentList)
159 {
160  AtomicString language;
161 #if ENABLE(VIDEO_TRACK)
162  if (is<WebVTTElement>(element))
163  language = downcast<WebVTTElement>(element).language();
164  else
165 #endif
166  language = element.computeInheritedLanguage();
167 
168  if (language.isEmpty())
169  return false;
170 
171  // Implement basic and extended filterings of given language tags
172  // as specified in www.ietf.org/rfc/rfc4647.txt.
173  StringView languageStringView = language.string();
174  unsigned languageLength = language.length();
175  for (const AtomicString& range : argumentList) {
176  if (range.isEmpty())
177  continue;
178 
179  if (range == "*")
180  return true;
181 
182  StringView rangeStringView = range.string();
183  if (equalIgnoringASCIICase(languageStringView, rangeStringView) && !languageStringView.contains('-'))
184  return true;
185 
186  unsigned rangeLength = rangeStringView.length();
187  unsigned rangeSubtagsStartIndex = 0;
188  unsigned rangeSubtagsEndIndex = rangeLength;
189  unsigned lastMatchedLanguageSubtagIndex = 0;
190 
191  bool matchedRange = true;
192  do {
193  if (rangeSubtagsStartIndex > 0)
194  rangeSubtagsStartIndex += 1;
195  if (rangeSubtagsStartIndex > languageLength)
196  return false;
197  rangeSubtagsEndIndex = std::min<unsigned>(rangeStringView.find('-', rangeSubtagsStartIndex), rangeLength);
198  StringView rangeSubtag = rangeStringView.substring(rangeSubtagsStartIndex, rangeSubtagsEndIndex - rangeSubtagsStartIndex);
199  if (!containslanguageSubtagMatchingRange(languageStringView, rangeSubtag, languageLength, lastMatchedLanguageSubtagIndex)) {
200  matchedRange = false;
201  break;
202  }
203  rangeSubtagsStartIndex = rangeSubtagsEndIndex;
204  } while (rangeSubtagsStartIndex < rangeLength);
205  if (matchedRange)
206  return true;
207  }
208  return false;
209 }

331 {

333  // element is an element in the document, the 'full-screen' pseudoclass applies to
