67  // Simulate the exchange of transport descriptions, and the gathering and
68  // exchange of ICE candidates.
69  void Connect(FakeTransportController* dest) {
70  for (const std::string& transport_name : transport_names_for_testing()) {
71  std::unique_ptr<rtc::SSLFingerprint> local_fingerprint;
72  std::unique_ptr<rtc::SSLFingerprint> remote_fingerprint;
73  if (certificate_for_testing()) {
74  local_fingerprint.reset(rtc::SSLFingerprint::CreateFromCertificate(
75  certificate_for_testing()));
76  }
77  if (dest->certificate_for_testing()) {
78  remote_fingerprint.reset(rtc::SSLFingerprint::CreateFromCertificate(
79  dest->certificate_for_testing()));
80  }
81  TransportDescription local_desc(
82  std::vector<std::string>(),
83  rtc::CreateRandomString(cricket::ICE_UFRAG_LENGTH),
84  rtc::CreateRandomString(cricket::ICE_PWD_LENGTH),
85  cricket::ICEMODE_FULL, cricket::CONNECTIONROLE_NONE,
86  local_fingerprint.get());
87  TransportDescription remote_desc(
88  std::vector<std::string>(),
89  rtc::CreateRandomString(cricket::ICE_UFRAG_LENGTH),
90  rtc::CreateRandomString(cricket::ICE_PWD_LENGTH),
91  cricket::ICEMODE_FULL, cricket::CONNECTIONROLE_NONE,
92  remote_fingerprint.get());
93  std::string err;
94  SetLocalTransportDescription(transport_name, local_desc,
95  cricket::CA_OFFER, &err);
96  dest->SetRemoteTransportDescription(transport_name, local_desc,
97  cricket::CA_OFFER, &err);
98  dest->SetLocalTransportDescription(transport_name, remote_desc,
99  cricket::CA_ANSWER, &err);
100  SetRemoteTransportDescription(transport_name, remote_desc,
101  cricket::CA_ANSWER, &err);
102  }
103  MaybeStartGathering();
104  dest->MaybeStartGathering();
105  network_thread()->Invoke<void>(
106  RTC_FROM_HERE,
107  rtc::Bind(&FakeTransportController::SetChannelDestinations_n, this,
108  dest));
109  }
