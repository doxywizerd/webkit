79 template<typename ViewClass>
80 inline JSObject* constructGenericTypedArrayViewFromIterator(ExecState* exec, Structure* structure, JSValue iterator)
81 {
82  VM& vm = exec->vm();
83  auto scope = DECLARE_THROW_SCOPE(vm);
84 
85  if (!iterator.isObject())
86  return throwTypeError(exec, scope, ASCIILiteral("Symbol.Iterator for the first argument did not return an object."));
87 
88  MarkedArgumentBuffer storage;
89  while (true) {
90  JSValue next = iteratorStep(exec, iterator);
91  RETURN_IF_EXCEPTION(scope, nullptr);
92 
93  if (next.isFalse())
94  break;
95 
96  JSValue nextItem = iteratorValue(exec, next);
97  RETURN_IF_EXCEPTION(scope, nullptr);
98 
99  storage.append(nextItem);
100  }
101 
102  ViewClass* result = ViewClass::createUninitialized(exec, structure, storage.size());
103  ASSERT(!!scope.exception() == !result);
104  if (UNLIKELY(!result))
105  return nullptr;
106 
107  for (unsigned i = 0; i < storage.size(); ++i) {
108  bool success = result->setIndex(exec, i, storage.at(i));
109  ASSERT(scope.exception() || success);
110  if (!success)
111  return nullptr;
112  }
113 
114  return result;
115 }

117 template<typename ViewClass>
118 inline JSObject* constructGenericTypedArrayViewWithArguments(ExecState* exec, Structure* structure, EncodedJSValue firstArgument, unsigned offset, std::optional<unsigned> lengthOpt)
119 {
120  VM& vm = exec->vm();
121  auto scope = DECLARE_THROW_SCOPE(vm);
122 
123  JSValue firstValue = JSValue::decode(firstArgument);
124 
125  if (JSArrayBuffer* jsBuffer = jsDynamicCast<JSArrayBuffer*>(vm, firstValue)) {
126  RefPtr<ArrayBuffer> buffer = jsBuffer->impl();
127  unsigned length = 0;
128 
129  if (lengthOpt)
130  length = lengthOpt.value();
131  else {
132  if ((buffer->byteLength() - offset) % ViewClass::elementSize)
133  return throwRangeError(exec, scope, ASCIILiteral("ArrayBuffer length minus the byteOffset is not a multiple of the element size"));
134  length = (buffer->byteLength() - offset) / ViewClass::elementSize;
135  }
136 
137  scope.release();
138  return ViewClass::create(exec, structure, WTFMove(buffer), offset, length);
139  }
140  ASSERT(!offset && !lengthOpt);
141 
142  if (ViewClass::TypedArrayStorageType == TypeDataView)
143  return throwTypeError(exec, scope, ASCIILiteral("Expected ArrayBuffer for the first argument."));
144 
145  // For everything but DataView, we allow construction with any of:
146  // - Another array. This creates a copy of the of that array.
147  // - An integer. This creates a new typed array of that length and zero-initializes it.
148 
149  if (JSObject* object = jsDynamicCast<JSObject*>(vm, firstValue)) {
150  unsigned length;
151 
152  if (isTypedView(object->classInfo(vm)->typedArrayStorageType))
153  length = jsCast<JSArrayBufferView*>(object)->length();
154  else {
155  // This getPropertySlot operation should not be observed by the Proxy.
156  // So we use VMInquiry. And purge the opaque object cases (proxy and namespace object) by isTaintedByOpaqueObject() guard.
157  PropertySlot lengthSlot(object, PropertySlot::InternalMethodType::VMInquiry);
158  object->getPropertySlot(exec, vm.propertyNames->length, lengthSlot);
159  RETURN_IF_EXCEPTION(scope, nullptr);
160 
161  JSValue iteratorFunc = object->get(exec, vm.propertyNames->iteratorSymbol);
162  RETURN_IF_EXCEPTION(scope, nullptr);
163 
164  // We would like not use the iterator as it is painfully slow. Fortunately, unless
165  // 1) The iterator is not a known iterator.
166  // 2) The base object does not have a length getter.
167  // 3) The base object might have indexed getters.
168  // it should not be observable that we do not use the iterator.
169 
170  if (!iteratorFunc.isUndefined()
171  && (iteratorFunc != object->globalObject()->arrayProtoValuesFunction()
172  || lengthSlot.isAccessor() || lengthSlot.isCustom() || lengthSlot.isTaintedByOpaqueObject()
173  || hasAnyArrayStorage(object->indexingType()))) {
174 
175  CallData callData;
176  CallType callType = getCallData(iteratorFunc, callData);
177  if (callType == CallType::None)
178  return throwTypeError(exec, scope, ASCIILiteral("Symbol.Iterator for the first argument cannot be called."));
179 
180  ArgList arguments;
181  JSValue iterator = call(exec, iteratorFunc, callType, callData, object, arguments);
182  RETURN_IF_EXCEPTION(scope, nullptr);
183 
184  scope.release();
185  return constructGenericTypedArrayViewFromIterator<ViewClass>(exec, structure, iterator);
186  }
187 
188  length = lengthSlot.isUnset() ? 0 : lengthSlot.getValue(exec, vm.propertyNames->length).toUInt32(exec);
189  RETURN_IF_EXCEPTION(scope, nullptr);
190  }
191 
192 
193  ViewClass* result = ViewClass::createUninitialized(exec, structure, length);
194  ASSERT(!!scope.exception() == !result);
195  if (UNLIKELY(!result))
196  return nullptr;
197 
198  scope.release();
199  if (!result->set(exec, 0, object, 0, length))
200  return nullptr;
201 
202  return result;
203  }
204 
205  if (!firstValue.isNumber())
206  return throwTypeError(exec, scope, ASCIILiteral("Invalid array length argument"));
207 
208  unsigned length = firstValue.toIndex(exec, "length");
209  RETURN_IF_EXCEPTION(scope, nullptr);
210  scope.release();
211  return ViewClass::create(exec, structure, length);
212 }

217 template<typename ViewClass>
218 EncodedJSValue JSC_HOST_CALL constructGenericTypedArrayView(ExecState* exec)
219 {
220  VM& vm = exec->vm();
221  auto scope = DECLARE_THROW_SCOPE(vm);
222 
223  InternalFunction* function = asInternalFunction(exec->jsCallee());
224  Structure* parentStructure = function->globalObject()->typedArrayStructure(ViewClass::TypedArrayStorageType);
225  Structure* structure = InternalFunction::createSubclassStructure(exec, exec->newTarget(), parentStructure);
226  RETURN_IF_EXCEPTION(scope, encodedJSValue());
227 
228  size_t argCount = exec->argumentCount();
229 
230  if (!argCount) {
231  if (ViewClass::TypedArrayStorageType == TypeDataView)
232  return throwVMTypeError(exec, scope, ASCIILiteral("DataView constructor requires at least one argument."));
233 
234  scope.release();
235  return JSValue::encode(ViewClass::create(exec, structure, 0));
236  }
237 
238  JSValue firstValue = exec->uncheckedArgument(0);
239  unsigned offset = 0;
240  std::optional<unsigned> length = std::nullopt;
241  if (jsDynamicCast<JSArrayBuffer*>(vm, firstValue) && argCount > 1) {
242  offset = exec->uncheckedArgument(1).toIndex(exec, "byteOffset");
243  RETURN_IF_EXCEPTION(scope, encodedJSValue());
244 
245  if (argCount > 2) {
246  if (ViewClass::TypedArrayStorageType == TypeDataView) {
247  // If the DataView byteLength is present but undefined, treat it as missing.
248  JSValue byteLengthValue = exec->uncheckedArgument(2);
249  if (!byteLengthValue.isUndefined()) {
250  length = byteLengthValue.toIndex(exec, "byteLength");
251  RETURN_IF_EXCEPTION(scope, encodedJSValue());
252  }
253  } else {
254  length = exec->uncheckedArgument(2).toIndex(exec, "length");
255  RETURN_IF_EXCEPTION(scope, encodedJSValue());
256  }
257  }
258  }
259 
260  scope.release();
261  return JSValue::encode(constructGenericTypedArrayViewWithArguments<ViewClass>(exec, structure, JSValue::encode(firstValue), offset, length));
262 }
