104 ALWAYS_INLINE bool JSStringJoiner::appendWithoutSideEffects(ExecState& state, JSValue value)
105 {
106  // The following code differs from using the result of JSValue::toString in the following ways:
107  // 1) It's inlined more than JSValue::toString is.
108  // 2) It includes conversion to WTF::String in a way that avoids allocating copies of substrings.
109  // 3) It doesn't create a JSString for numbers, true, or false.
110  // 4) It turns undefined and null into the empty string instead of "undefined" and "null".
111  // 5) It uses optimized code paths for all the cases known to be 8-bit and for the empty string.
112  // If we might make an effectful calls, return false. Otherwise return true.
113 
114  if (value.isCell()) {
115  JSString* jsString;
116  if (!value.asCell()->isString())
117  return false;
118  jsString = asString(value);
119  append(jsString->viewWithUnderlyingString(state));
120  return true;
121  }
122 
123  if (value.isInt32()) {
124  append8Bit(state.vm().numericStrings.add(value.asInt32()));
125  return true;
126  }
127  if (value.isDouble()) {
128  append8Bit(state.vm().numericStrings.add(value.asDouble()));
129  return true;
130  }
131  if (value.isTrue()) {
132  append8Bit(state.vm().propertyNames->trueKeyword.string());
133  return true;
134  }
135  if (value.isFalse()) {
136  append8Bit(state.vm().propertyNames->falseKeyword.string());
137  return true;
138  }
139  ASSERT(value.isUndefinedOrNull());
140  appendEmptyString();
141  return true;
142 }
