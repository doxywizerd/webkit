89 inline bool ReadableStreamDefaultController::enqueue(RefPtr<JSC::ArrayBuffer>&& buffer)
90 {
91  auto& globalObject = this->globalObject();
92  JSC::VM& vm = globalObject.vm();
93  JSC::JSLockHolder locker(vm);
94  auto scope = DECLARE_THROW_SCOPE(vm);
95  JSC::ExecState& state = globalExec();
96 
97  if (!buffer) {
98  error(state, createOutOfMemoryError(&state));
99  return false;
100  }
101  auto length = buffer->byteLength();
102  auto chunk = JSC::Uint8Array::create(WTFMove(buffer), 0, length);
103  ASSERT(chunk);
104  enqueue(state, toJS(&state, &globalObject, chunk.get()));
105  ASSERT_UNUSED(scope, !scope.exception());
106  return true;
107 }
