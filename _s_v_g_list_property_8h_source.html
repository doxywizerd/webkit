59  static void detachListWrappersAndResize(ListWrapperCache* wrappers, unsigned newListSize = 0)
60  {
61  // See SVGPropertyTearOff::detachWrapper() for an explanation about what's happening here.
62  ASSERT(wrappers);

289  ExceptionOr<ListItemType> replaceItemValues(const ListItemType& newItem, unsigned index)
290  {
291  auto result = canReplaceItem(index);
292  if (result.hasException())
293  return result.releaseException();
294  ASSERT(result.releaseReturnValue());
295 
296  // Spec: If newItem is already in a list, it is removed from its previous list before it is inserted into this list.
297  // Spec: If the item is already in this list, note that the index of the item to replace is before the removal of the item.
298  if (!processIncomingListItemValue(newItem, &index)) {
299  // Replacing the item with itself is a no-op.
300  return ListItemType { newItem };
301  }
302 
303  if (m_values->isEmpty()) {
304  // 'newItem' already lived in our list, we removed it, and now we're empty, which means there's nothing to replace.
305  return Exception { INDEX_SIZE_ERR };
306  }
307 
308  // Update the value at the desired position 'index'.
309  m_values->at(index) = newItem;

311  commitChange();

328  // Spec: If newItem is already in a list, it is removed from its previous list before it is inserted into this list.
