222 template<PromiseFunction promiseFunction, PromiseExecutionScope executionScope>
223 inline JSC::JSValue callPromiseFunction(JSC::ExecState& state)
224 {
225  JSC::VM& vm = state.vm();
226  auto scope = DECLARE_CATCH_SCOPE(vm);
227 
228  JSDOMGlobalObject& globalObject = *JSC::jsCast<JSDOMGlobalObject*>(state.lexicalGlobalObject());
229  JSC::JSPromiseDeferred* promiseDeferred = JSC::JSPromiseDeferred::create(&state, &globalObject);
230 
231  // promiseDeferred can be null when terminating a Worker abruptly.
232  if (executionScope == PromiseExecutionScope::WindowOrWorker && !promiseDeferred)
233  return JSC::jsUndefined();
234 
235  promiseFunction(state, DeferredPromise::create(globalObject, *promiseDeferred));
236 
237  rejectPromiseWithExceptionIfAny(state, globalObject, *promiseDeferred);
238  ASSERT_UNUSED(scope, !scope.exception());
239  return promiseDeferred->promise();
240 }
