313 template <typename T, typename std::enable_if<sizeof(T) == 8>::type* = nullptr>
314 ALWAYS_INLINE ConsumeDependency zeroWithConsumeDependency(T value)
315 {
316  uint64_t dependency;
317  uint64_t copy = bitwise_cast<uint64_t>(value);
318 #if CPU(ARM64)
319  // Create a magical zero value through inline assembly, whose computation
320  // isn't visible to the optimizer. This zero is then usable as an offset in
321  // further address computations: adding zero does nothing, but the compiler
322  // doesn't know it. It's magical because it creates an address dependency
323  // from the load of `location` to the uses of the dependency, which triggers
324  // the ARM ISA's address dependency rule, a.k.a. the mythical C++ consume
325  // ordering. This forces weak memory order CPUs to observe `location` and
326  // dependent loads in their store order without the reader using a barrier
327  // or an acquire load.
328  asm volatile("eor %x[dependency], %x[in], %x[in]"
329  : [dependency] "=r"(dependency)
330  : [in] "r"(copy)
331  // Lie about touching memory. Not strictly needed, but is
332  // likely to avoid unwanted load/store motion.
333  : "memory");
334 #elif CPU(ARM)
335  asm volatile("eor %[dependency], %[in], %[in]"
336  : [dependency] "=r"(dependency)
337  : [in] "r"(copy)
338  : "memory");
339 #else
340  // No dependency is needed for this architecture.
341  loadLoadFence();
342  dependency = 0;
343  (void)copy;
344 #endif
345  return static_cast<ConsumeDependency>(dependency);
346 }

391 // Consume load, returning the loaded `value` at `location` and a dependent-zero
392 // which creates an address dependency from the `location`.
393 //
394 // Usage notes:
395 //
396 // * Regarding control dependencies: merely branching based on `value` or
397 // `dependency` isn't sufficient to impose a dependency ordering: you must
398 // use `dependency` in the address computation of subsequent loads which
399 // should observe the store order w.r.t. `location`.
400 // * Regarding memory ordering: consume load orders the `location` load with
401 // susequent dependent loads *only*. It says nothing about ordering of other
402 // loads!
403 //
404 // Caveat emptor.
405 template <typename T>
406 ALWAYS_INLINE auto consumeLoad(const T* location)
407 {
408  typedef typename std::remove_cv<T>::type Returned;
409  Consumed<Returned> ret { };
410  // Force the read of `location` to occur exactly once and without fusing or
411  // forwarding using volatile. This is important because the compiler could
412  // otherwise rematerialize or find equivalent loads, or simply forward from
413  // a previous one, and lose the dependency we're trying so hard to
414  // create. Prevent tearing by using an atomic, but let it move around by
415  // using relaxed. We have at least a memory fence after this which prevents
416  // the load from moving too much.
417  ret.value = reinterpret_cast<const volatile std::atomic<Returned>*>(location)->load(std::memory_order_relaxed);
418  ret.dependency = zeroWithConsumeDependency(ret.value);
419  return ret;
420 }
