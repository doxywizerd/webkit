38 // This is a bitfield where each bit represents an type of array access that we have seen.
39 // There are 16 indexing types that use the lower bits.
40 // There are 9 typed array types taking the bits 16 to 25.
41 typedef unsigned ArrayModes;
42 
43 const ArrayModes Int8ArrayMode = 1 << 16;
44 const ArrayModes Int16ArrayMode = 1 << 17;
45 const ArrayModes Int32ArrayMode = 1 << 18;
46 const ArrayModes Uint8ArrayMode = 1 << 19;
47 const ArrayModes Uint8ClampedArrayMode = 1 << 20;
48 const ArrayModes Uint16ArrayMode = 1 << 21;
49 const ArrayModes Uint32ArrayMode = 1 << 22;
50 const ArrayModes Float32ArrayMode = 1 << 23;
51 const ArrayModes Float64ArrayMode = 1 << 24;

53 #define asArrayModes(type) \
54  (static_cast<unsigned>(1) << static_cast<unsigned>(type))

56 #define ALL_TYPED_ARRAY_MODES \

62  | Uint16ArrayMode \

68 #define ALL_NON_ARRAY_ARRAY_MODES \

88 inline ArrayModes arrayModeFromStructure(Structure* structure)
89 {
90  switch (structure->classInfo()->typedArrayStorageType) {
91  case TypeInt8:
92  return Int8ArrayMode;
93  case TypeUint8:
94  return Uint8ArrayMode;
95  case TypeUint8Clamped:
96  return Uint8ClampedArrayMode;
97  case TypeInt16:
98  return Int16ArrayMode;
99  case TypeUint16:
100  return Uint16ArrayMode;
101  case TypeInt32:
102  return Int32ArrayMode;
103  case TypeUint32:
104  return Uint32ArrayMode;
105  case TypeFloat32:
106  return Float32ArrayMode;
107  case TypeFloat64:
108  return Float64ArrayMode;
109  case TypeDataView:
110  case NotTypedArray:
111  break;
112  }
113  return asArrayModes(structure->indexingType());
114 }

119 inline bool mergeArrayModes(ArrayModes& left, ArrayModes right)
120 {
121  ArrayModes newModes = left | right;
