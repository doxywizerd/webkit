101  void applyTransform(const AffineTransform& localToAncestorTransform)
102  {
103  if (localToAncestorTransform.isIdentity())
104  return;
105 
106  context().concatCTM(localToAncestorTransform);
107 
108  if (rect.isInfinite())
109  return;
110 
111  FloatRect tranformedRect(localToAncestorTransform.inverse().value_or(AffineTransform()).mapRect(rect));
112  rect.setLocation(LayoutPoint(tranformedRect.location()));
113  rect.setSize(LayoutSize(tranformedRect.size()));
