136 inline WTFThreadData& wtfThreadData()
137 {
138  // WRT WebCore:
139  // WTFThreadData is used on main thread before it could possibly be used
140  // on secondary ones, so there is no need for synchronization here.
141  // WRT JavaScriptCore:
142  // wtfThreadData() is initially called from initializeThreading(), ensuring
143  // this is initially called in a pthread_once locked context.
144 #if !USE(PTHREAD_GETSPECIFIC_DIRECT)
145  if (!WTFThreadData::staticData)
146  WTFThreadData::staticData = new ThreadSpecific<WTFThreadData>;
147  return **WTFThreadData::staticData;
148 #else
149  if (WTFThreadData* data = static_cast<WTFThreadData*>(_pthread_getspecific_direct(WTFThreadData::directKey)))
150  return *data;
151  return WTFThreadData::createAndRegisterForGetspecificDirect();
152 #endif
153 }
