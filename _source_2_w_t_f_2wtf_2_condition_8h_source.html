37 // This is a condition variable that is suitable for use with any lock-like object, including
38 // our own WTF::Lock. It features standard wait()/notifyOne()/notifyAll() methods in addition to
39 // a variety of wait-with-timeout methods. This includes methods that use WTF's own notion of
40 // time, like wall-clock time (i.e. currentTime()) and monotonic time (i.e.
41 // monotonicallyIncreasingTime()). This is a very efficient condition variable. It only requires
42 // one byte of memory. notifyOne() and notifyAll() require just a load and branch for the fast
43 // case where no thread is waiting. This condition variable, when used with WTF::Lock, can
44 // outperform a system condition variable and lock by up to 58x.

46 // This is a struct without a constructor or destructor so that it can be statically initialized.

49  // Condition will accept any kind of time and convert it internally, but this typedef tells
50  // you what kind of time Condition would be able to use without conversions. However, if you
51  // are unlikely to be affected by the cost of conversions, it is better to use MonotonicTime.
52  typedef ParkingLot::Time Time;

54  // Wait on a parking queue while releasing the given lock. It will unlock the lock just before
55  // parking, and relock it upon wakeup. Returns true if we woke up due to some call to
56  // notifyOne() or notifyAll(). Returns false if we woke up due to a timeout. Note that this form
57  // of waitUntil() has some quirks:
58  //
59  // No spurious wake-up: in order for this to return before the timeout, some notifyOne() or
60  // notifyAll() call must have happened. No scenario other than timeout or notify can lead to this
61  // method returning. This means, for example, that you can't use pthread cancelation or signals to
62  // cause early return.
63  //
64  // Past timeout: it's possible for waitUntil() to be called with a timeout in the past. In that
65  // case, waitUntil() will still release the lock and reacquire it. waitUntil() will always return
66  // false in that case. This is subtly different from some pthread_cond_timedwait() implementations,
67  // which may not release the lock for past timeout. But, this behavior is consistent with OpenGroup
68  // documentation for timedwait().
69  template<typename LockType>
70  bool waitUntil(LockType& lock, const TimeWithDynamicClockType& timeout)
71  {
72  bool result;
73  if (timeout < timeout.nowWithSameClock()) {
74  lock.unlock();
75  result = false;
76  } else {
77  result = ParkingLot::parkConditionally(
78  &m_hasWaiters,
79  [this] () -> bool {
80  // Let everyone know that we will be waiting. Do this while we hold the queue lock,
81  // to prevent races with notifyOne().
82  m_hasWaiters.store(true);
83  return true;
84  },
85  [&lock] () { lock.unlock(); },
86  timeout).wasUnparked;
87  }
88  lock.lock();
89  return result;
90  }

92  // Wait until the given predicate is satisfied. Returns true if it is satisfied in the end.
93  // May return early due to timeout.
94  template<typename LockType, typename Functor>
95  bool waitUntil(
96  LockType& lock, const TimeWithDynamicClockType& timeout, const Functor& predicate)
97  {
98  while (!predicate()) {
99  if (!waitUntil(lock, timeout))
100  return predicate();
101  }
102  return true;
103  }

133  // Note that this method is extremely fast when nobody is waiting. It is not necessary to try to
134  // avoid calling this method. This returns true if someone was actually woken up.
135  bool notifyOne()
136  {
137  if (!m_hasWaiters.load()) {
138  // At this exact instant, there is nobody waiting on this condition. The way to visualize
139  // this is that if unparkOne() ran to completion without obstructions at this moment, it
140  // wouldn't wake anyone up. Hence, we have nothing to do!
141  return false;
142  }

144  bool didNotifyThread = false;
145  ParkingLot::unparkOne(
146  &m_hasWaiters,
147  [&] (ParkingLot::UnparkResult result) -> intptr_t {
148  if (!result.mayHaveMoreThreads)
149  m_hasWaiters.store(false);
150  didNotifyThread = result.didUnparkThread;
151  return 0;
152  });
153  return didNotifyThread;
154  }
155 
156  void notifyAll()
157  {
158  if (!m_hasWaiters.load()) {
159  // See above.
160  return;
161  }
162 
163  // It's totally safe for us to set this to false without any locking, because this thread is
164  // guaranteed to then unparkAll() anyway. So, if there is a race with some thread calling
165  // wait() just before this store happens, that thread is guaranteed to be awoken by the call to
166  // unparkAll(), below.
167  m_hasWaiters.store(false);
168 
169  ParkingLot::unparkAll(&m_hasWaiters);
170  }
