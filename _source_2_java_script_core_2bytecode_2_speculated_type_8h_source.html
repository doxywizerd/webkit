73 static const SpeculatedType SpecInt52Only = 1ull << 27; // It's definitely an Int52 and we intend it to unbox it. It's also definitely not an Int32.

78 static const SpeculatedType SpecDoublePureNaN = 1ull << 30; // It's definitely a NaN that is sae to tag (i.e. pure).
79 static const SpeculatedType SpecDoubleImpureNaN = 1ull << 31; // It's definitely a NaN that is unsafe to tag (i.e. impure).
80 static const SpeculatedType SpecDoubleNaN = SpecDoublePureNaN | SpecDoubleImpureNaN; // It's definitely some kind of NaN.
81 static const SpeculatedType SpecBytecodeDouble = SpecDoubleReal | SpecDoublePureNaN; // It's either a non-NaN or a NaN double, but it's definitely not impure NaN.

87 static const SpeculatedType SpecBoolean = 1ull << 32; // It's definitely a Boolean.

442 // Merge two predictions. Note that currently this just does left | right. It may
443 // seem tempting to do so directly, but you would be doing so at your own peril,
444 // since the merging protocol SpeculatedType may change at any time (and has already
445 // changed several times in its history).
446 inline SpeculatedType mergeSpeculations(SpeculatedType left, SpeculatedType right)
447 {
448  return left | right;
449 }
450 
451 template<typename T>
452 inline bool mergeSpeculation(T& left, SpeculatedType right)
453 {
454  SpeculatedType newSpeculation = static_cast<T>(mergeSpeculations(static_cast<SpeculatedType>(left), right));
455  bool result = newSpeculation != static_cast<SpeculatedType>(left);
456  left = newSpeculation;
457  return result;
458 }
