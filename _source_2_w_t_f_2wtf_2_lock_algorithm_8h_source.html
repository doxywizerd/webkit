36 // This is the algorithm used by WTF::Lock. You can use it to project one lock onto any atomic
37 // field. The limit of one lock is due to the use of the field's address as a key to find the lock's
38 // queue.

53  LockType oldValue = lock.load(std::memory_order_relaxed);
54  if (oldValue & isHeldBit)

66  {
67  for (;;) {

69  if (currentByteValue & isHeldBit)
70  return false;

118  NEVER_INLINE static void lockSlow(Atomic<LockType>& lock)
119  {
120  unsigned spinCount = 0;
121 
122  // This magic number turns out to be optimal based on past JikesRVM experiments.
123  const unsigned spinLimit = 40;
124 
125  for (;;) {
126  uint8_t currentByteValue = lock.load();
127 
128  // We allow ourselves to barge in.
129  if (!(currentByteValue & isHeldBit)
130  && lock.compareExchangeWeak(currentByteValue, currentByteValue | isHeldBit))
131  return;
132 
133  // If there is nobody parked and we haven't spun too much, we can just try to spin around.
134  if (!(currentByteValue & hasParkedBit) && spinCount < spinLimit) {
135  spinCount++;
136  std::this_thread::yield();
137  continue;
138  }
139 
140  // Need to park. We do this by setting the parked bit first, and then parking. We spin around
141  // if the parked bit wasn't set and we failed at setting it.
142  if (!(currentByteValue & hasParkedBit)
143  && !lock.compareExchangeWeak(currentByteValue, currentByteValue | hasParkedBit))
144  continue;
145 
146  // We now expect the value to be isHeld|hasParked. So long as that's the case, we can park.
147  ParkingLot::ParkResult parkResult =
148  ParkingLot::compareAndPark(&lock, currentByteValue | isHeldBit | hasParkedBit);
149  if (parkResult.wasUnparked) {
150  switch (static_cast<Token>(parkResult.token)) {
151  case DirectHandoff:
152  // The lock was never released. It was handed to us directly by the thread that did
153  // unlock(). This means we're done!
154  RELEASE_ASSERT(isLocked(lock));
155  return;
156  case BargingOpportunity:
157  // This is the common case. The thread that called unlock() has released the lock,
158  // and we have been woken up so that we may get an opportunity to grab the lock. But
159  // other threads may barge, so the best that we can do is loop around and try again.
160  break;
161  }
162  }
163 
164  // We have awoken, or we never parked because the byte value changed. Either way, we loop
165  // around and try again.
166  }
167  }

169  enum Fairness {
170  Fair,
171  Unfair
172  };
173  NEVER_INLINE static void unlockSlow(Atomic<LockType>& lock, Fairness fairness)
174  {
175  // We could get here because the weak CAS in unlock() failed spuriously, or because there is
176  // someone parked. So, we need a CAS loop: even if right now the lock is just held, it could
177  // be held and parked if someone attempts to lock just as we are unlocking.
178  for (;;) {
179  uint8_t oldByteValue = lock.load();
180  RELEASE_ASSERT(
181  (oldByteValue & mask) == isHeldBit
182  || (oldByteValue & mask) == (isHeldBit | hasParkedBit));
183 
184  if ((oldByteValue & mask) == isHeldBit) {
185  if (lock.compareExchangeWeak(oldByteValue, oldByteValue & ~isHeldBit))
186  return;
187  continue;
188  }
189 
190  // Someone is parked. Unpark exactly one thread. We may hand the lock to that thread
191  // directly, or we will unlock the lock at the same time as we unpark to allow for barging.
192  // When we unlock, we may leave the parked bit set if there is a chance that there are still
193  // other threads parked.
194  ASSERT((oldByteValue & mask) == (isHeldBit | hasParkedBit));
195  ParkingLot::unparkOne(
196  &lock,
197  [&] (ParkingLot::UnparkResult result) -> intptr_t {
198  // We are the only ones that can clear either the isHeldBit or the hasParkedBit,
199  // so we should still see both bits set right now.
200  ASSERT((lock.load() & mask) == (isHeldBit | hasParkedBit));
201 
202  if (result.didUnparkThread && (fairness == Fair || result.timeToBeFair)) {
203  // We don't unlock anything. Instead, we hand the lock to the thread that was
204  // waiting.
205  return DirectHandoff;
206  }
207 
208  lock.transaction(
209  [&] (LockType& value) {
210  value &= ~mask;
211  if (result.mayHaveMoreThreads)
212  value |= hasParkedBit;
213  });
214  return BargingOpportunity;
215  });
216  return;
217  }
218  }
