74 template<typename... T> struct Converter<IDLUnion<T...>> : DefaultConverter<IDLUnion<T...>> {
75  using Type = IDLUnion<T...>;
76  using TypeList = typename Type::TypeList;
77  using ReturnType = typename Type::ImplementationType;
78 
79  using NumericTypeList = brigand::filter<TypeList, IsIDLNumber<brigand::_1>>;
80  static constexpr size_t numberOfNumericTypes = brigand::size<NumericTypeList>::value;
81  static_assert(numberOfNumericTypes == 0 || numberOfNumericTypes == 1, "There can be 0 or 1 numeric types in an IDLUnion.");
82  using NumericType = ConditionalFront<NumericTypeList, numberOfNumericTypes != 0>;
83 
84  // FIXME: This should also check for IDLEnumeration<T>.
85  using StringTypeList = brigand::filter<TypeList, std::is_base_of<IDLString, brigand::_1>>;
86  static constexpr size_t numberOfStringTypes = brigand::size<StringTypeList>::value;
87  static_assert(numberOfStringTypes == 0 || numberOfStringTypes == 1, "There can be 0 or 1 string types in an IDLUnion.");
88  using StringType = ConditionalFront<StringTypeList, numberOfStringTypes != 0>;
89 
90  using SequenceTypeList = brigand::filter<TypeList, IsIDLSequence<brigand::_1>>;
91  static constexpr size_t numberOfSequenceTypes = brigand::size<SequenceTypeList>::value;
92  static_assert(numberOfSequenceTypes == 0 || numberOfSequenceTypes == 1, "There can be 0 or 1 sequence types in an IDLUnion.");
93  using SequenceType = ConditionalFront<SequenceTypeList, numberOfSequenceTypes != 0>;
94 
95  using FrozenArrayTypeList = brigand::filter<TypeList, IsIDLFrozenArray<brigand::_1>>;
96  static constexpr size_t numberOfFrozenArrayTypes = brigand::size<FrozenArrayTypeList>::value;
97  static_assert(numberOfFrozenArrayTypes == 0 || numberOfFrozenArrayTypes == 1, "There can be 0 or 1 FrozenArray types in an IDLUnion.");
98  using FrozenArrayType = ConditionalFront<FrozenArrayTypeList, numberOfFrozenArrayTypes != 0>;
99 
100  using DictionaryTypeList = brigand::filter<TypeList, IsIDLDictionary<brigand::_1>>;
101  static constexpr size_t numberOfDictionaryTypes = brigand::size<DictionaryTypeList>::value;
102  static_assert(numberOfDictionaryTypes == 0 || numberOfDictionaryTypes == 1, "There can be 0 or 1 dictionary types in an IDLUnion.");
103  static constexpr bool hasDictionaryType = numberOfDictionaryTypes != 0;
104  using DictionaryType = ConditionalFront<DictionaryTypeList, hasDictionaryType>;
105 
106  using RecordTypeList = brigand::filter<TypeList, IsIDLRecord<brigand::_1>>;
107  static constexpr size_t numberOfRecordTypes = brigand::size<RecordTypeList>::value;
108  static_assert(numberOfRecordTypes == 0 || numberOfRecordTypes == 1, "There can be 0 or 1 record types in an IDLUnion.");
109  static constexpr bool hasRecordType = numberOfRecordTypes != 0;
110  using RecordType = ConditionalFront<RecordTypeList, hasRecordType>;
111 
112  static constexpr bool hasObjectType = (numberOfSequenceTypes + numberOfFrozenArrayTypes + numberOfDictionaryTypes + numberOfRecordTypes) > 0;
113 
114  using InterfaceTypeList = brigand::filter<TypeList, IsIDLInterface<brigand::_1>>;

116  static ReturnType convert(JSC::ExecState& state, JSC::JSValue value)
117  {
118  JSC::VM& vm = state.vm();
119  auto scope = DECLARE_THROW_SCOPE(vm);
120 
121  // 1. If the union type includes a nullable type and V is null or undefined, then return the IDL value null.
122  constexpr bool hasNullType = brigand::any<TypeList, std::is_same<IDLNull, brigand::_1>>::value;
123  if (hasNullType) {
124  if (value.isUndefinedOrNull())
125  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, IDLNull, hasNullType>::convert(state, value).value());
126  }
127 
128  // 2. Let types be the flattened member types of the union type.
129  // NOTE: Union is expected to be pre-flattented.
130 
131  // 3. If V is null or undefined then:
132  if (hasDictionaryType || hasRecordType) {
133  if (value.isUndefinedOrNull()) {
134  // 1. If types includes a dictionary type, then return the result of converting V to that dictionary type.
135  if (hasDictionaryType)
136  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, DictionaryType, hasDictionaryType>::convert(state, value).value());
137 
138  // 2. If types includes a record type, then return the result of converting V to that record type.
139  if (hasRecordType)
140  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, RecordType, hasRecordType>::convert(state, value).value());
141  }
142  }
143 
144  // 4. If V is a platform object, then:
145  // 1. If types includes an interface type that V implements, then return the IDL value that is a reference to the object V.
146  // 2. If types includes object, then return the IDL value that is a reference to the object V.
147  // (FIXME: Add support for object and step 4.2)
148  if (brigand::any<TypeList, IsIDLInterface<brigand::_1>>::value) {
149  std::optional<ReturnType> returnValue;
150  brigand::for_each<InterfaceTypeList>([&](auto&& type) {
151  if (returnValue)
152  return;
153 
154  using Type = typename WTF::RemoveCVAndReference<decltype(type)>::type::type;
155  using ImplementationType = typename Type::ImplementationType;
156  using RawType = typename Type::RawType;
157  using WrapperType = typename JSDOMWrapperConverterTraits<RawType>::WrapperClass;
158 
159  auto castedValue = WrapperType::toWrapped(vm, value);
160  if (!castedValue)
161  return;
162 
163  returnValue = ReturnType(ImplementationType(castedValue));
164  });
165 
166  if (returnValue)
167  return WTFMove(returnValue.value());
168  }
169 
170  // FIXME: Add support for steps 5 - 10.
171 
172  // 11. If V is any kind of object, then:
173  if (hasObjectType) {
174  if (value.isCell()) {
175  JSC::JSCell* cell = value.asCell();
176  if (cell->isObject()) {
177  // FIXME: We should be able to optimize the following code by making use
178  // of the fact that we have proved that the value is an object.
179 
180  // 1. If types includes a sequence type, then:
181  // 1. Let method be the result of GetMethod(V, @@iterator).
182  // 2. ReturnIfAbrupt(method).
183  // 3. If method is not undefined, return the result of creating a
184  // sequence of that type from V and method.
185  constexpr bool hasSequenceType = numberOfSequenceTypes != 0;
186  if (hasSequenceType) {
187  bool hasIterator = JSC::hasIteratorMethod(state, value);
188  RETURN_IF_EXCEPTION(scope, ReturnType());
189  if (hasIterator)
190  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, SequenceType, hasSequenceType>::convert(state, value).value());
191  }
192 
193  // 2. If types includes a frozen array type, then:
194  // 1. Let method be the result of GetMethod(V, @@iterator).
195  // 2. ReturnIfAbrupt(method).
196  // 3. If method is not undefined, return the result of creating a
197  // frozen array of that type from V and method.
198  constexpr bool hasFrozenArrayType = numberOfFrozenArrayTypes != 0;
199  if (hasFrozenArrayType) {
200  bool hasIterator = JSC::hasIteratorMethod(state, value);
201  RETURN_IF_EXCEPTION(scope, ReturnType());
202  if (hasIterator)
203  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, FrozenArrayType, hasFrozenArrayType>::convert(state, value).value());
204  }
205 
206  // 3. If types includes a dictionary type, then return the result of
207  // converting V to that dictionary type.
208  if (hasDictionaryType)
209  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, DictionaryType, hasDictionaryType>::convert(state, value).value());
210 
211  // 4. If types includes a record type, then return the result of converting V to that record type.
212  if (hasRecordType)
213  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, RecordType, hasRecordType>::convert(state, value).value());
214 
215  // 5. If types includes a callback interface type, then return the result of converting V to that interface type.
216  // (FIXME: Add support for callback interface type and step 12.5)
217  // 6. If types includes object, then return the IDL value that is a reference to the object V.
218  // (FIXME: Add support for object and step 12.6)
219  }
220  }
221  }
222 
223  // 12. If V is a Boolean value, then:
224  // 1. If types includes a boolean, then return the result of converting V to boolean.
225  constexpr bool hasBooleanType = brigand::any<TypeList, std::is_same<IDLBoolean, brigand::_1>>::value;
226  if (hasBooleanType) {
227  if (value.isBoolean())
228  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, IDLBoolean, hasBooleanType>::convert(state, value).value());
229  }
230 
231  // 13. If V is a Number value, then:
232  // 1. If types includes a numeric type, then return the result of converting V to that numeric type.
233  constexpr bool hasNumericType = brigand::size<NumericTypeList>::value != 0;
234  if (hasNumericType) {
235  if (value.isNumber())
236  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, NumericType, hasNumericType>::convert(state, value).value());
237  }
238 
239  // 14. If types includes a string type, then return the result of converting V to that type.
240  constexpr bool hasStringType = brigand::size<StringTypeList>::value != 0;
241  if (hasStringType)
242  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, StringType, hasStringType>::convert(state, value).value());
243 
244  // 15. If types includes a numeric type, then return the result of converting V to that numeric type.
245  if (hasNumericType)
246  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, NumericType, hasNumericType>::convert(state, value).value());
247 
248  // 16. If types includes a boolean, then return the result of converting V to boolean.
249  if (hasBooleanType)
250  return std::move<WTF::CheckMoveParameter>(ConditionalConverter<ReturnType, IDLBoolean, hasBooleanType>::convert(state, value).value());
251 
252  // 17. Throw a TypeError.
253  throwTypeError(&state, scope);
254  return ReturnType();
255  }

269  {
270  auto index = variant.index();
271 
272  std::optional<JSC::JSValue> returnValue;
273  brigand::for_each<Sequence>([&](auto&& type) {
274  using I = typename WTF::RemoveCVAndReference<decltype(type)>::type::type;
275  if (I::value == index) {
276  ASSERT(!returnValue);
277  returnValue = toJS<brigand::at<TypeList, I>>(state, globalObject, WTF::get<I::value>(variant));
278  }
279  });
280 
281  ASSERT(returnValue);
282  return returnValue.value();
283  }
