33 // IndexSparseSet is an efficient set of integers that can only be valued
34 // between zero and size() - 1.
35 //
36 // The implementation is using Briggs Sparse Set representation. We allocate
37 // memory from 0 to size() - 1 to do mapping in O(1), but we never initialize
38 // that memory. When adding/removing values to the set, they are added in a list
39 // and the corresponding bucket is initialized to the position in the list.
40 //
41 // The assumption here is that we only need a sparse subset of number live at any
42 // time.
43

85 template<typename OverflowHandler>
86 inline bool IndexSparseSet<OverflowHandler>::remove(unsigned value)
87 {
88  unsigned position = m_map[value];
89  if (position >= m_values.size())
90  return false;
91 
92  if (m_values[position] == value) {
93  unsigned lastValue = m_values.last();
94  m_values[position] = lastValue;
95  m_map[lastValue] = position;
96  m_values.removeLast();
97  return true;
98  }
99 
100  return false;
101 }
