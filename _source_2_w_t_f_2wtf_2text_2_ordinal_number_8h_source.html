29 // An abstract number of element in a sequence. The sequence has a first element.
30 // This type should be used instead of integer because 2 contradicting traditions can
31 // call a first element '0' or '1' which makes integer type ambiguous.
32 class OrdinalNumber {
33 public:
34  static OrdinalNumber beforeFirst() { return OrdinalNumber(-1); }
35  static OrdinalNumber fromZeroBasedInt(int zeroBasedInt) { return OrdinalNumber(zeroBasedInt); }
36  static OrdinalNumber fromOneBasedInt(int oneBasedInt) { return OrdinalNumber(oneBasedInt - 1); }
37 
38  OrdinalNumber() : m_zeroBasedValue(0) { }
39 
40  int zeroBasedInt() const { return m_zeroBasedValue; }
41  int oneBasedInt() const { return m_zeroBasedValue + 1; }
42 
43  bool operator==(OrdinalNumber other) { return m_zeroBasedValue == other.m_zeroBasedValue; }
44  bool operator!=(OrdinalNumber other) { return !((*this) == other); }
45  bool operator>(OrdinalNumber other) { return m_zeroBasedValue > other.m_zeroBasedValue; }
46 
47 private:
48  OrdinalNumber(int zeroBasedInt) : m_zeroBasedValue(zeroBasedInt) { }
49  int m_zeroBasedValue;
50 };
