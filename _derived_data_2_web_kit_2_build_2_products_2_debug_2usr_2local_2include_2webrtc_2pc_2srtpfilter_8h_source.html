281  // For each different ssrc and error, we collect statistics separately.
282  struct FailureKey {
283  FailureKey()

285  mode(SrtpFilter::PROTECT),
286  error(SrtpFilter::ERROR_NONE) {
287  }
288  FailureKey(uint32_t in_ssrc,
289  SrtpFilter::Mode in_mode,
290  SrtpFilter::Error in_error)
291  : ssrc(in_ssrc), mode(in_mode), error(in_error) {}
292  bool operator <(const FailureKey& key) const {
293  return
294  (ssrc < key.ssrc) ||
295  (ssrc == key.ssrc && mode < key.mode) ||
296  (ssrc == key.ssrc && mode == key.mode && error < key.error);
297  }
