68 inline bool MarkedBlock::marksConveyLivenessDuringMarking(HeapVersion markingVersion)
69 {
70  // This returns true if any of these is true:
71  // - We just created the block and so the bits are clear already.
72  // - This block has objects marked during the last GC, and so its version was up-to-date just
73  // before the current collection did beginMarking(). This means that any objects that have
74  // their mark bit set are valid objects that were never deleted, and so are candidates for
75  // marking in any conservative scan. Using our jargon, they are "live".
76  // - We did ~2^32 collections and rotated the version back to null, so we needed to hard-reset
77  // everything. If the marks had been stale, we would have cleared them. So, we can be sure that
78  // any set mark bit reflects objects marked during last GC, i.e. "live" objects.
79  // It would be absurd to use this method when not collecting, since this special "one version
80  // back" state only makes sense when we're in a concurrent collection and have to be
81  // conservative.
82  ASSERT(space()->isMarking());
83  if (heap()->collectionScope() != CollectionScope::Full)
84  return false;
85  return m_markingVersion == MarkedSpace::nullVersion
86  || MarkedSpace::nextVersion(m_markingVersion) == markingVersion;
87 }

89 inline bool MarkedBlock::Handle::isLive(HeapVersion markingVersion, bool isMarking, const HeapCell* cell)
90 {
91  ASSERT(!isFreeListed());
92 
93  if (UNLIKELY(hasAnyNewlyAllocated())) {
94  if (isNewlyAllocated(cell))
95  return true;

107  return false;
108  }

143 {
144  if (specialize) {
145  emptyMode = specializedEmptyMode;
146  sweepMode = specializedSweepMode;
147  destructionMode = specializedDestructionMode;
148  scribbleMode = specializedScribbleMode;
149  newlyAllocatedMode = specializedNewlyAllocatedMode;
150  marksMode = specializedMarksMode;
151  }
152 
153  RELEASE_ASSERT(!(destructionMode == BlockHasNoDestructors && sweepMode == SweepOnly));
154 
155  SuperSamplerScope superSamplerScope(false);
156 
157  MarkedBlock& block = this->block();
158 
159  if (false)
160  dataLog(RawPointer(this), "/", RawPointer(&block), ": MarkedBlock::Handle::specializedSweep!\n");
161 
162  if (Options::useBumpAllocator()
163  && emptyMode == IsEmpty
164  && newlyAllocatedMode == DoesNotHaveNewlyAllocated) {
165 
166  // This is an incredibly powerful assertion that checks the sanity of our block bits.
167  if (marksMode == MarksNotStale && !block.m_marks.isEmpty()) {
168  WTF::dataFile().atomically(
169  [&] (PrintStream& out) {
170  out.print("Block ", RawPointer(&block), ": marks not empty!\n");
171  out.print("Block lock is held: ", block.m_lock.isHeld(), "\n");
172  out.print("Marking version of block: ", block.m_markingVersion, "\n");
173  out.print("Marking version of heap: ", space()->markingVersion(), "\n");
174  UNREACHABLE_FOR_PLATFORM();
175  });
176  }
177 
178  char* startOfLastCell = static_cast<char*>(cellAlign(block.atoms() + m_endAtom - 1));
179  char* payloadEnd = startOfLastCell + cellSize();
180  RELEASE_ASSERT(payloadEnd - MarkedBlock::blockSize <= bitwise_cast<char*>(&block));
181  char* payloadBegin = bitwise_cast<char*>(block.atoms() + firstAtom());
182  if (scribbleMode == Scribble)
183  scribble(payloadBegin, payloadEnd - payloadBegin);
184  if (sweepMode == SweepToFreeList)
185  setIsFreeListed();
186  else
187  m_allocator->setIsEmpty(NoLockingNecessary, this, true);
188  if (space()->isMarking())
189  block.m_lock.unlock();
190  FreeList result = FreeList::bump(payloadEnd, payloadEnd - payloadBegin);
191  if (false)
192  dataLog("Quickly swept block ", RawPointer(this), " with cell size ", cellSize(), " and attributes ", m_attributes, ": ", result, "\n");
193  return result;
194  }
195 
196  // This produces a free list that is ordered in reverse through the block.
197  // This is fine, since the allocation code makes no assumptions about the
198  // order of the free list.
199  FreeCell* head = 0;
200  size_t count = 0;
201  bool isEmpty = true;
202  Vector<size_t> deadCells;
203  VM& vm = *this->vm();
204  auto handleDeadCell = [&] (size_t i) {
205  HeapCell* cell = reinterpret_cast_ptr<HeapCell*>(&block.atoms()[i]);
206 
207  if (destructionMode != BlockHasNoDestructors && emptyMode == NotEmpty) {
208  JSCell* jsCell = static_cast<JSCell*>(cell);
209  if (!jsCell->isZapped()) {
210  destroyFunc(vm, jsCell);
211  jsCell->zap();
212  }
213  }
214 
215  if (sweepMode == SweepToFreeList) {
216  FreeCell* freeCell = reinterpret_cast_ptr<FreeCell*>(cell);
217  if (scribbleMode == Scribble)
218  scribble(freeCell, cellSize());
219  freeCell->next = head;
220  head = freeCell;
221  ++count;
222  }
223  };
224  for (size_t i = firstAtom(); i < m_endAtom; i += m_atomsPerCell) {
225  if (emptyMode == NotEmpty
226  && ((marksMode == MarksNotStale && block.m_marks.get(i))
227  || (newlyAllocatedMode == HasNewlyAllocated && m_newlyAllocated.get(i)))) {
228  isEmpty = false;
229  continue;
230  }
231 
232  if (destructionMode == BlockHasDestructorsAndCollectorIsRunning)
233  deadCells.append(i);
234  else
235  handleDeadCell(i);
236  }
237 
238  // We only want to discard the newlyAllocated bits if we're creating a FreeList,
239  // otherwise we would lose information on what's currently alive.
240  if (sweepMode == SweepToFreeList && newlyAllocatedMode == HasNewlyAllocated)
241  m_newlyAllocatedVersion = MarkedSpace::nullVersion;
242 
243  if (space()->isMarking())
244  block.m_lock.unlock();
245 
246  if (destructionMode == BlockHasDestructorsAndCollectorIsRunning) {
247  for (size_t i : deadCells)
248  handleDeadCell(i);
249  }
250 
251  FreeList result = FreeList::list(head, count * cellSize());
252  if (sweepMode == SweepToFreeList)
253  setIsFreeListed();
254  else if (isEmpty)
255  m_allocator->setIsEmpty(NoLockingNecessary, this, true);
256  if (false)
257  dataLog("Slowly swept block ", RawPointer(&block), " with cell size ", cellSize(), " and attributes ", m_attributes, ": ", result, "\n");
258  return result;
259 }

261 template<typename DestroyFunc>
262 FreeList MarkedBlock::Handle::finishSweepKnowingSubspace(SweepMode sweepMode, const DestroyFunc& destroyFunc)
263 {
264  SweepDestructionMode destructionMode = this->sweepDestructionMode();
265  EmptyMode emptyMode = this->emptyMode();
266  ScribbleMode scribbleMode = this->scribbleMode();
267  NewlyAllocatedMode newlyAllocatedMode = this->newlyAllocatedMode();
268  MarksMode marksMode = this->marksMode();
269 
270  FreeList result;
271  auto trySpecialized = [&] () -> bool {
272  if (sweepMode != SweepToFreeList)
273  return false;
274  if (scribbleMode != DontScribble)
275  return false;
276  if (newlyAllocatedMode != DoesNotHaveNewlyAllocated)
277  return false;
278  if (destructionMode != BlockHasDestructors)
279  return false;
280  if (emptyMode == IsEmpty)
281  return false;
282 
283  switch (marksMode) {
284  case MarksNotStale:
285  result = specializedSweep<true, NotEmpty, SweepToFreeList, BlockHasDestructors, DontScribble, DoesNotHaveNewlyAllocated, MarksNotStale>(IsEmpty, SweepToFreeList, BlockHasDestructors, DontScribble, DoesNotHaveNewlyAllocated, MarksNotStale, destroyFunc);
286  return true;
287  case MarksStale:
288  result = specializedSweep<true, NotEmpty, SweepToFreeList, BlockHasDestructors, DontScribble, DoesNotHaveNewlyAllocated, MarksStale>(IsEmpty, SweepToFreeList, BlockHasDestructors, DontScribble, DoesNotHaveNewlyAllocated, MarksStale, destroyFunc);
289  return true;
290  }
291 
292  return false;
293  };
294 
295  if (trySpecialized())
296  return result;
297 
298  // The template arguments don't matter because the first one is false.
299  return specializedSweep<false, IsEmpty, SweepOnly, BlockHasNoDestructors, DontScribble, HasNewlyAllocated, MarksStale>(emptyMode, sweepMode, destructionMode, scribbleMode, newlyAllocatedMode, marksMode, destroyFunc);

312 inline MarkedBlock::Handle::EmptyMode MarkedBlock::Handle::emptyMode()
313 {
314  // It's not obvious, but this is the only way to know if the block is empty. It's the only
315  // bit that captures these caveats:
316  // - It's true when the block is freshly allocated.
317  // - It's true if the block had been swept in the past, all destructors were called, and that
318  // sweep proved that the block is empty.
319  // - It's false if there are any destructors that need to be called, even if the block has no
320  // live objects.
321  return m_allocator->isEmpty(NoLockingNecessary, this) ? IsEmpty : NotEmpty;
322 }
