77 inline void HandleStack::zapTo(Frame& lastFrame)
78 {
79 #ifdef NDEBUG

82  const Vector<HandleSlot>& blocks = m_blockStack.blocks();
83 
84  if (lastFrame.m_end != m_frame.m_end) { // Zapping to a frame in a different block.
85  int i = blocks.size() - 1;
86  for ( ; blocks[i] + m_blockStack.blockLength != lastFrame.m_end; --i) {
87  for (int j = m_blockStack.blockLength - 1; j >= 0; --j)
88  blocks[i][j] = JSValue();
89  }
90 
91  for (HandleSlot it = blocks[i] + m_blockStack.blockLength - 1; it != lastFrame.m_next - 1; --it)
92  *it = JSValue();
93 
94  return;
95  }
96
