19 template <typename T>
20 typename AudioEncoderIsacT<T>::Config CreateIsacConfig(
21  const CodecInst& codec_inst,
22  const rtc::scoped_refptr<LockedIsacBandwidthInfo>& bwinfo) {
23  typename AudioEncoderIsacT<T>::Config config;
24  config.bwinfo = bwinfo;
25  config.payload_type = codec_inst.pltype;
26  config.sample_rate_hz = codec_inst.plfreq;
27  config.frame_size_ms =
28  rtc::CheckedDivExact(1000 * codec_inst.pacsize, config.sample_rate_hz);
29  config.adaptive_mode = (codec_inst.rate == -1);
30  if (codec_inst.rate != -1)
31  config.bit_rate = codec_inst.rate;
32  return config;
33 }

35 template <typename T>
36 bool AudioEncoderIsacT<T>::Config::IsOk() const {
37  if (max_bit_rate < 32000 && max_bit_rate != -1)
38  return false;
39  if (max_payload_size_bytes < 120 && max_payload_size_bytes != -1)
40  return false;
41  if (adaptive_mode && !bwinfo)
42  return false;
43  switch (sample_rate_hz) {
44  case 16000:
45  if (max_bit_rate > 53400)
46  return false;
47  if (max_payload_size_bytes > 400)
48  return false;
49  return (frame_size_ms == 30 || frame_size_ms == 60) &&
50  (bit_rate == 0 || (bit_rate >= 10000 && bit_rate <= 32000));
51  case 32000:
52  if (max_bit_rate > 160000)
53  return false;
54  if (max_payload_size_bytes > 600)
55  return false;
56  return T::has_swb &&
57  (frame_size_ms == 30 &&
58  (bit_rate == 0 || (bit_rate >= 10000 && bit_rate <= 56000)));
59  default:
60  return false;
61  }
62 }

110 template <typename T>
111 AudioEncoder::EncodedInfo AudioEncoderIsacT<T>::EncodeImpl(
112  uint32_t rtp_timestamp,
113  rtc::ArrayView<const int16_t> audio,
114  rtc::Buffer* encoded) {
115  if (!packet_in_progress_) {
116  // Starting a new packet; remember the timestamp for later.
117  packet_in_progress_ = true;
118  packet_timestamp_ = rtp_timestamp;
119  }
120  if (bwinfo_) {
121  IsacBandwidthInfo bwinfo = bwinfo_->Get();
122  T::SetBandwidthInfo(isac_state_, &bwinfo);
123  }
124 
125  size_t encoded_bytes = encoded->AppendData(
126  kSufficientEncodeBufferSizeBytes,
127  [&] (rtc::ArrayView<uint8_t> encoded) {
128  int r = T::Encode(isac_state_, audio.data(), encoded.data());
129 
130  RTC_CHECK_GE(r, 0) << "Encode failed (error code "
131  << T::GetErrorCode(isac_state_) << ")";
132 
133  return static_cast<size_t>(r);
134  });
135 
136  if (encoded_bytes == 0)
137  return EncodedInfo();
138 
139  // Got enough input to produce a packet. Return the saved timestamp from
140  // the first chunk of input that went into the packet.
141  packet_in_progress_ = false;
142  EncodedInfo info;
143  info.encoded_bytes = encoded_bytes;
144  info.encoded_timestamp = packet_timestamp_;
145  info.payload_type = config_.payload_type;
146  info.encoder_type = CodecType::kIsac;
147  return info;
148 }

155 template <typename T>
156 void AudioEncoderIsacT<T>::RecreateEncoderInstance(const Config& config) {
157  RTC_CHECK(config.IsOk());
158  packet_in_progress_ = false;
159  bwinfo_ = config.bwinfo;
160  if (isac_state_)
161  RTC_CHECK_EQ(0, T::Free(isac_state_));
162  RTC_CHECK_EQ(0, T::Create(&isac_state_));
163  RTC_CHECK_EQ(0, T::EncoderInit(isac_state_, config.adaptive_mode ? 0 : 1));
164  RTC_CHECK_EQ(0, T::SetEncSampRate(isac_state_, config.sample_rate_hz));
165  const int bit_rate = config.bit_rate == 0 ? kDefaultBitRate : config.bit_rate;
166  if (config.adaptive_mode) {
167  RTC_CHECK_EQ(0, T::ControlBwe(isac_state_, bit_rate, config.frame_size_ms,
168  config.enforce_frame_size));
169  } else {
170  RTC_CHECK_EQ(0, T::Control(isac_state_, bit_rate, config.frame_size_ms));
171  }
172  if (config.max_payload_size_bytes != -1)
173  RTC_CHECK_EQ(
174  0, T::SetMaxPayloadSize(isac_state_, config.max_payload_size_bytes));
175  if (config.max_bit_rate != -1)
176  RTC_CHECK_EQ(0, T::SetMaxRate(isac_state_, config.max_bit_rate));
177 
178  // Set the decoder sample rate even though we just use the encoder. This
179  // doesn't appear to be necessary to produce a valid encoding, but without it
180  // we get an encoding that isn't bit-for-bit identical with what a combined
181  // encoder+decoder object produces.
182  RTC_CHECK_EQ(0, T::SetDecSampRate(isac_state_, config.sample_rate_hz));
183 
184  config_ = config;
185 }
