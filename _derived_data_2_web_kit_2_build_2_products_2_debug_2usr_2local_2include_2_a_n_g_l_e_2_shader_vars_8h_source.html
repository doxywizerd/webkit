58  // All of the shader's variables are described using nested data
59  // structures. This is needed in order to disambiguate similar looking
60  // types, such as two structs containing the same fields, but in
61  // different orders. "findInfoByMappedName" provides an easy query for
62  // users to dive into the data structure and fetch the unique variable
63  // instance corresponding to a dereferencing chain of the top-level
64  // variable.
65  // Given a mapped name like 'a[0].b.c[0]', return the ShaderVariable
66  // that defines 'c' in |leafVar|, and the original name 'A[0].B.C[0]'
67  // in |originalName|, based on the assumption that |this| defines 'a'.
68  // If no match is found, return false.
69  bool findInfoByMappedName(const std::string &mappedFullName,
70  const ShaderVariable **leafVar,
71  std::string* originalFullName) const;
72 
73  bool isBuiltIn() const { return name.compare(0, 3, "gl_") == 0; }
