34 // ASSERT_VALID_CODE_POINTER checks that ptr is a non-null pointer, and that it is a valid
35 // instruction address on the platform (for example, check any alignment requirements).
36 #if CPU(ARM_THUMB2) && ENABLE(JIT)
37 // ARM instructions must be 16-bit aligned. Thumb2 code pointers to be loaded into
38 // into the processor are decorated with the bottom bit set, while traditional ARM has
39 // the lower bit clear. Since we don't know what kind of pointer, we check for both
40 // decorated and undecorated null.
41 #define ASSERT_VALID_CODE_POINTER(ptr) \
42  ASSERT(reinterpret_cast<intptr_t>(ptr) & ~1)
43 #define ASSERT_VALID_CODE_OFFSET(offset) \
44  ASSERT(!(offset & 1)) // Must be multiple of 2.
45 #else
46 #define ASSERT_VALID_CODE_POINTER(ptr) \
47  ASSERT(ptr)
48 #define ASSERT_VALID_CODE_OFFSET(offset) // Anything goes!
49 #endif
