44  // These methods are symmetric; reserveUncommitted allocates VM in an uncommitted state,
45  // releaseDecommitted should be called on a region of VM allocated by a single reservation,
46  // the memory must all currently be in a decommitted state.
47  WTF_EXPORT_PRIVATE static void* reserveUncommitted(size_t, Usage = UnknownUsage, bool writable = true, bool executable = false, bool includesGuardPages = false);
48  WTF_EXPORT_PRIVATE static void releaseDecommitted(void*, size_t);
49 
50  // These methods are symmetric; they commit or decommit a region of VM (uncommitted VM should
51  // never be accessed, since the OS may not have attached physical memory for these regions).
52  // Clients should only call commit on uncommitted regions and decommit on committed regions.
53  WTF_EXPORT_PRIVATE static void commit(void*, size_t, bool writable, bool executable);
54  WTF_EXPORT_PRIVATE static void decommit(void*, size_t);
55 
56  // These methods are symmetric; reserveAndCommit allocates VM in an committed state,
57  // decommitAndRelease should be called on a region of VM allocated by a single reservation,
58  // the memory must all currently be in a committed state.
59  WTF_EXPORT_PRIVATE static void* reserveAndCommit(size_t, Usage = UnknownUsage, bool writable = true, bool executable = false, bool includesGuardPages = false);
60  static void decommitAndRelease(void* base, size_t size);
61 
62  // These methods are akin to reserveAndCommit/decommitAndRelease, above - however rather than
63  // committing/decommitting the entire region additional parameters allow a subregion to be
64  // specified.
65  WTF_EXPORT_PRIVATE static void* reserveAndCommit(size_t reserveSize, size_t commitSize, Usage = UnknownUsage, bool writable = true, bool executable = false);
66 
67  // Reallocate an existing, committed allocation.
68  // The prior allocation must be fully comitted, and the new size will also be fully committed.
69  // This interface is provided since it may be possible to optimize this operation on some platforms.
70  template<typename T>
71  static T* reallocateCommitted(T*, size_t oldSize, size_t newSize, Usage = UnknownUsage, bool writable = true, bool executable = false);
72 
73  // Hint to the OS that an address range is not expected to be accessed anytime soon.
74  WTF_EXPORT_PRIVATE static void hintMemoryNotNeededSoon(void*, size_t);
75 };

77 inline void* OSAllocator::reserveAndCommit(size_t reserveSize, size_t commitSize, Usage usage, bool writable, bool executable)
78 {
79  void* base = reserveUncommitted(reserveSize, usage, writable, executable);
80  commit(base, commitSize, writable, executable);
81  return base;
82 }

84 inline void OSAllocator::decommitAndRelease(void* releaseBase, size_t releaseSize)
85 {
86  releaseDecommitted(releaseBase, releaseSize);
87 }

89 template<typename T>
90 inline T* OSAllocator::reallocateCommitted(T* oldBase, size_t oldSize, size_t newSize, Usage usage, bool writable, bool executable)
91 {
92  void* newBase = reserveAndCommit(newSize, usage, writable, executable);
93  memcpy(newBase, oldBase, std::min(oldSize, newSize));
94  decommitAndRelease(oldBase, oldSize);
95  return static_cast<T*>(newBase);
96 }
