103 /* EVP_tls_cbc_get_padding determines the padding from the decrypted, TLS, CBC
104  * record in |in|. This decrypted record should not include any "decrypted"
105  * explicit IV. If the record is publicly invalid, it returns zero. Otherwise,
106  * it returns one and sets |*out_padding_ok| to all ones (0xfff..f) if the
107  * padding is valid and zero otherwise. It then sets |*out_len| to the length
108  * with the padding removed or |in_len| if invalid.
109  *
110  * If the function returns one, it runs in time independent of the contents of
111  * |in|. It is also guaranteed that |*out_len| >= |mac_size|, satisfying
112  * |EVP_tls_cbc_copy_mac|'s precondition. */
113 int EVP_tls_cbc_remove_padding(unsigned *out_padding_ok, unsigned *out_len,
114  const uint8_t *in, unsigned in_len,
115  unsigned block_size, unsigned mac_size);

133 /* EVP_tls_cbc_digest_record computes the MAC of a decrypted, padded TLS
134  * record.
135  *
136  * md: the hash function used in the HMAC.
137  * EVP_tls_cbc_record_digest_supported must return true for this hash.
138  * md_out: the digest output. At most EVP_MAX_MD_SIZE bytes will be written.
139  * md_out_size: the number of output bytes is written here.
140  * header: the 13-byte, TLS record header.
141  * data: the record data itself
142  * data_plus_mac_size: the secret, reported length of the data and MAC
143  * once the padding has been removed.
144  * data_plus_mac_plus_padding_size: the public length of the whole
145  * record, including padding.
146  *
147  * On entry: by virtue of having been through one of the remove_padding
148  * functions, above, we know that data_plus_mac_size is large enough to contain
149  * a padding byte and MAC. (If the padding was invalid, it might contain the
150  * padding too. ) */
