19 // The std library doesn't provide a binary max_exponent for integers, however
20 // we can compute one by adding one to the number of non-sign bits. This allows
21 // for accurate range comparisons between floating point and integer types.
22 template <typename NumericType>
23 struct MaxExponent
24 {
25  static_assert(std::is_arithmetic<NumericType>::value, "Argument must be numeric.");
26  static const int value =
27  std::numeric_limits<NumericType>::is_iec559
28  ? std::numeric_limits<NumericType>::max_exponent
29  : (sizeof(NumericType) * CHAR_BIT + 1 - std::numeric_limits<NumericType>::is_signed);
30 };

95 enum RangeConstraint : unsigned char
96 {
97  RANGE_VALID = 0x0, // Value can be represented by the destination type.
98  RANGE_UNDERFLOW = 0x1, // Value would overflow.
99  RANGE_OVERFLOW = 0x2, // Value would underflow.

112 // This function creates a RangeConstraint from an upper and lower bound
113 // check by taking advantage of the fact that only NaN can be out of range in
114 // both directions at once.
115 constexpr inline RangeConstraint GetRangeConstraint(bool is_in_upper_bound, bool is_in_lower_bound)
116 {
117  return GetRangeConstraint((is_in_upper_bound ? 0 : RANGE_OVERFLOW) |
118  (is_in_lower_bound ? 0 : RANGE_UNDERFLOW));
119 }

121 // The following helper template addresses a corner case in range checks for
122 // conversion from a floating-point type to an integral type of smaller range
123 // but larger precision (e.g. float -> unsigned). The problem is as follows:
124 // 1. Integral maximum is always one less than a power of two, so it must be
125 // truncated to fit the mantissa of the floating point. The direction of
126 // rounding is implementation defined, but by default it's always IEEE
127 // floats, which round to nearest and thus result in a value of larger
128 // magnitude than the integral value.
129 // Example: float f = UINT_MAX; // f is 4294967296f but UINT_MAX
130 // // is 4294967295u.
131 // 2. If the floating point value is equal to the promoted integral maximum
132 // value, a range check will erroneously pass.
133 // Example: (4294967296f <= 4294967295u) // This is true due to a precision
134 // // loss in rounding up to float.
135 // 3. When the floating point value is then converted to an integral, the
136 // resulting value is out of range for the target integral type and
137 // thus is implementation defined.
138 // Example: unsigned u = (float)INT_MAX; // u will typically overflow to 0.
139 // To fix this bug we manually truncate the maximum value when the destination
140 // type is an integral of larger precision than the source floating-point type,
141 // such that the resulting maximum is represented exactly as a floating point.
142 template <typename Dst, typename Src>
143 struct NarrowingRange
144 {
145  typedef typename std::numeric_limits<Src> SrcLimits;
146  typedef typename std::numeric_limits<Dst> DstLimits;
147  // The following logic avoids warnings where the max function is
148  // instantiated with invalid values for a bit shift (even though
149  // such a function can never be called).
150  static const int shift = (MaxExponent<Src>::value > MaxExponent<Dst>::value &&
151  SrcLimits::digits < DstLimits::digits &&
152  SrcLimits::is_iec559 &&
153  DstLimits::is_integer)
154  ? (DstLimits::digits - SrcLimits::digits)
155  : 0;
156 
157  static constexpr Dst max()
158  {
159  // We use UINTMAX_C below to avoid compiler warnings about shifting floating
160  // points. Since it's a compile time calculation, it shouldn't have any
161  // performance impact.
162  return DstLimits::max() - static_cast<Dst>((UINTMAX_C(1) << shift) - 1);
163  }
164 
165  static constexpr Dst min()
166  {
167  return std::numeric_limits<Dst>::is_iec559 ? -DstLimits::max() : DstLimits::min();
168  }
169 };

254  value >= static_cast<Src>(0));
