154  const webrtc::CodecInst& codec)) {
155  WEBRTC_CHECK_CHANNEL(channel);
156  Channel* ch = channels_[channel];
157  // Check if something else already has this slot.
158  if (codec.pltype != -1) {
159  for (std::vector<webrtc::CodecInst>::iterator it =
160  ch->recv_codecs.begin(); it != ch->recv_codecs.end(); ++it) {
161  if (it->pltype == codec.pltype &&
162  _stricmp(it->plname, codec.plname) != 0) {
163  return -1;
164  }
165  }
166  }
167  // Otherwise try to find this codec and update its payload type.
168  int result = -1; // not found
169  for (std::vector<webrtc::CodecInst>::iterator it = ch->recv_codecs.begin();
170  it != ch->recv_codecs.end(); ++it) {
171  if (strcmp(it->plname, codec.plname) == 0 &&
172  it->plfreq == codec.plfreq &&
173  it->channels == codec.channels) {
174  it->pltype = codec.pltype;
175  result = 0;
176  }
177  }
178  return result;
179  }
