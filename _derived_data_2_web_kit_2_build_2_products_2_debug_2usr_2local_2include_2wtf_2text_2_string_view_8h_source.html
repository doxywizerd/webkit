409 inline StringView StringView::substring(unsigned start, unsigned length) const
410 {
411  if (start >= this->length())
412  return empty();
413  unsigned maxLength = this->length() - start;
414 
415  if (length >= maxLength) {
416  if (!start)
417  return *this;
418  length = maxLength;
419  }
420 
421  if (is8Bit()) {
422  StringView result(characters8() + start, length);
423  result.setUnderlyingString(*this);
424  return result;
425  }
426  StringView result(characters16() + start, length);
427  result.setUnderlyingString(*this);
428  return result;
