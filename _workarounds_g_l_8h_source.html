32  // When writing a float to a normalized integer framebuffer, desktop OpenGL is allowed to write
33  // one of the two closest normalized integer representations (although round to nearest is
34  // preferred) (see section 2.3.5.2 of the GL 4.5 core specification). OpenGL ES requires that
35  // round-to-nearest is used (see "Conversion from Floating-Point to Framebuffer Fixed-Point" in
36  // section 2.1.2 of the OpenGL ES 2.0.25 spec). This issue only shows up on Intel and AMD
37  // drivers on framebuffer formats that have 1-bit alpha, work around this by using higher
38  // precision formats instead.
39  bool avoid1BitAlphaTextureFormats;
40 
41  // On some older Intel drivers, GL_RGBA4 is not color renderable, glCheckFramebufferStatus
42  // returns GL_FRAMEBUFFER_UNSUPPORTED. Work around this by using a known color-renderable
43  // format.
44  bool rgba4IsNotSupportedForColorRendering;
45 
46  // When clearing a framebuffer on Intel or AMD drivers, when GL_FRAMEBUFFER_SRGB is enabled, the
47  // driver clears to the linearized clear color despite the framebuffer not supporting SRGB
48  // blending. It only seems to do this when the framebuffer has only linear attachments, mixed
49  // attachments appear to get the correct clear color.
50  bool doesSRGBClearsOnLinearFramebufferAttachments;
51 
52  // On Mac some GLSL constructs involving do-while loops cause GPU hangs, such as the following:
53  // int i = 1;
54  // do {
55  // i --;
56  // continue;
57  // } while (i > 0)
58  // Work around this by rewriting the do-while to use another GLSL construct (block + while)
59  bool doWhileGLSLCausesGPUHang;
60 
61  // Calling glFinish doesn't cause all queries to report that the result is available on some
62  // (NVIDIA) drivers. It was found that enabling GL_DEBUG_OUTPUT_SYNCHRONOUS before the finish
63  // causes it to fully finish.
64  bool finishDoesNotCauseQueriesToBeAvailable;
65 
66  // Always call useProgram after a successful link to avoid a driver bug.
67  // This workaround is meant to reproduce the use_current_program_after_successful_link
68  // workaround in Chromium (http://crbug.com/110263). It has been shown that this workaround is
69  // not necessary for MacOSX 10.9 and higher (http://crrev.com/39eb535b).
70  bool alwaysCallUseProgramAfterLink;
71 
72  // In the case of unpacking from a pixel unpack buffer, unpack overlapping rows row by row.
73  bool unpackOverlappingRowsSeparatelyUnpackBuffer;
74  // In the case of packing to a pixel pack buffer, pack overlapping rows row by row.
75  bool packOverlappingRowsSeparatelyPackBuffer;
76 
77  // During initialization, assign the current vertex attributes to the spec-mandated defaults.
78  bool initializeCurrentVertexAttributes;
79 
80  // abs(i) where i is an integer returns unexpected result on Intel Mac.
81  // Emulate abs(i) with i * sign(i).
82  bool emulateAbsIntFunction;
83 
84  // On Intel Mac, calculation of loop conditions in for and while loop has bug.
85  // Add "&& true" to the end of the condition expression to work around the bug.
86  bool addAndTrueToLoopCondition;
87 
88  // When uploading textures from an unpack buffer, some drivers count an extra row padding when
89  // checking if the pixel unpack buffer is big enough. Tracking bug: http://anglebug.com/1512
90  // For example considering the pixel buffer below where in memory, each row data (D) of the
91  // texture is followed by some unused data (the dots):
92  // +-------+--+
93  // |DDDDDDD|..|
94  // |DDDDDDD|..|
95  // |DDDDDDD|..|
96  // |DDDDDDD|..|
97  // +-------A--B
98  // The last pixel read will be A, but the driver will think it is B, causing it to generate an
99  // error when the pixel buffer is just big enough.
100  bool unpackLastRowSeparatelyForPaddingInclusion;
101 
102  // Equivalent workaround when uploading data from a pixel pack buffer.
103  bool packLastRowSeparatelyForPaddingInclusion;
104 
105  // On some Intel drivers, using isnan() on highp float will get wrong answer. To work around
106  // this bug, we use an expression to emulate function isnan().
107  // Tracking bug: http://crbug.com/650547
108  bool emulateIsnanFloat;
109 
110  // On Mac with OpenGL version 4.1, unused std140 or shared uniform blocks will be
111  // treated as inactive which is not consistent with WebGL2.0 spec. Reference all members in a
112  // unused std140 or shared uniform block at the beginning of main to work around it.
113  bool useUnusedBlocksWithStandardOrSharedLayout;
114 };
