118 inline bool SelectorChecker::isCommonPseudoClassSelector(const CSSSelector* selector)
119 {
120  if (selector->match() != CSSSelector::PseudoClass)
121  return false;
122  CSSSelector::PseudoClassType pseudoType = selector->pseudoClassType();
123  return pseudoType == CSSSelector::PseudoClassLink
124  || pseudoType == CSSSelector::PseudoClassAnyLink
125  || pseudoType == CSSSelector::PseudoClassAnyLinkDeprecated
126  || pseudoType == CSSSelector::PseudoClassVisited
127  || pseudoType == CSSSelector::PseudoClassFocus;
