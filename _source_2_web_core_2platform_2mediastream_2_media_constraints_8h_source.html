136  bool nearlyEqual(double a, double b) const
137  {
138  // Don't require strict equality when comparing constraints, or many floating point constraint values,
139  // e.g. "aspectRatio: 1.333", will never match.
140  const double epsilon = 0.00001;
141  return std::abs(a - b) <= epsilon;
142  }
143 
144  double fitnessDistance(ValueType rangeMin, ValueType rangeMax) const
145  {
146  // https://w3c.github.io/mediacapture-main/#dfn-applyconstraints
147  // 1. If the constraint is not supported by the browser, the fitness distance is 0.
148  if (isEmpty())
149  return 0;
150 
151  // 2. If the constraint is required ('min', 'max', or 'exact'), and the settings
152  // dictionary's value for the constraint does not satisfy the constraint, the
153  // fitness distance is positive infinity.
154  bool valid = validForRange(rangeMin, rangeMax);
155  if (m_exact) {
156  if (valid && m_min && m_exact.value() < m_min.value())
157  valid = false;
158  if (valid && m_max && m_exact.value() > m_max.value())
159  valid = false;
160  if (!valid)
161  return std::numeric_limits<double>::infinity();
162  }
163 
164  if (m_min) {
165  if (valid && m_max && m_min.value() > m_max.value())
166  valid = false;
167  if (!valid)
168  return std::numeric_limits<double>::infinity();
169  }
170 
171  if (m_max) {
172  if (valid && m_min && m_max.value() < m_min.value())
173  valid = false;
174  if (!valid)
175  return std::numeric_limits<double>::infinity();
176  }
177 
178  // 3. If no ideal value is specified, the fitness distance is 0.
179  if (!m_ideal)
180  return 0;
181 
182  // 4. For all positive numeric non-required constraints (such as height, width, frameRate,
183  // aspectRatio, sampleRate and sampleSize), the fitness distance is the result of the formula
184  //
185  // (actual == ideal) ? 0 : |actual - ideal| / max(|actual|,|ideal|)
186  ValueType ideal = m_ideal.value();
187  if (ideal >= rangeMin && ideal <= rangeMax)
188  return 0;
189 
190  ideal = ideal > std::max(rangeMin, rangeMax) ? rangeMax : rangeMin;
191  return static_cast<double>(std::abs(ideal - m_ideal.value())) / std::max(std::abs(ideal), std::abs(m_ideal.value()));
192  }
193 
194  bool validForRange(ValueType rangeMin, ValueType rangeMax) const
195  {
196  if (isEmpty())
197  return false;
198 
199  if (m_exact) {
200  const ValueType exact = m_exact.value();
201  if (exact < rangeMin && !nearlyEqual(exact, rangeMin))
202  return false;
203  if (exact > rangeMax && !nearlyEqual(exact, rangeMax))
204  return false;
205  }
206 
207  if (m_min) {
208  const ValueType constraintMin = m_min.value();
209  if (constraintMin > rangeMax && !nearlyEqual(constraintMin, rangeMax))
210  return false;
211  }
212 
213  if (m_max) {
214  const ValueType constraintMax = m_max.value();
215  if (constraintMax < rangeMin && !nearlyEqual(constraintMax, rangeMin))
216  return false;
217  }
218 
219  return true;
220  }
221 
222  ValueType find(std::function<bool(ValueType)> function) const
223  {
224  if (m_min && function(m_min.value()))
225  return m_min.value();
226 
227  if (m_max && function(m_max.value()))
228  return m_max.value();
229 
230  if (m_exact && function(m_exact.value()))
231  return m_exact.value();
232 
233  if (m_ideal && function(m_ideal.value()))
234  return m_ideal.value();
235 
236  return 0;
237  }
238 
239  ValueType valueForCapabilityRange(ValueType current, ValueType capabilityMin, ValueType capabilityMax) const
240  {
241  ValueType value;
242  ValueType min = capabilityMin;
243  ValueType max = capabilityMax;
244 
245  if (m_exact) {
246  ASSERT(validForRange(capabilityMin, capabilityMax));
247  return m_exact.value();
248  }
249 
250  if (m_min) {
251  value = m_min.value();
252  ASSERT(validForRange(value, capabilityMax));
253  if (value > min)
254  min = value;
255 
256  // If there is no ideal, don't change if minimum is smaller than current.
257  if (!m_ideal && value < current)
258  value = current;
259  }
260 
261  if (m_max) {
262  value = m_max.value();
263  ASSERT(validForRange(capabilityMin, value));
264  if (value < max)
265  max = value;
266  }
267 
268  if (m_ideal)
269  value = std::max(min, std::min(max, m_ideal.value()));
270 
271  return value;
272  }
273 
274  bool isEmpty() const { return !m_min && !m_max && !m_exact && !m_ideal; }
275  bool isMandatory() const { return m_min || m_max || m_exact; }

312  void innerMerge(const NumericConstraint& other)
313  {
314  if (other.isEmpty())
315  return;
316 
317  ValueType value;
318  if (other.getExact(value))
319  m_exact = value;
320 
321  if (other.getMin(value))
322  m_min = value;
323 
324  if (other.getMax(value))
325  m_max = value;
326 
327  // https://w3c.github.io/mediacapture-main/#constrainable-interface
328  // When processing advanced constraints:
329  // ... the User Agent must attempt to apply, individually, any 'ideal' constraints or
330  // a constraint given as a bare value for the property. Of these properties, it must
331  // satisfy the largest number that it can, in any order.
332  if (other.getIdeal(value)) {
333  if (!m_ideal || value > m_ideal.value())
334  m_ideal = value;
335  }
336  }
337 
338  std::optional<ValueType> m_min;
339  std::optional<ValueType> m_max;
340  std::optional<ValueType> m_exact;
341  std::optional<ValueType> m_ideal;
