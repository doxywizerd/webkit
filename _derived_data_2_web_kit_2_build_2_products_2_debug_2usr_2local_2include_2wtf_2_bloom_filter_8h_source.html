34 // Bloom filter with k=2. Uses 2^keyBits/8 bytes of memory.
35 // False positive rate is approximately (1-e^(-2n/m))^2, where n is the number of unique
36 // keys and m is the table size (==2^keyBits).
37 // See http://en.wikipedia.org/wiki/Bloom_filter
38 template <unsigned keyBits>
39 class BloomFilter {
40  WTF_MAKE_FAST_ALLOCATED;
41 public:
42  static const size_t tableSize = 1 << keyBits;
43 
44  BloomFilter();
45 
46  void add(unsigned hash);
47  // For example SHA1::Digest.
48  template <size_t hashSize> void add(const std::array<uint8_t, hashSize>&);
49 
50  void add(const BloomFilter&);
51 
52  // The filter may give false positives (claim it may contain a key it doesn't)
53  // but never false negatives (claim it doesn't contain a key it does).
54  bool mayContain(unsigned hash) const;
55  template <size_t hashSize> bool mayContain(const std::array<uint8_t, hashSize>&) const;
56 
57  void clear();
58 
59  void add(const AtomicString& string) { add(string.impl()->existingHash()); }
60  void add(const String& string) { add(string.impl()->hash()); }
61  bool mayContain(const AtomicString& string) const { return mayContain(string.impl()->existingHash()); }
62  bool mayContain(const String& string) const { return mayContain(string.impl()->hash()); }
63 
64 private:
65  static const unsigned bitsPerPosition = 8 * sizeof(unsigned);
66  static const unsigned keyMask = (1 << keyBits) - 1;
67  static unsigned arrayIndex(unsigned key) { return key / bitsPerPosition; }
68  static unsigned bitMask(unsigned key) { return 1 << (key % bitsPerPosition); }
69  template <size_t hashSize> static std::pair<unsigned, unsigned> keysFromHash(const std::array<uint8_t, hashSize>&);
70 
71  bool isBitSet(unsigned key) const;
72  void setBit(unsigned key);
73 
74  std::array<unsigned, tableSize / bitsPerPosition> m_bitArray;
75 };

136 {

156 // Counting bloom filter with 8 bit counters. Uses 2^keyBits bytes of memory. Error rates as above.
157 // See http://en.wikipedia.org/wiki/Bloom_filter#Counting_filters
158 template <unsigned keyBits>
159 class CountingBloomFilter {
160  WTF_MAKE_FAST_ALLOCATED;
161 public:
162  static const size_t tableSize = 1 << keyBits;
163  static unsigned maximumCount() { return std::numeric_limits<uint8_t>::max(); }

171  // but never false negatives (claim it doesn't contain a key it does).
172  bool mayContain(unsigned hash) const { return firstBucket(hash) && secondBucket(hash); }

175  // Otherwise overflowed keys will stick around.

222 {
223  auto& first = firstBucket(hash);
224  auto& second = secondBucket(hash);
225  ASSERT(first);
226  ASSERT(second);
227  // In case of an overflow, the bucket sticks in the table until clear().
228  if (LIKELY(first < maximumCount()))
229  --first;
230  if (LIKELY(second < maximumCount()))
