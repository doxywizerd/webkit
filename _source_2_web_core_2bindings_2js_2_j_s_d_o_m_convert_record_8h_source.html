62 template<typename K, typename V> struct Converter<IDLRecord<K, V>> : DefaultConverter<IDLRecord<K, V>> {
63  using ReturnType = typename IDLRecord<K, V>::ImplementationType;
64  using KeyType = typename K::ImplementationType;
65  using ValueType = typename V::ImplementationType;
66 
67  static ReturnType convert(JSC::ExecState& state, JSC::JSValue value)
68  {
69  auto& vm = state.vm();
70  auto scope = DECLARE_THROW_SCOPE(vm);
71 
72  // 1. Let result be a new empty instance of record<K, V>.
73  // 2. If Type(O) is Undefined or Null, return result.
74  if (value.isUndefinedOrNull())
75  return { };
76 
77  // 3. If Type(O) is not Object, throw a TypeError.
78  if (!value.isObject()) {
79  throwTypeError(&state, scope);
80  return { };
81  }
82 
83  JSC::JSObject* object = JSC::asObject(value);
84 
85  ReturnType result;
86 
87  // 4. Let keys be ? O.[[OwnPropertyKeys]]().
88  JSC::PropertyNameArray keys(&vm, JSC::PropertyNameMode::Strings);
89  object->getOwnPropertyNames(object, &state, keys, JSC::EnumerationMode());
90  RETURN_IF_EXCEPTION(scope, { });
91 
92  // 5. Repeat, for each element key of keys in List order:
93  for (auto& key : keys) {
94  // 1. Let desc be ? O.[[GetOwnProperty]](key).
95  JSC::PropertyDescriptor descriptor;
96  bool didGetDescriptor = object->getOwnPropertyDescriptor(&state, key, descriptor);
97  RETURN_IF_EXCEPTION(scope, { });
98 
99  if (!didGetDescriptor)
100  continue;
101 
102  // 2. If desc is not undefined and desc.[[Enumerable]] is true:
103 
104  // FIXME: Do we need to check for enumerable / undefined, or is this handled by the default
105  // enumeration mode?
106 
107  if (!descriptor.value().isUndefined() && descriptor.enumerable()) {
108  // 1. Let typedKey be key converted to an IDL value of type K.
109  auto typedKey = Detail::IdentifierConverter<K>::convert(state, key);
110 
111  // 2. Let value be ? Get(O, key).
112  auto subValue = object->get(&state, key);
113  RETURN_IF_EXCEPTION(scope, { });
114 
115  // 3. Let typedValue be value converted to an IDL value of type V.
116  auto typedValue = Converter<V>::convert(state, subValue);
117  RETURN_IF_EXCEPTION(scope, { });
118 
119  // 4. If typedKey is already a key in result, set its value to typedValue.
120  // Note: This can happen when O is a proxy object.
121  // FIXME: Handle this case.
122 
123  // 5. Otherwise, append to result a mapping (typedKey, typedValue).
124  result.append({ typedKey, typedValue });
125  }
126  }
127 
128  // 6. Return result.
129  return result;
130  }
131 };

137  template<typename MapType>
138  static JSC::JSValue convert(JSC::ExecState& state, JSDOMGlobalObject& globalObject, const MapType& map)
139  {
140  auto& vm = state.vm();
141 
142  // 1. Let result be ! ObjectCreate(%ObjectPrototype%).
143  auto result = constructEmptyObject(&state);
144 
145  // 2. Repeat, for each mapping (key, value) in D:
146  for (const auto& keyValuePair : map) {
147  // 1. Let esKey be key converted to an ECMAScript value.
148  // Note, this step is not required, as we need the key to be
149  // an Identifier, not a JSValue.
150 
151  // 2. Let esValue be value converted to an ECMAScript value.
152  auto esValue = toJS<V>(state, globalObject, keyValuePair.value);
153 
154  // 3. Let created be ! CreateDataProperty(result, esKey, esValue).
155  bool created = result->putDirect(vm, JSC::Identifier::fromString(&vm, keyValuePair.key), esValue);
156 
157  // 4. Assert: created is true.
158  ASSERT_UNUSED(created, created);
159  }
160 
161  // 3. Return result.
162  return result;
163  }
