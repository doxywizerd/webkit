332  void computeLexicallyCapturedVariablesAndPurgeCandidates()
333  {
334  // Because variables may be defined at any time in the range of a lexical scope, we must
335  // track lexical variables that might be captured. Then, when we're preparing to pop the top
336  // lexical scope off the stack, we should find which variables are truly captured, and which
337  // variable still may be captured in a parent scope.
338  if (m_lexicalVariables.size() && m_closedVariableCandidates.size()) {
339  for (UniquedStringImpl* impl : m_closedVariableCandidates)
340  m_lexicalVariables.markVariableAsCapturedIfDefined(impl);
341  }
342 
343  // We can now purge values from the captured candidates because they're captured in this scope.
344  {
345  for (auto entry : m_lexicalVariables) {
346  if (entry.value.isCaptured())
347  m_closedVariableCandidates.remove(entry.key.get());
348  }
349  }
350  }

588  void collectFreeVariables(Scope* nestedScope, bool shouldTrackClosedVariables)
589  {
590  if (nestedScope->m_usesEval)
591  m_usesEval = true;
592 
593  {
594  UniquedStringImplPtrSet& destinationSet = m_usedVariables.last();
595  for (const UniquedStringImplPtrSet& usedVariablesSet : nestedScope->m_usedVariables) {
596  for (UniquedStringImpl* impl : usedVariablesSet) {
597  if (nestedScope->m_declaredVariables.contains(impl) || nestedScope->m_lexicalVariables.contains(impl))
598  continue;
599 
600  // "arguments" reference should be resolved at function boudary.
601  if (nestedScope->isFunctionBoundary() && nestedScope->hasArguments() && impl == m_vm->propertyNames->arguments.impl() && !nestedScope->isArrowFunctionBoundary())
602  continue;
603 
604  destinationSet.add(impl);
605  // We don't want a declared variable that is used in an inner scope to be thought of as captured if
606  // that inner scope is both a lexical scope and not a function. Only inner functions and "catch"
607  // statements can cause variables to be captured.
608  if (shouldTrackClosedVariables && (nestedScope->m_isFunctionBoundary || !nestedScope->m_isLexicalScope))
609  m_closedVariableCandidates.add(impl);
610  }
611  }

1197  // for backwards compatibility. This allows us to declare functions with the same name more than once.
