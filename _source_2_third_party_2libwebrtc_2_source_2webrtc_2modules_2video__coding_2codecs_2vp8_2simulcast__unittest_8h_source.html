75  virtual Result OnEncodedImage(const EncodedImage& encoded_image,
76  const CodecSpecificInfo* codec_specific_info,
77  const RTPFragmentationHeader* fragmentation) {
78  // Only store the base layer.
79  if (codec_specific_info->codecSpecific.VP8.simulcastIdx == 0) {
80  if (encoded_image._frameType == kVideoFrameKey) {
81  delete[] encoded_key_frame_._buffer;
82  encoded_key_frame_._buffer = new uint8_t[encoded_image._size];
83  encoded_key_frame_._size = encoded_image._size;
84  encoded_key_frame_._length = encoded_image._length;
85  encoded_key_frame_._frameType = kVideoFrameKey;
86  encoded_key_frame_._completeFrame = encoded_image._completeFrame;
87  memcpy(encoded_key_frame_._buffer, encoded_image._buffer,
88  encoded_image._length);
89  } else {
90  delete[] encoded_frame_._buffer;
91  encoded_frame_._buffer = new uint8_t[encoded_image._size];
92  encoded_frame_._size = encoded_image._size;
93  encoded_frame_._length = encoded_image._length;
94  memcpy(encoded_frame_._buffer, encoded_image._buffer,
95  encoded_image._length);
96  }
97  }
98  picture_id_ = codec_specific_info->codecSpecific.VP8.pictureId;
99  layer_sync_[codec_specific_info->codecSpecific.VP8.simulcastIdx] =
100  codec_specific_info->codecSpecific.VP8.layerSync;
101  temporal_layer_[codec_specific_info->codecSpecific.VP8.simulcastIdx] =
102  codec_specific_info->codecSpecific.VP8.temporalIdx;
103  return Result(Result::OK, encoded_image._timeStamp);
104  }

503  void SwitchingToOneStream(int width, int height) {
504  // Disable all streams except the last and set the bitrate of the last to
505  // 100 kbps. This verifies the way GTP switches to screenshare mode.
506  settings_.VP8()->numberOfTemporalLayers = 1;
507  settings_.maxBitrate = 100;
508  settings_.startBitrate = 100;
509  settings_.width = width;
510  settings_.height = height;
511  for (int i = 0; i < settings_.numberOfSimulcastStreams - 1; ++i) {
512  settings_.simulcastStream[i].maxBitrate = 0;
513  settings_.simulcastStream[i].width = settings_.width;
514  settings_.simulcastStream[i].height = settings_.height;
515  }
516  // Setting input image to new resolution.
517  int half_width = (settings_.width + 1) / 2;
518  input_buffer_ = I420Buffer::Create(settings_.width, settings_.height,
519  settings_.width, half_width, half_width);
520  input_buffer_->InitializeData();
521 
522  input_frame_.reset(
523  new VideoFrame(input_buffer_, 0, 0, webrtc::kVideoRotation_0));
524 
525  // The for loop above did not set the bitrate of the highest layer.
526  settings_.simulcastStream[settings_.numberOfSimulcastStreams - 1]
527  .maxBitrate = 0;
528  // The highest layer has to correspond to the non-simulcast resolution.
529  settings_.simulcastStream[settings_.numberOfSimulcastStreams - 1].width =
530  settings_.width;
531  settings_.simulcastStream[settings_.numberOfSimulcastStreams - 1].height =
532  settings_.height;
533  SetUpRateAllocator();
534  EXPECT_EQ(0, encoder_->InitEncode(&settings_, 1, 1200));
535 
536  // Encode one frame and verify.
537  SetRates(kMaxBitrates[0] + kMaxBitrates[1], 30);
538  std::vector<FrameType> frame_types(kNumberOfSimulcastStreams,
539  kVideoFrameDelta);

541  encoder_callback_,
542  OnEncodedImage(AllOf(Field(&EncodedImage::_frameType, kVideoFrameKey),
