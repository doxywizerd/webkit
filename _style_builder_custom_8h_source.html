394 inline void StyleBuilderCustom::applyValueSize(StyleResolver& styleResolver, CSSValue& value)
395 {
396  styleResolver.style()->resetPageSizeType();
397 
398  if (!is<CSSValueList>(value))
399  return;
400 
401  Length width;
402  Length height;
403  PageSizeType pageSizeType = PAGE_SIZE_AUTO;
404 
405  auto& valueList = downcast<CSSValueList>(value);
406  switch (valueList.length()) {
407  case 2: {
408  auto firstValue = valueList.itemWithoutBoundsCheck(0);
409  auto secondValue = valueList.itemWithoutBoundsCheck(1);
410  // <length>{2} | <page-size> <orientation>
411  if (!is<CSSPrimitiveValue>(*firstValue) || !is<CSSPrimitiveValue>(*secondValue))
412  return;
413  auto& firstPrimitiveValue = downcast<CSSPrimitiveValue>(*firstValue);
414  auto& secondPrimitiveValue = downcast<CSSPrimitiveValue>(*secondValue);
415  if (firstPrimitiveValue.isLength()) {
416  // <length>{2}
417  if (!secondPrimitiveValue.isLength())
418  return;
419  CSSToLengthConversionData conversionData = styleResolver.state().cssToLengthConversionData().copyWithAdjustedZoom(1.0f);
420  width = firstPrimitiveValue.computeLength<Length>(conversionData);
421  height = secondPrimitiveValue.computeLength<Length>(conversionData);
422  } else {
423  // <page-size> <orientation>
424  // The value order is guaranteed. See CSSParser::parseSizeParameter.
425  if (!getPageSizeFromName(&firstPrimitiveValue, &secondPrimitiveValue, width, height))
426  return;
427  }
428  pageSizeType = PAGE_SIZE_RESOLVED;
429  break;
430  }
431  case 1: {
432  auto value = valueList.itemWithoutBoundsCheck(0);
433  // <length> | auto | <page-size> | [ portrait | landscape]
434  if (!is<CSSPrimitiveValue>(*value))
435  return;
436  auto& primitiveValue = downcast<CSSPrimitiveValue>(*value);
437  if (primitiveValue.isLength()) {
438  // <length>
439  pageSizeType = PAGE_SIZE_RESOLVED;
440  width = height = primitiveValue.computeLength<Length>(styleResolver.state().cssToLengthConversionData().copyWithAdjustedZoom(1.0f));
441  } else {
442  switch (primitiveValue.valueID()) {
443  case 0:
444  return;
445  case CSSValueAuto:
446  pageSizeType = PAGE_SIZE_AUTO;
447  break;
448  case CSSValuePortrait:
449  pageSizeType = PAGE_SIZE_AUTO_PORTRAIT;
450  break;
451  case CSSValueLandscape:
452  pageSizeType = PAGE_SIZE_AUTO_LANDSCAPE;
453  break;
454  default:
455  // <page-size>
456  pageSizeType = PAGE_SIZE_RESOLVED;
457  if (!getPageSizeFromName(&primitiveValue, nullptr, width, height))
458  return;
459  }
460  }
461  break;
462  }
463  default:
464  return;
465  }
466  styleResolver.style()->setPageSizeType(pageSizeType);
467  styleResolver.style()->setPageSize({ WTFMove(width), WTFMove(height) });
468 }

861  bool oldFamilyUsedFixedDefaultSize = fontDescription.useFixedDefaultSize();

864  families.reserveInitialCapacity(valueList.length());
865

874  isGenericFamily = fontFamily.fromSystemFontID;

881  family = serifFamily;

886  isGenericFamily = true;
887  break;
888  case CSSValueCursive:
889  family = cursiveFamily;
890  isGenericFamily = true;
891  break;
892  case CSSValueFantasy:
893  family = fantasyFamily;
894  isGenericFamily = true;
895  break;
896  case CSSValueMonospace:
897  family = monospaceFamily;
898  isGenericFamily = true;
899  break;
900  case CSSValueWebkitPictograph:
901  family = pictographFamily;
902  isGenericFamily = true;
903  break;
904  case CSSValueSystemUi:
905  family = systemUiFamily;
906  isGenericFamily = true;
907  break;
908  default:

918  }
