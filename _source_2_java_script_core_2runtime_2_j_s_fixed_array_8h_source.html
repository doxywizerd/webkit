46  ALWAYS_INLINE static JSFixedArray* createFromArray(ExecState* exec, VM& vm, JSArray* array)
47  {
48  IndexingType indexingType = array->indexingType() & IndexingShapeMask;
49  unsigned length = array->length();
50  JSFixedArray* result = JSFixedArray::create(vm, vm.fixedArrayStructure.get(), length);
51 
52  if (!length)
53  return result;
54 
55  if (indexingType == ContiguousShape || indexingType == Int32Shape) {
56  for (unsigned i = 0; i < length; i++) {
57  JSValue value = array->butterfly()->contiguous()[i].get();
58  value = !!value ? value : jsUndefined();
59  result->buffer()[i].set(vm, result, value);
60  }
61  return result;
62  }
63 
64  if (indexingType == DoubleShape) {
65  for (unsigned i = 0; i < length; i++) {
66  double d = array->butterfly()->contiguousDouble()[i];
67  JSValue value = std::isnan(d) ? jsUndefined() : JSValue(JSValue::EncodeAsDouble, d);
68  result->buffer()[i].set(vm, result, value);
69  }
70  return result;
71  }
72 
73 
74  auto throwScope = DECLARE_THROW_SCOPE(vm);
75  for (unsigned i = 0; i < length; i++) {
76  JSValue value = array->getDirectIndex(exec, i);
77  if (!value) {
78  // When we see a hole, we assume that it's safe to assume the get would have returned undefined.
79  // We may still call into this function when !globalObject->isArrayIteratorProtocolFastAndNonObservable(),
80  // however, if we do that, we ensure we're calling in with an array with all self properties between
81  // [0, length).
82  //
83  // We may also call into this during OSR exit to materialize a phantom fixed array.
84  // We may be creating a fixed array during OSR exit even after the iterator protocol changed.
85  // But, when the phantom would have logically been created, the protocol hadn't been
86  // changed. Therefore, it is sound to assume empty indices are jsUndefined().
87  value = jsUndefined();
88  }
89  RETURN_IF_EXCEPTION(throwScope, nullptr);
90  result->buffer()[i].set(vm, result, value);
91  }
92  return result;
93  }
