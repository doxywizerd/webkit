77 template<typename T>
78 float Distance(CartesianPoint<T> a, CartesianPoint<T> b) {
79  return std::sqrt((a.x() - b.x()) * (a.x() - b.x()) +
80  (a.y() - b.y()) * (a.y() - b.y()) +
81  (a.z() - b.z()) * (a.z() - b.z()));
82 }
