70 inline Butterfly* Butterfly::tryCreate(VM& vm, JSCell*, size_t preCapacity, size_t propertyCapacity, bool hasIndexingHeader, const IndexingHeader& indexingHeader, size_t indexingPayloadSizeInBytes)
71 {
72  size_t size = totalSize(preCapacity, propertyCapacity, hasIndexingHeader, indexingPayloadSizeInBytes);
73  void* base = vm.auxiliarySpace.tryAllocate(size);
74  if (!base)
75  return nullptr;
76  Butterfly* result = fromBase(base, preCapacity, propertyCapacity);
77  if (hasIndexingHeader)
78  *result->indexingHeader() = indexingHeader;
79  memset(result->propertyStorage() - propertyCapacity, 0, propertyCapacity * sizeof(EncodedJSValue));

103 inline Butterfly* Butterfly::createOrGrowPropertyStorage(
104  Butterfly* oldButterfly, VM& vm, JSCell* intendedOwner, Structure* structure, size_t oldPropertyCapacity, size_t newPropertyCapacity)
105 {
106  RELEASE_ASSERT(newPropertyCapacity > oldPropertyCapacity);
107  if (!oldButterfly)
108  return create(vm, intendedOwner, 0, newPropertyCapacity, false, IndexingHeader(), 0);
109 
110  size_t preCapacity = oldButterfly->indexingHeader()->preCapacity(structure);
111  size_t indexingPayloadSizeInBytes = oldButterfly->indexingHeader()->indexingPayloadSizeInBytes(structure);
112  bool hasIndexingHeader = structure->hasIndexingHeader(intendedOwner);
113  Butterfly* result = createUninitialized(
114  vm, intendedOwner, preCapacity, newPropertyCapacity, hasIndexingHeader, indexingPayloadSizeInBytes);
115  memcpy(
116  result->propertyStorage() - oldPropertyCapacity,
117  oldButterfly->propertyStorage() - oldPropertyCapacity,
118  totalSize(0, oldPropertyCapacity, hasIndexingHeader, indexingPayloadSizeInBytes));
119  memset(
120  result->propertyStorage() - newPropertyCapacity,
121  0,
122  (newPropertyCapacity - oldPropertyCapacity) * sizeof(EncodedJSValue));
123  return result;
124 }

200 inline Butterfly* Butterfly::unshift(Structure* structure, size_t numberOfSlots)
201 {
202  ASSERT(hasAnyArrayStorage(structure->indexingType()));
203  ASSERT(numberOfSlots <= indexingHeader()->preCapacity(structure));
204  unsigned propertyCapacity = structure->outOfLineCapacity();
205  // FIXME: It would probably be wise to rewrite this as a loop since (1) we know in which
206  // direction we're moving memory so we don't need the extra check of memmove and (2) we're
207  // moving a small amount of memory in the common case so the throughput of memmove won't
208  // amortize the overhead of calling it. And no, we cannot rely on the C++ compiler to
209  // inline memmove (particularly since the size argument is likely to be variable), nor can
210  // we rely on the compiler to recognize the ordering of the pointer arguments (since
211  // propertyCapacity is variable and could cause wrap-around as far as the compiler knows).
212  memmove(
213  propertyStorage() - numberOfSlots - propertyCapacity,
214  propertyStorage() - propertyCapacity,
215  sizeof(EncodedJSValue) * propertyCapacity + sizeof(IndexingHeader) + ArrayStorage::sizeFor(0));
216  return IndexingHeader::fromEndOf(propertyStorage() - numberOfSlots)->butterfly();
217 }
