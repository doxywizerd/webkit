45  BigInteger(const char* decimals, size_t length) : count_(1) {
46  RAPIDJSON_ASSERT(length > 0);
47  digits_[0] = 0;
48  size_t i = 0;
49  const size_t kMaxDigitPerIteration = 19; // 2^64 = 18446744073709551616 > 10^19
50  while (length >= kMaxDigitPerIteration) {
51  AppendDecimal64(decimals + i, decimals + i + kMaxDigitPerIteration);
52  length -= kMaxDigitPerIteration;
53  i += kMaxDigitPerIteration;
54  }
55 
56  if (length > 0)
57  AppendDecimal64(decimals + i, decimals + i + length);
58  }

66  BigInteger& operator+=(uint64_t u) {
67  Type backup = digits_[0];
68  digits_[0] += u;
69  for (size_t i = 0; i < count_ - 1; i++) {
70  if (digits_[i] >= backup)
71  return *this; // no carry
72  backup = digits_[i + 1];
73  digits_[i + 1] += 1;
74  }

124  BigInteger& operator<<=(size_t shift) {
125  if (IsZero() || shift == 0) return *this;
126 
127  size_t offset = shift / kTypeBit;
128  size_t interShift = shift % kTypeBit;
129  RAPIDJSON_ASSERT(count_ + offset <= kCapacity);
130 
131  if (interShift == 0) {
132  std::memmove(&digits_[count_ - 1 + offset], &digits_[count_ - 1], count_ * sizeof(Type));
133  count_ += offset;
134  }
135  else {
136  digits_[count_] = 0;
137  for (size_t i = count_; i > 0; i--)
138  digits_[i + offset] = (digits_[i] << interShift) | (digits_[i - 1] >> (kTypeBit - interShift));
139  digits_[offset] = digits_[0] << interShift;
140  count_ += offset;
141  if (digits_[count_])
142  count_++;
143  }
144 
145  std::memset(digits_, 0, offset * sizeof(Type));
146 
147  return *this;
148  }

180  // Compute absolute difference of this and rhs.
181  // Return false if this < rhs
182  bool Difference(const BigInteger& rhs, BigInteger* out) const {
183  int cmp = Compare(rhs);
184  if (cmp == 0) {
185  *out = BigInteger(0);
186  return false;
187  }
188  const BigInteger *a, *b; // Makes a > b
189  bool ret;
190  if (cmp < 0) { a = &rhs; b = this; ret = true; }
191  else { a = this; b = &rhs; ret = false; }
192 
193  Type borrow = 0;
194  for (size_t i = 0; i < a->count_; i++) {
195  Type d = a->digits_[i] - borrow;
196  if (i < b->count_)
197  d -= b->digits_[i];
198  borrow = (d > a->digits_[i]) ? 1 : 0;
199  out->digits_[i] = d;
200  if (d != 0)
201  out->count_ = i + 1;
202  }
203 
204  return ret;
205  }

223  void AppendDecimal64(const char* begin, const char* end) {
224  uint64_t u = ParseUint64(begin, end);
225  if (IsZero())
226  *this = u;

238  static uint64_t ParseUint64(const char* begin, const char* end) {
239  uint64_t r = 0;
240  for (const char* p = begin; p != end; ++p) {
241  RAPIDJSON_ASSERT(*p >= '0' && *p <= '9');
242  r = r * 10 + (*p - '0');
243  }
244  return r;

247  // Assume a * b + k < 2^128
248  static uint64_t MulAdd64(uint64_t a, uint64_t b, uint64_t k, uint64_t* outHigh) {
249 #if defined(_MSC_VER) && defined(_M_AMD64)
250  uint64_t low = _umul128(a, b, outHigh) + k;
251  if (low < k)
252  (*outHigh)++;
253  return low;
254 #elif (__GNUC__ > 4 || (__GNUC__ == 4 && __GNUC_MINOR__ >= 6)) && defined(__x86_64__)
255  unsigned __int128 p = static_cast<unsigned __int128>(a) * static_cast<unsigned __int128>(b);
256  p += k;
257  *outHigh = p >> 64;
258  return static_cast<uint64_t>(p);
259 #else
260  const uint64_t a0 = a & 0xFFFFFFFF, a1 = a >> 32, b0 = b & 0xFFFFFFFF, b1 = b >> 32;
261  uint64_t x0 = a0 * b0, x1 = a0 * b1, x2 = a1 * b0, x3 = a1 * b1;
262  x1 += (x0 >> 32); // can't give carry
263  x1 += x2;
264  if (x1 < x2)
265  x3 += (static_cast<uint64_t>(1) << 32);
266  uint64_t lo = (x1 << 32) + (x0 & 0xFFFFFFFF);
267  uint64_t hi = x3 + (x1 >> 32);
268 
269  lo += k;
270  if (lo < k)
271  hi++;
272  *outHigh = hi;
273  return lo;
274 #endif
275  }

278  Type c = a + b + (inCarry ? 1 : 0);
