333 
334  // Codecs should be in preference order (most preferred codec first).
335  const std::vector<C>& codecs() const { return codecs_; }
336  void set_codecs(const std::vector<C>& codecs) { codecs_ = codecs; }
337  virtual bool has_codecs() const { return !codecs_.empty(); }
338  bool HasCodec(int id) {
339  bool found = false;
340  for (typename std::vector<C>::iterator iter = codecs_.begin();
341  iter != codecs_.end(); ++iter) {
342  if (iter->id == id) {
343  found = true;
344  break;
345  }
346  }
347  return found;
348  }
349  void AddCodec(const C& codec) {
350  codecs_.push_back(codec);
351  }
352  void AddOrReplaceCodec(const C& codec) {
353  for (typename std::vector<C>::iterator iter = codecs_.begin();
354  iter != codecs_.end(); ++iter) {
355  if (iter->id == codec.id) {
356  *iter = codec;
357  return;
358  }
359  }
360  AddCodec(codec);
361  }
362  void AddCodecs(const std::vector<C>& codecs) {
363  typename std::vector<C>::const_iterator codec;
364  for (codec = codecs.begin(); codec != codecs.end(); ++codec) {
365  AddCodec(*codec);
366  }
367  }
368 
369  private:
370  std::vector<C> codecs_;
371 };
