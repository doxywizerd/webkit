261 /* CRYPTO ByteBuilder.
262  *
263  * |CBB| objects allow one to build length-prefixed serialisations. A |CBB|
264  * object is associated with a buffer and new buffers are created with
265  * |CBB_init|. Several |CBB| objects can point at the same buffer when a
266  * length-prefix is pending, however only a single |CBB| can be 'current' at
267  * any one time. For example, if one calls |CBB_add_u8_length_prefixed| then
268  * the new |CBB| points at the same buffer as the original. But if the original
269  * |CBB| is used then the length prefix is written out and the new |CBB| must
270  * not be used again.
271  *
272  * If one needs to force a length prefix to be written out because a |CBB| is
273  * going out of scope, use |CBB_flush|. If an operation on a |CBB| fails, it is
274  * in an undefined state and must not be used except to call |CBB_cleanup|. */
275 
276 struct cbb_buffer_st {
277  uint8_t *buf;
278  size_t len; /* The number of valid bytes. */
279  size_t cap; /* The size of buf. */
280  char can_resize; /* One iff |buf| is owned by this object. If not then |buf|
281  cannot be resized. */
282  char error; /* One iff there was an error writing to this CBB. All future
283  operations will fail. */
284 };
285 
286 struct cbb_st {
287  struct cbb_buffer_st *base;
288  /* child points to a child CBB if a length-prefix is pending. */
289  CBB *child;
290  /* offset is the number of bytes from the start of |base->buf| to this |CBB|'s
291  * pending length prefix. */
292  size_t offset;
293  /* pending_len_len contains the number of bytes in this |CBB|'s pending
294  * length-prefix, or zero if no length-prefix is pending. */
295  uint8_t pending_len_len;
296  char pending_is_asn1;
297  /* is_top_level is true iff this is a top-level |CBB| (as opposed to a child
298  * |CBB|). Top-level objects are valid arguments for |CBB_finish|. */
299  char is_top_level;
300 };
