28 inline IndexingType JSArray::mergeIndexingTypeForCopying(IndexingType other)
29 {
30  IndexingType type = indexingType();
31  if (!(type & IsArray && other & IsArray))
32  return NonArray;
33 
34  if (hasAnyArrayStorage(type) || hasAnyArrayStorage(other))
35  return NonArray;
36 
37  if (type == ArrayWithUndecided)
38  return other;
39 
40  if (other == ArrayWithUndecided)
41  return type;
42 
43  // We can memcpy an Int32 and a Contiguous into a Contiguous array since
44  // both share the same memory layout for Int32 numbers.
45  if ((type == ArrayWithInt32 || type == ArrayWithContiguous)
46  && (other == ArrayWithInt32 || other == ArrayWithContiguous)) {
47  if (other == ArrayWithContiguous)
48  return other;
49  return type;
50  }
51 
52  if (type != other)
53  return NonArray;
54 
55  return type;
56 }

58 inline bool JSArray::canFastCopy(VM& vm, JSArray* otherArray)
59 {
60  if (hasAnyArrayStorage(indexingType()) || hasAnyArrayStorage(otherArray->indexingType()))
61  return false;
62  // FIXME: We should have a watchpoint for indexed properties on Array.prototype and Object.prototype
63  // instead of walking the prototype chain. https://bugs.webkit.org/show_bug.cgi?id=155592
64  if (structure(vm)->holesMustForwardToPrototype(vm)
65  || otherArray->structure(vm)->holesMustForwardToPrototype(vm))
66  return false;
67  return true;
68 }
