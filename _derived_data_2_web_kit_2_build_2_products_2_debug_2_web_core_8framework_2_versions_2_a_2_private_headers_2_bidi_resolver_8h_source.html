309 void BidiResolverBase<Iterator, Run, DerivedClass>::appendRunInternal()
310 {
311  if (!m_emptyRun && !m_eor.atEnd()) {
312  unsigned startOffset = m_sor.offset();
313  unsigned endOffset = m_eor.offset();
314 
315  if (!endOfLine.atEnd() && endOffset >= endOfLine.offset()) {

357 template<typename Iterator, typename Run, typename DerivedClass>
358 void BidiResolverBase<Iterator, Run, DerivedClass>::lowerExplicitEmbeddingLevel(UCharDirection from)
359 {
360  if (!m_emptyRun && m_eor != m_last) {
361  checkDirectionInLowerRaiseEmbeddingLevel();
362  // bidi.sor ... bidi.eor ... bidi.last eor; need to append the bidi.sor-bidi.eor run or extend it through bidi.last
363  if (from == U_LEFT_TO_RIGHT) {
364  // bidi.sor ... bidi.eor ... bidi.last L
365  if (m_status.eor == U_EUROPEAN_NUMBER) {
366  if (m_status.lastStrong != U_LEFT_TO_RIGHT) {
367  m_direction = U_EUROPEAN_NUMBER;
368  appendRun();
369  }
370  } else if (m_status.eor == U_ARABIC_NUMBER) {
371  m_direction = U_ARABIC_NUMBER;
372  appendRun();
373  } else if (m_status.lastStrong != U_LEFT_TO_RIGHT) {
374  appendRun();
375  m_direction = U_LEFT_TO_RIGHT;
376  }
377  } else if (m_status.eor == U_EUROPEAN_NUMBER || m_status.eor == U_ARABIC_NUMBER || m_status.lastStrong == U_LEFT_TO_RIGHT) {
378  appendRun();
379  m_direction = U_RIGHT_TO_LEFT;
380  }
381  m_eor = m_last;
382  }
383 
384  appendRun();
385  m_emptyRun = true;
386 
387  // sor for the new run is determined by the higher level (rule X10)
388  setLastDir(from);
389  setLastStrongDir(from);
390  m_eor = Iterator();
391 }

395 {
396  if (!m_emptyRun && m_eor != m_last) {
397  checkDirectionInLowerRaiseEmbeddingLevel();

399  if (to == U_LEFT_TO_RIGHT) {
400  // bidi.sor ... bidi.eor ... bidi.last L

403  m_direction = U_EUROPEAN_NUMBER;

405  }
406  } else if (m_status.eor == U_ARABIC_NUMBER) {
407  m_direction = U_ARABIC_NUMBER;
408  appendRun();

416  appendRun();
417  m_direction = U_RIGHT_TO_LEFT;

419  m_eor = m_last;
420  }

427  m_eor = Iterator();

430 template<typename Iterator, typename Run, typename DerivedClass>
431 bool BidiResolverBase<Iterator, Run, DerivedClass>::commitExplicitEmbedding()
432 {
433  // When we're "inIsolate()" we're resolving the parent context which
434  // ignores (skips over) the isolated content, including embedding levels.
435  // We should never accrue embedding levels while skipping over isolated content.
436  ASSERT(!inIsolate() || m_currentExplicitEmbeddingSequence.isEmpty());
437 
438  auto fromLevel = context()->level();
439  RefPtr<BidiContext> toContext = context();
440 
441  for (auto& embedding : m_currentExplicitEmbeddingSequence) {
442  if (embedding.direction == U_POP_DIRECTIONAL_FORMAT) {
443  if (auto* parentContext = toContext->parent())
444  toContext = parentContext;
445  } else {
446  UCharDirection direction = (embedding.direction == U_RIGHT_TO_LEFT_EMBEDDING || embedding.direction == U_RIGHT_TO_LEFT_OVERRIDE) ? U_RIGHT_TO_LEFT : U_LEFT_TO_RIGHT;
447  bool override = embedding.direction == U_LEFT_TO_RIGHT_OVERRIDE || embedding.direction == U_RIGHT_TO_LEFT_OVERRIDE;
448  unsigned char level = toContext->level();
449  if (direction == U_RIGHT_TO_LEFT)
450  level = nextGreaterOddLevel(level);
451  else
452  level = nextGreaterEvenLevel(level);
453  if (level < 61)
454  toContext = BidiContext::create(level, direction, override, embedding.source, toContext.get());
455  }
456  }
457 
458  auto toLevel = toContext->level();
459 
460  if (toLevel > fromLevel)
461  raiseExplicitEmbeddingLevel(fromLevel % 2 ? U_RIGHT_TO_LEFT : U_LEFT_TO_RIGHT, toLevel % 2 ? U_RIGHT_TO_LEFT : U_LEFT_TO_RIGHT);
462  else if (toLevel < fromLevel)
463  lowerExplicitEmbeddingLevel(fromLevel % 2 ? U_RIGHT_TO_LEFT : U_LEFT_TO_RIGHT);
464 
465  setContext(WTFMove(toContext));
466 
467  m_currentExplicitEmbeddingSequence.clear();
468 
469  return fromLevel != toLevel;
470 }

472 template<typename Iterator, typename Run, typename DerivedClass>
473 inline void BidiResolverBase<Iterator, Run, DerivedClass>::updateStatusLastFromCurrentDirection(UCharDirection dirCurrent)
474 {
475  switch (dirCurrent) {
476  case U_EUROPEAN_NUMBER_TERMINATOR:
477  if (m_status.last != U_EUROPEAN_NUMBER)
478  m_status.last = U_EUROPEAN_NUMBER_TERMINATOR;
479  break;
480  case U_EUROPEAN_NUMBER_SEPARATOR:
481  case U_COMMON_NUMBER_SEPARATOR:
482  case U_SEGMENT_SEPARATOR:
483  case U_WHITE_SPACE_NEUTRAL:
484  case U_OTHER_NEUTRAL:
485  switch (m_status.last) {
486  case U_LEFT_TO_RIGHT:
487  case U_RIGHT_TO_LEFT:
488  case U_RIGHT_TO_LEFT_ARABIC:
489  case U_EUROPEAN_NUMBER:
490  case U_ARABIC_NUMBER:
491  m_status.last = dirCurrent;
492  break;
493  default:
494  m_status.last = U_OTHER_NEUTRAL;
495  }
496  break;
497  case U_DIR_NON_SPACING_MARK:
498  case U_BOUNDARY_NEUTRAL:
499  case U_RIGHT_TO_LEFT_EMBEDDING:
500  case U_LEFT_TO_RIGHT_EMBEDDING:
501  case U_RIGHT_TO_LEFT_OVERRIDE:
502  case U_LEFT_TO_RIGHT_OVERRIDE:
503  case U_POP_DIRECTIONAL_FORMAT:
504  // ignore these
505  break;
506  case U_EUROPEAN_NUMBER:

508  default:
509  m_status.last = dirCurrent;

513 template<typename Iterator, typename Run, typename DerivedClass>
514 inline void BidiResolverBase<Iterator, Run, DerivedClass>::reorderRunsFromLevels()
515 {
516  unsigned char levelLow = 128;
517  unsigned char levelHigh = 0;
518  for (Run* run = m_runs.firstRun(); run; run = run->next()) {
519  levelHigh = std::max(run->level(), levelHigh);
520  levelLow = std::min(run->level(), levelLow);
521  }
522 
523  // This implements reordering of the line (L2 according to Bidi spec):
524  // http://unicode.org/reports/tr9/#L2
525  // L2. From the highest level found in the text to the lowest odd level on each line,
526  // reverse any contiguous sequence of characters that are at that level or higher.
527 
528  // Reversing is only done up to the lowest odd level.
529  if (!(levelLow % 2))
530  levelLow++;
531 
532  unsigned count = m_runs.runCount() - 1;
533 
534  while (levelHigh >= levelLow) {
535  unsigned i = 0;
536  Run* run = m_runs.firstRun();
537  while (i < count) {
538  for (;i < count && run && run->level() < levelHigh; i++)
539  run = run->next();
540  unsigned start = i;
541  for (;i <= count && run && run->level() >= levelHigh; i++)
542  run = run->next();
543  unsigned end = i - 1;
544  m_runs.reverseRuns(start, end);
545  }
546  levelHigh--;
547  }
548 }

550 template<typename Iterator, typename Run, typename DerivedClass>
551 void BidiResolverBase<Iterator, Run, DerivedClass>::createBidiRunsForLine(const Iterator& end, VisualDirectionOverride override, bool hardLineBreak)
552 {
553  ASSERT(m_direction == U_OTHER_NEUTRAL);
554 
555  if (override != NoVisualOverride) {
556  m_emptyRun = false;
557  m_sor = m_current;
558  m_eor = Iterator();
559  while (m_current != end && !m_current.atEnd()) {
560  m_eor = m_current;
561  increment();
562  }
563  m_direction = override == VisualLeftToRightOverride ? U_LEFT_TO_RIGHT : U_RIGHT_TO_LEFT;
564  appendRun();
565  m_runs.setLogicallyLastRun(m_runs.lastRun());
566  if (override == VisualRightToLeftOverride && m_runs.runCount())
567  m_runs.reverseRuns(0, m_runs.runCount() - 1);
568  return;
569  }
570 
571  m_emptyRun = true;
572 
573  m_eor = Iterator();
574 
575  m_last = m_current;
576  bool pastEnd = false;
577  BidiResolverBase<Iterator, Run, DerivedClass> stateAtEnd;
578 
579  while (true) {
580  UCharDirection dirCurrent;
581  if (pastEnd && (hardLineBreak || m_current.atEnd())) {
582  BidiContext* c = context();
583  if (hardLineBreak) {
584  // A deviation from the Unicode Bidi Algorithm in order to match
585  // WinIE and user expectations: hard line breaks reset bidi state
586  // coming from unicode bidi control characters, but not those from
587  // DOM nodes with specified directionality
588  stateAtEnd.setContext(c->copyStackRemovingUnicodeEmbeddingContexts());
589 
590  dirCurrent = stateAtEnd.context()->dir();
591  stateAtEnd.setEorDir(dirCurrent);
592  stateAtEnd.setLastDir(dirCurrent);
593  stateAtEnd.setLastStrongDir(dirCurrent);
594  } else {
595  while (c->parent())
596  c = c->parent();
597  dirCurrent = c->dir();
598  }
599  } else {
600  dirCurrent = m_current.direction();
601  if (context()->override()
602  && dirCurrent != U_RIGHT_TO_LEFT_EMBEDDING
603  && dirCurrent != U_LEFT_TO_RIGHT_EMBEDDING
604  && dirCurrent != U_RIGHT_TO_LEFT_OVERRIDE
605  && dirCurrent != U_LEFT_TO_RIGHT_OVERRIDE
606  && dirCurrent != U_POP_DIRECTIONAL_FORMAT)
607  dirCurrent = context()->dir();
608  else if (dirCurrent == U_DIR_NON_SPACING_MARK)
609  dirCurrent = m_status.last;
610  }
611 
612 #if PLATFORM(WIN)
613  // Our Windows build hasn't updated its headers from ICU 6.1, which doesn't have these symbols.
614  const UCharDirection U_FIRST_STRONG_ISOLATE = static_cast<UCharDirection>(19);
615  const UCharDirection U_LEFT_TO_RIGHT_ISOLATE = static_cast<UCharDirection>(20);
616  const UCharDirection U_RIGHT_TO_LEFT_ISOLATE = static_cast<UCharDirection>(21);
617  const UCharDirection U_POP_DIRECTIONAL_ISOLATE = static_cast<UCharDirection>(22);
618 #endif
619  // We ignore all character directionality while in unicode-bidi: isolate spans.
620  // We'll handle ordering the isolated characters in a second pass.
621  if (inIsolate() || dirCurrent == U_FIRST_STRONG_ISOLATE || dirCurrent == U_LEFT_TO_RIGHT_ISOLATE || dirCurrent == U_RIGHT_TO_LEFT_ISOLATE || dirCurrent == U_POP_DIRECTIONAL_ISOLATE)
622  dirCurrent = U_OTHER_NEUTRAL;
623 
624  ASSERT(m_status.eor != U_OTHER_NEUTRAL || m_eor.atEnd());
625  switch (dirCurrent) {
626 
627  // embedding and overrides (X1-X9 in the Bidi specs)
628  case U_RIGHT_TO_LEFT_EMBEDDING:
629  case U_LEFT_TO_RIGHT_EMBEDDING:
630  case U_RIGHT_TO_LEFT_OVERRIDE:
631  case U_LEFT_TO_RIGHT_OVERRIDE:
632  case U_POP_DIRECTIONAL_FORMAT:
633  embed(dirCurrent, FromUnicode);
634  commitExplicitEmbedding();
635  break;
636 
637  // strong types
638  case U_LEFT_TO_RIGHT:
639  switch(m_status.last) {
640  case U_RIGHT_TO_LEFT:
641  case U_RIGHT_TO_LEFT_ARABIC:
642  case U_EUROPEAN_NUMBER:
643  case U_ARABIC_NUMBER:
644  if (m_status.last != U_EUROPEAN_NUMBER || m_status.lastStrong != U_LEFT_TO_RIGHT)
645  appendRun();
646  break;
647  case U_LEFT_TO_RIGHT:
648  break;
649  case U_EUROPEAN_NUMBER_SEPARATOR:
650  case U_EUROPEAN_NUMBER_TERMINATOR:
651  case U_COMMON_NUMBER_SEPARATOR:
652  case U_BOUNDARY_NEUTRAL:
653  case U_BLOCK_SEPARATOR:
654  case U_SEGMENT_SEPARATOR:
655  case U_WHITE_SPACE_NEUTRAL:
656  case U_OTHER_NEUTRAL:
657  if (m_status.eor == U_EUROPEAN_NUMBER) {
658  if (m_status.lastStrong != U_LEFT_TO_RIGHT) {
659  // the numbers need to be on a higher embedding level, so let's close that run
660  m_direction = U_EUROPEAN_NUMBER;
661  appendRun();
662  if (context()->dir() != U_LEFT_TO_RIGHT) {
663  // the neutrals take the embedding direction, which is R
664  m_eor = m_last;
665  m_direction = U_RIGHT_TO_LEFT;
666  appendRun();
667  }
668  }
669  } else if (m_status.eor == U_ARABIC_NUMBER) {
670  // Arabic numbers are always on a higher embedding level, so let's close that run
671  m_direction = U_ARABIC_NUMBER;
672  appendRun();
673  if (context()->dir() != U_LEFT_TO_RIGHT) {
674  // the neutrals take the embedding direction, which is R
675  m_eor = m_last;
676  m_direction = U_RIGHT_TO_LEFT;
677  appendRun();
678  }
679  } else if (m_status.lastStrong != U_LEFT_TO_RIGHT) {
680  //last stuff takes embedding dir
681  if (context()->dir() == U_RIGHT_TO_LEFT) {
682  m_eor = m_last;
683  m_direction = U_RIGHT_TO_LEFT;
684  }
685  appendRun();
686  }
687  break;
688  default:
689  break;
690  }
691  m_eor = m_current;
692  m_status.eor = U_LEFT_TO_RIGHT;
693  m_status.lastStrong = U_LEFT_TO_RIGHT;
694  m_direction = U_LEFT_TO_RIGHT;
695  break;
696  case U_RIGHT_TO_LEFT_ARABIC:
697  case U_RIGHT_TO_LEFT:
698  switch (m_status.last) {
699  case U_LEFT_TO_RIGHT:
700  case U_EUROPEAN_NUMBER:
701  case U_ARABIC_NUMBER:
702  appendRun();
703  FALLTHROUGH;
704  case U_RIGHT_TO_LEFT:
705  case U_RIGHT_TO_LEFT_ARABIC:
706  break;
707  case U_EUROPEAN_NUMBER_SEPARATOR:
708  case U_EUROPEAN_NUMBER_TERMINATOR:
709  case U_COMMON_NUMBER_SEPARATOR:
710  case U_BOUNDARY_NEUTRAL:
711  case U_BLOCK_SEPARATOR:
712  case U_SEGMENT_SEPARATOR:
713  case U_WHITE_SPACE_NEUTRAL:
714  case U_OTHER_NEUTRAL:
715  if (m_status.eor == U_EUROPEAN_NUMBER) {
716  if (m_status.lastStrong == U_LEFT_TO_RIGHT && context()->dir() == U_LEFT_TO_RIGHT)
717  m_eor = m_last;
718  appendRun();
719  } else if (m_status.eor == U_ARABIC_NUMBER)
720  appendRun();
721  else if (m_status.lastStrong == U_LEFT_TO_RIGHT) {
722  if (context()->dir() == U_LEFT_TO_RIGHT)
723  m_eor = m_last;
724  appendRun();
725  }
726  break;
727  default:
728  break;
729  }
730  m_eor = m_current;
731  m_status.eor = U_RIGHT_TO_LEFT;
732  m_status.lastStrong = dirCurrent;
733  m_direction = U_RIGHT_TO_LEFT;
734  break;
735 
736  // weak types:
737 
738  case U_EUROPEAN_NUMBER:
739  if (m_status.lastStrong != U_RIGHT_TO_LEFT_ARABIC) {
740  // if last strong was AL change EN to AN
741  switch (m_status.last) {
742  case U_EUROPEAN_NUMBER:
743  case U_LEFT_TO_RIGHT:
744  break;
745  case U_RIGHT_TO_LEFT:
746  case U_RIGHT_TO_LEFT_ARABIC:
747  case U_ARABIC_NUMBER:
748  m_eor = m_last;
749  appendRun();
750  m_direction = U_EUROPEAN_NUMBER;
751  break;
752  case U_EUROPEAN_NUMBER_SEPARATOR:
753  case U_COMMON_NUMBER_SEPARATOR:
754  if (m_status.eor == U_EUROPEAN_NUMBER)
755  break;
756  FALLTHROUGH;
757  case U_EUROPEAN_NUMBER_TERMINATOR:
758  case U_BOUNDARY_NEUTRAL:
759  case U_BLOCK_SEPARATOR:
760  case U_SEGMENT_SEPARATOR:
761  case U_WHITE_SPACE_NEUTRAL:
762  case U_OTHER_NEUTRAL:
763  if (m_status.eor == U_EUROPEAN_NUMBER) {
764  if (m_status.lastStrong == U_RIGHT_TO_LEFT) {
765  // ENs on both sides behave like Rs, so the neutrals should be R.
766  // Terminate the EN run.
767  appendRun();
768  // Make an R run.
769  m_eor = m_status.last == U_EUROPEAN_NUMBER_TERMINATOR ? m_lastBeforeET : m_last;
770  m_direction = U_RIGHT_TO_LEFT;
771  appendRun();
772  // Begin a new EN run.
773  m_direction = U_EUROPEAN_NUMBER;
774  }
775  } else if (m_status.eor == U_ARABIC_NUMBER) {
776  // Terminate the AN run.
777  appendRun();
778  if (m_status.lastStrong == U_RIGHT_TO_LEFT || context()->dir() == U_RIGHT_TO_LEFT) {
779  // Make an R run.
780  m_eor = m_status.last == U_EUROPEAN_NUMBER_TERMINATOR ? m_lastBeforeET : m_last;
781  m_direction = U_RIGHT_TO_LEFT;
782  appendRun();
783  // Begin a new EN run.
784  m_direction = U_EUROPEAN_NUMBER;
785  }
786  } else if (m_status.lastStrong == U_RIGHT_TO_LEFT) {
787  // Extend the R run to include the neutrals.
788  m_eor = m_status.last == U_EUROPEAN_NUMBER_TERMINATOR ? m_lastBeforeET : m_last;
789  m_direction = U_RIGHT_TO_LEFT;
790  appendRun();
791  // Begin a new EN run.
792  m_direction = U_EUROPEAN_NUMBER;
793  }
794  break;
795  default:
796  break;
797  }
798  m_eor = m_current;
799  m_status.eor = U_EUROPEAN_NUMBER;
800  if (m_direction == U_OTHER_NEUTRAL)
801  m_direction = U_LEFT_TO_RIGHT;
802  break;
803  }
804  FALLTHROUGH;
805  case U_ARABIC_NUMBER:
806  dirCurrent = U_ARABIC_NUMBER;
807  switch (m_status.last) {
808  case U_LEFT_TO_RIGHT:
809  if (context()->dir() == U_LEFT_TO_RIGHT)
810  appendRun();
811  break;
812  case U_ARABIC_NUMBER:
813  break;
814  case U_RIGHT_TO_LEFT:
815  case U_RIGHT_TO_LEFT_ARABIC:
816  case U_EUROPEAN_NUMBER:
817  m_eor = m_last;
818  appendRun();
819  break;
820  case U_COMMON_NUMBER_SEPARATOR:
821  if (m_status.eor == U_ARABIC_NUMBER)
822  break;
823  FALLTHROUGH;
824  case U_EUROPEAN_NUMBER_SEPARATOR:
825  case U_EUROPEAN_NUMBER_TERMINATOR:
826  case U_BOUNDARY_NEUTRAL:
827  case U_BLOCK_SEPARATOR:
828  case U_SEGMENT_SEPARATOR:
829  case U_WHITE_SPACE_NEUTRAL:
830  case U_OTHER_NEUTRAL:
831  if (m_status.eor == U_ARABIC_NUMBER
832  || (m_status.eor == U_EUROPEAN_NUMBER && (m_status.lastStrong == U_RIGHT_TO_LEFT || context()->dir() == U_RIGHT_TO_LEFT))
833  || (m_status.eor != U_EUROPEAN_NUMBER && m_status.lastStrong == U_LEFT_TO_RIGHT && context()->dir() == U_RIGHT_TO_LEFT)) {
834  // Terminate the run before the neutrals.
835  appendRun();
836  // Begin an R run for the neutrals.
837  m_direction = U_RIGHT_TO_LEFT;
838  } else if (m_direction == U_OTHER_NEUTRAL)
839  m_direction = m_status.lastStrong == U_LEFT_TO_RIGHT ? U_LEFT_TO_RIGHT : U_RIGHT_TO_LEFT;
840  m_eor = m_last;
841  appendRun();
842  break;
843  default:
844  break;
845  }
846  m_eor = m_current;
847  m_status.eor = U_ARABIC_NUMBER;
848  if (m_direction == U_OTHER_NEUTRAL)
849  m_direction = U_ARABIC_NUMBER;
850  break;
851  case U_EUROPEAN_NUMBER_SEPARATOR:
852  case U_COMMON_NUMBER_SEPARATOR:
853  break;
854  case U_EUROPEAN_NUMBER_TERMINATOR:
855  if (m_status.last == U_EUROPEAN_NUMBER) {
856  dirCurrent = U_EUROPEAN_NUMBER;
857  m_eor = m_current;
858  m_status.eor = dirCurrent;
859  } else if (m_status.last != U_EUROPEAN_NUMBER_TERMINATOR)
860  m_lastBeforeET = m_emptyRun ? m_eor : m_last;
861  break;
862 
863  // boundary neutrals should be ignored
864  case U_BOUNDARY_NEUTRAL:
865  if (m_eor == m_last)
866  m_eor = m_current;
867  break;
868  // neutrals
869  case U_BLOCK_SEPARATOR:
870  // FIXME: What do we do with newline and paragraph separators that come to here?
871  break;
872  case U_SEGMENT_SEPARATOR:
873  // FIXME: Implement rule L1.
874  break;
875  case U_WHITE_SPACE_NEUTRAL:
876  break;
877  case U_OTHER_NEUTRAL:
878  break;
879  default:
880  break;
881  }
882 
883  if (pastEnd && m_eor == m_current) {
884  if (!m_reachedEndOfLine) {
885  m_eor = endOfLine;
886  switch (m_status.eor) {
887  case U_LEFT_TO_RIGHT:
888  case U_RIGHT_TO_LEFT:
889  case U_ARABIC_NUMBER:
890  m_direction = m_status.eor;
891  break;
892  case U_EUROPEAN_NUMBER:
893  m_direction = m_status.lastStrong == U_LEFT_TO_RIGHT ? U_LEFT_TO_RIGHT : U_EUROPEAN_NUMBER;
894  break;
895  default:
896  ASSERT_NOT_REACHED();
897  }
898  appendRun();
899  }
900  m_current = end;
901  m_status = stateAtEnd.m_status;
902  m_sor = stateAtEnd.m_sor;
903  m_eor = stateAtEnd.m_eor;
904  m_last = stateAtEnd.m_last;
905  m_reachedEndOfLine = stateAtEnd.m_reachedEndOfLine;
906  m_lastBeforeET = stateAtEnd.m_lastBeforeET;
907  m_emptyRun = stateAtEnd.m_emptyRun;
908  m_direction = U_OTHER_NEUTRAL;
909  break;
910  }
911 
912  updateStatusLastFromCurrentDirection(dirCurrent);
913  m_last = m_current;
914 
915  if (m_emptyRun) {
916  m_sor = m_current;
917  m_emptyRun = false;
918  }
919 
920  increment();
921  if (!m_currentExplicitEmbeddingSequence.isEmpty()) {
922  bool committed = commitExplicitEmbedding();
923  if (committed && pastEnd) {
924  m_current = end;
925  m_status = stateAtEnd.m_status;
926  m_sor = stateAtEnd.m_sor;
927  m_eor = stateAtEnd.m_eor;
928  m_last = stateAtEnd.m_last;
929  m_reachedEndOfLine = stateAtEnd.m_reachedEndOfLine;
930  m_lastBeforeET = stateAtEnd.m_lastBeforeET;
931  m_emptyRun = stateAtEnd.m_emptyRun;
932  m_direction = U_OTHER_NEUTRAL;
933  break;
934  }
935  }
936 
937  if (!pastEnd && (m_current == end || m_current.atEnd())) {
938  if (m_emptyRun)
939  break;
940  stateAtEnd.m_status = m_status;
941  stateAtEnd.m_sor = m_sor;
942  stateAtEnd.m_eor = m_eor;
943  stateAtEnd.m_last = m_last;
944  stateAtEnd.m_reachedEndOfLine = m_reachedEndOfLine;
945  stateAtEnd.m_lastBeforeET = m_lastBeforeET;
946  stateAtEnd.m_emptyRun = m_emptyRun;
947  endOfLine = m_last;
948  pastEnd = true;
949  }
950  }
951 
952  m_runs.setLogicallyLastRun(m_runs.lastRun());
953  reorderRunsFromLevels();
954  endOfLine = Iterator();
955 }
