47 // Crash with a SIGTRAP i.e EXC_BREAKPOINT.
48 // We are not using __builtin_trap because it is only guaranteed to abort, but not necessarily
49 // trigger a SIGTRAP. Instead, we use inline asm to ensure that we trigger the SIGTRAP.
50 #define BCRASH() do { \
51  BBreakpointTrap(); \
52  __builtin_unreachable(); \
53  } while (false)
54 
55 #else // not defined(NDEBUG) && BOS(DARWIN)
56 
57 #define BCRASH() do { \
58  *(int*)0xbbadbeef = 0; \
59 } while (0);
60 
61 #endif // defined(NDEBUG) && BOS(DARWIN)
