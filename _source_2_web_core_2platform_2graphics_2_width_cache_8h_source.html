57  template<typename CharacterType> SmallStringKey(CharacterType* characters, unsigned short length)
58  : m_length(length)
59  {
60  ASSERT(length <= s_capacity);
61 
62  StringHasher hasher;
63 
64  bool remainder = length & 1;
65  length >>= 1;
66 
67  unsigned i = 0;
68  while (length--) {
69  m_characters[i] = characters[i];
70  m_characters[i + 1] = characters[i + 1];
71  hasher.addCharactersAssumingAligned(characters[i], characters[i + 1]);
72  i += 2;
73  }
74 
75  if (remainder) {
76  m_characters[i] = characters[i];
77  hasher.addCharacter(characters[i]);
78  }
79 
80  m_hash = hasher.hash();
81  }

83  const UChar* characters() const { return m_characters; }
84  unsigned short length() const { return m_length; }
85  unsigned hash() const { return m_hash; }
86 
87  bool isHashTableDeletedValue() const { return m_length == s_deletedValueLength; }
88  bool isHashTableEmptyValue() const { return m_length == s_emptyValueLength; }
89 
90  private:
91  static const unsigned s_capacity = 15;
92  static const unsigned s_emptyValueLength = s_capacity + 1;
93  static const unsigned s_deletedValueLength = s_capacity + 2;
94 
95  unsigned m_hash;
96  unsigned short m_length;
97  UChar m_characters[s_capacity];

137  float* add(const TextRun& run, float entry, bool hasKerningOrLigatures, bool hasWordSpacingOrLetterSpacing, GlyphOverflow* glyphOverflow)
138  {
139  if (MemoryPressureHandler::singleton().isUnderMemoryPressure())
140  return nullptr;
141  // The width cache is not really profitable unless we're doing expensive glyph transformations.
142  if (!hasKerningOrLigatures)
143  return 0;
144  // Word spacing and letter spacing can change the width of a word.
145  if (hasWordSpacingOrLetterSpacing)
146  return 0;
147  // Since this is just a width cache, we don't have enough information to satisfy glyph queries.
148  if (glyphOverflow)
149  return 0;
150  // If we allow tabs and a tab occurs inside a word, the width of the word varies based on its position on the line.
151  if (run.allowTabs())
152  return 0;

154  return 0;
155 
156  if (m_countdown > 0) {
157  --m_countdown;
158  return 0;

172  float* addSlowCase(StringView text, float entry)
173  {
174  int length = text.length();
175  bool isNewEntry;
176  float* value;
177  if (length == 1) {
178  SingleCharMap::AddResult addResult = m_singleCharMap.fastAdd(text[0], entry);
179  isNewEntry = addResult.isNewEntry;
180  value = &addResult.iterator->value;
181  } else {
182  SmallStringKey smallStringKey;
183  if (text.is8Bit())
184  smallStringKey = SmallStringKey(text.characters8(), length);
185  else
186  smallStringKey = SmallStringKey(text.characters16(), length);
187 
188  Map::AddResult addResult = m_map.fastAdd(smallStringKey, entry);
189  isNewEntry = addResult.isNewEntry;
190  value = &addResult.iterator->value;
191  }
192 
193  // Cache hit: ramp up by sampling the next few words.
194  if (!isNewEntry) {
195  m_interval = s_minInterval;
196  return value;
197  }
198 
199  // Cache miss: ramp down by increasing our sampling interval.
200  if (m_interval < s_maxInterval)
201  ++m_interval;
202  m_countdown = m_interval;
203 
204  if ((m_singleCharMap.size() + m_map.size()) < s_maxSize)
205  return value;
206 
207  // No need to be fancy: we're just trying to avoid pathological growth.
208  m_singleCharMap.clear();
209  m_map.clear();
210  return 0;
211  }
