239 template<typename CSSValueType>
240 inline bool compareCSSValueVector(const Vector<Ref<CSSValueType>>& firstVector, const Vector<Ref<CSSValueType>>& secondVector)
241 {
242  size_t size = firstVector.size();
243  if (size != secondVector.size())
244  return false;
245 
246  for (size_t i = 0; i < size; ++i) {
247  auto& firstPtr = firstVector[i];
248  auto& secondPtr = secondVector[i];
249  if (firstPtr.ptr() == secondPtr.ptr() || firstPtr->equals(secondPtr))
250  continue;
251  return false;
252  }
253  return true;
254 }
