63 class SourceParseModeSet {
64 public:
65  template<typename... Modes>
66  SourceParseModeSet(Modes... args)
67  : m_mask(mergeSourceParseModes(args...))
68  {
69  }
70 
71  ALWAYS_INLINE bool contains(SourceParseMode mode)
72  {
73  return static_cast<unsigned>(mode) & m_mask;
74  }

76 private:
77  ALWAYS_INLINE static unsigned mergeSourceParseModes(SourceParseMode mode)
78  {
79  return static_cast<unsigned>(mode);
80  }
81 
82  template<typename... Rest>
83  ALWAYS_INLINE static unsigned mergeSourceParseModes(SourceParseMode mode, Rest... rest)
84  {
85  return static_cast<unsigned>(mode) | mergeSourceParseModes(rest...);
86  }

88  const unsigned m_mask;

208 inline bool functionNameScopeIsDynamic(bool usesEval, bool isStrictMode)
209 {
210  // If non-strict eval is in play, a function gets a separate object in the scope chain for its name.
211  // This enables eval to declare and then delete a name that shadows the function's name.
212 
213  if (!usesEval)
214  return false;
215 
216  if (isStrictMode)
217  return false;
218 
219  return true;
