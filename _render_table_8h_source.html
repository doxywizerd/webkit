174  unsigned colToEffCol(unsigned column) const
175  {
176  if (!m_hasCellColspanThatDeterminesTableWidth)
177  return column;
178 
179  unsigned effColumn = 0;
180  unsigned numColumns = numEffCols();
181  for (unsigned c = 0; effColumn < numColumns && c + m_columns[effColumn].span - 1 < column; ++effColumn)
182  c += m_columns[effColumn].span;
183  return effColumn;
184  }

186  unsigned effColToCol(unsigned effCol) const

190

193  c += m_columns[i].span;
194  return c;
195  }
