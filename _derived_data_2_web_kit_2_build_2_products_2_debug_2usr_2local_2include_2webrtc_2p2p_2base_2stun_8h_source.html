287  EnsureAddressLength();
288  }

290 
291  virtual bool Read(rtc::ByteBufferReader* buf);
292  virtual bool Write(rtc::ByteBufferWriter* buf) const;
293 
294  private:
295  void EnsureAddressLength() {
296  switch (family()) {
