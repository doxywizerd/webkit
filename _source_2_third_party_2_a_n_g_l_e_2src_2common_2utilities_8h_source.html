65 // [OpenGL ES 3.0.2] Section 2.3.1 page 14
66 // Data Conversion For State-Setting Commands
67 // Floating-point values are rounded to the nearest integer, instead of truncated, as done by static_cast.
68 template <typename outT> outT iround(GLfloat value) { return static_cast<outT>(value > 0.0f ? floor(value + 0.5f) : ceil(value - 0.5f)); }
69 template <typename outT> outT uiround(GLfloat value) { return static_cast<outT>(value + 0.5f); }
