129  inline JSInterfaceJIT::Jump JSInterfaceJIT::emitLoadDouble(unsigned virtualRegisterIndex, FPRegisterID dst, RegisterID scratch)
130  {
131  ASSERT(static_cast<int>(virtualRegisterIndex) < FirstConstantRegisterIndex);
132  loadPtr(tagFor(virtualRegisterIndex), scratch);
133  Jump isDouble = branch32(Below, scratch, TrustedImm32(JSValue::LowestTag));
134  Jump notInt = branch32(NotEqual, scratch, TrustedImm32(JSValue::Int32Tag));
135  loadPtr(payloadFor(virtualRegisterIndex), scratch);
136  convertInt32ToDouble(scratch, dst);
137  Jump done = jump();
138  isDouble.link(this);
139  loadDouble(addressFor(virtualRegisterIndex), dst);
140  done.link(this);
141  return notInt;
142  }

174  inline JSInterfaceJIT::Jump JSInterfaceJIT::emitLoadDouble(unsigned virtualRegisterIndex, FPRegisterID dst, RegisterID scratch)
175  {
176  load64(addressFor(virtualRegisterIndex), scratch);
177  Jump notNumber = emitJumpIfNotNumber(scratch);
178  Jump notInt = branch64(Below, scratch, tagTypeNumberRegister);
179  convertInt32ToDouble(scratch, dst);

181  notInt.link(this);
