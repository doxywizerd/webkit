37 class SynchronizedFixedQueue {
38 public:

44  void open()
45  {
46  LockHolder lockHolder(m_mutex);
47  if (m_open)
48  return;
49 
50  // Restore the queue to its original state.
51  m_open = true;
52  m_queue.clear();
53  }

55  void close()
56  {
57  LockHolder lockHolder(m_mutex);
58  if (!m_open)
59  return;
60 
61  // Wake all the sleeping threads up with a closing state.
62  m_open = false;
63  m_condition.notifyAll();
64  }

66  bool isOpen()
67  {
68  LockHolder lockHolder(m_mutex);
69  return m_open;
70  }

72  bool enqueue(const T& value)
73  {
74  LockHolder lockHolder(m_mutex);
75 
76  // Wait for an empty place to be available in the queue.
77  m_condition.wait(m_mutex, [this]() { return !m_open || m_queue.size() < BufferSize; });
78 
79  // The queue is closing, exit immediately.
80  if (!m_open)
81  return false;
82 
83  // Add the item in the queue.
84  m_queue.append(value);
85 
86  // Notify the other threads that an item was added to the queue.
87  m_condition.notifyAll();
88  return true;
89  }
90 
91  bool dequeue(T& value)
92  {
93  LockHolder lockHolder(m_mutex);
94 
95  // Wait for an item to be added.
96  m_condition.wait(m_mutex, [this]() { return !m_open || m_queue.size(); });
97 
98  // The queue is closing, exit immediately.
99  if (!m_open)
100  return false;
101 
102  // Get a copy from m_queue.first and then remove it.
103  value = m_queue.first();
104  m_queue.removeFirst();
105 
106  // Notify the other threads that an item was removed from the queue.
107  m_condition.notifyAll();
108  return true;
109  }
110 
111 private:

113  Condition m_condition;
