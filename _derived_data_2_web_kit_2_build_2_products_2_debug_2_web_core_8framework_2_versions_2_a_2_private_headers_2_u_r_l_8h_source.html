238 void URL::encode(Encoder& encoder) const
239 {
240  encoder << m_string;
241  encoder << static_cast<bool>(m_isValid);
242  if (!m_isValid)

244  encoder << static_cast<bool>(m_protocolIsInHTTPFamily);
245  encoder << m_schemeEnd;
246  encoder << m_userStart;
247  encoder << m_userEnd;
248  encoder << m_passwordEnd;
249  encoder << m_hostEnd;
250  encoder << m_portEnd;
251  encoder << m_pathAfterLastSlash;
252  encoder << m_pathEnd;
253  encoder << m_queryEnd;
254  encoder << m_fragmentEnd;

257 template <class Decoder>
258 bool URL::decode(Decoder& decoder, URL& url)
259 {
260  if (!decoder.decode(url.m_string))
261  return false;
262  bool isValid;
263  if (!decoder.decode(isValid))
264  return false;
265  url.m_isValid = isValid;
266  if (!isValid)
267  return true;
268  bool protocolIsInHTTPFamily;
269  if (!decoder.decode(protocolIsInHTTPFamily))
270  return false;
271  url.m_protocolIsInHTTPFamily = protocolIsInHTTPFamily;
272  if (!decoder.decode(url.m_schemeEnd))
273  return false;
274  if (!decoder.decode(url.m_userStart))
275  return false;
276  if (!decoder.decode(url.m_userEnd))
277  return false;
278  if (!decoder.decode(url.m_passwordEnd))
279  return false;
280  if (!decoder.decode(url.m_hostEnd))
281  return false;
282  if (!decoder.decode(url.m_portEnd))
283  return false;
284  if (!decoder.decode(url.m_pathAfterLastSlash))
285  return false;
286  if (!decoder.decode(url.m_pathEnd))
287  return false;
288  if (!decoder.decode(url.m_queryEnd))
289  return false;
290  if (!decoder.decode(url.m_fragmentEnd))
291  return false;
292  return true;
293 }
