121 class FakeRTCCertificateGenerator
122  : public rtc::RTCCertificateGeneratorInterface,
123  public rtc::MessageHandler {
124  public:
125  typedef rtc::TypedMessageData<rtc::scoped_refptr<
126  rtc::RTCCertificateGeneratorCallback> > MessageData;

128  FakeRTCCertificateGenerator() : should_fail_(false) {}

130  void set_should_fail(bool should_fail) {
131  should_fail_ = should_fail;
132  }
133 
134  void use_original_key() { key_index_ = 0; }
135  void use_alternate_key() { key_index_ = 1; }
136 
137  int generated_certificates() { return generated_certificates_; }
138 
139  void GenerateCertificateAsync(
140  const rtc::KeyParams& key_params,
141  const rtc::Optional<uint64_t>& expires_ms,
142  const rtc::scoped_refptr<rtc::RTCCertificateGeneratorCallback>&
143  callback) override {
144  // The certificates are created from constant PEM strings and use its coded
145  // expiration time, we do not support modifying it.
146  RTC_DCHECK(!expires_ms);
147  MessageData* msg = new MessageData(
148  rtc::scoped_refptr<rtc::RTCCertificateGeneratorCallback>(callback));
149  uint32_t msg_id;
150  // Only supports RSA-1024-0x10001 and ECDSA-P256.
151  if (should_fail_) {
152  msg_id = MSG_FAILURE;
153  } else if (key_params.type() == rtc::KT_RSA) {
154  RTC_DCHECK(key_params.rsa_params().mod_size == 1024 &&
155  key_params.rsa_params().pub_exp == 0x10001);
156  msg_id = MSG_SUCCESS_RSA;
157  } else {
158  RTC_DCHECK(key_params.type() == rtc::KT_ECDSA &&
159  key_params.ec_curve() == rtc::EC_NIST_P256);
160  msg_id = MSG_SUCCESS_ECDSA;
161  }
162  rtc::Thread::Current()->Post(RTC_FROM_HERE, this, msg_id, msg);
163  }
164 
165  static rtc::scoped_refptr<rtc::RTCCertificate> GenerateCertificate() {
166  switch (rtc::KT_DEFAULT) {
167  case rtc::KT_RSA:
168  return rtc::RTCCertificate::FromPEM(kRsaPems[0]);
169  case rtc::KT_ECDSA:
170  return rtc::RTCCertificate::FromPEM(kEcdsaPems[0]);
171  default:
172  RTC_NOTREACHED();
173  return nullptr;
174  }
175  }
176 
177  private:
178  enum {
179  MSG_SUCCESS_RSA,
180  MSG_SUCCESS_ECDSA,
181  MSG_FAILURE,
182  };
183 
184  const rtc::RTCCertificatePEM& get_pem(const rtc::KeyType& key_type) const {
185  switch (key_type) {
186  case rtc::KT_RSA:
187  return kRsaPems[key_index_];
188  case rtc::KT_ECDSA:
189  return kEcdsaPems[key_index_];
190  default:
191  RTC_NOTREACHED();
192  return kEcdsaPems[key_index_];
193  }
194  }
195  const std::string& get_key(const rtc::KeyType& key_type) const {
196  return get_pem(key_type).private_key();
197  }
198  const std::string& get_cert(const rtc::KeyType& key_type) const {
199  return get_pem(key_type).certificate();
200  }
201 
202  // rtc::MessageHandler implementation.
203  void OnMessage(rtc::Message* msg) override {
204  MessageData* message_data = static_cast<MessageData*>(msg->pdata);
205  rtc::scoped_refptr<rtc::RTCCertificateGeneratorCallback> callback =
206  message_data->data();
207  rtc::scoped_refptr<rtc::RTCCertificate> certificate;
208  switch (msg->message_id) {
209  case MSG_SUCCESS_RSA:
210  case MSG_SUCCESS_ECDSA: {
211  rtc::KeyType key_type =
212  msg->message_id == MSG_SUCCESS_RSA ? rtc::KT_RSA : rtc::KT_ECDSA;
213  certificate = rtc::RTCCertificate::FromPEM(get_pem(key_type));
214  RTC_DCHECK(certificate);
215  ++generated_certificates_;
216  callback->OnSuccess(certificate);
217  break;
218  }
219  case MSG_FAILURE:
220  callback->OnFailure();
221  break;
222  }
223  delete message_data;
224  }
225 
226  bool should_fail_;
227  int key_index_ = 0;
228  int generated_certificates_ = 0;
229 };
