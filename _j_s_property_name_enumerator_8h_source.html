99 inline JSPropertyNameEnumerator* propertyNameEnumerator(ExecState* exec, JSObject* base)
100 {
101  VM& vm = exec->vm();
102  auto scope = DECLARE_THROW_SCOPE(vm);
103 
104  uint32_t indexedLength = base->methodTable(vm)->getEnumerableLength(exec, base);
105 
106  JSPropertyNameEnumerator* enumerator = nullptr;
107 
108  Structure* structure = base->structure(vm);
109  if (!indexedLength
110  && (enumerator = structure->cachedPropertyNameEnumerator())
111  && enumerator->cachedPrototypeChain() == structure->prototypeChain(exec))
112  return enumerator;
113 
114  uint32_t numberStructureProperties = 0;
115 
116  PropertyNameArray propertyNames(exec, PropertyNameMode::Strings);
117 
118  if (structure->canAccessPropertiesQuicklyForEnumeration() && indexedLength == base->getArrayLength()) {
119  base->methodTable(vm)->getStructurePropertyNames(base, exec, propertyNames, EnumerationMode());
120  ASSERT(!scope.exception());
121 
122  numberStructureProperties = propertyNames.size();
123 
124  base->methodTable(vm)->getGenericPropertyNames(base, exec, propertyNames, EnumerationMode());
125  ASSERT(!scope.exception());
126  } else {
127  // Generic property names vector contains all indexed property names.
128  // So disable indexed property enumeration phase by setting |indexedLength| to 0.
129  indexedLength = 0;
130  base->methodTable(vm)->getPropertyNames(base, exec, propertyNames, EnumerationMode());
131  RETURN_IF_EXCEPTION(scope, nullptr);
132  }
133 
134  ASSERT(propertyNames.size() < UINT32_MAX);
135 
136  normalizePrototypeChain(exec, structure);
137 
138  enumerator = JSPropertyNameEnumerator::create(vm, structure, indexedLength, numberStructureProperties, WTFMove(propertyNames));
139  enumerator->setCachedPrototypeChain(vm, structure->prototypeChain(exec));
140  if (!indexedLength && structure->canCachePropertyNameEnumerator())
141  structure->setCachedPropertyNameEnumerator(vm, enumerator);
142  return enumerator;
143 }
