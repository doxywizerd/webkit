38 // Gives you a compressed map between between bytecode indices and machine code
39 // entry points. The compression simply tries to use either 1, 2, or 4 bytes for
40 // any given offset. The largest offset that can be stored is 2^30.
41 
42 // Example use:
43 //
44 // CompactJITCodeMap::Encoder encoder(map);
45 // encoder.append(a, b);
46 // encoder.append(c, d); // preconditions: c >= a, d >= b
47 // auto map = encoder.finish();
48 //
49 // At some later time:
50 //
51 // Vector<BytecodeAndMachineOffset> decoded;
52 // map->decode(decoded);

110  unsigned decodeNumber(unsigned& index) const
111  {
112  uint8_t headValue = at(index++);
113  if (!(headValue & 128))
114  return headValue;
115  if (!(headValue & 64))
116  return (static_cast<unsigned>(headValue & ~128) << 8) | at(index++);
117  unsigned second = at(index++);
118  unsigned third = at(index++);
119  unsigned fourth = at(index++);
120  return (static_cast<unsigned>(headValue & ~(128 + 64)) << 24) | (second << 16) | (third << 8) | fourth;
121  }

200 inline void CompactJITCodeMap::Encoder::append(unsigned bytecodeIndex, unsigned machineCodeOffset)
201 {
202  ASSERT(bytecodeIndex >= m_previousBytecodeIndex);
203  ASSERT(machineCodeOffset >= m_previousMachineCodeOffset);
204  ensureCapacityFor(1);
205  encodeNumber(bytecodeIndex - m_previousBytecodeIndex);
206  encodeNumber(machineCodeOffset - m_previousMachineCodeOffset);
207  m_previousBytecodeIndex = bytecodeIndex;

232 inline void CompactJITCodeMap::Encoder::encodeNumber(uint32_t value)
233 {
234  ASSERT(m_size + 4 <= m_capacity);
235  ASSERT(value < (1 << 30));
236  if (value <= 127) {
237  uint8_t headValue = static_cast<uint8_t>(value);
238  ASSERT(!(headValue & 128));
239  appendByte(headValue);
240  } else if (value <= 16383) {
241  uint8_t headValue = static_cast<uint8_t>(value >> 8);
242  ASSERT(!(headValue & 128));
243  ASSERT(!(headValue & 64));
244  appendByte(headValue | 128);
245  appendByte(static_cast<uint8_t>(value));
246  } else {
247  uint8_t headValue = static_cast<uint8_t>(value >> 24);
248  ASSERT(!(headValue & 128));
249  ASSERT(!(headValue & 64));
250  appendByte(headValue | 128 | 64);
251  appendByte(static_cast<uint8_t>(value >> 16));
252  appendByte(static_cast<uint8_t>(value >> 8));
253  appendByte(static_cast<uint8_t>(value));
254  }
255 }
