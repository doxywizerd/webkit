91 static inline void* performJITMemcpy(void *dst, const void *src, size_t n)
92 {
93  if (reinterpret_cast<uintptr_t>(dst) >= startOfFixedExecutableMemoryPool && reinterpret_cast<uintptr_t>(dst) < endOfFixedExecutableMemoryPool) {
94 #if ENABLE(FAST_JIT_PERMISSIONS)
95  if (useFastPermisionsJITCopy) {
96  os_thread_self_restrict_rwx_to_rw();
97  memcpy(dst, src, n);
98  os_thread_self_restrict_rwx_to_rx();
99  return dst;
100  }
101 #endif
102 
103  if (jitWriteSeparateHeapsFunction) {
104  // Use execute-only write thunk for writes inside the JIT region. This is a variant of
105  // memcpy that takes an offset into the JIT region as its destination (first) parameter.
106  off_t offset = (off_t)((uintptr_t)dst - startOfFixedExecutableMemoryPool);
107  jitWriteSeparateHeapsFunction(offset, src, n);
108  return dst;
109  }
110  }
111 
112  // Use regular memcpy for writes outside the JIT region.
113  return memcpy(dst, src, n);
114 }
