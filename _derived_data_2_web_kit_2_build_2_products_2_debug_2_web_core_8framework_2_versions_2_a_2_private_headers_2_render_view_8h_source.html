112  {
113  if (m_layoutState) {
114  m_layoutState->m_layoutDelta += delta;
115 #if !ASSERT_DISABLED

276  // These functions may only be accessed by LayoutStateMaintainer.
277  bool pushLayoutState(RenderBox& renderer, const LayoutSize& offset, LayoutUnit pageHeight = 0, bool pageHeightChanged = false)
278  {
279  // We push LayoutState even if layoutState is disabled because it stores layoutDelta too.
280  if (!doingFullRepaint() || m_layoutState->isPaginated() || renderer.flowThreadContainingBlock()
281  || m_layoutState->lineGrid() || (renderer.style().lineGrid() != RenderStyle::initialLineGrid() && renderer.isRenderBlockFlow())) {
282  m_layoutState = std::make_unique<LayoutState>(WTFMove(m_layoutState), &renderer, offset, pageHeight, pageHeightChanged);
283  pushLayoutStateForCurrentFlowThread(renderer);
284  return true;
285  }
286  return false;
287  }

401 // Stack-based class to assist with LayoutState push/pop
402 class LayoutStateMaintainer {
403  WTF_MAKE_NONCOPYABLE(LayoutStateMaintainer);
404 public:
405  // Constructor to push now.
406  explicit LayoutStateMaintainer(RenderView& view, RenderBox& root, LayoutSize offset, bool disableState = false, LayoutUnit pageHeight = 0, bool pageHeightChanged = false)
407  : m_view(view)
408  , m_disabled(disableState)
409  {
410  push(root, offset, pageHeight, pageHeightChanged);
411  }
412 
413  // Constructor to maybe push later.
414  explicit LayoutStateMaintainer(RenderView& view)
415  : m_view(view)
416  {
417  }
418 
419  ~LayoutStateMaintainer()
420  {
421  ASSERT(!m_didCallPush || m_didCallPush == m_didCallPop);
422  }
423 
424  void push(RenderBox& root, LayoutSize offset, LayoutUnit pageHeight = 0, bool pageHeightChanged = false)
425  {
426  ASSERT(!m_didCallPush);
427  m_didCallPush = true;
428  // We push state even if disabled, because we still need to store layoutDelta
429  m_didPushLayoutState = m_view.pushLayoutState(root, offset, pageHeight, pageHeightChanged);
430  if (!m_didPushLayoutState)
431  return;
432  if (m_disabled)
433  m_view.disableLayoutState();
434  }
435 
436  void pop()
437  {
438  ASSERT(!m_didCallPop);
439  m_didCallPop = true;
440  if (!m_didCallPush)
441  return;
442  if (!m_didPushLayoutState)
443  return;
444  m_view.popLayoutState();
445  if (m_disabled)
446  m_view.enableLayoutState();
447  }
448 
449  bool didPush() const { return m_didCallPush; }
450 
451 private:
452  RenderView& m_view;
453  bool m_disabled { false };
454  bool m_didCallPush { false };
455  bool m_didCallPop { false };
456  bool m_didPushLayoutState { false };
457 };
