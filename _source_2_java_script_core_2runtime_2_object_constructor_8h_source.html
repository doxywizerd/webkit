95 // Section 6.2.4.4 of the ES6 specification.
96 // https://tc39.github.io/ecma262/#sec-frompropertydescriptor
97 inline JSObject* constructObjectFromPropertyDescriptor(ExecState* exec, const PropertyDescriptor& descriptor)
98 {
99  VM& vm = exec->vm();
100  auto scope = DECLARE_THROW_SCOPE(vm);
101  JSObject* description = constructEmptyObject(exec);
102  RETURN_IF_EXCEPTION(scope, nullptr);
103 
104  if (!descriptor.isAccessorDescriptor()) {
105  description->putDirect(vm, vm.propertyNames->value, descriptor.value() ? descriptor.value() : jsUndefined(), 0);
106  description->putDirect(vm, vm.propertyNames->writable, jsBoolean(descriptor.writable()), 0);
107  } else {
108  ASSERT(descriptor.getter() || descriptor.setter());
109  if (descriptor.getter())
110  description->putDirect(vm, vm.propertyNames->get, descriptor.getter(), 0);
111  if (descriptor.setter())
112  description->putDirect(vm, vm.propertyNames->set, descriptor.setter(), 0);
113  }
114 
115  description->putDirect(vm, vm.propertyNames->enumerable, jsBoolean(descriptor.enumerable()), 0);
116  description->putDirect(vm, vm.propertyNames->configurable, jsBoolean(descriptor.configurable()), 0);
117 
118  return description;
119 }
