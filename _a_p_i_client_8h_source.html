41  // Helper class that can return an std::array of element sizes in a tuple.
42  template<typename> struct InterfaceSizes;
43  template<typename... Interfaces> struct InterfaceSizes<std::tuple<Interfaces...>> {
44  static std::array<size_t, sizeof...(Interfaces)> sizes()
45  {
46  return { { sizeof(Interfaces)... } };

48  };

61  void initialize(const ClientInterface* client)
62  {
63  if (client && client->version == latestClientVersion) {
64  m_client = *reinterpret_cast<const LatestClientInterface*>(client);
65  return;
66  }
67 
68  memset(&m_client, 0, sizeof(m_client));
69 
70  if (client && client->version < latestClientVersion) {
71  auto interfaceSizes = InterfaceSizes<ClientVersions>::sizes();
72 
73  memcpy(&m_client, client, interfaceSizes[client->version]);
74  }
75  }
