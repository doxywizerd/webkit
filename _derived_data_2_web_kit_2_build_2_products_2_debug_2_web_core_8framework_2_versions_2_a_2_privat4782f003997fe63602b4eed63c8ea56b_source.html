316  // Used to track the path down the tree for replica layers.
317  struct ReplicaState {
318  static const size_t maxReplicaDepth = 16;
319  enum ReplicaBranchType { ChildBranch = 0, ReplicaBranch = 1 };
320  ReplicaState(ReplicaBranchType firstBranch)
321  : m_replicaDepth(0)
322  {
323  push(firstBranch);
324  }
325 
326  // Called as we walk down the tree to build replicas.
327  void push(ReplicaBranchType branchType)
328  {
329  m_replicaBranches.append(branchType);
330  if (branchType == ReplicaBranch)
331  ++m_replicaDepth;
332  }
333 
334  void setBranchType(ReplicaBranchType branchType)
335  {
336  ASSERT(!m_replicaBranches.isEmpty());
337 
338  if (m_replicaBranches.last() != branchType) {
339  if (branchType == ReplicaBranch)
340  ++m_replicaDepth;
341  else
342  --m_replicaDepth;
343  }
344 
345  m_replicaBranches.last() = branchType;
346  }
347 
348  void pop()
349  {
350  if (m_replicaBranches.last() == ReplicaBranch)
351  --m_replicaDepth;
352  m_replicaBranches.removeLast();
353  }
354 
355  size_t depth() const { return m_replicaBranches.size(); }
356  size_t replicaDepth() const { return m_replicaDepth; }
357 
358  CloneID cloneID() const;
359 
360  private:
361  Vector<ReplicaBranchType> m_replicaBranches;
362  size_t m_replicaDepth;
363  };
