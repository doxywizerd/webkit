27 // Abstract base class for RTCStats-derived dictionaries, see
28 // https://w3c.github.io/webrtc-stats/.
29 //
30 // All derived classes must have the following static variable defined:
31 // static const char kType[];
32 // It is used as a unique class identifier and a string representation of the
33 // class type, see https://w3c.github.io/webrtc-stats/#rtcstatstype-str*.
34 // Use the |WEBRTC_RTCSTATS_IMPL| macro when implementing subclasses, see macro
35 // for details.
36 //
37 // Derived classes list their dictionary members, RTCStatsMember<T>, as public
38 // fields, allowing the following:
39 //
40 // RTCFooStats foo("fooId", GetCurrentTime());
41 // foo.bar = 42;
42 // foo.baz = std::vector<std::string>();
43 // foo.baz->push_back("hello world");
44 // uint32_t x = *foo.bar;
45 //
46 // Pointers to all the members are available with |Members|, allowing iteration:
47 //
48 // for (const RTCStatsMemberInterface* member : foo.Members()) {
49 // printf("%s = %s\n", member->name(), member->ValueToString().c_str());
50 // }
51 class RTCStats {
52  public:
53  RTCStats(const std::string& id, int64_t timestamp_us)
54  : id_(id), timestamp_us_(timestamp_us) {}
55  RTCStats(std::string&& id, int64_t timestamp_us)
56  : id_(std::move(id)), timestamp_us_(timestamp_us) {}
57  virtual ~RTCStats() {}
58 
59  virtual std::unique_ptr<RTCStats> copy() const = 0;
60 
61  const std::string& id() const { return id_; }
62  // Time relative to the UNIX epoch (Jan 1, 1970, UTC), in microseconds.
63  int64_t timestamp_us() const { return timestamp_us_; }
64  // Returns the static member variable |kType| of the implementing class.
65  virtual const char* type() const = 0;
66  // Returns a vector of pointers to all the |RTCStatsMemberInterface| members
67  // of this class. This allows for iteration of members. For a given class,
68  // |Members| always returns the same members in the same order.
69  std::vector<const RTCStatsMemberInterface*> Members() const;
70  // Checks if the two stats objects are of the same type and have the same
71  // member values. Timestamps are not compared. These operators are exposed for
72  // testing.
73  bool operator==(const RTCStats& other) const;
74  bool operator!=(const RTCStats& other) const;
75 
76  // Creates a human readable string representation of the stats object, listing
77  // all of its members (names and values).
78  std::string ToString() const;
79 
80  // Downcasts the stats object to an |RTCStats| subclass |T|. DCHECKs that the
81  // object is of type |T|.
82  template<typename T>
83  const T& cast_to() const {
84  RTC_DCHECK_EQ(type(), T::kType);
85  return static_cast<const T&>(*this);
86  }
87 
88  protected:
89  // Gets a vector of all members of this |RTCStats| object, including members
90  // derived from parent classes. |additional_capacity| is how many more members
91  // shall be reserved in the vector (so that subclasses can allocate a vector
92  // with room for both parent and child members without it having to resize).
93  virtual std::vector<const RTCStatsMemberInterface*>
94  MembersOfThisObjectAndAncestors(
95  size_t additional_capacity) const;
96 
97  std::string const id_;
98  int64_t timestamp_us_;
99 };

101 // All |RTCStats| classes should use these macros.
102 // |WEBRTC_RTCSTATS_DECL| is placed in a public section of the class definition.
103 // |WEBRTC_RTCSTATS_IMPL| is placed outside the class definition (in a .cc).
104 //
105 // These macros declare (in _DECL) and define (in _IMPL) the static |kType| and
106 // overrides methods as required by subclasses of |RTCStats|: |copy|, |type| and
107 // |MembersOfThisObjectAndAncestors|. The |...| argument is a list of addresses
108 // to each member defined in the implementing class. The list must have at least
109 // one member.
110 //
111 // (Since class names need to be known to implement these methods this cannot be
112 // part of the base |RTCStats|. While these methods could be implemented using
113 // templates, that would only work for immediate subclasses. Subclasses of
114 // subclasses also have to override these methods, resulting in boilerplate
115 // code. Using a macro avoids this and works for any |RTCStats| class, including
116 // grandchildren.)
117 //
118 // Sample usage:
119 //
120 // rtcfoostats.h:
121 // class RTCFooStats : public RTCStats {
122 // public:
123 // WEBRTC_RTCSTATS_DECL();
124 //
125 // RTCFooStats(const std::string& id, int64_t timestamp_us);
126 //
127 // RTCStatsMember<int32_t> foo;
128 // RTCStatsMember<int32_t> bar;
129 // };
130 //
131 // rtcfoostats.cc:
132 // WEBRTC_RTCSTATS_IMPL(RTCFooStats, RTCStats, "foo-stats"
133 // &foo,
134 // &bar);
135 //
136 // RTCFooStats::RTCFooStats(const std::string& id, int64_t timestamp_us)
137 // : RTCStats(id, timestamp_us),
138 // foo("foo"),
139 // bar("bar") {
140 // }
141 //
142 #define WEBRTC_RTCSTATS_DECL() \
143  public: \
144  WEBRTC_DYLIB_EXPORT static const char kType[]; \
145  \
146  std::unique_ptr<webrtc::RTCStats> copy() const override; \
147  const char* type() const override; \
148  \
149  protected: \
150  std::vector<const webrtc::RTCStatsMemberInterface*> \
151  MembersOfThisObjectAndAncestors( \
152  size_t local_var_additional_capacity) const override; \
153  \
154  public:
155 
156 #define WEBRTC_RTCSTATS_IMPL(this_class, parent_class, type_str, ...) \
157  const char this_class::kType[] = type_str; \
158  \
159  std::unique_ptr<webrtc::RTCStats> this_class::copy() const { \
160  return std::unique_ptr<webrtc::RTCStats>(new this_class(*this)); \
161  } \
162  \
163  const char* this_class::type() const { \
164  return this_class::kType; \
165  } \
166  \

168  this_class::MembersOfThisObjectAndAncestors( \
169  size_t local_var_additional_capacity) const { \
170  const webrtc::RTCStatsMemberInterface* local_var_members[] = { \
171  __VA_ARGS__ \
172  }; \
173  size_t local_var_members_count = \
174  sizeof(local_var_members) / sizeof(local_var_members[0]); \
175  std::vector<const webrtc::RTCStatsMemberInterface*> local_var_members_vec =\
176  parent_class::MembersOfThisObjectAndAncestors( \
177  local_var_members_count + local_var_additional_capacity); \
178  RTC_DCHECK_GE( \
179  local_var_members_vec.capacity() - local_var_members_vec.size(), \
180  local_var_members_count + local_var_additional_capacity); \
181  local_var_members_vec.insert(local_var_members_vec.end(), \
182  &local_var_members[0], \
183  &local_var_members[local_var_members_count]); \
184  return local_var_members_vec; \
185  }

187 // Interface for |RTCStats| members, which have a name and a value of a type
188 // defined in a subclass. Only the types listed in |Type| are supported, these

241 // Template implementation of |RTCStatsMemberInterface|. Every possible |T| is
242 // specialized in rtcstats.cc, using a different |T| results in a linker error
243 // (undefined reference to |kType|). The supported types are the ones described
244 // by |RTCStatsMemberInterface::Type|.
245 template<typename T>
246 class RTCStatsMember : public RTCStatsMemberInterface {
247  public:
248  static const Type kType;
249 
250  explicit RTCStatsMember(const char* name)
251  : RTCStatsMemberInterface(name, false),
252  value_() {}
253  RTCStatsMember(const char* name, const T& value)
254  : RTCStatsMemberInterface(name, true),
255  value_(value) {}
256  RTCStatsMember(const char* name, T&& value)
257  : RTCStatsMemberInterface(name, true),
258  value_(std::move(value)) {}
259  explicit RTCStatsMember(const RTCStatsMember<T>& other)
260  : RTCStatsMemberInterface(other.name_, other.is_defined_),
261  value_(other.value_) {}
262  explicit RTCStatsMember(RTCStatsMember<T>&& other)
263  : RTCStatsMemberInterface(other.name_, other.is_defined_),
264  value_(std::move(other.value_)) {}
265 
266  Type type() const override { return kType; }
267  bool is_sequence() const override;
268  bool is_string() const override;
269  bool operator==(const RTCStatsMemberInterface& other) const override {
270  if (type() != other.type())
271  return false;
272  const RTCStatsMember<T>& other_t =
273  static_cast<const RTCStatsMember<T>&>(other);
274  if (!is_defined_)
275  return !other_t.is_defined();
276  if (!other.is_defined())
277  return false;
278  return value_ == other_t.value_;
279  }
280  std::string ValueToString() const override;
281 
282  // Assignment operators.
283  T& operator=(const T& value) {
284  value_ = value;
285  is_defined_ = true;
286  return value_;
287  }
288  T& operator=(const T&& value) {
289  value_ = std::move(value);
290  is_defined_ = true;
291  return value_;
292  }
293  T& operator=(const RTCStatsMember<T>& other) {
294  RTC_DCHECK(other.is_defined_);
295  value_ = other.is_defined_;
296  is_defined_ = true;
297  return value_;
298  }
299 
300  // Value getters.
301  T& operator*() {
302  RTC_DCHECK(is_defined_);
303  return value_;
304  }
305  const T& operator*() const {
306  RTC_DCHECK(is_defined_);
307  return value_;
308  }
309 
310  // Value getters, arrow operator.
311  T* operator->() {
312  RTC_DCHECK(is_defined_);
313  return &value_;
314  }
315  const T* operator->() const {
316  RTC_DCHECK(is_defined_);
317  return &value_;
318  }
319 
320  private:
321  T value_;
322 };
