35 ALWAYS_INLINE JSArray* tryCreateUninitializedRegExpMatchesArray(VM& vm, GCDeferralContext* deferralContext, Structure* structure, unsigned initialLength)
36 {
37  unsigned vectorLength = initialLength;
38  if (vectorLength > MAX_STORAGE_VECTOR_LENGTH)
39  return 0;
40 
41  void* temp = vm.auxiliarySpace.tryAllocate(deferralContext, Butterfly::totalSize(0, structure->outOfLineCapacity(), true, vectorLength * sizeof(EncodedJSValue)));
42  if (!temp)
43  return nullptr;
44  Butterfly* butterfly = Butterfly::fromBase(temp, 0, structure->outOfLineCapacity());
45  butterfly->setVectorLength(vectorLength);
46  butterfly->setPublicLength(initialLength);
47 
48  for (unsigned i = initialLength; i < vectorLength; ++i)
49  butterfly->contiguous()[i].clear();
50 
51  return JSArray::createWithButterfly(vm, deferralContext, structure, butterfly);
52 }

54 ALWAYS_INLINE JSArray* createRegExpMatchesArray(
55  VM& vm, JSGlobalObject* globalObject, JSString* input, const String& inputValue,
56  RegExp* regExp, unsigned startOffset, MatchResult& result)
57 {
58  Vector<int, 32> subpatternResults;
59  int position = regExp->matchInline(vm, inputValue, startOffset, subpatternResults);
60  if (position == -1) {
61  result = MatchResult::failed();
62  return nullptr;
63  }
64 
65  result.start = position;
66  result.end = subpatternResults[1];
67 
68  JSArray* array;
69 
70  // FIXME: This should handle array allocation errors gracefully.
71  // https://bugs.webkit.org/show_bug.cgi?id=155144
72 
73  auto setProperties = [&] () {
74  array->putDirect(vm, RegExpMatchesArrayIndexPropertyOffset, jsNumber(result.start));
75  array->putDirect(vm, RegExpMatchesArrayInputPropertyOffset, input);
76  };
77 
78  unsigned numSubpatterns = regExp->numSubpatterns();
79 
80  GCDeferralContext deferralContext(vm.heap);
81 
82  if (UNLIKELY(globalObject->isHavingABadTime())) {
83  array = JSArray::tryCreateForInitializationPrivate(vm, &deferralContext, globalObject->regExpMatchesArrayStructure(), numSubpatterns + 1);
84 
85  setProperties();
86 
87  array->initializeIndexWithoutBarrier(0, jsSubstringOfResolved(vm, &deferralContext, input, result.start, result.end - result.start));
88 
89  for (unsigned i = 1; i <= numSubpatterns; ++i) {
90  int start = subpatternResults[2 * i];
91  JSValue value;
92  if (start >= 0)
93  value = JSRopeString::createSubstringOfResolved(vm, &deferralContext, input, start, subpatternResults[2 * i + 1] - start);
94  else
95  value = jsUndefined();
96  array->initializeIndexWithoutBarrier(i, value);
97  }
98  } else {
99  array = tryCreateUninitializedRegExpMatchesArray(vm, &deferralContext, globalObject->regExpMatchesArrayStructure(), numSubpatterns + 1);
100  RELEASE_ASSERT(array);
101 
102  setProperties();
103 
104  // Now the object is safe to scan by GC.
105 
106  array->initializeIndexWithoutBarrier(0, jsSubstringOfResolved(vm, &deferralContext, input, result.start, result.end - result.start), ArrayWithContiguous);
107 
108  for (unsigned i = 1; i <= numSubpatterns; ++i) {
109  int start = subpatternResults[2 * i];
110  JSValue value;
111  if (start >= 0)
112  value = JSRopeString::createSubstringOfResolved(vm, &deferralContext, input, start, subpatternResults[2 * i + 1] - start);
113  else
114  value = jsUndefined();
115  array->initializeIndexWithoutBarrier(i, value, ArrayWithContiguous);
116  }
117  }
118  return array;
119 }
