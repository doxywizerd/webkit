380 template<typename T> bool Connection::sendSync(T&& message, typename T::Reply&& reply, uint64_t destinationID, Seconds timeout, OptionSet<SendSyncOption> sendSyncOptions)
381 {
382  COMPILE_ASSERT(T::isSync, SyncMessageExpected);
383 
384  uint64_t syncRequestID = 0;
385  std::unique_ptr<Encoder> encoder = createSyncMessageEncoder(T::receiverName(), T::name(), destinationID, syncRequestID);
386 
387  if (sendSyncOptions.contains(SendSyncOption::UseFullySynchronousModeForTesting)) {
388  encoder->setFullySynchronousModeForTesting();
389  m_fullySynchronousModeIsAllowedForTesting = true;
390  }
391 
392  // Encode the rest of the input arguments.
393  encoder->encode(message.arguments());
394 
395  // Now send the message and wait for a reply.
396  std::unique_ptr<Decoder> replyDecoder = sendSyncMessage(syncRequestID, WTFMove(encoder), timeout, sendSyncOptions);
397  if (!replyDecoder)
398  return false;
399 
400  // Decode the reply.
401  return replyDecoder->decode(reply);
402 }
