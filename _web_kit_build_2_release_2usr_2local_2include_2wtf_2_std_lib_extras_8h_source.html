213 template<typename ArrayElementType, typename KeyType, typename ArrayType, typename ExtractKey, BinarySearchMode mode>
214 inline ArrayElementType* binarySearchImpl(ArrayType& array, size_t size, KeyType key, const ExtractKey& extractKey = ExtractKey())
215 {
216  size_t offset = 0;
217  while (size > 1) {
218  size_t pos = (size - 1) >> 1;
219  KeyType val = extractKey(&array[offset + pos]);
220 
221  if (val == key)
222  return &array[offset + pos];
223  // The item we are looking for is smaller than the item being check; reduce the value of 'size',
224  // chopping off the right hand half of the array.
225  if (key < val)
226  size = pos;
227  // Discard all values in the left hand half of the array, up to and including the item at pos.
228  else {
229  size -= (pos + 1);
230  offset += (pos + 1);
231  }
232 
233  ASSERT(mode != KeyMustBePresentInArray || size);
234  }
235 
236  if (mode == KeyMightNotBePresentInArray && !size)
237  return 0;
238 
239  ArrayElementType* result = &array[offset];
240 
241  if (mode == KeyMightNotBePresentInArray && key != extractKey(result))
242  return 0;
243 
244  if (mode == KeyMustBePresentInArray) {
245  ASSERT(size == 1);
246  ASSERT(key == extractKey(result));
247  }
248 
249  return result;
250 }
