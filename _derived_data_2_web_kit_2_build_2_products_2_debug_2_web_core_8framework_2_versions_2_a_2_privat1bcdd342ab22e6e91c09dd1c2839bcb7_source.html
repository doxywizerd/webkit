109 inline JSC::JSObject* JSEventListener::jsFunction(ScriptExecutionContext* scriptExecutionContext) const
110 {
111  // initializeJSFunction can trigger code that deletes this event listener
112  // before we're done. It should always return 0 in this case.
113  auto protect = makeRef(const_cast<JSEventListener&>(*this));
114  JSC::Strong<JSC::JSObject> wrapper(m_isolatedWorld->vm(), m_wrapper.get());
115 
116  if (!m_jsFunction) {
117  JSC::JSObject* function = initializeJSFunction(scriptExecutionContext);
118  JSC::JSObject* wrapper = m_wrapper.get();
119  if (wrapper)
120  JSC::Heap::heap(wrapper)->writeBarrier(wrapper, function);
121  m_jsFunction = JSC::Weak<JSC::JSObject>(function);
122  }
123 
124  // Verify that we have a valid wrapper protecting our function from
125  // garbage collection. That is except for when we're not in the normal
126  // world and can have zombie m_jsFunctions.
127  ASSERT(!m_isolatedWorld->isNormal() || m_wrapper || !m_jsFunction);
128 
129  // If m_wrapper is 0, then m_jsFunction is zombied, and should never be accessed.
130  if (!m_wrapper)
131  return 0;
132 
133  // Try to verify that m_jsFunction wasn't recycled. (Not exact, since an
134  // event listener can be almost anything, but this makes test-writing easier).
135  ASSERT(!m_jsFunction || static_cast<JSC::JSCell*>(m_jsFunction.get())->isObject());
136 
137  return m_jsFunction.get();
138 }
