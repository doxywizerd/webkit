22 // InputType/OutputType can be any class that implements the capturing/rendering
23 // part of the AudioDeviceGeneric API.
24 // Construction and destruction must be done on one and the same thread. Each
25 // internal implementation of InputType and OutputType will RTC_DCHECK if that
26 // is not the case. All implemented methods must also be called on the same
27 // thread. See comments in each InputType/OutputType class for more info.
28 // It is possible to call the two static methods (SetAndroidAudioDeviceObjects
29 // and ClearAndroidAudioDeviceObjects) from a different thread but both will
30 // RTC_CHECK that the calling thread is attached to a Java VM.
31

56  InitStatus Init() override {
57  LOG(INFO) << __FUNCTION__;
58  RTC_DCHECK(thread_checker_.CalledOnValidThread());
59  RTC_DCHECK(!initialized_);
60  if (!audio_manager_->Init()) {
61  return InitStatus::OTHER_ERROR;
62  }
63  if (output_.Init() != 0) {
64  audio_manager_->Close();
65  return InitStatus::PLAYOUT_ERROR;
66  }
67  if (input_.Init() != 0) {
68  output_.Terminate();
69  audio_manager_->Close();
70  return InitStatus::RECORDING_ERROR;
71  }
72  initialized_ = true;
73  return InitStatus::OK;
74  }

76  int32_t Terminate() override {
77  LOG(INFO) << __FUNCTION__;
78  RTC_DCHECK(thread_checker_.CalledOnValidThread());
79  int32_t err = input_.Terminate();
80  err |= output_.Terminate();
81  err |= !audio_manager_->Close();
82  initialized_ = false;
83  RTC_DCHECK_EQ(err, 0);
84  return err;
85  }

485  // Returns true if the device both supports built in AEC and the device
486  // is not blacklisted.
487  // Currently, if OpenSL ES is used in both directions, this method will still
488  // report the correct value and it has the correct effect. As an example:
489  // a device supports built in AEC and this method returns true. Libjingle
490  // will then disable the WebRTC based AEC and that will work for all devices
491  // (mainly Nexus) even when OpenSL ES is used for input since our current
492  // implementation will enable built-in AEC by default also for OpenSL ES.
493  // The only "bad" thing that happens today is that when Libjingle calls
494  // OpenSLESRecorder::EnableBuiltInAEC() it will not have any real effect and
495  // a "Not Implemented" log will be filed. This non-perfect state will remain
496  // until I have added full support for audio effects based on OpenSL ES APIs.
497  bool BuiltInAECIsAvailable() const override {
498  LOG(INFO) << __FUNCTION__;
499  return audio_manager_->IsAcousticEchoCancelerSupported();
500  }
