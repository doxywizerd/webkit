108 struct HashTable {
109  int numberOfValues;
110  int indexMask;
111  bool hasSetterOrReadonlyProperties;
112 
113  const HashTableValue* values; // Fixed values generated by script.
114  const CompactHashIndex* index;

116  // Find an entry in the table, and return the entry.
117  ALWAYS_INLINE const HashTableValue* entry(PropertyName propertyName) const
118  {
119  if (propertyName.isSymbol())
120  return nullptr;
121 
122  auto uid = propertyName.uid();
123  if (!uid)
124  return nullptr;
125 
126  int indexEntry = IdentifierRepHash::hash(uid) & indexMask;
127  int valueIndex = index[indexEntry].value;
128  if (valueIndex == -1)
129  return nullptr;
130 
131  while (true) {
132  if (WTF::equal(uid, values[valueIndex].m_key))
133  return &values[valueIndex];
134 
135  indexEntry = index[indexEntry].next;
136  if (indexEntry == -1)
137  return nullptr;
138  valueIndex = index[indexEntry].value;
139  ASSERT(valueIndex != -1);
140  };
141  }

223 inline bool getStaticPropertySlotFromTable(VM& vm, const HashTable& table, JSObject* thisObject, PropertyName propertyName, PropertySlot& slot)
224 {
225  if (thisObject->staticPropertiesReified())
226  return false;
227 
228  auto* entry = table.entry(propertyName);
229  if (!entry)
230  return false;
231 
232  if (entry->attributes() & BuiltinOrFunctionOrAccessorOrLazyProperty)
233  return setUpStaticFunctionSlot(vm, entry, thisObject, propertyName, slot);
234 
235  if (entry->attributes() & ConstantInteger) {
236  slot.setValue(thisObject, attributesForStructure(entry->attributes()), jsNumber(entry->constantInteger()));
237  return true;
238  }
239 
240  if (entry->attributes() & DOMJITAttribute) {
241  DOMJIT::GetterSetter* domJIT = entry->domJIT();
242  slot.setCacheableCustom(thisObject, attributesForStructure(entry->attributes()), domJIT->getter(), domJIT);
243  return true;
244  }
245 
246  slot.setCacheableCustom(thisObject, attributesForStructure(entry->attributes()), entry->propertyGetter());
247  return true;
248 }

261 // 'base' means the object holding the property (possibly in the prototype chain of the object put was called on).
262 // 'thisValue' is the object that put is being applied to (in the case of a proxy, the proxy target).
263 // 'slot.thisValue()' is the object the put was originally performed on (in the case of a proxy, the proxy itself).
264 inline bool putEntry(ExecState* exec, const HashTableValue* entry, JSObject* base, JSObject* thisValue, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
265 {
266  VM& vm = exec->vm();
267  auto scope = DECLARE_THROW_SCOPE(vm);
268 
269  if (entry->attributes() & BuiltinOrFunctionOrLazyProperty) {
270  if (!(entry->attributes() & ReadOnly)) {
271  // If this is a function or lazy property put then we just do the put because
272  // logically the object already had the property, so this is just a replace.
273  if (JSObject* thisObject = jsDynamicCast<JSObject*>(vm, thisValue))
274  thisObject->putDirect(vm, propertyName, value);
275  return true;
276  }
277  return typeError(exec, scope, slot.isStrictMode(), ASCIILiteral(ReadonlyPropertyWriteError));
278  }
279 
280  if (entry->attributes() & Accessor)
281  return typeError(exec, scope, slot.isStrictMode(), ASCIILiteral(ReadonlyPropertyWriteError));
282 
283  if (!(entry->attributes() & ReadOnly)) {
284  ASSERT_WITH_MESSAGE(!(entry->attributes() & DOMJITAttribute), "DOMJITAttribute supports readonly attributes currently.");
285  bool isAccessor = entry->attributes() & CustomAccessor;
286  JSValue updateThisValue = entry->attributes() & CustomAccessor ? slot.thisValue() : JSValue(base);
287  bool result = callCustomSetter(exec, entry->propertyPutter(), isAccessor, updateThisValue, value);
288  if (isAccessor)
289  slot.setCustomAccessor(base, entry->propertyPutter());
290  else
291  slot.setCustomValue(base, entry->propertyPutter());
292  return result;
293  }
294 
295  return typeError(exec, scope, slot.isStrictMode(), ASCIILiteral(ReadonlyPropertyWriteError));
296 }

314 inline void reifyStaticProperty(VM& vm, const PropertyName& propertyName, const HashTableValue& value, JSObject& thisObj)
315 {
316  if (value.attributes() & Builtin) {
317  if (value.attributes() & Accessor)
318  reifyStaticAccessor(vm, value, thisObj, propertyName);
319  else
320  thisObj.putDirectBuiltinFunction(vm, thisObj.globalObject(), propertyName, value.builtinGenerator()(vm), attributesForStructure(value.attributes()));
321  return;
322  }
323 
324  if (value.attributes() & Function) {
325  if (value.attributes() & DOMJITFunction) {
326  thisObj.putDirectNativeFunction(
327  vm, thisObj.globalObject(), propertyName, value.functionLength(),
328  value.function(), value.intrinsic(), value.signature(), attributesForStructure(value.attributes()));
329  return;
330  }
331  thisObj.putDirectNativeFunction(
332  vm, thisObj.globalObject(), propertyName, value.functionLength(),
333  value.function(), value.intrinsic(), attributesForStructure(value.attributes()));
334  return;
335  }
336 
337  if (value.attributes() & ConstantInteger) {
338  thisObj.putDirect(vm, propertyName, jsNumber(value.constantInteger()), attributesForStructure(value.attributes()));
339  return;
340  }
341 
342  if (value.attributes() & Accessor) {
343  reifyStaticAccessor(vm, value, thisObj, propertyName);
344  return;
345  }
346 
347  if (value.attributes() & CellProperty) {
348  LazyCellProperty* property = bitwise_cast<LazyCellProperty*>(
349  bitwise_cast<char*>(&thisObj) + value.lazyCellPropertyOffset());
350  JSCell* result = property->get(&thisObj);
351  thisObj.putDirect(vm, propertyName, result, attributesForStructure(value.attributes()));
352  return;
353  }
354 
355  if (value.attributes() & ClassStructure) {
356  LazyClassStructure* structure = bitwise_cast<LazyClassStructure*>(
357  bitwise_cast<char*>(&thisObj) + value.lazyClassStructureOffset());
358  structure->get(jsCast<JSGlobalObject*>(&thisObj));
359  return;
360  }
361 
362  if (value.attributes() & PropertyCallback) {
363  JSValue result = value.lazyPropertyCallback()(vm, &thisObj);
364  thisObj.putDirect(vm, propertyName, result, attributesForStructure(value.attributes()));
365  return;
366  }
367 
368  if (value.attributes() & DOMJITAttribute) {
369  DOMJIT::GetterSetter* domJIT = value.domJIT();
370  CustomGetterSetter* customGetterSetter = CustomGetterSetter::create(vm, domJIT->getter(), domJIT->setter(), domJIT);
371  thisObj.putDirectCustomAccessor(vm, propertyName, customGetterSetter, attributesForStructure(value.attributes()));
372  return;
373  }
374 
375  CustomGetterSetter* customGetterSetter = CustomGetterSetter::create(vm, value.propertyGetter(), value.propertyPutter());
376  thisObj.putDirectCustomAccessor(vm, propertyName, customGetterSetter, attributesForStructure(value.attributes()));
377 }
