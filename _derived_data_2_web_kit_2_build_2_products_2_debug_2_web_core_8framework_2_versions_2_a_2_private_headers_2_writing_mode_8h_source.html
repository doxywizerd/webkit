149 inline PhysicalBoxSide mapLogicalSideToPhysicalSide(TextFlow textflow, LogicalBoxSide logicalSide)
150 {
151  PhysicalBoxSide physicalSide = static_cast<PhysicalBoxSide>(logicalSide);
152  bool horizontalSide = isHorizontalPhysicalSide(physicalSide);
153 
154  if (isVerticalTextFlow(textflow))
155  physicalSide = rotatePhysicalSide(physicalSide);
156 
157  if ((horizontalSide && isReversedTextFlow(textflow)) || (!horizontalSide && isFlippedTextFlow(textflow)))
158  physicalSide = mirrorPhysicalSide(physicalSide);
159 
160  return physicalSide;
161 }
