456  // The high bits of 'hash' are always empty, but we prefer to store our flags
457  // in the low bits because it makes them slightly more efficient to access.
458  // So, we shift left and right when setting and getting our hash code.
459  void setHash(unsigned hash) const
460  {
461  ASSERT(!hasHash());
462  // Multiple clients assume that StringHasher is the canonical string hash function.
463  ASSERT(hash == (is8Bit() ? StringHasher::computeHashAndMaskTop8Bits(m_data8, m_length) : StringHasher::computeHashAndMaskTop8Bits(m_data16, m_length)));
464  ASSERT(!(hash & (s_flagMask << (8 * sizeof(hash) - s_flagCount)))); // Verify that enough high bits are empty.
465 
466  hash <<= s_flagCount;
467  ASSERT(!(hash & m_hashAndFlags)); // Verify that enough low bits are empty after shift.
468  ASSERT(hash); // Verify that 0 is a valid sentinel hash value.
469 
470  m_hashAndFlags |= hash; // Store hash with flags in low bits.
471  }
