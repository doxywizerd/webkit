160  // Two conditions are compatible if they are identical or if they speak of different uids or
161  // different objects. If false is returned, you have to decide how to resolve the conflict -
162  // for example if there is a Presence and an Equivalence then in some cases you'll want the
163  // more general of the two while in other cases you'll want the more specific of the two. This
164  // will also return false for contradictions, like Presence and Absence on the same
165  // object/uid. By convention, invalid conditions aren't compatible with anything.
166  bool isCompatibleWith(const ObjectPropertyCondition& other) const
167  {
168  if (!*this || !other)
169  return false;
170  return *this == other || uid() != other.uid() || object() != other.object();
171  }
