289  // Checks that this kind is valid for use with B3::Value.
290  ALWAYS_INLINE static void checkKind(Kind kind, unsigned numArgs)
291  {
292  switch (kind.opcode()) {
293  case FramePointer:
294  case Nop:
295  case Phi:
296  case Jump:
297  case Oops:
298  case EntrySwitch:
299  if (UNLIKELY(numArgs))
300  badKind(kind, numArgs);
301  break;
302  case Return:
303  if (UNLIKELY(numArgs > 1))
304  badKind(kind, numArgs);
305  break;
306  case Identity:
307  case Neg:
308  case Clz:
309  case Abs:
310  case Ceil:
311  case Floor:
312  case Sqrt:
313  case SExt8:
314  case SExt16:
315  case Trunc:
316  case SExt32:
317  case ZExt32:
318  case FloatToDouble:
319  case IToD:
320  case DoubleToFloat:
321  case IToF:
322  case BitwiseCast:
323  case Branch:
324  if (UNLIKELY(numArgs != 1))
325  badKind(kind, numArgs);
326  break;
327  case Add:
328  case Sub:
329  case Mul:
330  case Div:
331  case UDiv:
332  case Mod:
333  case UMod:
334  case BitAnd:
335  case BitOr:
336  case BitXor:
337  case Shl:
338  case SShr:
339  case ZShr:
340  case RotR:
341  case RotL:
342  case Equal:
343  case NotEqual:
344  case LessThan:
345  case GreaterThan:
346  case LessEqual:
347  case GreaterEqual:
348  case Above:
349  case Below:
350  case AboveEqual:
351  case BelowEqual:
352  case EqualOrUnordered:
353  if (UNLIKELY(numArgs != 2))
354  badKind(kind, numArgs);
355  break;
356  case Select:
357  if (UNLIKELY(numArgs != 3))
358  badKind(kind, numArgs);
359  break;
360  default:
361  badKind(kind, numArgs);
362  break;
363  }
364  }
