33 inline double FastPath(double significand, int exp) {
34  if (exp < -308)
35  return 0.0;
36  else if (exp >= 0)
37  return significand * internal::Pow10(exp);
38  else
39  return significand / internal::Pow10(-exp);
40 }

61 inline int CheckWithinHalfULP(double b, const BigInteger& d, int dExp, bool* adjustToNegative) {
62  const Double db(b);
63  const uint64_t bInt = db.IntegerSignificand();
64  const int bExp = db.IntegerExponent();
65  const int hExp = bExp - 1;
66 
67  int dS_Exp2 = 0, dS_Exp5 = 0, bS_Exp2 = 0, bS_Exp5 = 0, hS_Exp2 = 0, hS_Exp5 = 0;
68 
69  // Adjust for decimal exponent
70  if (dExp >= 0) {
71  dS_Exp2 += dExp;
72  dS_Exp5 += dExp;
73  }
74  else {
75  bS_Exp2 -= dExp;
76  bS_Exp5 -= dExp;
77  hS_Exp2 -= dExp;
78  hS_Exp5 -= dExp;
79  }
80 
81  // Adjust for binary exponent
82  if (bExp >= 0)
83  bS_Exp2 += bExp;
84  else {
85  dS_Exp2 -= bExp;
86  hS_Exp2 -= bExp;
87  }
88 
89  // Adjust for half ulp exponent
90  if (hExp >= 0)
91  hS_Exp2 += hExp;
92  else {
93  dS_Exp2 -= hExp;
94  bS_Exp2 -= hExp;
95  }
96 
97  // Remove common power of two factor from all three scaled values
98  int common_Exp2 = Min3(dS_Exp2, bS_Exp2, hS_Exp2);
99  dS_Exp2 -= common_Exp2;
100  bS_Exp2 -= common_Exp2;
101  hS_Exp2 -= common_Exp2;
102 
103  BigInteger dS = d;
104  dS.MultiplyPow5(dS_Exp5) <<= dS_Exp2;
105 
106  BigInteger bS(bInt);
107  bS.MultiplyPow5(bS_Exp5) <<= bS_Exp2;
108 
109  BigInteger hS(1);
110  hS.MultiplyPow5(hS_Exp5) <<= hS_Exp2;
111 
112  BigInteger delta(0);
113  *adjustToNegative = dS.Difference(bS, &delta);
114 
115  int cmp = delta.Compare(hS);
116  // If delta is within 1/2 ULP, check for special case when significand is power of two.
117  // In this case, need to compare with 1/2h in the lower bound.
118  if (cmp < 0 && *adjustToNegative && // within and dS < bS
119  db.IsNormal() && (bInt & (bInt - 1)) == 0 && // Power of 2
120  db.Uint64Value() != RAPIDJSON_UINT64_C2(0x00100000, 0x00000000)) // minimum normal number must not do this
121  {
122  delta <<= 1;
123  return delta.Compare(hS);
124  }
125  return cmp;
126 }

128 inline bool StrtodFast(double d, int p, double* result) {
129  // Use fast path for string-to-double conversion if possible
130  // see http://www.exploringbinary.com/fast-path-decimal-to-floating-point-conversion/
131  if (p > 22 && p < 22 + 16) {
132  // Fast Path Cases In Disguise
133  d *= internal::Pow10(p - 22);
134  p = 22;
135  }
136 
137  if (p >= -22 && p <= 22 && d <= 9007199254740991.0) { // 2^53 - 1
138  *result = FastPath(d, p);
139  return true;
140  }
141  else
142  return false;
143 }

145 // Compute an approximation and see if it is within 1/2 ULP
146 inline bool StrtodDiyFp(const char* decimals, size_t length, size_t decimalPosition, int exp, double* result) {
147  uint64_t significand = 0;
148  size_t i = 0; // 2^64 - 1 = 18446744073709551615, 1844674407370955161 = 0x1999999999999999
149  for (; i < length; i++) {
150  if (significand > RAPIDJSON_UINT64_C2(0x19999999, 0x99999999) ||
151  (significand == RAPIDJSON_UINT64_C2(0x19999999, 0x99999999) && decimals[i] > '5'))
152  break;
153  significand = significand * 10 + (decimals[i] - '0');
154  }
155 
156  if (i < length && decimals[i] >= '5') // Rounding
157  significand++;
158 
159  size_t remaining = length - i;
160  const unsigned kUlpShift = 3;
161  const unsigned kUlp = 1 << kUlpShift;
162  int error = (remaining == 0) ? 0 : kUlp / 2;
163 
164  DiyFp v(significand, 0);
165  v = v.Normalize();
166  error <<= -v.e;
167 
168  const int dExp = (int)decimalPosition - (int)i + exp;
169 
170  int actualExp;
171  DiyFp cachedPower = GetCachedPower10(dExp, &actualExp);
172  if (actualExp != dExp) {
173  static const DiyFp kPow10[] = {
174  DiyFp(RAPIDJSON_UINT64_C2(0xa0000000, 00000000), -60), // 10^1
175  DiyFp(RAPIDJSON_UINT64_C2(0xc8000000, 00000000), -57), // 10^2
176  DiyFp(RAPIDJSON_UINT64_C2(0xfa000000, 00000000), -54), // 10^3
177  DiyFp(RAPIDJSON_UINT64_C2(0x9c400000, 00000000), -50), // 10^4
178  DiyFp(RAPIDJSON_UINT64_C2(0xc3500000, 00000000), -47), // 10^5
179  DiyFp(RAPIDJSON_UINT64_C2(0xf4240000, 00000000), -44), // 10^6
180  DiyFp(RAPIDJSON_UINT64_C2(0x98968000, 00000000), -40) // 10^7
181  };
182  int adjustment = dExp - actualExp - 1;
183  RAPIDJSON_ASSERT(adjustment >= 0 && adjustment < 7);
184  v = v * kPow10[adjustment];
185  if (length + adjustment > 19) // has more digits than decimal digits in 64-bit
186  error += kUlp / 2;
187  }
188 
189  v = v * cachedPower;
190 
191  error += kUlp + (error == 0 ? 0 : 1);
192 
193  const int oldExp = v.e;
194  v = v.Normalize();
195  error <<= oldExp - v.e;
196 
197  const unsigned effectiveSignificandSize = Double::EffectiveSignificandSize(64 + v.e);
198  unsigned precisionSize = 64 - effectiveSignificandSize;
199  if (precisionSize + kUlpShift >= 64) {
200  unsigned scaleExp = (precisionSize + kUlpShift) - 63;
201  v.f >>= scaleExp;
202  v.e += scaleExp;
203  error = (error >> scaleExp) + 1 + kUlp;
204  precisionSize -= scaleExp;
205  }
206 
207  DiyFp rounded(v.f >> precisionSize, v.e + precisionSize);
208  const uint64_t precisionBits = (v.f & ((uint64_t(1) << precisionSize) - 1)) * kUlp;
209  const uint64_t halfWay = (uint64_t(1) << (precisionSize - 1)) * kUlp;
210  if (precisionBits >= halfWay + error)
211  rounded.f++;
212 
213  *result = rounded.ToDouble();
214 
215  return halfWay - error >= precisionBits || precisionBits >= halfWay + error;
216 }

218 inline double StrtodBigInteger(double approx, const char* decimals, size_t length, size_t decimalPosition, int exp) {
219  const BigInteger dInt(decimals, length);
220  const int dExp = (int)decimalPosition - (int)length + exp;
221  Double a(approx);
222  for (int i = 0; i < 10; i++) {
223  bool adjustToNegative;
224  int cmp = CheckWithinHalfULP(a.Value(), dInt, dExp, &adjustToNegative);
225  if (cmp < 0)
226  return a.Value(); // within half ULP
227  else if (cmp == 0) {
228  // Round towards even
229  if (a.Significand() & 1)
230  return adjustToNegative ? a.PreviousPositiveDouble() : a.NextPositiveDouble();
231  else
232  return a.Value();
233  }
234  else // adjustment
235  a = adjustToNegative ? a.PreviousPositiveDouble() : a.NextPositiveDouble();
236  }
237 
238  // This should not happen, but in case there is really a bug, break the infinite-loop
239  return a.Value();
240 }

242 inline double StrtodFullPrecision(double d, int p, const char* decimals, size_t length, size_t decimalPosition, int exp) {
243  RAPIDJSON_ASSERT(d >= 0.0);
244  RAPIDJSON_ASSERT(length >= 1);
245 
246  double result;
247  if (StrtodFast(d, p, &result))
248  return result;
249 
250  // Trim leading zeros
251  while (*decimals == '0' && length > 1) {
252  length--;
253  decimals++;
254  decimalPosition--;
255  }
256 
257  // Trim trailing zeros
258  while (decimals[length - 1] == '0' && length > 1) {
259  length--;
260  decimalPosition--;
261  exp++;
262  }
263 
264  // Trim right-most digits
265  const int kMaxDecimalDigit = 780;
266  if ((int)length > kMaxDecimalDigit) {
267  exp += (int(length) - kMaxDecimalDigit);
268  length = kMaxDecimalDigit;
269  }
270 
271  // If too small, underflow to zero
272  if (int(length) + exp < -324)
273  return 0.0;
274 
275  if (StrtodDiyFp(decimals, length, decimalPosition, exp, &result))
276  return result;
277 
278  // Use approximation from StrtodDiyFp and make adjustment with BigInteger comparison
279  return StrtodBigInteger(result, decimals, length, decimalPosition, exp);
280 }
