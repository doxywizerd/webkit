185  template<typename AnimValType>
186  void executeAction(AnimationAction action, const SVGElementAnimatedPropertyList& animatedTypes, unsigned whichProperty, typename AnimValType::ContentType* type = 0)
187  {
188  // FIXME: Can't use SVGElement::InstanceUpdateBlocker because of circular header dependency. Would be nice to untangle this.
189  setInstanceUpdatesBlocked(*animatedTypes[0].element, true);
190 
191  for (auto& animatedType : animatedTypes) {
192  ASSERT_WITH_SECURITY_IMPLICATION(whichProperty < animatedType.properties.size());
193  AnimValType* property = castAnimatedPropertyToActualType<AnimValType>(animatedType.properties[whichProperty].get());
194 
195  switch (action) {
196  case StartAnimationAction:
197  ASSERT(type);
198  if (!property->isAnimating())
199  property->animationStarted(type);
200  break;
201  case StopAnimationAction:
202  ASSERT(!type);
203  if (property->isAnimating())
204  property->animationEnded();
205  break;
206  case AnimValWillChangeAction:
207  ASSERT(!type);
208  property->animValWillChange();
209  break;
210  case AnimValDidChangeAction:
211  ASSERT(!type);
212  property->animValDidChange();
213  break;
214  }
215  }
216 
217  setInstanceUpdatesBlocked(*animatedTypes[0].element, false);
218  }
