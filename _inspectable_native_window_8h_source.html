70  // setNewClientSize is used by the WinRT size change handler. It isn't used by the rest of ANGLE.
71  void setNewClientSize(const Size &newWindowSize)
72  {
73  // If the client doesn't support swapchain resizing then we should have already unregistered from size change handler
74  ASSERT(mSupportsSwapChainResize);
75 
76  if (mSupportsSwapChainResize)
77  {
78  // If the swapchain size was specified then we should ignore this call too
79  if (!mSwapChainSizeSpecified)
80  {
81  mNewClientRect = clientRect(newWindowSize);
82  mClientRectChanged = true;
83 
84  // If a scale was specified, then now is the time to apply the scale matrix for the new swapchain size and window size
85  if (mSwapChainScaleSpecified)
86  {
87  scaleSwapChain(newWindowSize, mNewClientRect);
88  }
89  }
90 
91  // Even if the swapchain size was fixed, the window might have changed size.
92  // In this case, we should recalculate the scale matrix to account for the new window size
93  if (mSwapChainSizeSpecified)
94  {
95  scaleSwapChain(newWindowSize, mClientRect);
96  }
97  }
98  }
