284 inline LayoutUnit RenderTableCell::logicalHeightForRowSizing() const
285 {
286  // FIXME: This function does too much work, and is very hot during table layout!
287  LayoutUnit adjustedLogicalHeight = logicalHeight() - (intrinsicPaddingBefore() + intrinsicPaddingAfter());
288  if (!style().logicalHeight().isSpecified())
289  return adjustedLogicalHeight;
290  LayoutUnit styleLogicalHeight = valueForLength(style().logicalHeight(), 0);
291  // In strict mode, box-sizing: content-box do the right thing and actually add in the border and padding.
292  // Call computedCSSPadding* directly to avoid including implicitPadding.
293  if (!document().inQuirksMode() && style().boxSizing() != BORDER_BOX)
294  styleLogicalHeight += computedCSSPaddingBefore() + computedCSSPaddingAfter() + borderBefore() + borderAfter();
295  return std::max(styleLogicalHeight, adjustedLogicalHeight);
296 }
