163  unsigned hash() const
164  {
165  unsigned result = WTF::PtrHash<UniquedStringImpl*>::hash(m_uid) + static_cast<unsigned>(m_kind);
166  switch (m_kind) {
167  case Presence:
168  result ^= u.presence.offset;
169  result ^= u.presence.attributes;
170  break;
171  case Absence:
172  case AbsenceOfSetter:
173  result ^= WTF::PtrHash<JSObject*>::hash(u.absence.prototype);
174  break;
175  case Equivalence:
176  result ^= EncodedJSValueHash::hash(u.equivalence.value);
177  break;
178  }
179  return result;
180  }

182  bool operator==(const PropertyCondition& other) const
183  {
184  if (m_uid != other.m_uid)
185  return false;
186  if (m_kind != other.m_kind)
187  return false;
188  switch (m_kind) {

192  case Absence:
193  case AbsenceOfSetter:

195  case Equivalence:

207  // Two conditions are compatible if they are identical or if they speak of different uids. If
208  // false is returned, you have to decide how to resolve the conflict - for example if there is
209  // a Presence and an Equivalence then in some cases you'll want the more general of the two
210  // while in other cases you'll want the more specific of the two. This will also return false
211  // for contradictions, like Presence and Absence on the same uid. By convention, invalid
212  // conditions aren't compatible with anything.
213  bool isCompatibleWith(const PropertyCondition& other) const
214  {
215  if (!*this || !other)
216  return false;
217  return *this == other || uid() != other.uid();

236  // In some cases, the condition is not watchable, but could be made watchable by enabling the
237  // appropriate watchpoint. For example, replacement watchpoints are enabled only when some
238  // access is cached on the property in some structure. This is mainly to save space for
239  // dictionary properties or properties that never get very hot. But, it's always safe to
240  // enable watching, provided that this is called from the main thread.
241  enum WatchabilityEffort {
242  // This is the default. It means that we don't change the state of any Structure or
243  // object, and implies that if the property happens not to be watchable then we don't make
244  // it watchable. This is mandatory if calling from a JIT thread. This is also somewhat
245  // preferable when first deciding whether to watch a condition for the first time (i.e.
246  // not from a watchpoint fire that causes us to see if we should adapt), since a
247  // watchpoint not being initialized for watching implies that maybe we don't know enough
248  // yet to make it profitable to watch -- as in, the thing being watched may not have
249  // stabilized yet. We prefer to only assume that a condition will hold if it has been
250  // known to hold for a while already.
251  MakeNoChanges,
252 
253  // Do what it takes to ensure that the property can be watched, if doing so has no
254  // user-observable effect. For now this just means that we will ensure that a property
255  // replacement watchpoint is enabled if it hadn't been enabled already. Do not use this
256  // from JIT threads, since the act of enabling watchpoints is not thread-safe.
257  EnsureWatchability
258  };
