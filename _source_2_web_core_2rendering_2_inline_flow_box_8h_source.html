60  // Internet Explorer and Firefox always create a marker for list items, even when the list-style-type is none. We do not make a marker
61  // in the list-style-type: none case, since it is wasteful to do so. However, in order to match other browsers we have to pretend like
62  // an invisible marker exists. The side effect of having an invisible marker is that the quirks mode behavior of shrinking lines with no
63  // text children must not apply. This change also means that gaps will exist between image bullet list items. Even when the list bullet
64  // is an image, the line is still considered to be immune from the quirk.
65  m_hasTextChildren = renderer.style().display() == LIST_ITEM;
66  m_hasTextDescendants = m_hasTextChildren;

221  // Line visual and layout overflow are in the coordinate space of the block. This means that they aren't purely physical directions.
222  // For horizontal-tb and vertical-lr they will match physical directions, but for horizontal-bt and vertical-rl, the top/bottom and left/right
223  // respectively are flipped when compared to their physical counterparts. For example minX is on the left in vertical-lr, but it is on the right in vertical-rl.
224  LayoutRect layoutOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
225  {
226  return m_overflow ? m_overflow->layoutOverflowRect() : enclosingLayoutRect(frameRectIncludingLineHeight(lineTop, lineBottom));
227  }
228  LayoutUnit logicalTopLayoutOverflow(LayoutUnit lineTop) const
229  {
230  if (m_overflow)
231  return isHorizontal() ? m_overflow->layoutOverflowRect().y() : m_overflow->layoutOverflowRect().x();
232  return lineTop;
233  }
234  LayoutUnit logicalBottomLayoutOverflow(LayoutUnit lineBottom) const
235  {
236  if (m_overflow)
237  return isHorizontal() ? m_overflow->layoutOverflowRect().maxY() : m_overflow->layoutOverflowRect().maxX();
238  return lineBottom;
239  }
240  LayoutRect logicalLayoutOverflowRect(LayoutUnit lineTop, LayoutUnit lineBottom) const
241  {
242  LayoutRect result = layoutOverflowRect(lineTop, lineBottom);

245  return result;
246  }
