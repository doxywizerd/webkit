298  // Nodes belonging to this document increase referencingNodeCount -
299  // these are enough to keep the document from being destroyed, but
300  // not enough to keep it from removing its children. This allows a
301  // node that outlives its document to still have a valid document
302  // pointer without introducing reference cycles.

305  ASSERT(!m_deletionHasBegun);
306  ++m_referencingNodeCount;
307  }

309  void decrementReferencingNodeCount()
310  {
311  ASSERT(!m_deletionHasBegun || !m_referencingNodeCount);
312  --m_referencingNodeCount;
313  if (!m_referencingNodeCount && !refCount()) {
314 #if !ASSERT_DISABLED
315  m_deletionHasBegun = true;
316 #endif
317  delete this;
318  }
319  }
