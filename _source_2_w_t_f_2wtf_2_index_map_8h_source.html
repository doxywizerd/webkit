32 // This is a map for keys that have an index(). It's super efficient for BasicBlocks. It's only
33 // efficient for Values if you don't create too many of these maps, since Values can have very
34 // sparse indices and there are a lot of Values.
35 
36 template<typename Key, typename Value>
37 class IndexMap {
38 public:
39  explicit IndexMap(size_t size = 0)
40  {
41  m_vector.fill(Value(), size);
42  }

47  }

52  }
