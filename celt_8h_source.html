172 static OPUS_INLINE int toOpus(unsigned char c)
173 {
174  int ret=0;
175  if (c<0xA0)
176  ret = toOpusTable[c>>3];
177  if (ret == 0)
178  return -1;
179  else
180  return ret|(c&0x7);
181 }
