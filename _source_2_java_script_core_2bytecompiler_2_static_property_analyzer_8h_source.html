110 inline void StaticPropertyAnalyzer::kill(int dst)
111 {
112  // We observe kills in order to avoid piling on properties to an object after
113  // its bytecode register has been recycled.
114 
115  // Consider these cases:
116 
117  // (1) Aliased temporary
118  // var o1 = { name: name };
119  // var o2 = { name: name };
120 
121  // (2) Aliased local -- no control flow
122  // var local;
123  // local = new Object;
124  // local.name = name;
125  // ...
126 
127  // local = lookup();
128  // local.didLookup = true;
129  // ...
130 
131  // (3) Aliased local -- control flow
132  // var local;
133  // if (condition)
134  // local = { };
135  // else {
136  // local = new Object;
137  // }
138  // local.name = name;
139 
140  // (Note: our default codegen for "new Object" looks like case (3).)
141 
142  // Case (1) is easy because temporaries almost never survive across control flow.
143 
144  // Cases (2) and (3) are hard. Case (2) should kill "local", while case (3) should
145  // not. There is no great way to solve these cases with simple static analysis.
146 
147  // Since this is a simple static analysis, we just try to catch the simplest cases,
148  // so we accept kills to any registers except for registers that have no inferred
149  // properties yet.
150 
151  AnalysisMap::iterator it = m_analyses.find(dst);
152  if (it == m_analyses.end())
153  return;
154  if (!it->value->propertyIndexCount())
155  return;
156 
157  kill(it->value.get());
158  m_analyses.remove(it);
159 }
