37 inline DoubleFormatState mergeDoubleFormatStates(DoubleFormatState a, DoubleFormatState b)
38 {
39  switch (a) {
40  case EmptyDoubleFormatState:
41  return b;
42  case UsingDoubleFormat:
43  switch (b) {
44  case EmptyDoubleFormatState:
45  case UsingDoubleFormat:
46  return UsingDoubleFormat;
47  case NotUsingDoubleFormat:
48  case CantUseDoubleFormat:
49  return CantUseDoubleFormat;
50  }
51  case NotUsingDoubleFormat:
52  switch (b) {
53  case EmptyDoubleFormatState:
54  case NotUsingDoubleFormat:
55  return NotUsingDoubleFormat;
56  case UsingDoubleFormat:
57  case CantUseDoubleFormat:
58  return CantUseDoubleFormat;
59  }
60  case CantUseDoubleFormat:
61  return CantUseDoubleFormat;
62  }
63  RELEASE_ASSERT_NOT_REACHED();
64  return CantUseDoubleFormat;
65 }

67 inline bool mergeDoubleFormatState(DoubleFormatState& dest, DoubleFormatState src)
68 {
69  DoubleFormatState newState = mergeDoubleFormatStates(dest, src);
