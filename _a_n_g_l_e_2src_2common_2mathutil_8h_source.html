86 template <typename DestT, typename SrcT>
87 inline DestT clampCast(SrcT value)
88 {
89  static const DestT destLo = std::numeric_limits<DestT>::min();
90  static const DestT destHi = std::numeric_limits<DestT>::max();
91  static const SrcT srcLo = static_cast<SrcT>(destLo);
92  static const SrcT srcHi = static_cast<SrcT>(destHi);
93 
94  // When value is outside of or equal to the limits for DestT we use the DestT limit directly.
95  // This avoids undefined behaviors due to loss of precision when converting from floats to
96  // integers:
97  // destHi for ints is 2147483647 but the closest float number is around 2147483648, so when
98  // doing a conversion from float to int we run into an UB because the float is outside of the
99  // range representable by the int.
100  if (value <= srcLo)
101  {
102  return destLo;
103  }
104  else if (value >= srcHi)
105  {
106  return destHi;
107  }
108  else
109  {
110  return static_cast<DestT>(value);
111  }
112 }

185 inline unsigned short float32ToFloat16(float fp32)
186 {
187  unsigned int fp32i = bitCast<unsigned int>(fp32);
188  unsigned int sign = (fp32i & 0x80000000) >> 16;
189  unsigned int abs = fp32i & 0x7FFFFFFF;
190 
191  if(abs > 0x47FFEFFF) // Infinity
192  {
193  return static_cast<unsigned short>(sign | 0x7FFF);
194  }
195  else if(abs < 0x38800000) // Denormal
196  {
197  unsigned int mantissa = (abs & 0x007FFFFF) | 0x00800000;
198  int e = 113 - (abs >> 23);
199 
200  if(e < 24)
201  {
202  abs = mantissa >> e;
203  }
204  else
205  {
206  abs = 0;
207  }
208 
209  return static_cast<unsigned short>(sign | (abs + 0x00000FFF + ((abs >> 13) & 1)) >> 13);
210  }
211  else
212  {
213  return static_cast<unsigned short>(sign | (abs + 0xC8000000 + 0x00000FFF + ((abs >> 13) & 1)) >> 13);
214  }
215 }

222 inline unsigned short float32ToFloat11(float fp32)
223 {
224  const unsigned int float32MantissaMask = 0x7FFFFF;
225  const unsigned int float32ExponentMask = 0x7F800000;
226  const unsigned int float32SignMask = 0x80000000;
227  const unsigned int float32ValueMask = ~float32SignMask;
228  const unsigned int float32ExponentFirstBit = 23;
229  const unsigned int float32ExponentBias = 127;
230 
231  const unsigned short float11Max = 0x7BF;
232  const unsigned short float11MantissaMask = 0x3F;
233  const unsigned short float11ExponentMask = 0x7C0;
234  const unsigned short float11BitMask = 0x7FF;
235  const unsigned int float11ExponentBias = 14;
236 
237  const unsigned int float32Maxfloat11 = 0x477E0000;
238  const unsigned int float32Minfloat11 = 0x38800000;
239 
240  const unsigned int float32Bits = bitCast<unsigned int>(fp32);
241  const bool float32Sign = (float32Bits & float32SignMask) == float32SignMask;
242 
243  unsigned int float32Val = float32Bits & float32ValueMask;
244 
245  if ((float32Val & float32ExponentMask) == float32ExponentMask)
246  {
247  // INF or NAN
248  if ((float32Val & float32MantissaMask) != 0)
249  {
250  return float11ExponentMask | (((float32Val >> 17) | (float32Val >> 11) | (float32Val >> 6) | (float32Val)) & float11MantissaMask);
251  }
252  else if (float32Sign)
253  {
254  // -INF is clamped to 0 since float11 is positive only
255  return 0;
256  }
257  else
258  {
259  return float11ExponentMask;
260  }
261  }
262  else if (float32Sign)
263  {
264  // float11 is positive only, so clamp to zero
265  return 0;
266  }
267  else if (float32Val > float32Maxfloat11)
268  {
269  // The number is too large to be represented as a float11, set to max
270  return float11Max;
271  }
272  else
273  {
274  if (float32Val < float32Minfloat11)
275  {
276  // The number is too small to be represented as a normalized float11
277  // Convert it to a denormalized value.
278  const unsigned int shift = (float32ExponentBias - float11ExponentBias) - (float32Val >> float32ExponentFirstBit);
279  float32Val = ((1 << float32ExponentFirstBit) | (float32Val & float32MantissaMask)) >> shift;
280  }
281  else
282  {
283  // Rebias the exponent to represent the value as a normalized float11
284  float32Val += 0xC8000000;
285  }
286 
287  return ((float32Val + 0xFFFF + ((float32Val >> 17) & 1)) >> 17) & float11BitMask;
288  }
289 }

360 inline float float11ToFloat32(unsigned short fp11)
361 {
362  unsigned short exponent = (fp11 >> 6) & 0x1F;
363  unsigned short mantissa = fp11 & 0x3F;
364 
365  if (exponent == 0x1F)
366  {
367  // INF or NAN
368  return bitCast<float>(0x7f800000 | (mantissa << 17));
369  }
370  else
371  {
372  if (exponent != 0)
373  {
374  // normalized
375  }
376  else if (mantissa != 0)
377  {
378  // The value is denormalized
379  exponent = 1;
380 
381  do
382  {
383  exponent--;
384  mantissa <<= 1;
385  }
386  while ((mantissa & 0x40) == 0);
387 
388  mantissa = mantissa & 0x3F;
389  }
390  else // The value is zero
391  {
392  exponent = static_cast<unsigned short>(-112);
393  }
394 
395  return bitCast<float>(((exponent + 112) << 23) | (mantissa << 17));
396  }
397 }

488 inline unsigned int CountLeadingZeros(uint32_t x)
489 {
490  // Use binary search to find the amount of leading zeros.
491  unsigned int zeros = 32u;
492  uint32_t y;
493 
494  y = x >> 16u;
495  if (y != 0)
496  {
497  zeros = zeros - 16u;
498  x = y;
499  }
500  y = x >> 8u;
501  if (y != 0)
502  {
503  zeros = zeros - 8u;
504  x = y;
505  }
506  y = x >> 4u;
507  if (y != 0)
508  {
509  zeros = zeros - 4u;
510  x = y;
511  }
512  y = x >> 2u;
513  if (y != 0)
514  {
515  zeros = zeros - 2u;
516  x = y;
517  }
518  y = x >> 1u;
519  if (y != 0)
520  {
521  return zeros - 2u;
522  }

635 // First, both normalized floating-point values are converted into 16-bit integer values.
636 // Then, the results are packed into the returned 32-bit unsigned integer.
637 // The first float value will be written to the least significant bits of the output;
638 // the last float value will be written to the most significant bits.
639 // The conversion of each value to fixed point is done as follows :
640 // packSnorm2x16 : round(clamp(c, -1, +1) * 32767.0)
641 inline uint32_t packSnorm2x16(float f1, float f2)
642 {
643  int16_t leastSignificantBits = static_cast<int16_t>(roundf(clamp(f1, -1.0f, 1.0f) * 32767.0f));
644  int16_t mostSignificantBits = static_cast<int16_t>(roundf(clamp(f2, -1.0f, 1.0f) * 32767.0f));
645  return static_cast<uint32_t>(mostSignificantBits) << 16 |
646  (static_cast<uint32_t>(leastSignificantBits) & 0xFFFF);
647 }

650 // component is converted to a normalized floating-point value to generate the returned two float values.
651 // The first float value will be extracted from the least significant bits of the input;

653 // The conversion for unpacked fixed-point value to floating point is done as follows:
654 // unpackSnorm2x16 : clamp(f / 32767.0, -1, +1)

656 {

691 // floating-point representation found in the OpenGL ES Specification, and then packing these

768 // Sum, difference and multiplication operations for signed ints that wrap on 32-bit overflow.
769 //
770 // Unsigned types are defined to do arithmetic modulo 2^n in C++. For signed types, overflow
771 // behavior is undefined.
772 
773 template <typename T>
774 inline T WrappingSum(T lhs, T rhs)
775 {
776  uint32_t lhsUnsigned = static_cast<uint32_t>(lhs);
777  uint32_t rhsUnsigned = static_cast<uint32_t>(rhs);
778  return static_cast<T>(lhsUnsigned + rhsUnsigned);
779 }

783 {
784  uint32_t lhsUnsigned = static_cast<uint32_t>(lhs);
785  uint32_t rhsUnsigned = static_cast<uint32_t>(rhs);
786  return static_cast<T>(lhsUnsigned - rhsUnsigned);

789 inline int32_t WrappingMul(int32_t lhs, int32_t rhs)
790 {
791  int64_t lhsWide = static_cast<int64_t>(lhs);
792  int64_t rhsWide = static_cast<int64_t>(rhs);
793  // The multiplication is guaranteed not to overflow.
794  int64_t resultWide = lhsWide * rhsWide;
795  // Implement the desired wrapping behavior by masking out the high-order 32 bits.
796  resultWide = resultWide & 0xffffffffll;
797  // Casting to a narrower signed type is fine since the casted value is representable in the
798  // narrower type.
799  return static_cast<int32_t>(resultWide);
