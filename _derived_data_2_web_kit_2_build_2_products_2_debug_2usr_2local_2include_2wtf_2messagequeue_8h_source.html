139  template<typename Predicate>
140  inline auto MessageQueue<DataType>::waitForMessageFilteredWithTimeout(MessageQueueWaitResult& result, Predicate&& predicate, WallTime absoluteTime) -> std::unique_ptr<DataType>
141  {
142  LockHolder lock(m_mutex);
143  bool timedOut = false;
144 
145  auto found = m_queue.end();
146  while (!m_killed && !timedOut) {
147  found = m_queue.findIf([&predicate](const std::unique_ptr<DataType>& ptr) -> bool {
148  ASSERT(ptr);
149  return predicate(*ptr);
150  });
151  if (found != m_queue.end())
152  break;
153 
154  timedOut = !m_condition.waitUntil(m_mutex, absoluteTime);
155  }
156 
157  ASSERT(!timedOut || absoluteTime != WallTime::infinity());
158 
159  if (m_killed) {
160  result = MessageQueueTerminated;
161  return nullptr;
162  }
163 
164  if (timedOut) {
165  result = MessageQueueTimeout;
166  return nullptr;
167  }
168 
169  ASSERT(found != m_queue.end());
170  std::unique_ptr<DataType> message = WTFMove(*found);
171  m_queue.remove(found);
172  result = MessageQueueMessageReceived;
173  return message;
174  }
