126  // ImplicitAddress:
127  //
128  // This class is used for explicit 'load' and 'store' operations
129  // (as opposed to situations in which a memory operand is provided
130  // to a generic operation, such as an integer arithmetic instruction).
131  //
132  // In the case of a load (or store) operation we want to permit
133  // addresses to be implicitly constructed, e.g. the two calls:
134  //
135  // load32(Address(addrReg), destReg);

137  //
138  // Are equivalent, and the explicit wrapping of the Address in the former
139  // is unnecessary.

574  void link(AbstractMacroAssemblerType* masm) const
575  {
576  masm->invalidateAllTempRegisters();
577 
578 #if ENABLE(DFG_REGISTER_ALLOCATION_VALIDATION)
579  masm->checkRegisterAllocationAgainstBranchRange(m_label.m_offset, masm->debugOffset());
580 #endif
581 
582 #if CPU(ARM_THUMB2)
583  masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition);
584 #elif CPU(ARM64)
585  if ((m_type == ARM64Assembler::JumpCompareAndBranch) || (m_type == ARM64Assembler::JumpCompareAndBranchFixedSize))
586  masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition, m_is64Bit, m_compareRegister);
587  else if ((m_type == ARM64Assembler::JumpTestBit) || (m_type == ARM64Assembler::JumpTestBitFixedSize))
588  masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition, m_bitNumber, m_compareRegister);
589  else
590  masm->m_assembler.linkJump(m_label, masm->m_assembler.label(), m_type, m_condition);
591 #else
592  masm->m_assembler.linkJump(m_label, masm->m_assembler.label());
593 #endif
594  }
595
