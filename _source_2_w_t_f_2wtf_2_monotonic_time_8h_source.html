37 // The current time according to a monotonic clock. Monotonic clocks don't go backwards and
38 // possibly don't count downtime. This uses floating point internally so that you can reason about
39 // infinity and other things that arise in math. It's acceptable to use this to wrap NaN times,
40 // negative times, and infinite times, so long as they are all relative to the same clock.
41 // Specifically, MonotonicTime should be used in agreement with the principle that
42 // MonotonicTime::now().secondsSinceEpoch().value() is the same as
43 // WTF::monotonicallyIncreasingTime().
44 class MonotonicTime {
45 public:
46  static const ClockType clockType = ClockType::Monotonic;
47 
48  // This is the epoch. So, x.secondsSinceEpoch() should be the same as x - MonotonicTime().
49  MonotonicTime() { }

51  // Call this if you know for sure that the double represents time according to
52  // WTF::monotonicallyIncreasingTime(). It must be in seconds and it must be from the same time
53  // source.
54  static MonotonicTime fromRawSeconds(double value)
55  {
56  MonotonicTime result;
57  result.m_value = value;
58  return result;
59  }
