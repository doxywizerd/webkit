89  enum Role : int8_t {
90  // Use means that the Inst will read from this value before doing anything else.
91  //
92  // For Tmp: The Inst will read this Tmp.
93  // For Arg::addr and friends: The Inst will load from this address.
94  // For Arg::imm and friends: The Inst will materialize and use this immediate.
95  // For RelCond/ResCond/Special: This is the only valid role for these kinds.
96  //
97  // Note that Use of an address does not mean escape. It only means that the instruction will
98  // load from the address before doing anything else. This is a bit tricky; for example
99  // Specials could theoretically squirrel away the address and effectively escape it. However,
100  // this is not legal. On the other hand, any address other than Stack is presumed to be
101  // always escaping, and Stack is presumed to be always escaping if it's Locked.
102  Use,
103 
104  // Exactly like Use, except that it also implies that the use is cold: that is, replacing the
105  // use with something on the stack is free.
106  ColdUse,
107 
108  // LateUse means that the Inst will read from this value after doing its Def's. Note that LateUse
109  // on an Addr or Index still means Use on the internal temporaries. Note that specifying the
110  // same Tmp once as Def and once as LateUse has undefined behavior: the use may happen before
111  // the def, or it may happen after it.
112  LateUse,
113 
114  // Combination of LateUse and ColdUse.
115  LateColdUse,
116 
117  // Def means that the Inst will write to this value after doing everything else.
118  //
119  // For Tmp: The Inst will write to this Tmp.
120  // For Arg::addr and friends: The Inst will store to this address.
121  // This isn't valid for any other kinds.
122  //
123  // Like Use of address, Def of address does not mean escape.
124  Def,
125 
126  // This is a special variant of Def that implies that the upper bits of the target register are
127  // zero-filled. Specifically, if the Width of a ZDef is less than the largest possible width of
128  // the argument (for example, we're on a 64-bit machine and we have a Width32 ZDef of a GPR) then
129  // this has different implications for the upper bits (i.e. the top 32 bits in our example)
130  // depending on the kind of the argument:
131  //
132  // For register: the upper bits are zero-filled.
133  // For anonymous stack slot: the upper bits are zero-filled.
134  // For address: the upper bits are not touched (i.e. we do a 32-bit store in our example).
135  // For tmp: either the upper bits are not touched or they are zero-filled, and we won't know
136  // which until we lower the tmp to either a StackSlot or a Reg.
137  //
138  // The behavior of ZDef is consistent with what happens when you perform 32-bit operations on a
139  // 64-bit GPR. It's not consistent with what happens with 8-bit or 16-bit Defs on x86 GPRs, or
140  // what happens with float Defs in ARM NEON or X86 SSE. Hence why we have both Def and ZDef.
141  ZDef,
142 
143  // This is a combined Use and Def. It means that both things happen.
144  UseDef,
145 
146  // This is a combined Use and ZDef. It means that both things happen.
147  UseZDef,
148 
149  // This is like Def, but implies that the assignment occurs before the start of the Inst's
150  // execution rather than after. Note that specifying the same Tmp once as EarlyDef and once
151  // as Use has undefined behavior: the use may happen before the def, or it may happen after
152  // it.
153  EarlyDef,
154 
155  // Some instructions need a scratch register. We model this by saying that the temporary is
156  // defined early and used late. This role implies that.
157  Scratch,
158 
159  // This is a special kind of use that is only valid for addresses. It means that the
160  // instruction will evaluate the address expression and consume the effective address, but it
161  // will neither load nor store. This is an escaping use, because now the address may be
162  // passed along to who-knows-where. Note that this isn't really a Use of the Arg, but it does
163  // imply that we're Use'ing any registers that the Arg contains.
164  UseAddr
165  };

457  // If you don't pass a Width, this optimistically assumes that you're using the right width.
458  static bool isValidScale(unsigned scale, std::optional<Width> width = std::nullopt)
459  {
460  switch (scale) {
461  case 1:
462  if (isX86() || isARM64())
463  return true;
464  return false;
465  case 2:
466  case 4:
467  case 8:
468  if (isX86())
469  return true;
470  if (isARM64()) {
471  if (!width)
472  return true;
473  return scale == 1 || scale == bytes(*width);
474  }
475  return false;
476  default:
477  return false;
478  }
479  }
