120  unsigned hash() const
121  {
122  Vector<unsigned> hashCodes;
123  hashCodes.append(static_cast<unsigned>(m_type));
124  hashCodes.append(m_isNull ? 1 : 0);
125  hashCodes.append(m_isDeletedValue ? 1 : 0);
126  switch (m_type) {
127  case KeyType::Invalid:
128  case KeyType::Max:
129  case KeyType::Min:
130  break;
131  case KeyType::Number:
132  case KeyType::Date:
133  hashCodes.append(StringHasher::hashMemory<sizeof(double)>(&WTF::get<double>(m_value)));
134  break;
135  case KeyType::String:
136  hashCodes.append(StringHash::hash(WTF::get<String>(m_value)));
137  break;
138  case KeyType::Binary: {
139  auto* data = WTF::get<ThreadSafeDataBuffer>(m_value).data();
140  if (!data)
141  hashCodes.append(0);
142  else
143  hashCodes.append(StringHasher::hashMemory(data->data(), data->size()));
144  break;
145  }
146  case KeyType::Array:
147  for (auto& key : WTF::get<Vector<IDBKeyData>>(m_value))
148  hashCodes.append(key.hash());
149  break;
150  }
151 
152  return StringHasher::hashMemory(hashCodes.data(), hashCodes.size() * sizeof(unsigned));
153  }
