436  // -----------------------------------------------------------------------------
437  // Notification of document structure changes (see ContainerNode.h for more notification methods)
438  //
439  // At first, WebKit notifies the node that it has been inserted into the document. This is called during document parsing, and also
440  // when a node is added through the DOM methods insertBefore(), appendChild() or replaceChild(). The call happens _after_ the node has been added to the tree.
441  // This is similar to the DOMNodeInsertedIntoDocument DOM event, but does not require the overhead of event
442  // dispatching.
443  //
444  // WebKit notifies this callback regardless if the subtree of the node is a document tree or a floating subtree.
445  // Implementation can determine the type of subtree by seeing insertionPoint->isConnected().
446  // For a performance reason, notifications are delivered only to ContainerNode subclasses if the insertionPoint is out of document.
447  //
448  // There is another callback named finishedInsertingSubtree(), which is called after all descendants are notified.
449  // Only a few subclasses actually need this. To utilize this, the node should return InsertionShouldCallFinishedInsertingSubtree
450  // from insrtedInto().
451  //
452  enum InsertionNotificationRequest {
453  InsertionDone,
454  InsertionShouldCallFinishedInsertingSubtree
455  };
