78 class CodeCacheMap {
79 public:
80  typedef HashMap<SourceCodeKey, SourceCodeValue, SourceCodeKey::Hash, SourceCodeKey::HashTraits> MapType;
81  typedef MapType::iterator iterator;
82  typedef MapType::AddResult AddResult;
83 
84  CodeCacheMap()
85  : m_size(0)
86  , m_sizeAtLastPrune(0)
87  , m_timeAtLastPrune(monotonicallyIncreasingTime())
88  , m_minCapacity(0)
89  , m_capacity(0)
90  , m_age(0)
91  {
92  }
93 
94  SourceCodeValue* findCacheAndUpdateAge(const SourceCodeKey& key)
95  {
96  prune();
97 
98  iterator findResult = m_map.find(key);
99  if (findResult == m_map.end())
100  return nullptr;
101 
102  int64_t age = m_age - findResult->value.age;
103  if (age > m_capacity) {
104  // A requested object is older than the cache's capacity. We can
105  // infer that requested objects are subject to high eviction probability,
106  // so we grow the cache to improve our hit rate.
107  m_capacity += recencyBias * oldObjectSamplingMultiplier * key.length();
108  } else if (age < m_capacity / 2) {
109  // A requested object is much younger than the cache's capacity. We can
110  // infer that requested objects are subject to low eviction probability,
111  // so we shrink the cache to save memory.
112  m_capacity -= recencyBias * key.length();
113  if (m_capacity < m_minCapacity)
114  m_capacity = m_minCapacity;
115  }
116 
117  findResult->value.age = m_age;
118  m_age += key.length();
119 
120  return &findResult->value;
121  }
122 
123  AddResult addCache(const SourceCodeKey& key, const SourceCodeValue& value)
124  {
125  prune();
126 
127  AddResult addResult = m_map.add(key, value);
128  ASSERT(addResult.isNewEntry);
129 
130  m_size += key.length();
131  m_age += key.length();
132  return addResult;
133  }
134 
135  void remove(iterator it)
136  {
137  m_size -= it->key.length();
138  m_map.remove(it);
139  }
140 
141  void clear()
142  {
143  m_size = 0;
144  m_age = 0;
145  m_map.clear();
146  }
147 
148  int64_t age() { return m_age; }
149 
150 private:
151  // This constant factor biases cache capacity toward allowing a minimum
152  // working set to enter the cache before it starts evicting.
153  static const double workingSetTime;
154  static const int64_t workingSetMaxBytes = 16000000;
155  static const size_t workingSetMaxEntries = 2000;
156 
157  // This constant factor biases cache capacity toward recent activity. We
158  // want to adapt to changing workloads.
159  static const int64_t recencyBias = 4;
160 
161  // This constant factor treats a sampled event for one old object as if it
162  // happened for many old objects. Most old objects are evicted before we can
163  // sample them, so we need to extrapolate from the ones we do sample.
164  static const int64_t oldObjectSamplingMultiplier = 32;
165 
166  size_t numberOfEntries() const { return static_cast<size_t>(m_map.size()); }
167  bool canPruneQuickly() const { return numberOfEntries() < workingSetMaxEntries; }
168 
169  void pruneSlowCase();
170  void prune()
171  {
172  if (m_size <= m_capacity && canPruneQuickly())
173  return;
174 
175  if (monotonicallyIncreasingTime() - m_timeAtLastPrune < workingSetTime
176  && m_size - m_sizeAtLastPrune < workingSetMaxBytes
177  && canPruneQuickly())
178  return;
179 
180  pruneSlowCase();
181  }
182 
183  MapType m_map;
184  int64_t m_size;
185  int64_t m_sizeAtLastPrune;
186  double m_timeAtLastPrune;
187  int64_t m_minCapacity;
188  int64_t m_capacity;
189  int64_t m_age;
190 };
