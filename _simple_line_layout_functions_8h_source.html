74 inline LayoutUnit computeFlowHeight(const RenderBlockFlow& flow, const Layout& layout)
75 {
76  auto flowHeight = lineHeightFromFlow(flow) * layout.lineCount();
77  if (!layout.hasLineStruts())
78  return flowHeight;
79  for (auto& strutEntry : layout.struts())
80  flowHeight += strutEntry.offset;
81  return flowHeight;
82 }

111 inline bool containsCaretOffset(const RenderText&, const Layout& layout, unsigned offset)
112 {
113  for (unsigned i = 0; i < layout.runCount(); ++i) {
114  auto& run = layout.runAt(i);
115  if (offset < run.start)
116  return false;
117  if (offset <= run.end)
118  return true;
119  }
120  return false;
121 }
