37 template<typename T>
38 inline bool WARN_UNUSED_RETURN decodeUInt(const uint8_t* bytes, size_t length, size_t& offset, T& result)
39 {
40  const size_t numBits = sizeof(T) * CHAR_BIT;
41  const size_t maxByteLength = (numBits - 1) / 7 + 1; // numBits / 7 rounding up.
42  if (length <= offset)
43  return false;
44  result = 0;
45  unsigned shift = 0;
46  size_t last = std::min(maxByteLength, length - offset) - 1;
47  for (unsigned i = 0; true; ++i) {
48  uint8_t byte = bytes[offset++];
49  result |= static_cast<T>(byte & 0x7f) << shift;
50  shift += 7;
51  if (!(byte & 0x80))
52  return true;
53  if (i == last)
54  return false;
55  }
56  RELEASE_ASSERT_NOT_REACHED();
57  return true;
58 }

60 template<typename T>
61 inline bool WARN_UNUSED_RETURN decodeInt(const uint8_t* bytes, size_t length, size_t& offset, T& result)
62 {
63  const size_t numBits = sizeof(T) * CHAR_BIT;
64  const size_t maxByteLength = (numBits - 1) / 7 + 1; // numBits / 7 rounding up.
65  if (length <= offset)
66  return false;
67  result = 0;
68  unsigned shift = 0;
69  size_t last = std::min(maxByteLength, length - offset) - 1;
70  uint8_t byte;
71  for (unsigned i = 0; true; ++i) {
72  byte = bytes[offset++];
73  result |= static_cast<T>(byte & 0x7f) << shift;
74  shift += 7;
75  if (!(byte & 0x80))
76  break;
77  if (i == last)
78  return false;
79  }
80 
81  using UnsignedT = typename std::make_unsigned<T>::type;
82  if (shift < numBits && (byte & 0x40))
83  result = static_cast<T>(static_cast<UnsignedT>(result) | (static_cast<UnsignedT>(-1) << shift));
84  return true;
85 }
