238 {
239  // 0x180E used to be in Zs category before Unicode 6.3, and EcmaScript says that we should keep treating it as such.

338 template <typename T>
339 ALWAYS_INLINE JSTokenType Lexer<T>::lexExpectIdentifier(JSToken* tokenRecord, unsigned lexerFlags, bool strictMode)
340 {
341  JSTokenData* tokenData = &tokenRecord->m_data;
342  JSTokenLocation* tokenLocation = &tokenRecord->m_location;
343  ASSERT((lexerFlags & LexerFlagsIgnoreReservedWords));
344  const T* start = m_code;
345  const T* ptr = start;
346  const T* end = m_codeEnd;
347  JSTextPosition startPosition = currentPosition();
348  if (ptr >= end) {
349  ASSERT(ptr == end);
350  goto slowCase;
351  }
352  if (!WTF::isASCIIAlpha(*ptr))
353  goto slowCase;
354  ++ptr;
355  while (ptr < end) {
356  if (!WTF::isASCIIAlphanumeric(*ptr))
357  break;
358  ++ptr;
359  }
360 
361  // Here's the shift
362  if (ptr < end) {
363  if ((!WTF::isASCII(*ptr)) || (*ptr == '\\') || (*ptr == '_') || (*ptr == '$'))
364  goto slowCase;
365  m_current = *ptr;
366  } else
367  m_current = 0;
368 
369  m_code = ptr;
370  ASSERT(currentOffset() >= currentLineStartOffset());
371 
372  // Create the identifier if needed
373  if (lexerFlags & LexexFlagsDontBuildKeywords
374 #if !ASSERT_DISABLED
375  && !m_parsingBuiltinFunction
376 #endif
377  )
378  tokenData->ident = 0;
379  else
380  tokenData->ident = makeLCharIdentifier(start, ptr - start);
381 
382  tokenLocation->line = m_lineNumber;
383  tokenLocation->lineStartOffset = currentLineStartOffset();
384  tokenLocation->startOffset = offsetFromSourcePtr(start);
385  tokenLocation->endOffset = currentOffset();
386  ASSERT(tokenLocation->startOffset >= tokenLocation->lineStartOffset);
387  tokenRecord->m_startPosition = startPosition;
388  tokenRecord->m_endPosition = currentPosition();
389 #if !ASSERT_DISABLED
390  if (m_parsingBuiltinFunction) {
391  if (!isSafeBuiltinIdentifier(*m_vm, tokenData->ident))
392  return ERRORTOK;
393  }
394 #endif
395 
396  m_lastToken = IDENT;
397  return IDENT;
398 
399 slowCase:
400  return lex(tokenRecord, lexerFlags, strictMode);
401 }
