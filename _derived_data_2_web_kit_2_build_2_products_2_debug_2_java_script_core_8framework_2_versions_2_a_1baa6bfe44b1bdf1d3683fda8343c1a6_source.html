35 void createListFromArrayLike(ExecState* exec, JSValue arrayLikeValue, RuntimeTypeMask legalTypesFilter, const String& errorMessage, AddFunction addFunction)
36 {
37  VM& vm = exec->vm();
38  auto scope = DECLARE_THROW_SCOPE(vm);
39 
40  Vector<JSValue> result;
41  JSValue lengthProperty = arrayLikeValue.get(exec, vm.propertyNames->length);
42  RETURN_IF_EXCEPTION(scope, void());
43  double lengthAsDouble = lengthProperty.toLength(exec);
44  RETURN_IF_EXCEPTION(scope, void());
45  RELEASE_ASSERT(lengthAsDouble >= 0.0 && lengthAsDouble == std::trunc(lengthAsDouble));
46  uint64_t length = static_cast<uint64_t>(lengthAsDouble);
47  for (uint64_t index = 0; index < length; index++) {
48  JSValue next = arrayLikeValue.get(exec, index);
49  RETURN_IF_EXCEPTION(scope, void());
50 
51  RuntimeType type = runtimeTypeForValue(next);
52  if (!(type & legalTypesFilter)) {
53  throwTypeError(exec, scope, errorMessage);
54  return;
55  }
56 
57  bool exitEarly = addFunction(next, type);
58  if (exitEarly)
59  return;
60  }

63 ALWAYS_INLINE bool JSObject::canPerformFastPutInline(ExecState* exec, VM& vm, PropertyName propertyName)
64 {
65  if (UNLIKELY(propertyName == exec->propertyNames().underscoreProto))
66  return false;
67 
68  // Check if there are any setters or getters in the prototype chain
69  JSValue prototype;
70  JSObject* obj = this;
71  while (true) {
72  if (obj->structure(vm)->hasReadOnlyOrGetterSetterPropertiesExcludingProto() || obj->type() == ProxyObjectType)
73  return false;
74 
75  prototype = obj->getPrototypeDirect();
76  if (prototype.isNull())
77  return true;
78 
79  obj = asObject(prototype);
80  }
81 
82  ASSERT_NOT_REACHED();
83 }

102 ALWAYS_INLINE bool JSObject::getPropertySlot(ExecState* exec, unsigned propertyName, PropertySlot& slot)
103 {
104  VM& vm = exec->vm();
105  auto scope = DECLARE_THROW_SCOPE(vm);
106  auto& structureIDTable = vm.heap.structureIDTable();
107  JSObject* object = this;
108  MethodTable::GetPrototypeFunctionPtr defaultGetPrototype = JSObject::getPrototype;
109  while (true) {
110  Structure* structure = structureIDTable.get(object->structureID());
111  if (structure->classInfo()->methodTable.getOwnPropertySlotByIndex(object, exec, propertyName, slot))
112  return true;
113  RETURN_IF_EXCEPTION(scope, false);
114  JSValue prototype;
115  if (LIKELY(structure->classInfo()->methodTable.getPrototype == defaultGetPrototype || slot.internalMethodType() == PropertySlot::InternalMethodType::VMInquiry))
116  prototype = structure->storedPrototype();
117  else {
118  prototype = object->getPrototype(vm, exec);
119  RETURN_IF_EXCEPTION(scope, false);
120  }
121  if (!prototype.isObject())
122  return false;
123  object = asObject(prototype);
124  }

174 ALWAYS_INLINE PropertyOffset JSObject::prepareToPutDirectWithoutTransition(VM& vm, PropertyName propertyName, unsigned attributes, StructureID structureID, Structure* structure)
175 {
176  unsigned oldOutOfLineCapacity = structure->outOfLineCapacity();
177  PropertyOffset result;
178  structure->addPropertyWithoutTransition(
179  vm, propertyName, attributes,
180  [&] (const GCSafeConcurrentJSLocker&, PropertyOffset offset, PropertyOffset newLastOffset) {
181  unsigned newOutOfLineCapacity = Structure::outOfLineCapacity(newLastOffset);
182  if (newOutOfLineCapacity != oldOutOfLineCapacity) {
183  Butterfly* butterfly = allocateMoreOutOfLineStorage(vm, oldOutOfLineCapacity, newOutOfLineCapacity);
184  nukeStructureAndSetButterfly(vm, structureID, butterfly);
185  structure->setLastOffset(newLastOffset);
186  WTF::storeStoreFence();
187  setStructureIDDirectly(structureID);
188  } else
189  structure->setLastOffset(newLastOffset);
190  result = offset;
191  });
192  return result;
193 }

195 // ECMA 8.6.2.2
196 ALWAYS_INLINE bool JSObject::putInline(JSCell* cell, ExecState* exec, PropertyName propertyName, JSValue value, PutPropertySlot& slot)
197 {
198  VM& vm = exec->vm();
199  auto scope = DECLARE_THROW_SCOPE(vm);
200 
201  JSObject* thisObject = jsCast<JSObject*>(cell);
202  ASSERT(value);
203  ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(thisObject));
204 
205  if (UNLIKELY(isThisValueAltered(slot, thisObject)))
206  return ordinarySetSlow(exec, thisObject, propertyName, value, slot.thisValue(), slot.isStrictMode());
207 
208  // Try indexed put first. This is required for correctness, since loads on property names that appear like
209  // valid indices will never look in the named property storage.
210  if (std::optional<uint32_t> index = parseIndex(propertyName))
211  return putByIndex(thisObject, exec, index.value(), value, slot.isStrictMode());
212 
213  if (thisObject->canPerformFastPutInline(exec, vm, propertyName)) {
214  ASSERT(!thisObject->structure(vm)->prototypeChainMayInterceptStoreTo(vm, propertyName));
215  if (!thisObject->putDirectInternal<PutModePut>(vm, propertyName, value, 0, slot))
216  return typeError(exec, scope, slot.isStrictMode(), ASCIILiteral(ReadonlyPropertyWriteError));
217  return true;
218  }
219 
220  return thisObject->putInlineSlow(exec, propertyName, value, slot);
221 }

245 template<JSObject::PutMode mode>
246 ALWAYS_INLINE bool JSObject::putDirectInternal(VM& vm, PropertyName propertyName, JSValue value, unsigned attributes, PutPropertySlot& slot)
247 {
248  ASSERT(value);
249  ASSERT(value.isGetterSetter() == !!(attributes & Accessor));
250  ASSERT(!Heap::heap(value) || Heap::heap(value) == Heap::heap(this));
251  ASSERT(!parseIndex(propertyName));
252 
253  StructureID structureID = this->structureID();
254  Structure* structure = vm.heap.structureIDTable().get(structureID);
255  if (structure->isDictionary()) {
256  ASSERT(!structure->hasInferredTypes());
257 
258  unsigned currentAttributes;
259  PropertyOffset offset = structure->get(vm, propertyName, currentAttributes);
260  if (offset != invalidOffset) {
261  if ((mode == PutModePut) && currentAttributes & ReadOnly)
262  return false;
263 
264  putDirect(vm, offset, value);
265  structure->didReplaceProperty(offset);
266  slot.setExistingProperty(this, offset);
267 
268  if ((attributes & Accessor) != (currentAttributes & Accessor) || (attributes & CustomAccessor) != (currentAttributes & CustomAccessor)) {
269  ASSERT(!(attributes & ReadOnly));
270  setStructure(vm, Structure::attributeChangeTransition(vm, structure, propertyName, attributes));
271  }
272  return true;
273  }
274 
275  if ((mode == PutModePut) && !isStructureExtensible())
276  return false;
277 
278  offset = prepareToPutDirectWithoutTransition(vm, propertyName, attributes, structureID, structure);
279  validateOffset(offset);
280  putDirect(vm, offset, value);
281  slot.setNewProperty(this, offset);
282  if (attributes & ReadOnly)
283  this->structure()->setContainsReadOnlyProperties();
284  return true;
285  }
286 
287  PropertyOffset offset;
288  size_t currentCapacity = this->structure()->outOfLineCapacity();
289  Structure* newStructure = Structure::addPropertyTransitionToExistingStructure(
290  structure, propertyName, attributes, offset);
291  if (newStructure) {
292  newStructure->willStoreValueForExistingTransition(
293  vm, propertyName, value, slot.context() == PutPropertySlot::PutById);
294 
295  Butterfly* newButterfly = butterfly();
296  if (currentCapacity != newStructure->outOfLineCapacity()) {
297  ASSERT(newStructure != this->structure());
298  newButterfly = allocateMoreOutOfLineStorage(vm, currentCapacity, newStructure->outOfLineCapacity());
299  nukeStructureAndSetButterfly(vm, structureID, newButterfly);
300  }
301 
302  validateOffset(offset);
303  ASSERT(newStructure->isValidOffset(offset));
304  putDirect(vm, offset, value);
305  setStructure(vm, newStructure);
306  slot.setNewProperty(this, offset);
307  return true;
308  }
309 
310  unsigned currentAttributes;
311  bool hasInferredType;
312  offset = structure->get(vm, propertyName, currentAttributes, hasInferredType);
313  if (offset != invalidOffset) {
314  if ((mode == PutModePut) && currentAttributes & ReadOnly)
315  return false;
316 
317  structure->didReplaceProperty(offset);
318  if (UNLIKELY(hasInferredType)) {
319  structure->willStoreValueForReplace(
320  vm, propertyName, value, slot.context() == PutPropertySlot::PutById);
321  }
322 
323  slot.setExistingProperty(this, offset);
324  putDirect(vm, offset, value);
325 
326  if ((attributes & Accessor) != (currentAttributes & Accessor) || (attributes & CustomAccessor) != (currentAttributes & CustomAccessor)) {
327  ASSERT(!(attributes & ReadOnly));
328  setStructure(vm, Structure::attributeChangeTransition(vm, structure, propertyName, attributes));
329  }
330  return true;
331  }
332 
333  if ((mode == PutModePut) && !isStructureExtensible())
334  return false;
335 
336  // We want the structure transition watchpoint to fire after this object has switched
337  // structure. This allows adaptive watchpoints to observe if the new structure is the one
338  // we want.
339  DeferredStructureTransitionWatchpointFire deferredWatchpointFire;
340 
341  newStructure = Structure::addNewPropertyTransition(
342  vm, structure, propertyName, attributes, offset, slot.context(), &deferredWatchpointFire);
343  newStructure->willStoreValueForNewTransition(
344  vm, propertyName, value, slot.context() == PutPropertySlot::PutById);
345 
346  validateOffset(offset);
347  ASSERT(newStructure->isValidOffset(offset));
348  size_t oldCapacity = structure->outOfLineCapacity();
349  size_t newCapacity = newStructure->outOfLineCapacity();
350  ASSERT(oldCapacity <= newCapacity);
351  if (oldCapacity != newCapacity) {
352  Butterfly* newButterfly = allocateMoreOutOfLineStorage(vm, oldCapacity, newCapacity);
353  nukeStructureAndSetButterfly(vm, structureID, newButterfly);
354  }
355  putDirect(vm, offset, value);
356  setStructure(vm, newStructure);
357  slot.setNewProperty(this, offset);
358  if (attributes & ReadOnly)
359  newStructure->setContainsReadOnlyProperties();
360  return true;
361 }
