170  BackwardsGraph(Graph& graph)
171  : m_graph(graph)
172  {
173  GraphNodeWorklist<typename Graph::Node, typename Graph::Set> worklist;
174 
175  auto addRootSuccessor = [&] (typename Graph::Node node) {
176  if (worklist.push(node)) {
177  m_rootSuccessorList.append(node);
178  m_rootSuccessorSet.add(node);
179  while (typename Graph::Node node = worklist.pop())
180  worklist.pushAll(graph.predecessors(node));
181  }
182  };
183 
184  for (unsigned i = 0; i < graph.numNodes(); ++i) {
185  if (typename Graph::Node node = graph.node(i)) {
186  if (!graph.successors(node).size())
187  addRootSuccessor(node);
188  }
189  }
190 
191  // At this point there will be some nodes in the graph that aren't known to the worklist. We
192  // could add any or all of them to the root successors list. Adding all of them would be a bad
193  // pessimisation. Ideally we would pick the ones that have backward edges but no forward
194  // edges. That would require thinking, so we just use a rough heuristic: add the highest
195  // numbered nodes first, which is totally fine if the input program is already sorted nicely.
196  for (unsigned i = graph.numNodes(); i--;) {
197  if (typename Graph::Node node = graph.node(i))
198  addRootSuccessor(node);
199  }
200  }
