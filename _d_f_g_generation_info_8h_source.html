38 // === GenerationInfo ===
39 //
40 // This class is used to track the current status of live values during code generation.
41 // Can provide information as to whether a value is in machine registers, and if so which,
42 // whether a value has been spilled to the RegisterFile, and if so may be able to provide
43 // details of the format in memory (all values are spilled in a boxed form, but we may be
44 // able to track the type of box), and tracks how many outstanding uses of a value remain,
45 // so that we know when the value is dead and the machine registers associated with it
46 // may be released.

153  void noticeOSRBirth(VariableEventStream& stream, Node* node, VirtualRegister virtualRegister)
154  {
155  if (m_node != node)
156  return;
157  if (!alive())
158  return;
159  if (m_bornForOSR)
160  return;

162  m_bornForOSR = true;
163 
164  if (m_isConstant)
165  appendBirth(stream);
166  else if (m_registerFormat != DataFormatNone)
167  appendFill(BirthToFill, stream);
168  else if (m_spillFormat != DataFormatNone)
169  appendSpill(BirthToSpill, stream, virtualRegister);

272  // Called when a VirtualRegister is being spilled to the JSStack for the first time.
273  void spill(VariableEventStream& stream, VirtualRegister virtualRegister, DataFormat spillFormat)
274  {
275  // We shouldn't be spill values that don't need spilling.
276  ASSERT(!m_canFill);
277  ASSERT(m_spillFormat == DataFormatNone);
278  // We should only be spilling values that are currently in machine registers.
279  ASSERT(m_registerFormat != DataFormatNone);

380  ValueRecovery recovery(VirtualRegister spillSlot) const
381  {
382  if (m_isConstant)
383  return ValueRecovery::constant(m_node->constant()->value());
384 
385  if (m_registerFormat == DataFormatDouble)
386  return ValueRecovery::inFPR(u.fpr, DataFormatDouble);
387 
388 #if USE(JSVALUE32_64)

391  return ValueRecovery::inPair(u.v.tagGPR, u.v.payloadGPR);

394 #endif

396  return ValueRecovery::inGPR(u.gpr, m_registerFormat);
