44 // Note: WeakReference is an implementation detail, and should not be used directly.
45 template<typename T>
46 class WeakReference : public ThreadSafeRefCounted<WeakReference<T>> {
47  WTF_MAKE_NONCOPYABLE(WeakReference<T>);
48  WTF_MAKE_FAST_ALLOCATED;
49 public:
50  T* get() const
51  {
52 #if USE(WEB_THREAD)
53  ASSERT(canAccessThreadLocalDataForThread(m_boundThread));
54 #else
55  ASSERT(m_boundThread == currentThread());
56 #endif
57  return m_ptr;
58  }
59 
60  void clear()
61  {
62 #if USE(WEB_THREAD)
63  ASSERT(canAccessThreadLocalDataForThread(m_boundThread));
64 #else
65  ASSERT(m_boundThread == currentThread());
66 #endif
67  m_ptr = nullptr;
68  }
69 
70 private:
71  friend class WeakPtr<T>;
72  friend class WeakPtrFactory<T>;
73 
74  static Ref<WeakReference<T>> create(T* ptr) { return adoptRef(*new WeakReference(ptr)); }
75 
76  explicit WeakReference(T* ptr)
77  : m_ptr(ptr)
78 #ifndef NDEBUG
79  , m_boundThread(currentThread())
80 #endif
81  {
82  }
83 
84  T* m_ptr;
85 #ifndef NDEBUG
86  ThreadIdentifier m_boundThread;
87 #endif
88 };
