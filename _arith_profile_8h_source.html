69 struct ArithProfile {
70 private:
71  static const uint32_t numberOfFlagBits = 5;
72  static const uint32_t rhsResultTypeShift = numberOfFlagBits;
73  static const uint32_t lhsResultTypeShift = rhsResultTypeShift + ResultType::numBitsNeeded;
74  static const uint32_t rhsObservedTypeShift = lhsResultTypeShift + ResultType::numBitsNeeded;
75  static const uint32_t lhsObservedTypeShift = rhsObservedTypeShift + ObservedType::numBitsNeeded;
76 
77  static_assert(ObservedType::numBitsNeeded == 3, "We make a hard assumption about that here.");
78  static const uint32_t clearRhsObservedTypeBitMask = static_cast<uint32_t>(~((1 << rhsObservedTypeShift) | (1 << (rhsObservedTypeShift + 1)) | (1 << (rhsObservedTypeShift + 2))));
79  static const uint32_t clearLhsObservedTypeBitMask = static_cast<uint32_t>(~((1 << lhsObservedTypeShift) | (1 << (lhsObservedTypeShift + 1)) | (1 << (lhsObservedTypeShift + 2))));
80 
81  static const uint32_t resultTypeMask = (1 << ResultType::numBitsNeeded) - 1;
82  static const uint32_t observedTypeMask = (1 << ObservedType::numBitsNeeded) - 1;
83 public:
84  static const uint32_t specialFastPathBit = 1 << (lhsObservedTypeShift + ObservedType::numBitsNeeded);
85  static_assert((lhsObservedTypeShift + ObservedType::numBitsNeeded) <= (sizeof(uint32_t) * 8) - 1, "Should fit in a uint32_t.");
86  static_assert(!(specialFastPathBit & ~clearLhsObservedTypeBitMask), "These bits should not intersect.");
87  static_assert(specialFastPathBit & clearLhsObservedTypeBitMask, "These bits should intersect.");
88  static_assert(specialFastPathBit > ~clearLhsObservedTypeBitMask, "These bits should not intersect and specialFastPathBit should be a higher bit.");
89 
90  ArithProfile(ResultType arg)
91  {
92  m_bits = (arg.bits() << lhsResultTypeShift);
93  ASSERT(lhsResultType().bits() == arg.bits());
94  ASSERT(lhsObservedType().isEmpty());
95  ASSERT(rhsObservedType().isEmpty());
96  }
97 
98  ArithProfile(ResultType lhs, ResultType rhs)
99  {
100  m_bits = (lhs.bits() << lhsResultTypeShift) | (rhs.bits() << rhsResultTypeShift);
101  ASSERT(lhsResultType().bits() == lhs.bits() && rhsResultType().bits() == rhs.bits());
102  ASSERT(lhsObservedType().isEmpty());
103  ASSERT(rhsObservedType().isEmpty());
104  }
105  ArithProfile() = default;
