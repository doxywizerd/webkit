38 // ECMA 262-3 8.6.1
39 // Property attributes
40 enum Attribute {
41  None = 0,
42  ReadOnly = 1 << 1, // property can be only read, not written
43  DontEnum = 1 << 2, // property doesn't appear in (for .. in ..)
44  DontDelete = 1 << 3, // property can't be deleted
45  Accessor = 1 << 4, // property is a getter/setter
46  CustomAccessor = 1 << 5,
47 
48  // Things that are used by static hashtables are not in the attributes byte in PropertyMapEntry.
49  Function = 1 << 8, // property is a function - only used by static hashtables
50  Builtin = 1 << 9, // property is a builtin function - only used by static hashtables
51  ConstantInteger = 1 << 10, // property is a constant integer - only used by static hashtables
52  CellProperty = 1 << 11, // property is a lazy property - only used by static hashtables
53  ClassStructure = 1 << 12, // property is a lazy class structure - only used by static hashtables
54  PropertyCallback = 1 << 13, // property that is a lazy property callback - only used by static hashtables
55  DOMJITAttribute = 1 << 14, // property is a DOM JIT attribute - only used by static hashtables
56  DOMJITFunction = 1 << 15, // property is a DOM JIT function - only used by static hashtables
57  BuiltinOrFunction = Builtin | Function, // helper only used by static hashtables
58  BuiltinOrFunctionOrLazyProperty = Builtin | Function | CellProperty | ClassStructure | PropertyCallback, // helper only used by static hashtables
59  BuiltinOrFunctionOrAccessorOrLazyProperty = Builtin | Function | Accessor | CellProperty | ClassStructure | PropertyCallback, // helper only used by static hashtables
60  BuiltinOrFunctionOrAccessorOrLazyPropertyOrConstant = Builtin | Function | Accessor | CellProperty | ClassStructure | PropertyCallback | ConstantInteger // helper only used by static hashtables
61 };
