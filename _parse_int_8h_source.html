35 ALWAYS_INLINE static int parseDigit(unsigned short c, int radix)
36 {
37  int digit = -1;
38 
39  if (isASCIIDigit(c))
40  digit = c - '0';

42  digit = c - 'A' + 10;
43  else if (isASCIILower(c))
44  digit = c - 'a' + 10;
45 
46  if (digit >= radix)
47  return -1;
48  return digit;

52 {
53  double number = 0.0;
54  double radixMultiplier = 1.0;
55 
56  for (const LChar* p = s + length - 1; p >= s; p--) {

124 template <typename CharType>
125 ALWAYS_INLINE
126 static double parseInt(StringView s, const CharType* data, int radix)
127 {
128  // 1. Let inputString be ToString(string).
129  // 2. Let S be a newly created substring of inputString consisting of the first character that is not a
130  // StrWhiteSpaceChar and all characters following that character. (In other words, remove leading white
131  // space.) If inputString does not contain any such characters, let S be the empty string.
132  int length = s.length();
133  int p = 0;
134  while (p < length && isStrWhiteSpace(data[p]))
135  ++p;
136 
137  // 3. Let sign be 1.
138  // 4. If S is not empty and the first character of S is a minus sign -, let sign be -1.
139  // 5. If S is not empty and the first character of S is a plus sign + or a minus sign -, then remove the first character from S.
140  double sign = 1;
141  if (p < length) {
142  if (data[p] == '+')
143  ++p;
144  else if (data[p] == '-') {
145  sign = -1;
146  ++p;
147  }
148  }
149 
150  // 6. Let R = ToInt32(radix).
151  // 7. Let stripPrefix be true.
152  // 8. If R != 0,then
153  // b. If R != 16, let stripPrefix be false.
154  // 9. Else, R == 0
155  // a. LetR = 10.
156  // 10. If stripPrefix is true, then
157  // a. If the length of S is at least 2 and the first two characters of S are either ―0x or ―0X,
158  // then remove the first two characters from S and let R = 16.
159  // 11. If S contains any character that is not a radix-R digit, then let Z be the substring of S
160  // consisting of all characters before the first such character; otherwise, let Z be S.
161  if ((radix == 0 || radix == 16) && length - p >= 2 && data[p] == '0' && (data[p + 1] == 'x' || data[p + 1] == 'X')) {
162  radix = 16;
163  p += 2;
164  } else if (radix == 0)
165  radix = 10;
166 
167  // 8.a If R < 2 or R > 36, then return NaN.
168  if (radix < 2 || radix > 36)
169  return PNaN;
170 
171  // 13. Let mathInt be the mathematical integer value that is represented by Z in radix-R notation, using the letters
172  // A-Z and a-z for digits with values 10 through 35. (However, if R is 10 and Z contains more than 20 significant
173  // digits, every significant digit after the 20th may be replaced by a 0 digit, at the option of the implementation;
174  // and if R is not 2, 4, 8, 10, 16, or 32, then mathInt may be an implementation-dependent approximation to the
175  // mathematical integer value that is represented by Z in radix-R notation.)
176  // 14. Let number be the Number value for mathInt.
177  int firstDigitPosition = p;
178  bool sawDigit = false;
179  double number = 0;
180  while (p < length) {
181  int digit = parseDigit(data[p], radix);
182  if (digit == -1)
183  break;
184  sawDigit = true;
185  number *= radix;
186  number += digit;
187  ++p;
188  }
189 
190  // 12. If Z is empty, return NaN.
191  if (!sawDigit)
192  return PNaN;
193 
194  // Alternate code path for certain large numbers.
195  if (number >= mantissaOverflowLowerBound) {
196  if (radix == 10) {
197  size_t parsedLength;
198  number = parseDouble(s.substring(firstDigitPosition, p - firstDigitPosition), parsedLength);
199  } else if (radix == 2 || radix == 4 || radix == 8 || radix == 16 || radix == 32)
200  number = parseIntOverflow(s.substring(firstDigitPosition, p - firstDigitPosition), radix);
201  }
202 
203  // 15. Return sign x number.
204  return sign * number;
205 }
