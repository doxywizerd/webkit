22 // This is a work around for the Chrome tab full-screen behavior: Chrome
23 // creates a new window in full-screen mode to show a tab full-screen and
24 // minimizes the old window. To continue capturing in this case, we try to
25 // find the new full-screen window using these criteria:
26 // 0. The original shared window is minimized.
27 // 1. The original shared window's owner application name is "Google Chrome".
28 // 2. The original window and the new window have the same title and owner
29 // pid.
30 // 3. The new window is full-screen.
31 // 4. The new window didn't exist at least 500 millisecond ago.
32 
33 class FullScreenChromeWindowDetector {
34  public:
35  FullScreenChromeWindowDetector();
36 
37  void AddRef() { ++ref_count_; }
38  void Release() {
39  if (--ref_count_ == 0)
40  delete this;
41  }
42 
43  // Returns the full-screen window in place of the original window if all the
44  // criteria are met, or kCGNullWindowID if no such window found.
45  CGWindowID FindFullScreenWindow(CGWindowID original_window);
46 
47  // The caller should call this function periodically, no less than twice per
48  // second.
49  void UpdateWindowListIfNeeded(CGWindowID original_window);
50 
51  private:
52  ~FullScreenChromeWindowDetector();
53 
54  Atomic32 ref_count_;
55 
56  // We cache the last two results of the window list, so
57  // |previous_window_list_| is taken at least 500ms before the next Capture()
58  // call. If we only save the last result, we may get false positive (i.e.
59  // full-screen window exists in the list) if Capture() is called too soon.
60  DesktopCapturer::SourceList current_window_list_;
61  DesktopCapturer::SourceList previous_window_list_;
62  int64_t last_update_time_ns_;
63 
64  RTC_DISALLOW_COPY_AND_ASSIGN(FullScreenChromeWindowDetector);
65 };
