78 struct TriggerHash {

80  {
81  unsigned hash = trigger.urlFilterIsCaseSensitive ? 10619863 : 40960001;
82  if (!trigger.urlFilter.isNull())
83  hash ^= StringHash::hash(trigger.urlFilter);
84  hash = WTF::pairIntHash(hash, DefaultHash<ResourceFlags>::Hash::hash(trigger.flags));
