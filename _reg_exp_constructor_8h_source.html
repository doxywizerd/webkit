95 /*
96  To facilitate result caching, exec(), test(), match(), search(), and replace() dipatch regular
97  expression matching through the performMatch function. We use cached results to calculate,
98  e.g., RegExp.lastMatch and RegExp.leftParen.
99 */
100 ALWAYS_INLINE MatchResult RegExpConstructor::performMatch(VM& vm, RegExp* regExp, JSString* string, const String& input, int startOffset, int** ovector)
101 {
102  int position = regExp->match(vm, input, startOffset, m_ovector);
103 
104  if (ovector)
105  *ovector = m_ovector.data();
106 
107  if (position == -1)
108  return MatchResult::failed();
109 
110  ASSERT(!m_ovector.isEmpty());
111  ASSERT(m_ovector[0] == position);
112  ASSERT(m_ovector[1] >= position);
113  size_t end = m_ovector[1];
114 
115  m_cachedResult.record(vm, this, regExp, string, MatchResult(position, end));
116 
117  return MatchResult(position, end);
118 }
119 ALWAYS_INLINE MatchResult RegExpConstructor::performMatch(VM& vm, RegExp* regExp, JSString* string, const String& input, int startOffset)
120 {
121  MatchResult result = regExp->match(vm, input, startOffset);
122  if (result)
123  m_cachedResult.record(vm, this, regExp, string, result);
124  return result;
125 }

133 ALWAYS_INLINE bool isRegExp(VM& vm, ExecState* exec, JSValue value)
134 {
135  auto scope = DECLARE_THROW_SCOPE(vm);
136  if (!value.isObject())
137  return false;
138 
139  JSObject* object = asObject(value);
140  JSValue matchValue = object->get(exec, vm.propertyNames->matchSymbol);
141  RETURN_IF_EXCEPTION(scope, false);
142  if (!matchValue.isUndefined())
143  return matchValue.toBoolean(exec);
144 
145  return object->inherits(vm, RegExpObject::info());
