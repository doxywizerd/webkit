34 // VisiblePosition default affinity is downstream because
35 // the callers do not really care (they just want the
36 // deep position without regard to line position), and this
37 // is cheaper than UPSTREAM
38 #define VP_DEFAULT_AFFINITY DOWNSTREAM
39 
40 // Callers who do not know where on the line the position is,
41 // but would like UPSTREAM if at a line break or DOWNSTREAM
42 // otherwise, need a clear way to specify that. The
43 // constructors auto-correct UPSTREAM to DOWNSTREAM if the
44 // position is not at a line break.
45 #define VP_UPSTREAM_IF_POSSIBLE UPSTREAM
