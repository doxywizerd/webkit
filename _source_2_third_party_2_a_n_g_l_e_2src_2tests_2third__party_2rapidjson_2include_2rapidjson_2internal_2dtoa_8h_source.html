39 inline void GrisuRound(char* buffer, int len, uint64_t delta, uint64_t rest, uint64_t ten_kappa, uint64_t wp_w) {
40  while (rest < wp_w && delta - rest >= ten_kappa &&
41  (rest + ten_kappa < wp_w ||
42  wp_w - rest > rest + ten_kappa - wp_w)) {
43  buffer[len - 1]--;
44  rest += ten_kappa;
45  }
46 }

48 inline unsigned CountDecimalDigit32(uint32_t n) {
49  // Simple pure C++ implementation was faster than __builtin_clz version in this situation.
50  if (n < 10) return 1;
51  if (n < 100) return 2;
52  if (n < 1000) return 3;
53  if (n < 10000) return 4;
54  if (n < 100000) return 5;
55  if (n < 1000000) return 6;
56  if (n < 10000000) return 7;
57  if (n < 100000000) return 8;
58  if (n < 1000000000) return 9;
59  return 10;
60 }

62 inline void DigitGen(const DiyFp& W, const DiyFp& Mp, uint64_t delta, char* buffer, int* len, int* K) {
63  static const uint32_t kPow10[] = { 1, 10, 100, 1000, 10000, 100000, 1000000, 10000000, 100000000, 1000000000 };
64  const DiyFp one(uint64_t(1) << -Mp.e, Mp.e);
65  const DiyFp wp_w = Mp - W;
66  uint32_t p1 = static_cast<uint32_t>(Mp.f >> -one.e);
67  uint64_t p2 = Mp.f & (one.f - 1);
68  int kappa = CountDecimalDigit32(p1);
69  *len = 0;
70 
71  while (kappa > 0) {
72  uint32_t d;
73  switch (kappa) {
74  case 10: d = p1 / 1000000000; p1 %= 1000000000; break;
75  case 9: d = p1 / 100000000; p1 %= 100000000; break;
76  case 8: d = p1 / 10000000; p1 %= 10000000; break;
77  case 7: d = p1 / 1000000; p1 %= 1000000; break;
78  case 6: d = p1 / 100000; p1 %= 100000; break;
79  case 5: d = p1 / 10000; p1 %= 10000; break;
80  case 4: d = p1 / 1000; p1 %= 1000; break;
81  case 3: d = p1 / 100; p1 %= 100; break;
82  case 2: d = p1 / 10; p1 %= 10; break;
83  case 1: d = p1; p1 = 0; break;
84  default:
85 #if defined(_MSC_VER)
86  __assume(0);
87 #elif __GNUC__ > 4 || (__GNUC__ == 4 && __GNUC_MINOR__ >= 5)
88  __builtin_unreachable();
89 #else
90  d = 0;
91 #endif
92  }
93  if (d || *len)
94  buffer[(*len)++] = static_cast<char>('0' + static_cast<char>(d));
95  kappa--;
96  uint64_t tmp = (static_cast<uint64_t>(p1) << -one.e) + p2;
97  if (tmp <= delta) {
98  *K += kappa;
99  GrisuRound(buffer, *len, delta, tmp, static_cast<uint64_t>(kPow10[kappa]) << -one.e, wp_w.f);
100  return;
101  }
102  }
103 
104  // kappa = 0
105  for (;;) {
106  p2 *= 10;
107  delta *= 10;
108  char d = static_cast<char>(p2 >> -one.e);
109  if (d || *len)
110  buffer[(*len)++] = static_cast<char>('0' + d);
111  p2 &= one.f - 1;
112  kappa--;
113  if (p2 < delta) {
114  *K += kappa;
115  GrisuRound(buffer, *len, delta, p2, one.f, wp_w.f * kPow10[-kappa]);
116  return;
117  }
118  }
119 }

121 inline void Grisu2(double value, char* buffer, int* length, int* K) {
122  const DiyFp v(value);
123  DiyFp w_m, w_p;
124  v.NormalizedBoundaries(&w_m, &w_p);
125 
126  const DiyFp c_mk = GetCachedPower(w_p.e, K);
127  const DiyFp W = v.Normalize() * c_mk;
128  DiyFp Wp = w_p * c_mk;
129  DiyFp Wm = w_m * c_mk;
130  Wm.f++;
131  Wp.f--;
132  DigitGen(W, Wp, Wp.f - Wm.f, buffer, length, K);
133 }

135 inline char* WriteExponent(int K, char* buffer) {
136  if (K < 0) {
137  *buffer++ = '-';
138  K = -K;
139  }
140 
141  if (K >= 100) {

143  K %= 100;
144  const char* d = GetDigitsLut() + K * 2;
145  *buffer++ = d[0];

150  *buffer++ = d[0];
151  *buffer++ = d[1];
152  }
153  else
154  *buffer++ = static_cast<char>('0' + static_cast<char>(K));
155 
156  return buffer;

159 inline char* Prettify(char* buffer, int length, int k) {
160  const int kk = length + k; // 10^(kk-1) <= v < 10^kk
161 
162  if (length <= kk && kk <= 21) {
163  // 1234e7 -> 12340000000
164  for (int i = length; i < kk; i++)
165  buffer[i] = '0';
166  buffer[kk] = '.';
167  buffer[kk + 1] = '0';
168  return &buffer[kk + 2];
169  }
170  else if (0 < kk && kk <= 21) {
171  // 1234e-2 -> 12.34
172  std::memmove(&buffer[kk + 1], &buffer[kk], length - kk);
173  buffer[kk] = '.';
174  return &buffer[length + 1];
175  }
176  else if (-6 < kk && kk <= 0) {
177  // 1234e-6 -> 0.001234
178  const int offset = 2 - kk;
179  std::memmove(&buffer[offset], &buffer[0], length);
180  buffer[0] = '0';
181  buffer[1] = '.';
182  for (int i = 2; i < offset; i++)
183  buffer[i] = '0';
184  return &buffer[length + offset];
185  }
186  else if (length == 1) {
187  // 1e30
188  buffer[1] = 'e';
189  return WriteExponent(kk - 1, &buffer[2]);
190  }
191  else {
192  // 1234e30 -> 1.234e33
193  std::memmove(&buffer[2], &buffer[1], length - 1);
194  buffer[1] = '.';
195  buffer[length + 1] = 'e';
196  return WriteExponent(kk - 1, &buffer[0 + length + 2]);
197  }
198 }

200 inline char* dtoa(double value, char* buffer) {
201  if (value == 0) {
202  buffer[0] = '0';
203  buffer[1] = '.';
204  buffer[2] = '0';
205  return &buffer[3];
206  }
207  else {
208  if (value < 0) {
209  *buffer++ = '-';
210  value = -value;
211  }
212  int length, K;
213  Grisu2(value, buffer, &length, &K);
214  return Prettify(buffer, length, K);
215  }
