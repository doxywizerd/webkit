43 // Some timing constants for optimal operation. See
44 // https://tango.0pointer.de/pipermail/pulseaudio-discuss/2008-January/001170.html
45 // for a good explanation of some of the factors that go into this.
46 
47 // Playback.
48 
49 // For playback, there is a round-trip delay to fill the server-side playback
50 // buffer, so setting too low of a latency is a buffer underflow risk. We will
51 // automatically increase the latency if a buffer underflow does occur, but we
52 // also enforce a sane minimum at start-up time. Anything lower would be
53 // virtually guaranteed to underflow at least once, so there's no point in
54 // allowing lower latencies.
55 const uint32_t WEBRTC_PA_PLAYBACK_LATENCY_MINIMUM_MSECS = 20;
56 
57 // Every time a playback stream underflows, we will reconfigure it with target
58 // latency that is greater by this amount.
59 const uint32_t WEBRTC_PA_PLAYBACK_LATENCY_INCREMENT_MSECS = 20;
60 
61 // We also need to configure a suitable request size. Too small and we'd burn
62 // CPU from the overhead of transfering small amounts of data at once. Too large
63 // and the amount of data remaining in the buffer right before refilling it
64 // would be a buffer underflow risk. We set it to half of the buffer size.
65 const uint32_t WEBRTC_PA_PLAYBACK_REQUEST_FACTOR = 2;
66 
67 // Capture.
68 
69 // For capture, low latency is not a buffer overflow risk, but it makes us burn
70 // CPU from the overhead of transfering small amounts of data at once, so we set
71 // a recommended value that we use for the kLowLatency constant (but if the user
72 // explicitly requests something lower then we will honour it).
73 // 1ms takes about 6-7% CPU. 5ms takes about 5%. 10ms takes about 4.x%.
74 const uint32_t WEBRTC_PA_LOW_CAPTURE_LATENCY_MSECS = 10;
75 
76 // There is a round-trip delay to ack the data to the server, so the
77 // server-side buffer needs extra space to prevent buffer overflow. 20ms is
78 // sufficient, but there is no penalty to making it bigger, so we make it huge.
79 // (750ms is libpulse's default value for the _total_ buffer size in the
80 // kNoLatencyRequirements case.)
81 const uint32_t WEBRTC_PA_CAPTURE_BUFFER_EXTRA_MSECS = 750;
82 
83 const uint32_t WEBRTC_PA_MSECS_PER_SEC = 1000;
84 
85 // Init _configuredLatencyRec/Play to this value to disable latency requirements
86 const int32_t WEBRTC_PA_NO_LATENCY_REQUIREMENTS = -1;
