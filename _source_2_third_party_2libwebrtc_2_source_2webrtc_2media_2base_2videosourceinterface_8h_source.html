21 struct VideoSinkWants {

23  // By default, any rotation must be applied by the sink.

27  bool black_frames = false;

30  rtc::Optional<int> max_pixel_count;
31  // Tells the source the desired number of pixels the sinks wants. This will
32  // typically be used when stepping the resolution up again when conditions
33  // have improved after an earlier downgrade. The source should select the
34  // closest resolution to this pixel count, but if max_pixel_count is set, it
35  // still sets the absolute upper bound.

37 };
