98 static inline bool nameShouldBeVisibleInDocumentAll(HTMLElement& element)
99 {
100  // The document.all collection returns only certain types of elements by name,
101  // although it returns any type of element by id.
102  return element.hasTagName(HTMLNames::appletTag)
103  || element.hasTagName(HTMLNames::embedTag)
104  || element.hasTagName(HTMLNames::formTag)
105  || element.hasTagName(HTMLNames::imgTag)
106  || element.hasTagName(HTMLNames::inputTag)
107  || element.hasTagName(HTMLNames::objectTag)
108  || element.hasTagName(HTMLNames::selectTag);
109 }

111 static inline bool nameShouldBeVisibleInDocumentAll(Element& element)
112 {
113  return is<HTMLElement>(element) && nameShouldBeVisibleInDocumentAll(downcast<HTMLElement>(element));

116 template <typename HTMLCollectionClass, CollectionTraversalType traversalType>
117 Element* CachedHTMLCollection<HTMLCollectionClass, traversalType>::namedItem(const AtomicString& name) const
118 {
119  // http://msdn.microsoft.com/workshop/author/dhtml/reference/methods/nameditem.asp
120  // This method first searches for an object with a matching id
121  // attribute. If a match is not found, the method then searches for an
122  // object with a matching name attribute, but only on those elements
123  // that are allowed a name attribute.
124 
125  if (name.isEmpty())
126  return nullptr;
127 
128  ContainerNode& root = rootNode();
129  if (traversalType != CollectionTraversalType::CustomForwardOnly && root.isInTreeScope()) {
130  Element* candidate = nullptr;
131 
132  TreeScope& treeScope = root.treeScope();
133  if (treeScope.hasElementWithId(*name.impl())) {
134  if (!treeScope.containsMultipleElementsWithId(name))
135  candidate = treeScope.getElementById(name);
136  } else if (treeScope.hasElementWithName(*name.impl())) {
137  if (!treeScope.containsMultipleElementsWithName(name)) {
138  if ((candidate = treeScope.getElementByName(name))) {
139  if (!is<HTMLElement>(*candidate))
140  candidate = nullptr;
141  else if (type() == DocAll && !nameShouldBeVisibleInDocumentAll(*candidate))
142  candidate = nullptr;
143  }
144  }
145  } else
146  return nullptr;
147 
148  if (candidate && collection().elementMatches(*candidate)) {
149  if (traversalType == CollectionTraversalType::ChildrenOnly ? candidate->parentNode() == &root : candidate->isDescendantOf(root))
150  return candidate;
151  }
152  }
153 
154  return namedItemSlow(name);
155 }
