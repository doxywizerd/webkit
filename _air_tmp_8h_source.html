39 // A Tmp is a generalization of a register. It can be used to refer to any GPR or FPR. It can also
40 // be used to refer to an unallocated register (i.e. a temporary). Like many Air classes, we use
41 // deliberately terse naming since we will have to use this name a lot.

252  static FPRReg decodeFPR(int value)
253  {
254  ASSERT(isEncodedFPR(value));
255  return static_cast<FPRReg>(
256  (encodeFPR(MacroAssembler::firstFPRegister()) - value) +
257  MacroAssembler::firstFPRegister());
258  }
259 
260  static unsigned decodeGPTmp(int value)
261  {
262  ASSERT(isEncodedGPTmp(value));
263  return value - (encodeGPR(MacroAssembler::lastRegister()) + 1);
264  }
265 
266  static unsigned decodeFPTmp(int value)
267  {
268  ASSERT(isEncodedFPTmp(value));
269  return (encodeFPR(MacroAssembler::lastFPRegister()) - 1) - value;
270  }

275  int m_value;
