38 enum ResourceRequestCachePolicy {
39  UseProtocolCachePolicy, // normal load, equivalent to fetch "default" cache mode.
40  ReloadIgnoringCacheData, // reload, equivalent to fetch "reload"cache mode.
41  ReturnCacheDataElseLoad, // back/forward or encoding change - allow stale data, equivalent to fetch "force-cache" cache mode.
42  ReturnCacheDataDontLoad, // results of a post - allow stale data and only use cache, equivalent to fetch "only-if-cached" cache mode.
43  DoNotUseAnyCache, // Bypass the cache entirely, equivalent to fetch "no-store" cache mode.
44  RefreshAnyCacheData, // Serve cache data only if revalidated, equivalent to fetch "no-cache" mode.
45 };

282 ALWAYS_INLINE bool ResourceRequestBase::decodeBase(Decoder& decoder)
283 {
284  if (!decoder.decode(m_url))
285  return false;

287  if (!decoder.decode(m_timeoutInterval))
288  return false;

292  return false;

294 
295  if (!decoder.decode(m_httpMethod))
296  return false;
297 
298  if (!decoder.decode(m_httpHeaderFields))
299  return false;
300 
301  if (!decoder.decode(m_responseContentDispositionEncodingFallbackArray))
302  return false;
303 
304  ResourceRequestCachePolicy cachePolicy;
305  if (!decoder.decodeEnum(cachePolicy))
306  return false;
307  m_cachePolicy = cachePolicy;
308 
309  bool allowCookies;
310  if (!decoder.decode(allowCookies))
311  return false;
312  m_allowCookies = allowCookies;
313 
314  ResourceLoadPriority priority;
315  if (!decoder.decodeEnum(priority))
316  return false;
317  m_priority = priority;
318 
319  if (!decoder.decodeEnum(m_requester))
320  return false;
321 
322  return true;
323 }
