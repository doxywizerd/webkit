36 // This is a mapping from nodes to values that is useful for flow-sensitive analysis. In such an
37 // analysis, at every point in the program we need to consider the values of nodes plus the shadow
38 // values of Phis. This makes it easy to do both of those things.
39 template<typename T>
40 class FlowMap {
41 public:
42  FlowMap(Graph& graph)
43  : m_graph(graph)
44  {
45  resize();
46  }
47 
48  // Call this if the number of nodes in the graph has changed. Note that this does not reset any
49  // entries.
50  void resize()
51  {
52  m_map.resize(m_graph.maxNodeCount());
53  m_shadowMap.resize(m_graph.maxNodeCount());
54  }
55 
56  Graph& graph() const { return m_graph; }
57 
58  T& at(unsigned nodeIndex)
59  {
60  return m_map[nodeIndex];
61  }
62 
63  T& at(Node* node)
64  {
65  return at(node->index());
66  }
67 
68  T& atShadow(unsigned nodeIndex)
69  {
70  return m_shadowMap[nodeIndex];
71  }
72 
73  T& atShadow(Node* node)
74  {
75  return atShadow(node->index());
76  }
77 
78  T& at(unsigned nodeIndex, NodeFlowProjection::Kind kind)
79  {
80  switch (kind) {
81  case NodeFlowProjection::Primary:
82  return at(nodeIndex);
83  case NodeFlowProjection::Shadow:
84  return atShadow(nodeIndex);
85  }
86  RELEASE_ASSERT_NOT_REACHED();
87  return *bitwise_cast<T*>(nullptr);
88  }
89 
90  T& at(Node* node, NodeFlowProjection::Kind kind)
91  {
92  return at(node->index(), kind);
93  }
94 
95  T& at(NodeFlowProjection projection)
96  {
97  return at(projection.node(), projection.kind());
98  }
99 
100  const T& at(unsigned nodeIndex) const { return const_cast<FlowMap*>(this)->at(nodeIndex); }
101  const T& at(Node* node) const { return const_cast<FlowMap*>(this)->at(node); }
102  const T& atShadow(unsigned nodeIndex) const { return const_cast<FlowMap*>(this)->atShadow(nodeIndex); }
103  const T& atShadow(Node* node) const { return const_cast<FlowMap*>(this)->atShadow(node); }
104  const T& at(unsigned nodeIndex, NodeFlowProjection::Kind kind) const { return const_cast<FlowMap*>(this)->at(nodeIndex, kind); }
105  const T& at(Node* node, NodeFlowProjection::Kind kind) const { return const_cast<FlowMap*>(this)->at(node, kind); }
106  const T& at(NodeFlowProjection projection) const { return const_cast<FlowMap*>(this)->at(projection); }
107 
108 private:
109  Graph& m_graph;
110  Vector<T, 0, UnsafeVectorOverflow> m_map;
111  Vector<T, 0, UnsafeVectorOverflow> m_shadowMap;
112 };

119 void printInternal(PrintStream& out, const JSC::DFG::FlowMap<T>& map)
120 {
121  CommaPrinter comma;
122  for (unsigned i = 0; i < map.graph().maxNodeCount(); ++i) {
123  if (JSC::DFG::Node* node = map.graph().nodeAt(i)) {
124  if (const T& value = map.at(node))
125  out.print(comma, node, "=>", value);
126  }
127  }
128  for (unsigned i = 0; i < map.graph().maxNodeCount(); ++i) {
129  if (JSC::DFG::Node* node = map.graph().nodeAt(i)) {
130  if (const T& value = map.atShadow(node))
131  out.print(comma, "shadow(", node, ")=>", value);
132  }
133  }
134 }
