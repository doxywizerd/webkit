35 // Custom pool allocator for exactly one type (type T). It has fast (O(1), only a few
36 // instructions) allocator, and a similarly fast free(). Recycling works if either of
37 // the following is true:
38 // - T has a trivial destructor. In that case you don't have to ever call free() on
39 // anything. You can just call freeAll() instead.
40 // - You call free() on all T's that you allocated, and never use freeAll().
41 
42 template<typename T>
43 class Allocator {
44 public:
45  Allocator();
46  ~Allocator();
47 
48  void* allocate(); // Use placement new to allocate, and avoid using this method.
49  void free(T*); // Call this method to delete; never use 'delete' directly.
50 
51  void freeAll(); // Only call this if you've either freed everything or if T has a trivial destructor.
52  void reset(); // Like freeAll(), but also returns all memory to the OS.
53 
54  unsigned indexOf(const T*);
55 
56  static Allocator* allocatorOf(const T*);
57 
58 private:
59  void* bumpAllocate();
60  void* freeListAllocate();
61  void* allocateSlow();

63  struct Region {
64  static size_t size() { return 64 * KB; }
65  static size_t headerSize() { return std::max(sizeof(Region), sizeof(T)); }
66  static unsigned numberOfThingsPerRegion() { return (size() - headerSize()) / sizeof(T); }
67  T* data() { return bitwise_cast<T*>(bitwise_cast<char*>(this) + headerSize()); }
68  bool isInThisRegion(const T* pointer) { return static_cast<unsigned>(pointer - data()) < numberOfThingsPerRegion(); }
69  static Region* regionFor(const T* pointer) { return bitwise_cast<Region*>(bitwise_cast<uintptr_t>(pointer) & ~(size() - 1)); }
70 
71  void* m_allocation;
72  Allocator* m_allocator;
73  Region* m_next;
74  };

77  void startBumpingIn(Region*);
78 
79  Region* m_regionHead;
80  void** m_freeListHead;
81  T* m_bumpEnd;
82  unsigned m_bumpRemaining;
83 };

118 template<typename T>
119 void Allocator<T>::freeAll()
120 {
121  if (!m_regionHead) {
122  ASSERT(!m_bumpRemaining);
123  ASSERT(!m_freeListHead);
124  return;
125  }
126 
127  // Since the caller is opting out of calling the destructor for any allocated thing,
128  // we have two choices, plus a continuum between: we can either just delete all regions
129  // (i.e. call reset()), or we can make all regions available for reuse. We do something
130  // that optimizes for (a) speed of freeAll(), (b) the assumption that if the user calls
131  // freeAll() then they will probably be calling allocate() in the near future. Namely,
132  // we free all but one region, and make the remaining region a bump allocation region.
133 
134  freeRegionsStartingAt(m_regionHead->m_next);
135 
136  m_regionHead->m_next = 0;
137  m_freeListHead = 0;
138  startBumpingIn(m_regionHead);
139 }

151 template<typename T>
152 unsigned Allocator<T>::indexOf(const T* object)
153 {
154  unsigned numRegions = 0;
155  for (Region* region = m_regionHead; region; region = region->m_next)
156  numRegions++;
157  unsigned regionIndex = 0;
158  for (Region* region = m_regionHead; region; region = region->m_next) {
159  if (region->isInThisRegion(object))
160  return (numRegions - 1 - regionIndex) * Region::numberOfThingsPerRegion() + (object - region->data());
161  regionIndex++;
162  }
163  CRASH();
164  return 0;
165 }

175 {

186 {

194 template<typename T>
195 void* Allocator<T>::allocateSlow()
196 {
197  ASSERT(!m_freeListHead);
198  ASSERT(!m_bumpRemaining);
199 
200  if (logCompilationChanges())
201  dataLog("Allocating another allocator region.\n");
202 
203  void* allocation = fastAlignedMalloc(Region::size(), Region::size());
204  Region* region = static_cast<Region*>(allocation);
205  region->m_allocation = allocation;
206  region->m_allocator = this;
207  startBumpingIn(region);
208  region->m_next = m_regionHead;
209  m_regionHead = region;
210 
211  void* result = bumpAllocate();
212  ASSERT(result);
213  return result;
214 }

216 template<typename T>
217 void Allocator<T>::freeRegionsStartingAt(typename Allocator<T>::Region* region)
218 {
219  while (region) {
220  Region* nextRegion = region->m_next;
221  fastAlignedFree(region->m_allocation);
222  region = nextRegion;
