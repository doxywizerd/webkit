273 inline PropertyTable::find_iterator PropertyTable::find(const KeyType& key)
274 {
275  ASSERT(key);
276  ASSERT(key->isAtomic() || key->isSymbol());
277  unsigned hash = IdentifierRepHash::hash(key);
278 
279 #if DUMP_PROPERTYMAP_STATS
280  ++propertyMapHashTableStats->numFinds;
281 #endif
282 
283  while (true) {
284  unsigned entryIndex = m_index[hash & m_indexMask];
285  if (entryIndex == EmptyEntryIndex)
286  return std::make_pair((ValueType*)0, hash & m_indexMask);
287  if (key == table()[entryIndex - 1].key)
288  return std::make_pair(&table()[entryIndex - 1], hash & m_indexMask);
289 
290 #if DUMP_PROPERTYMAP_STATS
291  ++propertyMapHashTableStats->numCollisions;
292 #endif
293 
294 #if DUMP_PROPERTYMAP_COLLISIONS
295  dataLog("PropertyTable collision for ", key, " (", hash, ")\n");
296  dataLog("Collided with ", table()[entryIndex - 1].key, "(", IdentifierRepHash::hash(table()[entryIndex - 1].key), ")\n");
297 #endif
298 
299  hash++;
300  }
301 }

303 inline PropertyTable::ValueType* PropertyTable::get(const KeyType& key)
304 {
305  ASSERT(key);
306  ASSERT(key->isAtomic() || key->isSymbol());
307 
308  if (!m_keyCount)
309  return nullptr;
310 
311  unsigned hash = IdentifierRepHash::hash(key);
312 
313 #if DUMP_PROPERTYMAP_STATS
314  ++propertyMapHashTableStats->numLookups;
315 #endif
316 
317  while (true) {
318  unsigned entryIndex = m_index[hash & m_indexMask];
319  if (entryIndex == EmptyEntryIndex)
320  return nullptr;
321  if (key == table()[entryIndex - 1].key)
322  return &table()[entryIndex - 1];
323 
324 #if DUMP_PROPERTYMAP_STATS
325  ++propertyMapHashTableStats->numLookupProbing;
326 #endif
327 
328  hash++;

332 inline std::pair<PropertyTable::find_iterator, bool> PropertyTable::add(const ValueType& entry, PropertyOffset& offset, EffectOnPropertyOffset offsetEffect)
333 {
334  // Look for a value with a matching key already in the array.
335  find_iterator iter = find(entry.key);
336  if (iter.first) {
337  RELEASE_ASSERT(iter.first->offset <= offset);
338  return std::make_pair(iter, false);
339  }
340 
341 #if DUMP_PROPERTYMAP_STATS
342  ++propertyMapHashTableStats->numAdds;
343 #endif
344 
345  // Ref the key
346  entry.key->ref();
347 
348  // ensure capacity is available.
349  if (!canInsert()) {
350  rehash(m_keyCount + 1);
351  iter = find(entry.key);
352  ASSERT(!iter.first);
353  }
354 
355  // Allocate a slot in the hashtable, and set the index to reference this.
356  unsigned entryIndex = usedCount() + 1;
357  m_index[iter.second] = entryIndex;
358  iter.first = &table()[entryIndex - 1];
359  *iter.first = entry;
360 
361  ++m_keyCount;
362 
363  if (offsetEffect == PropertyOffsetMayChange)
364  offset = std::max(offset, entry.offset);
365  else
366  RELEASE_ASSERT(offset >= entry.offset);
367 
368  return std::make_pair(iter, true);
369 }

371 inline void PropertyTable::remove(const find_iterator& iter)
372 {
373  // Removing a key that doesn't exist does nothing!
374  if (!iter.first)
375  return;
376 
377 #if DUMP_PROPERTYMAP_STATS
378  ++propertyMapHashTableStats->numRemoves;
379 #endif
380 
381  // Replace this one element with the deleted sentinel. Also clear out
382  // the entry so we can iterate all the entries as needed.
383  m_index[iter.second] = deletedEntryIndex();
384  iter.first->key->deref();
385  iter.first->key = PROPERTY_MAP_DELETED_ENTRY_KEY;
386 
387  ASSERT(m_keyCount >= 1);
388  --m_keyCount;
389  ++m_deletedCount;
390 
391  if (m_deletedCount * 4 >= m_indexSize)
392  rehash(m_keyCount);
393 }

488 inline void PropertyTable::rehash(unsigned newCapacity)
489 {
490 #if DUMP_PROPERTYMAP_STATS
491  ++propertyMapHashTableStats->numRehashes;
492 #endif
493 
494  unsigned* oldEntryIndices = m_index;
495  iterator iter = this->begin();
496  iterator end = this->end();
497 
498  m_indexSize = sizeForCapacity(newCapacity);
499  m_indexMask = m_indexSize - 1;
500  m_keyCount = 0;
501  m_deletedCount = 0;
502  m_index = static_cast<unsigned*>(fastZeroedMalloc(dataSize()));
503 
504  for (; iter != end; ++iter) {
505  ASSERT(canInsert());
506  reinsert(*iter);
507  }
508 
509  fastFree(oldEntryIndices);
510 }
