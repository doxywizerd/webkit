57 // Each state may transition to the state immediately above or below it.
58 // SessionState transitions are only allowed when SegmentState is Unloaded.
59 enum class SessionState {
60  Capturing,
61  // Neither capturing or replaying. m_currentPosition is not valid in this state.
62  Inactive,
63  Replaying,
64 };
65 
66 // Each state may transition to the state immediately above or below it.
67 enum class SegmentState {
68  // Inputs can be appended into an unassociated session segment.
69  // We can stop capturing, which reverts to the Unloaded state.
70  Appending,
71  // No session segment is loaded.
72  // We can start capturing, or load a segment (and then replay it).
73  Unloaded,
74  // A session segment is loaded.
75  // We can unload the segment, or begin playback from m_currentPosition.
76  Loaded,
77  // The controller is actively dispatching event loop inputs.
78  // We can pause or cancel playback, which reverts to the Loaded state.
79  Dispatching,
80 };
