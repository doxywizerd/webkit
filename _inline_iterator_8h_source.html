150 template <class Observer>
151 static inline void notifyObserverEnteredObject(Observer* observer, RenderObject* object)
152 {
153  if (!observer || !object || !object->isRenderInline())
154  return;
155 
156  const RenderStyle& style = object->style();
157  EUnicodeBidi unicodeBidi = style.unicodeBidi();
158  if (unicodeBidi == UBNormal) {
159  // http://dev.w3.org/csswg/css3-writing-modes/#unicode-bidi
160  // "The element does not open an additional level of embedding with respect to the bidirectional algorithm."
161  // Thus we ignore any possible dir= attribute on the span.
162  return;
163  }
164  if (isIsolated(unicodeBidi)) {
165  // Make sure that explicit embeddings are committed before we enter the isolated content.
166  observer->commitExplicitEmbedding();
167  observer->enterIsolate();
168  // Embedding/Override characters implied by dir= will be handled when
169  // we process the isolated span, not when laying out the "parent" run.
170  return;
171  }
172 
173  if (!observer->inIsolate())
174  observer->embed(embedCharFromDirection(style.direction(), unicodeBidi), FromStyleOrDOM);
175 }

177 template <class Observer>
178 static inline void notifyObserverWillExitObject(Observer* observer, RenderObject* object)
179 {
180  if (!observer || !object || !object->isRenderInline())
181  return;
182 
183  EUnicodeBidi unicodeBidi = object->style().unicodeBidi();
184  if (unicodeBidi == UBNormal)
185  return; // Nothing to do for unicode-bidi: normal
186  if (isIsolated(unicodeBidi)) {
187  observer->exitIsolate();
188  return;
189  }
190 
191  // Otherwise we pop any embed/override character we added when we opened this tag.
192  if (!observer->inIsolate())
193  observer->embed(U_POP_DIRECTIONAL_FORMAT, FromStyleOrDOM);
194 }

208 static bool isEmptyInline(const RenderInline& renderer)
209 {
210  for (auto& current : childrenOfType<RenderObject>(renderer)) {
211  if (current.isFloatingOrOutOfFlowPositioned())
212  continue;
213  if (is<RenderText>(current)) {

224 // FIXME: This function is misleadingly named. It has little to do with bidi.
225 // This function will iterate over inlines within a block, optionally notifying
226 // a bidi resolver as it enters/exits inlines (so it can push/pop embedding levels).
227 template <class Observer>
228 static inline RenderObject* bidiNextShared(RenderElement& root, RenderObject* current, Observer* observer = nullptr, EmptyInlineBehavior emptyInlineBehavior = SkipEmptyInlines, bool* endOfInlinePtr = nullptr)
229 {
230  RenderObject* next = nullptr;
231  // oldEndOfInline denotes if when we last stopped iterating if we were at the end of an inline.
232  bool oldEndOfInline = endOfInlinePtr ? *endOfInlinePtr : false;
233  bool endOfInline = false;
234 
235  while (current) {
236  next = nullptr;
237  if (!oldEndOfInline && !isIteratorTarget(current)) {
238  next = downcast<RenderElement>(*current).firstChild();
239  notifyObserverEnteredObject(observer, next);
240  }
241 
242  // We hit this when either current has no children, or when current is not a renderer we care about.
243  if (!next) {
244  // If it is a renderer we care about, and we're doing our inline-walk, return it.
245  if (emptyInlineBehavior == IncludeEmptyInlines && !oldEndOfInline && is<RenderInline>(*current)) {
246  next = current;
247  endOfInline = true;
248  break;
249  }
250 
251  while (current && current != &root) {
252  notifyObserverWillExitObject(observer, current);
253 
254  next = current->nextSibling();
255  if (next) {
256  notifyObserverEnteredObject(observer, next);
257  break;
258  }
259 
260  current = current->parent();
261  if (emptyInlineBehavior == IncludeEmptyInlines && current && current != &root && is<RenderInline>(*current)) {
262  next = current;
263  endOfInline = true;
264  break;
265  }
266  }
267  }
268 
269  if (!next)
270  break;
271 
272  if (isIteratorTarget(next)
273  || (is<RenderInline>(*next) && (emptyInlineBehavior == IncludeEmptyInlines || isEmptyInline(downcast<RenderInline>(*next)))))
274  break;
275  current = next;
276  }
277 
278  if (endOfInlinePtr)
279  *endOfInlinePtr = endOfInline;
280 
281  return next;
282 }

304 static inline RenderObject* bidiFirstSkippingEmptyInlines(RenderElement& root, InlineBidiResolver* resolver = nullptr)
305 {
306  RenderObject* renderer = root.firstChild();
307  if (!renderer)
308  return nullptr;
309 
310  if (is<RenderInline>(*renderer)) {
311  notifyObserverEnteredObject(resolver, renderer);
312  if (!isEmptyInline(downcast<RenderInline>(*renderer)))
313  renderer = bidiNextSkippingEmptyInlines(root, renderer, resolver);
314  else {
315  // Never skip empty inlines.
316  if (resolver)
317  resolver->commitExplicitEmbedding();
318  return renderer;
319  }
320  }
321 
322  // FIXME: Unify this with the bidiNext call above.
323  if (renderer && !isIteratorTarget(renderer))
324  renderer = bidiNextSkippingEmptyInlines(root, renderer, resolver);
325 
326  if (resolver)
327  resolver->commitExplicitEmbedding();
328  return renderer;
329 }

527  void addFakeRunIfNecessary(RenderObject& obj, unsigned pos, unsigned end, RenderElement& root, InlineBidiResolver& resolver)
528  {
529  // We only need to add a fake run for a given isolated span once during each call to createBidiRunsForLine.
530  // We'll be called for every span inside the isolated span so we just ignore subsequent calls.
531  // We also avoid creating a fake run until we hit a child that warrants one, e.g. we skip floats.
532  if (RenderBlock::shouldSkipCreatingRunsForObject(obj))
533  return;
534  if (!m_haveAddedFakeRunForRootIsolate) {
535  // obj and pos together denote a single position in the inline, from which the parsing of the isolate will start.
536  // We don't need to mark the end of the run because this is implicit: it is either endOfLine or the end of the
537  // isolate, when we call createBidiRunsForLine it will stop at whichever comes first.
538  addPlaceholderRunForIsolatedInline(resolver, obj, pos, root);
539  }
540  m_haveAddedFakeRunForRootIsolate = true;
541  RenderBlockFlow::appendRunsForObject(nullptr, pos, end, obj, resolver);
542  }
543 
544 private:
545  unsigned m_nestedIsolateCount;
546  bool m_haveAddedFakeRunForRootIsolate;

549 template<>
550 inline void InlineBidiResolver::appendRunInternal()
551 {
552  if (!m_emptyRun && !m_eor.atEnd() && !m_reachedEndOfLine) {
553  // Keep track of when we enter/leave "unicode-bidi: isolate" inlines.
554  // Initialize our state depending on if we're starting in the middle of such an inline.
555  // FIXME: Could this initialize from this->inIsolate() instead of walking up the render tree?
556  IsolateTracker isolateTracker(numberOfIsolateAncestors(m_sor));
557  int start = m_sor.offset();
558  RenderObject* obj = m_sor.renderer();
559  while (obj && obj != m_eor.renderer() && obj != endOfLine.renderer()) {
560  if (isolateTracker.inIsolate())
561  isolateTracker.addFakeRunIfNecessary(*obj, start, obj->length(), *m_sor.root(), *this);
562  else
563  RenderBlockFlow::appendRunsForObject(&m_runs, start, obj->length(), *obj, *this);
564  // FIXME: start/obj should be an InlineIterator instead of two separate variables.
565  start = 0;
566  obj = bidiNextSkippingEmptyInlines(*m_sor.root(), obj, &isolateTracker);
567  }
568  if (obj) {
569  unsigned pos = obj == m_eor.renderer() ? m_eor.offset() : UINT_MAX;
570  if (obj == endOfLine.renderer() && endOfLine.offset() <= pos) {
571  m_reachedEndOfLine = true;
572  pos = endOfLine.offset();
573  }
574  // It's OK to add runs for zero-length RenderObjects, just don't make the run larger than it should be
575  int end = obj->length() ? pos + 1 : 0;
576  if (isolateTracker.inIsolate())
577  isolateTracker.addFakeRunIfNecessary(*obj, start, obj->length(), *m_sor.root(), *this);
578  else
579  RenderBlockFlow::appendRunsForObject(&m_runs, start, end, *obj, *this);
580  }
581 
582  m_eor.increment();
583  m_sor = m_eor;
584  }
585 
586  m_direction = U_OTHER_NEUTRAL;
587  m_status.eor = U_OTHER_NEUTRAL;
588 }
