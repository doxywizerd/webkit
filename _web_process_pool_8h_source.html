626 template<typename T>
627 void WebProcessPool::sendToAllProcesses(const T& message)
628 {
629  size_t processCount = m_processes.size();
630  for (size_t i = 0; i < processCount; ++i) {
631  WebProcessProxy* process = m_processes[i].get();
632  if (process->canSendMessage())
633  process->send(T(message), 0);
634  }
635 }
