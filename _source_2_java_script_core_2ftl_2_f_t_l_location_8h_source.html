135  unsigned hash() const
136  {
137  unsigned result = m_kind;
138 
139  switch (kind()) {
140  case Unprocessed:

142  break;
143 
144  case Register:
145  result ^= u.variable.regIndex;
146  break;
147 
148  case Indirect:
149  result ^= u.variable.regIndex;

152 
153  case Constant:
154  result ^= WTF::IntHash<int64_t>::hash(u.constant);
