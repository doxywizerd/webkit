55 template <typename CharacterType>

61 template <typename CharacterType>
62 inline bool skipOptionalSVGSpaces(const CharacterType*& ptr, const CharacterType* end)
63 {
64  while (ptr < end && isSVGSpace(*ptr))
65  ptr++;
66  return ptr < end;
67 }

69 template <typename CharacterType>
70 inline bool skipOptionalSVGSpacesOrDelimiter(const CharacterType*& ptr, const CharacterType* end, char delimiter = ',')
71 {
72  if (ptr < end && !isSVGSpace(*ptr) && *ptr != delimiter)
73  return false;
74  if (skipOptionalSVGSpaces(ptr, end)) {
75  if (ptr < end && *ptr == delimiter) {
76  ptr++;
77  skipOptionalSVGSpaces(ptr, end);
78  }
79  }
80  return ptr < end;
81 }
