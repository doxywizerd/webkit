38 // Parse RTCP packet of given type. Assumes RTCP header is valid and that there
39 // is excatly one packet of correct type in the buffer.
40 template <typename Packet>
41 bool ParseSinglePacket(const uint8_t* buffer, size_t size, Packet* packet) {
42  rtcp::CommonHeader header;
43  RTC_CHECK(header.Parse(buffer, size));
44  RTC_CHECK_EQ(size, header.NextPacket() - buffer);
45  return packet->Parse(header);
46 }
47 // Same function, but takes raw buffer as single argument instead of pair.
48 template <typename Packet>
49 bool ParseSinglePacket(rtc::ArrayView<const uint8_t> buffer, Packet* packet) {
50  return ParseSinglePacket(buffer.data(), buffer.size(), packet);
51 }
