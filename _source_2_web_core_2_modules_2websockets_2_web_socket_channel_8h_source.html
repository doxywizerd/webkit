133  // It is allowed to send a Blob as a binary frame if hybi-10 protocol is in use. Sending a Blob
134  // can be delayed because it must be read asynchronously. Other types of data (String or
135  // ArrayBuffer) may also be blocked by preceding sending request of a Blob.
136  //
137  // To address this situation, messages to be sent need to be stored in a queue. Whenever a new
138  // data frame is going to be sent, it first must go to the queue. Items in the queue are processed
139  // in the order they were put into the queue. Sending request of a Blob blocks further processing
140  // until the Blob is completely read and sent to the socket stream.
141  enum QueuedFrameType {
142  QueuedFrameTypeString,
143  QueuedFrameTypeVector,
144  QueuedFrameTypeBlob
145  };
146  struct QueuedFrame {
147  WebSocketFrame::OpCode opCode;
148  QueuedFrameType frameType;
149  // Only one of the following items is used, according to the value of frameType.
150  CString stringData;
151  Vector<char> vectorData;
152  RefPtr<Blob> blobData;
153  };
154  void enqueueTextFrame(const CString&);
155  void enqueueRawFrame(WebSocketFrame::OpCode, const char* data, size_t dataLength);
156  void enqueueBlobFrame(WebSocketFrame::OpCode, Blob&);

158  void processOutgoingFrameQueue();

161  enum OutgoingFrameQueueStatus {
162  // It is allowed to put a new item into the queue.
163  OutgoingFrameQueueOpen,
164  // Close frame has already been put into the queue but may not have been sent yet;
165  // m_handle->close() will be called as soon as the queue is cleared. It is not
166  // allowed to put a new item into the queue.
167  OutgoingFrameQueueClosing,
168  // Close frame has been sent or the queue was aborted. It is not allowed to put
169  // a new item to the queue.
170  OutgoingFrameQueueClosed
171  };
