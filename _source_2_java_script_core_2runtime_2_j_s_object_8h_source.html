233  bool canGetIndexQuickly(unsigned i)
234  {
235  Butterfly* butterfly = m_butterfly.get();
236  switch (indexingType()) {
237  case ALL_BLANK_INDEXING_TYPES:
238  case ALL_UNDECIDED_INDEXING_TYPES:
239  return false;
240  case ALL_INT32_INDEXING_TYPES:
241  case ALL_CONTIGUOUS_INDEXING_TYPES:
242  return i < butterfly->vectorLength() && butterfly->contiguous()[i];
243  case ALL_DOUBLE_INDEXING_TYPES: {
244  if (i >= butterfly->vectorLength())
245  return false;
246  double value = butterfly->contiguousDouble()[i];
247  if (value != value)
248  return false;
249  return true;
250  }
251  case ALL_ARRAY_STORAGE_INDEXING_TYPES:
252  return i < butterfly->arrayStorage()->vectorLength() && butterfly->arrayStorage()->m_vector[i];
253  default:
254  RELEASE_ASSERT_NOT_REACHED();
255  return false;
256  }

278  {
279  Butterfly* butterfly = m_butterfly.get();
280  switch (indexingType()) {
281  case ALL_BLANK_INDEXING_TYPES:

297  break;

299  if (result != result)

302  }
303  case ALL_ARRAY_STORAGE_INDEXING_TYPES:
304  if (i < butterfly->arrayStorage()->vectorLength())
305  return butterfly->arrayStorage()->m_vector[i].get();
306  break;
307  default:
308  RELEASE_ASSERT_NOT_REACHED();

310  }

371  void setIndexQuickly(VM& vm, unsigned i, JSValue v)
372  {
373  Butterfly* butterfly = m_butterfly.get();
374  switch (indexingType()) {
375  case ALL_INT32_INDEXING_TYPES: {
376  ASSERT(i < butterfly->vectorLength());
377  if (!v.isInt32()) {
378  convertInt32ToDoubleOrContiguousWhilePerformingSetIndex(vm, i, v);
379  return;
380  }
381  FALLTHROUGH;
382  }
383  case ALL_CONTIGUOUS_INDEXING_TYPES: {
384  ASSERT(i < butterfly->vectorLength());
385  butterfly->contiguous()[i].set(vm, this, v);
386  if (i >= butterfly->publicLength())
387  butterfly->setPublicLength(i + 1);
388  break;
389  }
390  case ALL_DOUBLE_INDEXING_TYPES: {
391  ASSERT(i < butterfly->vectorLength());
392  if (!v.isNumber()) {
393  convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
394  return;
395  }
396  double value = v.asNumber();
397  if (value != value) {
398  convertDoubleToContiguousWhilePerformingSetIndex(vm, i, v);
399  return;
400  }
401  butterfly->contiguousDouble()[i] = value;
402  if (i >= butterfly->publicLength())
403  butterfly->setPublicLength(i + 1);
404  break;
405  }
406  case ALL_ARRAY_STORAGE_INDEXING_TYPES: {
407  ArrayStorage* storage = butterfly->arrayStorage();
408  WriteBarrier<Unknown>& x = storage->m_vector[i];
409  JSValue old = x.get();
410  x.set(vm, this, v);
411  if (!old) {
412  ++storage->m_numValuesInVector;
413  if (i >= storage->length())
414  storage->setLength(i + 1);
415  }
416  break;
417  }
418  default:
419  RELEASE_ASSERT_NOT_REACHED();
420  }

1364 {
