182 inline int compareBinaryKeyData(const Vector<uint8_t>& a, const Vector<uint8_t>& b)
183 {
184  size_t length = std::min(a.size(), b.size());
185 
186  for (size_t i = 0; i < length; ++i) {
187  if (a[i] > b[i])
188  return 1;
189  if (a[i] < b[i])
190  return -1;
191  }
192 
193  if (a.size() == b.size())
194  return 0;

202 inline int compareBinaryKeyData(const ThreadSafeDataBuffer& a, const ThreadSafeDataBuffer& b)
203 {
204  auto* aData = a.data();
205  auto* bData = b.data();
206 
207  // Covers the cases where both pointers are null as well as both pointing to the same buffer.
208  if (aData == bData)
209  return 0;
210 
211  if (aData && !bData)
212  return 1;
213  if (!aData && bData)
214  return -1;
215
