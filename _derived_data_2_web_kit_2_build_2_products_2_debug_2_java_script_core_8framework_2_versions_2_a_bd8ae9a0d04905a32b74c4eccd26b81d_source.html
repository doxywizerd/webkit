190 inline void Heap::decrementDeferralDepthAndGCIfNeeded()
191 {
192  ASSERT(!mayBeGCThread() || m_collectorBelievesThatTheWorldIsStopped);
193  m_deferralDepth--;
194 
195  if (UNLIKELY(m_didDeferGCWork)) {
196  decrementDeferralDepthAndGCIfNeededSlow();
197 
198  // Here are the possible relationships between m_deferralDepth and m_didDeferGCWork.
199  // Note that prior to the call to decrementDeferralDepthAndGCIfNeededSlow,
200  // m_didDeferGCWork had to have been true. Now it can be either false or true. There is
201  // nothing we can reliably assert.
202  //
203  // Possible arrangements of m_didDeferGCWork and !!m_deferralDepth:
204  //
205  // Both false: We popped out of all DeferGCs and we did whatever work was deferred.
206  //
207  // Only m_didDeferGCWork is true: We stopped for GC and the GC did DeferGC. This is
208  // possible because of how we handle the baseline JIT's worklist. It's also perfectly
209  // safe because it only protects reportExtraMemory. We can just ignore this.
210  //
211  // Only !!m_deferralDepth is true: m_didDeferGCWork had been set spuriously. It is only
212  // cleared by decrementDeferralDepthAndGCIfNeededSlow(). So, if we had deferred work but
213  // then decrementDeferralDepth()'d, then we might have the bit set even if we GC'd since
214  // then.
215  //
216  // Both true: We're in a recursive ~DeferGC. We wanted to do something about the
217  // deferred work, but were unable to.
218  }
219 }
