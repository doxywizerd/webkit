47  void setWithinEventLoopInputExtent(bool withinEventLoopInputExtent)
48  {
49  // We can be within two input extents when a nested run loop
50  // processes additional user inputs while the debugger is paused.
51  // However, the debugger should not pause when capturing, and we
52  // should not replay event loop inputs while in a nested run loop.
53  ASSERT(m_withinEventLoopInputExtent != withinEventLoopInputExtent || !(isCapturing() || isReplaying()));
54  m_withinEventLoopInputExtent = withinEventLoopInputExtent;
55  }
