301  EvqVaryingIn, // readonly, fragment shaders only
302  EvqVaryingOut, // vertex shaders only read/write
303  EvqUniform, // Readonly, vertex and fragment
304 
305  EvqVertexIn, // Vertex shader input
306  EvqFragmentOut, // Fragment shader output

387 struct TLayoutQualifier
388 {
389  int location;
390  unsigned int locationsSpecified;
391  TLayoutMatrixPacking matrixPacking;
392  TLayoutBlockStorage blockStorage;
393 
394  // Compute shader layout qualifiers.
395  sh::WorkGroupSize localSize;
396 
397  static TLayoutQualifier create()
398  {
399  TLayoutQualifier layoutQualifier;
400 
401  layoutQualifier.location = -1;
402  layoutQualifier.locationsSpecified = 0;
403  layoutQualifier.matrixPacking = EmpUnspecified;
404  layoutQualifier.blockStorage = EbsUnspecified;
405 
406  layoutQualifier.localSize.fill(-1);
407 
408  return layoutQualifier;
409  }
410 
411  bool isEmpty() const
412  {
413  return location == -1 && matrixPacking == EmpUnspecified &&
414  blockStorage == EbsUnspecified && !localSize.isAnyValueSet();
415  }
416 
417  bool isCombinationValid() const
418  {
419  bool workSizeSpecified = localSize.isAnyValueSet();
420  bool otherLayoutQualifiersSpecified =
421  (location != -1 || matrixPacking != EmpUnspecified || blockStorage != EbsUnspecified);
422 
423  // we can have either the work group size specified, or the other layout qualifiers
424  return !(workSizeSpecified && otherLayoutQualifiersSpecified);
425  }
426 
427  bool isLocalSizeEqual(const sh::WorkGroupSize &localSizeIn) const
428  {
429  return localSize.isWorkGroupSizeMatching(localSizeIn);
430  }
431 };
