43 // Must not grow beyond 3 bits, due to packing in StyleProperties.
44 enum CSSParserMode {
45  HTMLStandardMode,
46  HTMLQuirksMode,
47  // HTML attributes are parsed in quirks mode but also allows internal properties and values.
48  HTMLAttributeMode,
49  // SVG attributes are parsed in quirks mode but rules differ slightly.
50  SVGAttributeMode,
51  // @viewport rules are parsed in standards mode but CSSOM modifications (via StylePropertySet)
52  // must call parseViewportProperties so needs a special mode.
53  CSSViewportRuleMode,
54  // User agent stylesheets are parsed in standards mode but also allows internal properties and values.
55  UASheetMode
56 };
