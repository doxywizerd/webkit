74 // Unlinked instructions are packed in a simple stream format.
75 //
76 // The first byte is always the opcode.
77 // It's followed by an opcode-dependent number of argument values.
78 // The first 3 bits of each value determines the format:
79 //
80 // 5-bit positive integer (1 byte total)
81 // 5-bit negative integer (1 byte total)
82 // 13-bit positive integer (2 bytes total)
83 // 13-bit negative integer (2 bytes total)
84 // 5-bit constant register index, based at 0x40000000 (1 byte total)
85 // 13-bit constant register index, based at 0x40000000 (2 bytes total)
86 // 32-bit raw value (5 bytes total)
87 
88 enum PackedValueType {
89  Positive5Bit = 0,
90  Negative5Bit,
91  Positive13Bit,
92  Negative13Bit,
93  ConstantRegister5Bit,
94  ConstantRegister13Bit,
95  Full32Bit
96 };

109 ALWAYS_INLINE unsigned UnlinkedInstructionStream::Reader::read32()
110 {
111  const unsigned char* data = &m_stream.m_data.data()[m_index];
112  unsigned char type = data[0] >> 5;
113 
114  switch (type) {
115  case Positive5Bit:
116  m_index++;
117  return data[0];
118  case Negative5Bit:
119  m_index++;
120  return 0xffffffe0 | data[0];
121  case Positive13Bit:
122  m_index += 2;
123  return ((data[0] & 0x1F) << 8) | data[1];
124  case Negative13Bit:
125  m_index += 2;
126  return 0xffffe000 | ((data[0] & 0x1F) << 8) | data[1];
127  case ConstantRegister5Bit:
128  m_index++;
129  return 0x40000000 | (data[0] & 0x1F);
130  case ConstantRegister13Bit:
131  m_index += 2;
132  return 0x40000000 | ((data[0] & 0x1F) << 8) | data[1];
133  default:
134  ASSERT(type == Full32Bit);
135  m_index += 5;
136  return data[1] | data[2] << 8 | data[3] << 16 | data[4] << 24;
137  }
138 }

140 ALWAYS_INLINE const UnlinkedInstruction* UnlinkedInstructionStream::Reader::next()
141 {
142  m_unpackedBuffer[0].u.opcode = static_cast<OpcodeID>(read8());
143  unsigned opLength = opcodeLength(m_unpackedBuffer[0].u.opcode);
144  for (unsigned i = 1; i < opLength; ++i)
145  m_unpackedBuffer[i].u.unsignedValue = read32();
146  return m_unpackedBuffer;
