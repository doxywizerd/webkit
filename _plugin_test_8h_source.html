42 #define DEFINE_HAS_MEMBER_CHECK(member, returnType, argumentTypes) \
43 template<typename T, returnType (T::*member) argumentTypes> struct pmf_##member##_helper {}; \
44 template<typename T> no_tag has_member_##member##_helper(...); \
45 template<typename T> yes_tag has_member_##member##_helper(pmf_##member##_helper<T, &T::member >*); \

47 static const bool value = sizeof(has_member_##member##_helper<T>(0)) == sizeof(yes_tag); \
